138  private:
139  template<std::size_t... TIs>
140  [[nodiscard]] ALPAKA_FN_HOST_ACC auto subscript(Vec<DimInt<TDim>, TBufferIdx> index) const -> ReturnType
141  {
142 #if BOOST_COMP_GNUC
143 # pragma GCC diagnostic push
144 # pragma GCC diagnostic ignored "-Wcast-align"
145 #endif
146  auto bp = internal::asBytePtr(p);
147  for(std::size_t i = 0u; i < TDim; i++)
148  {
149  auto const pitch = i < TDim - 1 ? pitchesInBytes[i] : static_cast<TBufferIdx>(sizeof(TElem));
150  bp += index[i] * pitch;
151  }
152  return *reinterpret_cast<TElem*>(bp);
153 #if BOOST_COMP_GNUC
154 # pragma GCC diagnostic pop
155 #endif
156  }

160  {
161  return subscript(i);
162  }

213  template<
214  typename... TAccessModes,
215  typename TViewForwardRef,
216  std::size_t... TPitchIs,
217  std::size_t... TExtentIs>
218  ALPAKA_FN_HOST auto buildViewAccessor(
219  TViewForwardRef&& view,
220  std::index_sequence<TPitchIs...>,
221  std::index_sequence<TExtentIs...>)
222  {
223  using TView = std::decay_t<TViewForwardRef>;
224  static_assert(IsView<TView>::value);
225  using TBufferIdx = Idx<TView>;
226  constexpr auto dim = Dim<TView>::value;
227  using Elem = Elem<TView>;
228  auto p = getPtrNative(view);
229  static_assert(
230  std::is_same_v<decltype(p), Elem const*> || std::is_same_v<decltype(p), Elem*>,
231  "We assume that getPtrNative() returns a raw pointer to the view's elements");
232  static_assert(
233  !std::is_same_v<
234  decltype(p),
235  Elem const*> || std::is_same_v<std::tuple<TAccessModes...>, std::tuple<ReadAccess>>,
236  "When getPtrNative() returns a const raw pointer, the access mode must be ReadAccess");
237  using AccessModeList = typename BuildAccessModeList<TAccessModes...>::type;
238  return Accessor<Elem*, Elem, TBufferIdx, dim, AccessModeList>{
239  const_cast<Elem*>(p), // strip constness, this is handled the the access modes
240  {getPitchBytes<TPitchIs + 1>(view)...},
241  {getExtent<TExtentIs>(view)...}};
242  }
