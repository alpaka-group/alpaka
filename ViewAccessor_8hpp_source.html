126  template<typename TElem, typename TBufferIdx, std::size_t TDim, typename TAccessModes>

132  TElem* p_,

134  Vec<DimInt<TDim>, TBufferIdx> extents_)
135  : p(p_)

141  template<typename TOtherAccessModes>
142  ALPAKA_FN_HOST_ACC Accessor(const Accessor<TElem*, TElem, TBufferIdx, TDim, TOtherAccessModes>& other)
143  : p(other.p)
144  , pitchesInBytes(other.pitchesInBytes)
145  , extents(other.extents)
146  {
147  }
148 
149  private:
150  template<std::size_t... TIs>
151  [[nodiscard]] ALPAKA_FN_HOST_ACC auto subscript(Vec<DimInt<TDim>, TBufferIdx> index) const -> ReturnType
152  {
153 #if BOOST_COMP_GNUC
154 # pragma GCC diagnostic push
155 # pragma GCC diagnostic ignored "-Wcast-align"
156 #endif
157  auto bp = internal::asBytePtr(p);
158  for(std::size_t i = 0u; i < TDim; i++)
159  {
160  const auto pitch = i < TDim - 1 ? pitchesInBytes[i] : static_cast<TBufferIdx>(sizeof(TElem));
161  bp += index[i] * pitch;
162  }
163  return *reinterpret_cast<TElem*>(bp);
164 #if BOOST_COMP_GNUC
165 # pragma GCC diagnostic pop
166 #endif
167  }
168 
169  public:
170  ALPAKA_FN_HOST_ACC auto operator[](Vec<DimInt<TDim>, TBufferIdx> i) const -> ReturnType
171  {
172  return subscript(i);
173  }
174 
175  template<typename... Ts>
176  ALPAKA_FN_HOST_ACC auto operator()(Ts... i) const -> ReturnType
177  {
178  static_assert(sizeof...(Ts) == TDim, "You need to specify TDim indices.");
179  return subscript(Vec<DimInt<TDim>, TBufferIdx>{static_cast<TBufferIdx>(i)...});
180  }
181 
182  TElem* p;
183  Vec<DimInt<TDim - 1>, TBufferIdx> pitchesInBytes;
184  Vec<DimInt<TDim>, TBufferIdx> extents;
185  };

224  ALPAKA_NO_HOST_ACC_WARNING
225  template<
226  typename... TAccessModes,
227  typename TViewForwardRef,
228  std::size_t... TPitchIs,
229  std::size_t... TExtentIs>
230  ALPAKA_FN_HOST_ACC auto buildViewAccessor(
231  TViewForwardRef&& view,
232  std::index_sequence<TPitchIs...>,
233  std::index_sequence<TExtentIs...>)
234  {
235  using TView = std::decay_t<TViewForwardRef>;
236  static_assert(IsView<TView>::value);
237  using TBufferIdx = Idx<TView>;
238  constexpr auto dim = Dim<TView>::value;
239  using Elem = Elem<TView>;
240  auto p = getPtrNative(view);
241  static_assert(
242  std::is_same_v<decltype(p), const Elem*> || std::is_same_v<decltype(p), Elem*>,
243  "We assume that getPtrNative() returns a raw pointer to the view's elements");
244  static_assert(
245  !std::is_same_v<
246  decltype(p),
247  const Elem*> || std::is_same_v<std::tuple<TAccessModes...>, std::tuple<ReadAccess>>,
248  "When getPtrNative() returns a const raw pointer, the access mode must be ReadAccess");
249  using AccessModeList = typename BuildAccessModeList<TAccessModes...>::type;
250  return Accessor<Elem*, Elem, TBufferIdx, dim, AccessModeList>{
251  const_cast<Elem*>(p), // strip constness, this is handled the the access modes
252  {getPitchBytes<TPitchIs + 1>(view)...},
253  {getExtent<TExtentIs>(view)...}};
254  }
