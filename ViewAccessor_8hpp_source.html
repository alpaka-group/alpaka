143  template<typename TOtherAccessModes>
144  ALPAKA_FN_HOST_ACC Accessor(const Accessor<TElem*, TElem, TBufferIdx, TDim, TOtherAccessModes>& other)
145  : p(other.p)
146  , pitchesInBytes(other.pitchesInBytes)
147  , extents(other.extents)
148  {
149  }
150 
151  private:
152  template<std::size_t... TIs>
153  ALPAKA_FN_HOST_ACC auto subscript(Vec<DimInt<TDim>, TBufferIdx> index) const -> ReturnType
154  {
155 #if BOOST_COMP_GNUC
156 # pragma GCC diagnostic push
157 # pragma GCC diagnostic ignored "-Wcast-align"
158 #endif
159  auto bp = internal::asBytePtr(p);
160  for(std::size_t i = 0u; i < TDim; i++)
161  {
162  const auto pitch = i < TDim - 1 ? pitchesInBytes[i] : static_cast<TBufferIdx>(sizeof(TElem));
163  bp += index[i] * pitch;
164  }
165  return *reinterpret_cast<TElem*>(bp);
166 #if BOOST_COMP_GNUC
167 # pragma GCC diagnostic pop
168 #endif
169  }
170 
171  public:
172  ALPAKA_FN_HOST_ACC auto operator[](Vec<DimInt<TDim>, TBufferIdx> i) const -> ReturnType
173  {
174  return subscript(i);
175  }
176 
177  template<typename... Ts>
178  ALPAKA_FN_HOST_ACC auto operator()(Ts... i) const -> ReturnType
179  {
180  static_assert(sizeof...(Ts) == TDim, "You need to specify TDim indices.");
181  return subscript(Vec<DimInt<TDim>, TBufferIdx>{static_cast<TBufferIdx>(i)...});
182  }

185  Vec<DimInt<TDim - 1>, TBufferIdx> pitchesInBytes;
186  Vec<DimInt<TDim>, TBufferIdx> extents;
187  };

226  ALPAKA_NO_HOST_ACC_WARNING
227  template<
228  typename... TAccessModes,
229  typename TViewForwardRef,
230  std::size_t... TPitchIs,
231  std::size_t... TExtentIs>
232  ALPAKA_FN_HOST_ACC auto buildViewAccessor(
233  TViewForwardRef&& view,
234  std::index_sequence<TPitchIs...>,
235  std::index_sequence<TExtentIs...>)
236  {
237  using TView = std::decay_t<TViewForwardRef>;
238  static_assert(IsView<TView>::value, "");
239  using TBufferIdx = Idx<TView>;
240  constexpr auto dim = Dim<TView>::value;
241  using Elem = Elem<TView>;
242  auto p = getPtrNative(view);
243  static_assert(
244  std::is_same<decltype(p), const Elem*>::value || std::is_same<decltype(p), Elem*>::value,
245  "We assume that getPtrNative() returns a raw pointer to the view's elements");
246  static_assert(
247  !std::is_same<decltype(p), const Elem*>::value
248  || std::is_same<std::tuple<TAccessModes...>, std::tuple<ReadAccess>>::value,
249  "When getPtrNative() returns a const raw pointer, the access mode must be ReadAccess");
250  using AccessModeList = typename BuildAccessModeList<TAccessModes...>::type;
251  return Accessor<Elem*, Elem, TBufferIdx, dim, AccessModeList>{
252  const_cast<Elem*>(p), // strip constness, this is handled the the access modes
253  {getPitchBytes<TPitchIs + 1>(view)...},
254  {extent::getExtent<TExtentIs>(view)...}};
255  }
