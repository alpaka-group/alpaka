48  // Don't call this without locking first!
49  auto clean_queues() -> void
50  {
51  // Clean up dead queues
52  auto const start = std::begin(m_queues);
53  auto const old_end = std::end(m_queues);
54  auto const new_end = std::remove_if(start, old_end, [](auto q_ptr) { return q_ptr.expired(); });
55  m_queues.erase(new_end, old_end);
56  }
57 
58  auto register_queue(std::shared_ptr<QueueGenericSyclImpl> const& queue) -> void
59  {
60  std::lock_guard<std::shared_mutex> lock{m_mutex};
61 
62  clean_queues();
63  m_queues.emplace_back(queue);
64  }
65 
66  auto register_dependency(sycl::event event) -> void
67  {
68  std::shared_lock<std::shared_mutex> lock{m_mutex};
69 
70  for(auto& q_ptr : m_queues)
71  {
72  if(auto ptr = q_ptr.lock(); ptr != nullptr)
73  ptr->register_dependency(event);
74  }
75  }
76 
77  auto wait()
78  {
79  std::shared_lock<std::shared_mutex> lock{m_mutex};
80 
81  for(auto& q_ptr : m_queues)
82  {
83  if(auto ptr = q_ptr.lock(); ptr != nullptr)
84  ptr->wait();
85  }
86  }

89  {
