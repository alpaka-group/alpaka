40  namespace detail
41  {
42  class DevGenericSyclImpl
43  {
44  public:
45  DevGenericSyclImpl(sycl::device device, sycl::context context)
46  : m_device{std::move(device)}
47  , m_context{std::move(context)}
48  {
49  }
50 
51  // Don't call this without locking first!
52  auto clean_queues() -> void
53  {
54  // Clean up dead queues
55  auto const start = std::begin(m_queues);
56  auto const old_end = std::end(m_queues);
57  auto const new_end = std::remove_if(start, old_end, [](auto q_ptr) { return q_ptr.expired(); });
58  m_queues.erase(new_end, old_end);
59  }
60 
61  auto register_queue(std::shared_ptr<QueueGenericSyclImpl> const& queue) -> void
62  {
63  std::lock_guard<std::shared_mutex> lock{m_mutex};
64 
65  clean_queues();
66  m_queues.emplace_back(queue);
67  }
68 
69  auto register_dependency(sycl::event event) -> void
70  {
71  std::shared_lock<std::shared_mutex> lock{m_mutex};
72 
73  for(auto& q_ptr : m_queues)
74  {
75  if(auto ptr = q_ptr.lock(); ptr != nullptr)
76  ptr->register_dependency(event);
77  }
78  }
79 
80  auto wait()
81  {
82  std::shared_lock<std::shared_mutex> lock{m_mutex};
83 
84  for(auto& q_ptr : m_queues)
85  {
86  if(auto ptr = q_ptr.lock(); ptr != nullptr)
87  ptr->wait();
88  }
89  }
90 
91  auto get_device() const -> sycl::device
92  {
93  return m_device;
94  }
95 
96  auto get_context() const -> sycl::context
97  {
98  return m_context;
99  }
100 
101  private:
102  sycl::device m_device;
103  sycl::context m_context;
104  std::vector<std::weak_ptr<QueueGenericSyclImpl>> m_queues;
105  std::shared_mutex mutable m_mutex;
106  };
