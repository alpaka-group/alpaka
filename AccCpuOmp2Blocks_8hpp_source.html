55  //! The CPU OpenMP 2.0 block accelerator.
56  //!
57  //! This accelerator allows parallel kernel execution on a CPU device.
58  //! It uses OpenMP 2.0 to implement the grid block parallelism.
59  //! The block idx is restricted to 1x1x1.
60  template<
61  typename TDim,
62  typename TIdx>
63  class AccCpuOmp2Blocks final :
64  public WorkDivMembers<TDim, TIdx>,
65  public gb::IdxGbRef<TDim, TIdx>,
66  public bt::IdxBtZero<TDim, TIdx>,
67  public AtomicHierarchy<
68  AtomicStdLibLock<16>, // grid atomics
69  AtomicOmpBuiltIn, // block atomics
70  AtomicNoOp // thread atomics
71  >,
72  public math::MathStdLib,
73  public BlockSharedMemDynMember<>,
74  public BlockSharedMemStMember<>,
75  public BlockSyncNoOp,
76  public IntrinsicCpu,
77  public rand::RandStdLib,
78  public TimeOmp,
79  public warp::WarpSingleThread,
80  public concepts::Implements<ConceptAcc, AccCpuOmp2Blocks<TDim, TIdx>>
81  {
82  static_assert(
83  sizeof(TIdx) >= sizeof(int),
84  "Index type is not supported, consider using int or a larger type.");
85 
86  public:
87  // Partial specialization with the correct TDim and TIdx is not allowed.
88  template<typename TDim2, typename TIdx2, typename TKernelFnObj, typename... TArgs>
89  friend class ::alpaka::TaskKernelCpuOmp2Blocks;
90 
91  private:
92  template<typename TWorkDiv>
93  ALPAKA_FN_HOST AccCpuOmp2Blocks(TWorkDiv const& workDiv, std::size_t const& blockSharedMemDynSizeBytes)
94  : WorkDivMembers<TDim, TIdx>(workDiv)
95  , gb::IdxGbRef<TDim, TIdx>(m_gridBlockIdx)
96  , bt::IdxBtZero<TDim, TIdx>()
97  , AtomicHierarchy<
98  AtomicStdLibLock<16>, // atomics between grids
99  AtomicOmpBuiltIn, // atomics between blocks
