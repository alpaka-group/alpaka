57  //! The CPU OpenMP 2.0 block accelerator.
58  //!
59  //! This accelerator allows parallel kernel execution on a CPU device.
60  //! It uses OpenMP 2.0 to implement the grid block parallelism.
61  //! The block idx is restricted to 1x1x1.
62  template<
63  typename TDim,
64  typename TIdx>
65  class AccCpuOmp2Blocks final :
66  public WorkDivMembers<TDim, TIdx>,
67  public gb::IdxGbRef<TDim, TIdx>,
68  public bt::IdxBtZero<TDim, TIdx>,
69  public AtomicHierarchy<
70  AtomicCpu, // grid atomics
71  AtomicOmpBuiltIn, // block atomics
72  AtomicNoOp // thread atomics
73  >,
74  public math::MathStdLib,
75  public BlockSharedMemDynMember<>,
76  public BlockSharedMemStMember<>,
77  public BlockSyncNoOp,
78  public IntrinsicCpu,
79  public MemFenceOmp2Blocks,
80  public rand::RandStdLib,
81  public TimeOmp,
82  public warp::WarpSingleThread,
83  public concepts::Implements<ConceptAcc, AccCpuOmp2Blocks<TDim, TIdx>>
84  {
85  static_assert(
86  sizeof(TIdx) >= sizeof(int),
87  "Index type is not supported, consider using int or a larger type.");
88 
89  public:
90  // Partial specialization with the correct TDim and TIdx is not allowed.
91  template<typename TDim2, typename TIdx2, typename TKernelFnObj, typename... TArgs>
92  friend class ::alpaka::TaskKernelCpuOmp2Blocks;

94  AccCpuOmp2Blocks(AccCpuOmp2Blocks const&) = delete;

97  auto operator=(AccCpuOmp2Blocks&&) -> AccCpuOmp2Blocks& = delete;
98 
99  private:
100  template<typename TWorkDiv>
101  ALPAKA_FN_HOST AccCpuOmp2Blocks(TWorkDiv const& workDiv, std::size_t const& blockSharedMemDynSizeBytes)
102  : WorkDivMembers<TDim, TIdx>(workDiv)
103  , gb::IdxGbRef<TDim, TIdx>(m_gridBlockIdx)
104  , bt::IdxBtZero<TDim, TIdx>()

113  , BlockSyncNoOp()
