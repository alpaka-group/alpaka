43  //! The Omp5 memory copy trait.
44  template<
45  typename TDim,
46  typename TViewDst,
47  typename TViewSrc,
48  typename TExtent>
49  struct TaskCopyOmp5
50  {
51  static_assert(
52  !std::is_const<TViewDst>::value,
53  "The destination view can not be const!");
54 
55  static_assert(
56  Dim<TViewSrc>::value == TDim::value,
57  "The source view is required to have dimensionality TDim!");
58  static_assert(
59  Dim<TViewDst>::value == Dim<TViewSrc>::value,
60  "The source and the destination view are required to have the same dimensionality!");
61  static_assert(
62  Dim<TViewDst>::value == Dim<TExtent>::value,
63  "The views and the extent are required to have the same dimensionality!");
64  // TODO: Maybe check for Idx of TViewDst and TViewSrc to have greater or equal range than TExtent.
65  static_assert(
66  std::is_same<Elem<TViewDst>, typename std::remove_const<Elem<TViewSrc>>::type>::value,
67  "The source and the destination view are required to have the same element type!");
68 
69  using Idx = alpaka::Idx<TExtent>;
70 
71  //-----------------------------------------------------------------------------
72  ALPAKA_FN_HOST TaskCopyOmp5(
73  TViewDst & viewDst,
74  TViewSrc const & viewSrc,
75  TExtent const & extent,
76  int const & iDstDevice,
77  int const & iSrcDevice) :
78  m_iDstDevice(iDstDevice),
79  m_iSrcDevice(iSrcDevice),
80  m_extent(castVec<size_t>(extent::getExtentVec(extent))),
81  m_dstPitchBytes(castVec<size_t>(view::getPitchBytesVec(viewDst))),
82  m_srcPitchBytes(castVec<size_t>(view::getPitchBytesVec(viewSrc))),
83  m_dstMemNative(reinterpret_cast<void *>(view::getPtrNative(viewDst))),
84  m_srcMemNative(reinterpret_cast<void const *>(view::getPtrNative(viewSrc)))
85  {
86 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
87  const auto dstExtent(castVec<size_t>(extent::getExtentVec(viewDst)));
88  const auto srcExtent(castVec<size_t>(extent::getExtentVec(viewSrc)));
89  for(auto i = static_cast<decltype(TDim::value)>(0u); i < TDim::value; ++i)
90  {
91  ALPAKA_ASSERT(m_extent[i] <= dstExtent[i]);
92  ALPAKA_ASSERT(m_extent[i] <= srcExtent[i]);
93  }
94  std::cout << "TaskCopyOmp5<" << TDim::value << ",...>::ctor\tsrcExtent="
95  << srcExtent << ", dstExtent=" << dstExtent << std::endl;
96 #endif
97  }
98 
99 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
100  //-----------------------------------------------------------------------------
101  ALPAKA_FN_HOST auto printDebug() const
102  -> void
103  {
104  std::cout << __func__
105  << " ddev: " << m_iDstDevice
106  << " ew: " << m_extent
107  << " dptr: " << m_dstMemNative
108  << " sdev: " << m_iSrcDevice
109  << " sptr: " << m_srcMemNative
110  << std::endl;
111  }
112 #endif
113  int m_iDstDevice;
114  int m_iSrcDevice;
115  Vec<TDim, size_t> m_extent;
116  Vec<TDim, size_t> m_dstPitchBytes;
117  Vec<TDim, size_t> m_srcPitchBytes;
118  void * m_dstMemNative;
119  void const * m_srcMemNative;
120 
121  //-----------------------------------------------------------------------------
122  //! Executes the kernel function object.
123  ALPAKA_FN_HOST auto operator()() const
124  -> void
125  {
126  ALPAKA_DEBUG_FULL_LOG_SCOPE;
127 
128 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
129  printDebug();
130 #endif
131  constexpr auto lastDim = TDim::value - 1;
132 
133  if(m_extent.prod() > 0)
134  {
135  // offsets == 0 by ptr shift (?)
136  auto dstOffset(Vec<TDim, size_t>::zeros());
137  auto srcOffset(Vec<TDim, size_t>::zeros());
138 
139  auto dstExtentFull(Vec<TDim, size_t>::zeros());
140  auto srcExtentFull(Vec<TDim, size_t>::zeros());
141 
142  const size_t elementSize =
143  ( m_dstPitchBytes[0]%sizeof(Elem<TViewDst>) || m_srcPitchBytes[0]%sizeof(Elem<TViewDst>) )
144  ? 1 : sizeof(Elem<TViewDst>);
145 
146  dstExtentFull[lastDim] = m_dstPitchBytes[lastDim]/elementSize;
147  srcExtentFull[lastDim] = m_srcPitchBytes[lastDim]/elementSize;
148  for(int i = lastDim - 1; i >= 0; --i)
149  {
150  dstExtentFull[i] = m_dstPitchBytes[i]/m_dstPitchBytes[i+1];
151  srcExtentFull[i] = m_srcPitchBytes[i]/m_srcPitchBytes[i+1];
152  }
153 
154  ALPAKA_OMP5_CHECK(
155  omp_target_memcpy_rect(
156  m_dstMemNative, const_cast<void*>(m_srcMemNative),
157  sizeof(Elem<TViewDst>),
158  TDim::value,
159  reinterpret_cast<size_t const *>(&m_extent),
160  reinterpret_cast<size_t const *>(&dstOffset),
161  reinterpret_cast<size_t const *>(&srcOffset),
162  reinterpret_cast<size_t const *>(&dstExtentFull),
163  reinterpret_cast<size_t const *>(&srcExtentFull),
164  m_iDstDevice, m_iSrcDevice));
165  }
166  }
167  };

201  //-----------------------------------------------------------------------------
202  ALPAKA_FN_HOST TaskCopyOmp5(
203  TViewDst & viewDst,
204  TViewSrc const & viewSrc,
205  TExtent const & extent,
206  int const & iDstDevice,
207  int const & iSrcDevice) :
208  m_iDstDevice(iDstDevice),
209  m_iSrcDevice(iSrcDevice),
210 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
211  m_extentWidth(extent::getWidth(extent)),
212  m_dstWidth(static_cast<Idx>(extent::getWidth(viewDst))),
213  m_srcWidth(static_cast<Idx>(extent::getWidth(viewSrc))),
214 #endif
215  m_extentWidthBytes(extent::getWidth(extent) * static_cast<Idx>(sizeof(Elem<TViewDst>))),
216  m_dstMemNative(reinterpret_cast<void *>(view::getPtrNative(viewDst))),
217  m_srcMemNative(reinterpret_cast<void const *>(view::getPtrNative(viewSrc)))
218  {
219 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
220  ALPAKA_ASSERT(m_extentWidth <= m_dstWidth);
221  ALPAKA_ASSERT(m_extentWidth <= m_srcWidth);
222 #endif
223  }
224 
225 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
226  //-----------------------------------------------------------------------------
227  ALPAKA_FN_HOST auto printDebug() const
228  -> void
229  {
230  std::cout << __func__
231  << " ddev: " << m_iDstDevice
232  << " ew: " << m_extentWidth
233  << " ewb: " << m_extentWidthBytes
234  << " dw: " << m_dstWidth
235  << " dptr: " << m_dstMemNative
236  << " sdev: " << m_iSrcDevice
237  << " sw: " << m_srcWidth
238  << " sptr: " << m_srcMemNative
239  << std::endl;
240  }
241 #endif
242  int m_iDstDevice;
243  int m_iSrcDevice;
244 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
245  Idx m_extentWidth;
246  Idx m_dstWidth;
247  Idx m_srcWidth;
248 #endif
249  Idx m_extentWidthBytes;
250  void * m_dstMemNative;
251  void const * m_srcMemNative;
252 
253  //-----------------------------------------------------------------------------
254  //! Executes the kernel function object.
255  ALPAKA_FN_HOST auto operator()() const
256  -> void
257  {
258  ALPAKA_DEBUG_FULL_LOG_SCOPE;
259 
260 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
261  printDebug();
262 #endif
263  if(m_extentWidthBytes == 0)
264  {
265  return;
266  }
267 
268  ALPAKA_OMP5_CHECK(
269  omp_target_memcpy(
270  m_dstMemNative, const_cast<void*>(m_srcMemNative),
271  static_cast<std::size_t>(m_extentWidthBytes),
272  0,0, m_iDstDevice, m_iSrcDevice));
273  }
274  };
