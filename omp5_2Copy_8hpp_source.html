45  //! The Omp5 memory copy trait.
46  template<
47  typename TDim,
48  typename TViewDst,
49  typename TViewSrc,
50  typename TExtent>
51  struct TaskCopyOmp5
52  {
53  static_assert(
54  !std::is_const<TViewDst>::value,
55  "The destination view can not be const!");
56 
57  static_assert(
58  dim::Dim<TViewSrc>::value == TDim::value,
59  "The source view is required to have dimensionality TDim!");
60  static_assert(
61  dim::Dim<TViewDst>::value == dim::Dim<TViewSrc>::value,
62  "The source and the destination view are required to have the same dimensionality!");
63  static_assert(
64  dim::Dim<TViewDst>::value == dim::Dim<TExtent>::value,
65  "The views and the extent are required to have the same dimensionality!");
66  // TODO: Maybe check for Idx of TViewDst and TViewSrc to have greater or equal range than TExtent.
67  static_assert(
68  std::is_same<elem::Elem<TViewDst>, typename std::remove_const<elem::Elem<TViewSrc>>::type>::value,
69  "The source and the destination view are required to have the same element type!");
70 
71  using Idx = idx::Idx<TExtent>;
72 
73  //-----------------------------------------------------------------------------
74  ALPAKA_FN_HOST TaskCopyOmp5(
75  TViewDst & viewDst,
76  TViewSrc const & viewSrc,
77  TExtent const & extent,
78  int const & iDstDevice,
79  int const & iSrcDevice) :
80  m_iDstDevice(iDstDevice),
81  m_iSrcDevice(iSrcDevice),
82  m_extent(cast<size_t>(extent::getExtentVec(extent))),
83  m_dstPitchBytes(cast<size_t>(mem::view::getPitchBytesVec(viewDst))),
84  m_srcPitchBytes(cast<size_t>(mem::view::getPitchBytesVec(viewSrc))),
85  m_dstMemNative(reinterpret_cast<void *>(mem::view::getPtrNative(viewDst))),
86  m_srcMemNative(reinterpret_cast<void const *>(mem::view::getPtrNative(viewSrc)))
87  {
88 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
89  const auto dstExtent(cast<size_t>(extent::getExtentVec(viewDst)));
90  const auto srcExtent(cast<size_t>(extent::getExtentVec(viewSrc)));
91  for(auto i = static_cast<decltype(TDim::value)>(0u); i < TDim::value; ++i)
92  {
93  ALPAKA_ASSERT(m_extent[i] <= dstExtent[i]);
94  ALPAKA_ASSERT(m_extent[i] <= srcExtent[i]);
95  }
96  std::cout << "TaskCopyOmp5<" << TDim::value << ",...>::ctor\tsrcExtent="
97  << srcExtent << ", dstExtent=" << dstExtent << std::endl;
98 #endif
99  }
100 
101 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
102  //-----------------------------------------------------------------------------
103  ALPAKA_FN_HOST auto printDebug() const
104  -> void
105  {
106  std::cout << __func__
107  << " ddev: " << m_iDstDevice
108  << " ew: " << m_extent
109  << " dptr: " << m_dstMemNative
110  << " sdev: " << m_iSrcDevice
111  << " sptr: " << m_srcMemNative
112  << std::endl;
113  }
114 #endif
115  int m_iDstDevice;
116  int m_iSrcDevice;
117  Vec<TDim, size_t> m_extent;
118  Vec<TDim, size_t> m_dstPitchBytes;
119  Vec<TDim, size_t> m_srcPitchBytes;
120  void * m_dstMemNative;
121  void const * m_srcMemNative;
122 
123  //-----------------------------------------------------------------------------
124  //! Executes the kernel function object.
125  ALPAKA_FN_HOST auto operator()() const
126  -> void
127  {
128  ALPAKA_DEBUG_FULL_LOG_SCOPE;
129 
130 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
131  printDebug();
132 #endif
133  constexpr auto lastDim = TDim::value - 1;
134 
135  if(m_extent.prod() > 0)
136  {
137  // offsets == 0 by ptr shift (?)
138  auto dstOffset(Vec<TDim, size_t>::zeros());
139  auto srcOffset(Vec<TDim, size_t>::zeros());
140 
141  auto dstExtentFull(Vec<TDim, size_t>::zeros());
142  auto srcExtentFull(Vec<TDim, size_t>::zeros());
143 
144  const size_t elementSize =
145  ( m_dstPitchBytes[0]%sizeof(elem::Elem<TViewDst>) || m_srcPitchBytes[0]%sizeof(elem::Elem<TViewDst>) )
146  ? 1 : sizeof(elem::Elem<TViewDst>);
147 
148  dstExtentFull[lastDim] = m_dstPitchBytes[lastDim]/elementSize;
149  srcExtentFull[lastDim] = m_srcPitchBytes[lastDim]/elementSize;
150  for(int i = lastDim - 1; i >= 0; --i)
151  {
152  dstExtentFull[i] = m_dstPitchBytes[i]/m_dstPitchBytes[i+1];
153  srcExtentFull[i] = m_srcPitchBytes[i]/m_srcPitchBytes[i+1];
154  }
155 
156  ALPAKA_OMP5_CHECK(
157  omp_target_memcpy_rect(
158  m_dstMemNative, const_cast<void*>(m_srcMemNative),
159  sizeof(elem::Elem<TViewDst>),
160  TDim::value,
161  reinterpret_cast<size_t const *>(&m_extent),
162  reinterpret_cast<size_t const *>(&dstOffset),
163  reinterpret_cast<size_t const *>(&srcOffset),
164  reinterpret_cast<size_t const *>(&dstExtentFull),
165  reinterpret_cast<size_t const *>(&srcExtentFull),
166  m_iDstDevice, m_iSrcDevice));
167  }
168  }
169  };
