38  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
39  struct TaskCopyOmp5
40  {
41  template<typename TViewDstFwd>
42  ALPAKA_FN_HOST TaskCopyOmp5(
43  TViewDstFwd&& viewDst,
44  TViewSrc const& viewSrc,
45  TExtent const& extent,
46  int const& iDstDevice,
47  int const& iSrcDevice)
48  : m_iDstDevice(iDstDevice)
49  , m_iSrcDevice(iSrcDevice)
50  , m_extent(castVec<size_t>(getExtentVec(extent)))
51  , m_dstPitchBytes(castVec<size_t>(getPitchBytesVec(viewDst)))
52  , m_srcPitchBytes(castVec<size_t>(getPitchBytesVec(viewSrc)))
53  , m_dstMemNative(reinterpret_cast<void*>(getPtrNative(viewDst)))
54  , m_srcMemNative(reinterpret_cast<void const*>(getPtrNative(viewSrc)))
55  {
56 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
57  auto const dstExtent(castVec<size_t>(getExtentVec(viewDst)));
58  auto const srcExtent(castVec<size_t>(getExtentVec(viewSrc)));
59  for(auto i = static_cast<decltype(TDim::value)>(0u); i < TDim::value; ++i)
60  {
61  ALPAKA_ASSERT(m_extent[i] <= dstExtent[i]);
62  ALPAKA_ASSERT(m_extent[i] <= srcExtent[i]);
63  }
64  std::cout << "TaskCopyOmp5<" << TDim::value << ",...>::ctor\tsrcExtent=" << srcExtent
65  << ", dstExtent=" << dstExtent << std::endl;
66 # endif
67  }
68

70  ALPAKA_FN_HOST auto printDebug() const -> void
71  {
72  std::cout << __func__ << " ddev: " << m_iDstDevice << " ew: " << m_extent
73  << " dptr: " << m_dstMemNative << " sdev: " << m_iSrcDevice << " sptr: " << m_srcMemNative
74  << std::endl;
75  }
76 # endif
77  int m_iDstDevice;
78  int m_iSrcDevice;
79  Vec<TDim, size_t> m_extent;
80  Vec<TDim, size_t> m_dstPitchBytes;
81  Vec<TDim, size_t> m_srcPitchBytes;
82  void* m_dstMemNative;
83  void const* m_srcMemNative;
84 
85  //! Executes the kernel function object.
86  ALPAKA_FN_HOST auto operator()() const -> void
87  {
88  ALPAKA_DEBUG_FULL_LOG_SCOPE;
89 
90 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
91  printDebug();
92 # endif
93  constexpr auto lastDim = TDim::value - 1;
94 
95  if(m_extent.prod() > 0)
96  {
97  // offsets == 0 by ptr shift (?)
98  auto dstOffset(Vec<TDim, size_t>::zeros());
99  auto srcOffset(Vec<TDim, size_t>::zeros());
100 
101  auto dstExtentFull(Vec<TDim, size_t>::zeros());
102  auto srcExtentFull(Vec<TDim, size_t>::zeros());
103 
104  const size_t elementSize
105  = (m_dstPitchBytes[0] % sizeof(Elem<TViewDst>) || m_srcPitchBytes[0] % sizeof(Elem<TViewDst>))
106  ? 1
107  : sizeof(Elem<TViewDst>);
108 
109  dstExtentFull[lastDim] = m_dstPitchBytes[lastDim] / elementSize;
110  srcExtentFull[lastDim] = m_srcPitchBytes[lastDim] / elementSize;
111  for(int i = lastDim - 1; i >= 0; --i)
112  {
113  dstExtentFull[i] = m_dstPitchBytes[i] / m_dstPitchBytes[i + 1];
114  srcExtentFull[i] = m_srcPitchBytes[i] / m_srcPitchBytes[i + 1];
115  }
116 
117  ALPAKA_OMP5_CHECK(omp_target_memcpy_rect(
118  m_dstMemNative,
119  const_cast<void*>(m_srcMemNative),
120  sizeof(Elem<TViewDst>),
121  TDim::value,
122  reinterpret_cast<size_t const*>(&m_extent),
123  reinterpret_cast<size_t const*>(&dstOffset),
124  reinterpret_cast<size_t const*>(&srcOffset),
125  reinterpret_cast<size_t const*>(&dstExtentFull),
126  reinterpret_cast<size_t const*>(&srcExtentFull),
127  m_iDstDevice,
128  m_iSrcDevice));
129  }
130  }
