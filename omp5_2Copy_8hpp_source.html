37  //! The Omp5 memory copy trait.
38  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
39  struct TaskCopyOmp5
40  {
41  static_assert(!std::is_const_v<TViewDst>, "The destination view can not be const!");
42 
43  static_assert(
44  Dim<TViewSrc>::value == TDim::value,
45  "The source view is required to have dimensionality TDim!");
46  static_assert(
47  Dim<TViewDst>::value == Dim<TViewSrc>::value,
48  "The source and the destination views are required to have the same dimensionality!");
49  static_assert(
50  Dim<TViewDst>::value == Dim<TExtent>::value,
51  "The views and the extent are required to have the same dimensionality!");
52  // TODO: Maybe check for Idx of TViewDst and TViewSrc to have greater or equal range than TExtent.
53  static_assert(
54  std::is_same_v<Elem<TViewDst>, typename std::remove_const<Elem<TViewSrc>>::type>,
55  "The source and the destination views are required to have the same element type!");
56 
57  using Idx = alpaka::Idx<TExtent>;
58 
59  ALPAKA_FN_HOST TaskCopyOmp5(
60  TViewDst& viewDst,
61  TViewSrc const& viewSrc,
62  TExtent const& extent,
63  int const& iDstDevice,
64  int const& iSrcDevice)
65  : m_iDstDevice(iDstDevice)
66  , m_iSrcDevice(iSrcDevice)
67  , m_extent(castVec<size_t>(getExtentVec(extent)))
68  , m_dstPitchBytes(castVec<size_t>(getPitchBytesVec(viewDst)))
69  , m_srcPitchBytes(castVec<size_t>(getPitchBytesVec(viewSrc)))
70  , m_dstMemNative(reinterpret_cast<void*>(getPtrNative(viewDst)))
71  , m_srcMemNative(reinterpret_cast<void const*>(getPtrNative(viewSrc)))
72  {
73 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
74  const auto dstExtent(castVec<size_t>(getExtentVec(viewDst)));
75  const auto srcExtent(castVec<size_t>(getExtentVec(viewSrc)));
76  for(auto i = static_cast<decltype(TDim::value)>(0u); i < TDim::value; ++i)
77  {
78  ALPAKA_ASSERT(m_extent[i] <= dstExtent[i]);
79  ALPAKA_ASSERT(m_extent[i] <= srcExtent[i]);
80  }
81  std::cout << "TaskCopyOmp5<" << TDim::value << ",...>::ctor\tsrcExtent=" << srcExtent
82  << ", dstExtent=" << dstExtent << std::endl;
83 # endif
84  }
85 
86 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
87  ALPAKA_FN_HOST auto printDebug() const -> void
88  {
89  std::cout << __func__ << " ddev: " << m_iDstDevice << " ew: " << m_extent
90  << " dptr: " << m_dstMemNative << " sdev: " << m_iSrcDevice << " sptr: " << m_srcMemNative
91  << std::endl;
92  }
93 # endif
94  int m_iDstDevice;
95  int m_iSrcDevice;
96  Vec<TDim, size_t> m_extent;
97  Vec<TDim, size_t> m_dstPitchBytes;
98  Vec<TDim, size_t> m_srcPitchBytes;
99  void* m_dstMemNative;
100  void const* m_srcMemNative;
101 
102  //! Executes the kernel function object.
103  ALPAKA_FN_HOST auto operator()() const -> void
104  {
105  ALPAKA_DEBUG_FULL_LOG_SCOPE;
106 
107 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
108  printDebug();
109 # endif
110  constexpr auto lastDim = TDim::value - 1;
111 
112  if(m_extent.prod() > 0)
113  {
114  // offsets == 0 by ptr shift (?)
115  auto dstOffset(Vec<TDim, size_t>::zeros());
116  auto srcOffset(Vec<TDim, size_t>::zeros());
117 
118  auto dstExtentFull(Vec<TDim, size_t>::zeros());
119  auto srcExtentFull(Vec<TDim, size_t>::zeros());
120 
121  const size_t elementSize
122  = (m_dstPitchBytes[0] % sizeof(Elem<TViewDst>) || m_srcPitchBytes[0] % sizeof(Elem<TViewDst>))
123  ? 1
124  : sizeof(Elem<TViewDst>);
125 
126  dstExtentFull[lastDim] = m_dstPitchBytes[lastDim] / elementSize;
127  srcExtentFull[lastDim] = m_srcPitchBytes[lastDim] / elementSize;
128  for(int i = lastDim - 1; i >= 0; --i)
129  {
130  dstExtentFull[i] = m_dstPitchBytes[i] / m_dstPitchBytes[i + 1];
131  srcExtentFull[i] = m_srcPitchBytes[i] / m_srcPitchBytes[i + 1];
132  }
133 
134  ALPAKA_OMP5_CHECK(omp_target_memcpy_rect(
135  m_dstMemNative,
136  const_cast<void*>(m_srcMemNative),
137  sizeof(Elem<TViewDst>),
138  TDim::value,
139  reinterpret_cast<size_t const*>(&m_extent),
140  reinterpret_cast<size_t const*>(&dstOffset),
141  reinterpret_cast<size_t const*>(&srcOffset),
142  reinterpret_cast<size_t const*>(&dstExtentFull),
143  reinterpret_cast<size_t const*>(&srcExtentFull),
144  m_iDstDevice,
145  m_iSrcDevice));
146  }
147  }
148  };
