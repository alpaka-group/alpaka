25  //! The memory view to wrap plain pointers.

27  class ViewPlainPtr final : public internal::ViewAccessOps<ViewPlainPtr<TDev, TElem, TDim, TIdx>>
28  {

30 
31  using Dev = alpaka::Dev<TDev>;
32 
33  public:
34  template<typename TExtent>
35  ALPAKA_FN_HOST ViewPlainPtr(TElem* pMem, Dev dev, TExtent const& extent = TExtent())
36  : m_pMem(pMem)
37  , m_dev(std::move(dev))
38  , m_extentElements(getExtentVecEnd<TDim>(extent))
39  , m_pitchBytes(detail::calculatePitchesFromExtents<TElem>(m_extentElements))
40  {
41  }

45  : m_pMem(pMem)
46  , m_dev(dev)
47  , m_extentElements(getExtentVecEnd<TDim>(extent))
48  , m_pitchBytes(subVecEnd<TDim>(static_cast<Vec<TDim, TIdx>>(pitchBytes)))
49  {
50  }
