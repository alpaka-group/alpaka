30  /** Class basic Philox family counter-based PRNG
31  *
32  * Checks the validity of passed-in parameters and calls the \a TBackend methods to perform N rounds of the
33  * Philox shuffle.
34  *
35  * @tparam TBackend device-dependent backend, specifies the array types
36  * @tparam TParams Philox algorithm parameters \sa PhiloxParams
37  */

39  class PhiloxStateless : public PhiloxConstants<TParams>
40  {

43  return TParams::rounds;

62  using Counter = typename TBackend::Counter;
63  using Key = typename TBackend::Key;
64  using Constants = PhiloxConstants<TParams>;

66  protected:
67  /** Single round of the Philox shuffle
68  *
69  * @param counter state of the counter
70  * @param key value of the key
71  * @return shuffled counter
72  */
73  static ALPAKA_FN_HOST_ACC auto singleRound(Counter const& counter, Key const& key)
74  {
75  std::uint32_t H0, L0, H1, L1;
76  multiplyAndSplit64to32(counter[0], Constants::MULTIPLITER_4x32_0(), H0, L0);
77  multiplyAndSplit64to32(counter[2], Constants::MULTIPLITER_4x32_1(), H1, L1);
78  return Counter{H1 ^ counter[1] ^ key[0], L1, H0 ^ counter[3] ^ key[1], L0};
79  }
80 
81  /** Bump the \a key by the Weyl sequence step parameter
82  *
83  * @param key the key to be bumped
84  * @return the bumped key
85  */
86  static ALPAKA_FN_HOST_ACC auto bumpKey(Key const& key)
87  {
88  return Key{key[0] + Constants::WEYL_32_0(), key[1] + Constants::WEYL_32_1()};
89  }
90 
91  /** Performs N rounds of the Philox shuffle
92  *
93  * @param counter_in initial state of the counter
94  * @param key_in initial state of the key
95  * @return result of the PRNG shuffle; has the same size as the counter
96  */
97  static ALPAKA_FN_HOST_ACC auto nRounds(Counter const& counter_in, Key const& key_in) -> Counter
98  {
99  Key key{key_in};
100  Counter counter = singleRound(counter_in, key);
101 
102  ALPAKA_UNROLL(numRounds())
103  for(unsigned int n = 0; n < numRounds(); ++n)
104  {
105  key = bumpKey(key);
106  counter = singleRound(counter, key);
107  }
108 
109  return counter;
110  }
111 
112  public:
113  /** Generates a random number (\p TCounterSize x32-bit)
114  *
115  * @param counter initial state of the counter
116  * @param key initial state of the key
117  * @return result of the PRNG shuffle; has the same size as the counter
118  */
119  static ALPAKA_FN_HOST_ACC auto generate(Counter const& counter, Key const& key) -> Counter
120  {
121  return nRounds(counter, key);
122  }
123  };
