40  //! Dynamic block shared memory provider using fixed-size
41  //! member array to allocate memory on the stack or in shared
42  //! memory.
43  template<std::size_t TStaticAllocKiB = BlockSharedDynMemberAllocKiB>
44  class alignas(core::vectorization::defaultAlignment) BlockSharedMemDynMember
45  : public concepts::Implements<ConceptBlockSharedDyn, BlockSharedMemDynMember<TStaticAllocKiB>>
46  {
47  public:
48  BlockSharedMemDynMember(std::size_t sizeBytes) : m_dynPitch(getPitch(sizeBytes))
49  {
50  ALPAKA_ASSERT_OFFLOAD(static_cast<std::uint32_t>(sizeBytes) <= staticAllocBytes());
51  }
52 
53  auto dynMemBegin() const -> uint8_t*
54  {
55  return std::data(m_mem);
56  }
57 
58  /*! \return the pointer to the begin of data after the portion allocated as dynamical shared memory.

60  auto staticMemBegin() const -> uint8_t*
61  {
62  return std::data(m_mem) + m_dynPitch;
63  }
64 
65  /*! \return the remaining capacity for static block shared memory,
66  returns a 32-bit type for register efficiency on GPUs
67  */
68  auto staticMemCapacity() const -> std::uint32_t
69  {
70  return staticAllocBytes() - m_dynPitch;
71  }
72 
73  //! \return size of statically allocated memory available for both
74  //! dynamic and static shared memory. Value is of a 32-bit type
75  //! for register efficiency on GPUs
76  static constexpr auto staticAllocBytes() -> std::uint32_t
77  {
78  return detail::BlockSharedMemDynMemberStatic<TStaticAllocKiB>::staticAllocBytes;

83  {

85  return static_cast<std::uint32_t>((sizeBytes / alignment + (sizeBytes % alignment > 0u)) * alignment);
