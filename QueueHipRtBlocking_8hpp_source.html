53  //! The HIP RT blocking queue implementation.
54  class QueueHipRtBlockingImpl final
55  {
56  public:
57  //-----------------------------------------------------------------------------
58  ALPAKA_FN_HOST QueueHipRtBlockingImpl(
59  dev::DevHipRt const & dev) :
60  m_dev(dev),
61  m_HipQueue()
62  {
63  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
64 
65  // Set the current device.
66  ALPAKA_HIP_RT_CHECK(
67  hipSetDevice(
68  m_dev.m_iDevice));
69  // - hipStreamDefault: Default queue creation flag.
70  // - hipStreamNonBlocking: Specifies that work running in the created queue may run concurrently with work in queue 0 (the NULL queue),
71  // and that the created queue should perform no implicit synchronization with queue 0.
72  // Create the queue on the current device.
73  // NOTE: hipStreamNonBlocking is required to match the semantic implemented in the alpaka CPU queue.
74  // It would be too much work to implement implicit default queue synchronization on CPU.
75  ALPAKA_HIP_RT_CHECK(
76  hipStreamCreateWithFlags(
77  &m_HipQueue,
78  hipStreamNonBlocking));
79  }
80  //-----------------------------------------------------------------------------
81  QueueHipRtBlockingImpl(QueueHipRtBlockingImpl const &) = delete;
82  //-----------------------------------------------------------------------------
83  QueueHipRtBlockingImpl(QueueHipRtBlockingImpl &&) = delete;

85  auto operator=(QueueHipRtBlockingImpl const &) -> QueueHipRtBlockingImpl & = delete;
86  //-----------------------------------------------------------------------------
87  auto operator=(QueueHipRtBlockingImpl &&) -> QueueHipRtBlockingImpl & = delete;
88  //-----------------------------------------------------------------------------
89  ALPAKA_FN_HOST ~QueueHipRtBlockingImpl()
90  {
91  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
92 
93  // Set the current device. \TODO: Is setting the current device before hipStreamDestroy required?
94  ALPAKA_HIP_RT_CHECK(
95  hipSetDevice(
96  m_dev.m_iDevice));
97  // In case the device is still doing work in the queue when hipStreamDestroy() is called, the function will return immediately
98  // and the resources associated with queue will be released automatically once the device has completed all work in queue.
99  // -> No need to synchronize here.
100  ALPAKA_HIP_RT_CHECK(

102  m_HipQueue));
103  }

107  hipStream_t m_HipQueue;

203  //! The HIP RT blocking queue enqueue trait specialization.
204  template<
205  typename TTask>
206  struct Enqueue<
207  queue::QueueHipRtBlocking,
208  TTask>
209  {
210  //#############################################################################
211  enum class CallbackState
212  {
213  enqueued,
214  notified,
215  finished,
216  };
217 
218  //#############################################################################
219  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
220  {
221  std::mutex m_mutex;
222  std::condition_variable m_event;
223  CallbackState state = CallbackState::enqueued;
224  };
225 
226  //-----------------------------------------------------------------------------
227  static void HIPRT_CB hipRtCallback(hipStream_t /*queue*/, hipError_t /*status*/, void *arg)
228  {
229  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread has already finished.
230  const auto pCallbackSynchronizationData = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
231 
232  // Notify the executing thread.
233  {
234  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
235  pCallbackSynchronizationData->state = CallbackState::notified;
236  }
237  pCallbackSynchronizationData->m_event.notify_one();
238 
239  // Wait for the executing thread to finish the task if it has not already finished.
240  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
241  if(pCallbackSynchronizationData->state != CallbackState::finished)
242  {
243  pCallbackSynchronizationData->m_event.wait(
244  lock,
245  [pCallbackSynchronizationData](){
246  return pCallbackSynchronizationData->state == CallbackState::finished;
247  }
248  );
249  }
250  }
251 
252  //-----------------------------------------------------------------------------
253  ALPAKA_FN_HOST static auto enqueue(
254  queue::QueueHipRtBlocking & queue,
255  TTask const & task)
256  -> void
257  {
258 #if BOOST_COMP_HCC // NOTE: workaround for unwanted nonblocking hip streams for HCC (NVCC streams are blocking)
259  {
260  // thread-safe callee incrementing
261  std::lock_guard<std::mutex> guard(queue.m_spQueueImpl->m_mutex);
262  queue.m_spQueueImpl->m_callees += 1;
263  }
264 #endif
265 
266  auto pCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
267 
268  ALPAKA_HIP_RT_CHECK(hipStreamAddCallback(
269  queue.m_spQueueImpl->m_HipQueue,
270  hipRtCallback,
271  pCallbackSynchronizationData.get(),
272  0u));
273 
274  // We start a new std::thread which stores the task to be executed.
275  // This circumvents the limitation that it is not possible to call HIP methods within the HIP callback thread.
276  // The HIP thread signals the std::thread when it is ready to execute the task.
277  // The HIP thread is waiting for the std::thread to signal that it is finished executing the task
278  // before it executes the next task in the queue (HIP stream).
279  std::thread t(
280  [pCallbackSynchronizationData,
281  task
282 #if BOOST_COMP_HCC // NOTE: workaround for unwanted nonblocking hip streams for HCC (NVCC streams are blocking)
283  ,&queue
284 #endif
285  ](){
286 
287 #if BOOST_COMP_HCC // NOTE: workaround for unwanted nonblocking hip streams for HCC (NVCC streams are blocking)
288  // thread-safe task execution and callee decrementing
289  std::lock_guard<std::mutex> guard(queue.m_spQueueImpl->m_mutex);
290 #endif
291 
292  // If the callback has not yet been called, we wait for it.
293  {
294  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
295  if(pCallbackSynchronizationData->state != CallbackState::notified)
296  {
297  pCallbackSynchronizationData->m_event.wait(
298  lock,
299  [pCallbackSynchronizationData](){
300  return pCallbackSynchronizationData->state == CallbackState::notified;
301  }
302  );
303  }
304 
305  task();
306 
307  // Notify the waiting HIP thread.
308  pCallbackSynchronizationData->state = CallbackState::finished;
309  }
310  pCallbackSynchronizationData->m_event.notify_one();
311 
312 #if BOOST_COMP_HCC // NOTE: workaround for unwanted nonblocking hip streams for HCC (NVCC streams are blocking)
313  queue.m_spQueueImpl->m_callees -= 1;
314 #endif
315  }
316  );
317 
318  t.join();
319  }
320  };
