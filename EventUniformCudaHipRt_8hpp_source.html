49  //! The CUDA/HIP RT device event implementation.
50  class EventUniformCudaHipImpl final
51  {
52  public:
53  ALPAKA_FN_HOST EventUniformCudaHipImpl(DevUniformCudaHipRt const& dev, bool bBusyWait)
54  : m_dev(dev)
55  , m_UniformCudaHipEvent()
56  {
57  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
58 
59  // Set the current device.
60  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(m_dev.m_iDevice));
61 
62  // Create the event on the current device with the specified flags. Valid flags include:
63  // - cuda/hip-EventDefault: Default event creation flag.
64  // - cuda/hip-EventBlockingSync : Specifies that event should use blocking synchronization.
65  // A host thread that uses cuda/hip-EventSynchronize() to wait on an event created with this flag
66  // will block until the event actually completes.
67  // - cuda/hip-EventDisableTiming : Specifies that the created event does not need to record timing
68  // data.
69  // Events created with this flag specified and the cuda/hip-EventBlockingSync flag not specified
70  // will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
71  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(EventCreateWithFlags)(
72  &m_UniformCudaHipEvent,
73  (bBusyWait ? ALPAKA_API_PREFIX(EventDefault) : ALPAKA_API_PREFIX(EventBlockingSync))
74  | ALPAKA_API_PREFIX(EventDisableTiming)));
75  }
76  EventUniformCudaHipImpl(EventUniformCudaHipImpl const&) = delete;
77  EventUniformCudaHipImpl(EventUniformCudaHipImpl&&) = default;
78  auto operator=(EventUniformCudaHipImpl const&) -> EventUniformCudaHipImpl& = delete;
79  auto operator=(EventUniformCudaHipImpl&&) -> EventUniformCudaHipImpl& = delete;
80  ALPAKA_FN_HOST ~EventUniformCudaHipImpl()
81  {
82  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
83 
84  // Set the current device. \TODO: Is setting the current device before cuda/hip-EventDestroy
85  // required?
86 
87  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(m_dev.m_iDevice));
88  // In case event has been recorded but has not yet been completed when cuda/hip-EventDestroy() is
89  // called, the function will return immediately and the resources associated with event will be
90  // released automatically once the device has completed event.
91  // -> No need to synchronize here.
92  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(EventDestroy)(m_UniformCudaHipEvent));
93  }
94 
95  public:
96  DevUniformCudaHipRt const m_dev; //!< The device this event is bound to.
97 
98  ALPAKA_API_PREFIX(Event_t) m_UniformCudaHipEvent;
99  };
100  } // namespace detail
