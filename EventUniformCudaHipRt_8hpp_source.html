38  //! The CUDA/HIP RT device event implementation.
39  template<typename TApi>
40  class EventUniformCudaHipImpl final
41  {
42  public:
43  ALPAKA_FN_HOST EventUniformCudaHipImpl(DevUniformCudaHipRt<TApi> const& dev, bool bBusyWait)
44  : m_dev(dev)
45  , m_UniformCudaHipEvent()
46  {
47  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
48 
49  // Set the current device.
50  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::setDevice(m_dev.getNativeHandle()));
51 
52  // Create the event on the current device with the specified flags. Valid flags include:
53  // - cuda/hip-EventDefault: Default event creation flag.
54  // - cuda/hip-EventBlockingSync : Specifies that event should use blocking synchronization.
55  // A host thread that uses cuda/hip-EventSynchronize() to wait on an event created with this flag
56  // will block until the event actually completes.
57  // - cuda/hip-EventDisableTiming : Specifies that the created event does not need to record timing
58  // data.
59  // Events created with this flag specified and the cuda/hip-EventBlockingSync flag not specified
60  // will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
61  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::eventCreateWithFlags(
62  &m_UniformCudaHipEvent,
63  (bBusyWait ? TApi::eventDefault : TApi::eventBlockingSync) | TApi::eventDisableTiming));
64  }
65  EventUniformCudaHipImpl(EventUniformCudaHipImpl const&) = delete;
66  auto operator=(EventUniformCudaHipImpl const&) -> EventUniformCudaHipImpl& = delete;
67  ALPAKA_FN_HOST ~EventUniformCudaHipImpl()
68  {
69  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
70 
71  // In case event has been recorded but has not yet been completed when cuda/hip-EventDestroy() is
72  // called, the function will return immediately and the resources associated with event will be
73  // released automatically once the device has completed event.
74  // -> No need to synchronize here.
75  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(TApi::eventDestroy(m_UniformCudaHipEvent));
76  }
77 
78  [[nodiscard]] auto getNativeHandle() const noexcept
79  {
80  return m_UniformCudaHipEvent;
81  }
82 
83  public:
84  DevUniformCudaHipRt<TApi> const m_dev; //!< The device this event is bound to.
85 
86  private:
87  typename TApi::Event_t m_UniformCudaHipEvent;
88  };
89  } // namespace uniform_cuda_hip::detail
