33  //! The CUDA/HIP RT device event implementation.
34  template<typename TApi>
35  class EventUniformCudaHipImpl final
36  {
37  public:
38  ALPAKA_FN_HOST EventUniformCudaHipImpl(DevUniformCudaHipRt<TApi> const& dev, bool bBusyWait)
39  : m_dev(dev)
40  , m_UniformCudaHipEvent()
41  {
42  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
43 
44  // Set the current device.
45  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::setDevice(m_dev.getNativeHandle()));
46 
47  // Create the event on the current device with the specified flags. Valid flags include:
48  // - cuda/hip-EventDefault: Default event creation flag.
49  // - cuda/hip-EventBlockingSync : Specifies that event should use blocking synchronization.
50  // A host thread that uses cuda/hip-EventSynchronize() to wait on an event created with this flag
51  // will block until the event actually completes.
52  // - cuda/hip-EventDisableTiming : Specifies that the created event does not need to record timing
53  // data.
54  // Events created with this flag specified and the cuda/hip-EventBlockingSync flag not specified
55  // will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
56  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::eventCreateWithFlags(
57  &m_UniformCudaHipEvent,
58  (bBusyWait ? TApi::eventDefault : TApi::eventBlockingSync) | TApi::eventDisableTiming));
59  }
60  EventUniformCudaHipImpl(EventUniformCudaHipImpl const&) = delete;
61  auto operator=(EventUniformCudaHipImpl const&) -> EventUniformCudaHipImpl& = delete;
62  ALPAKA_FN_HOST ~EventUniformCudaHipImpl()
63  {
64  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
65 
66  // In case event has been recorded but has not yet been completed when cuda/hip-EventDestroy() is
67  // called, the function will return immediately and the resources associated with event will be
68  // released automatically once the device has completed event.
69  // -> No need to synchronize here.
70  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(TApi::eventDestroy(m_UniformCudaHipEvent));
71  }
72 
73  [[nodiscard]] auto getNativeHandle() const noexcept
74  {
75  return m_UniformCudaHipEvent;
76  }
77 
78  public:
79  DevUniformCudaHipRt<TApi> const m_dev; //!< The device this event is bound to.
80 
81  private:
82  typename TApi::Event_t m_UniformCudaHipEvent;
83  };
84  } // namespace uniform_cuda_hip::detail
