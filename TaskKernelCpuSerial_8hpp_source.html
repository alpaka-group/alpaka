39  //! The CPU serial execution task implementation.
40  template<typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
41  class TaskKernelCpuSerial final : public WorkDivMembers<TDim, TIdx>
42  {
43  public:
44  template<typename TWorkDiv>
45  ALPAKA_FN_HOST TaskKernelCpuSerial(TWorkDiv&& workDiv, TKernelFnObj kernelFnObj, TArgs&&... args)
46  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
47  , m_kernelFnObj(std::move(kernelFnObj))
48  , m_args(std::forward<TArgs>(args)...)
49  {
50  static_assert(
51  Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
52  "The work division and the execution task have to be of the same dimensionality!");
53  }
54 
55  //! Executes the kernel function object.
56  ALPAKA_FN_HOST auto operator()() const -> void
57  {
58  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
59 
60  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(*this);
61  auto const blockThreadExtent = getWorkDiv<Block, Threads>(*this);
62  auto const threadElemExtent = getWorkDiv<Thread, Elems>(*this);
63 
64  // Get the size of the block shared dynamic memory.
65  auto const blockSharedMemDynSizeBytes = std::apply(
66  [&](ALPAKA_DECAY_T(TArgs) const&... args)
67  {
68  return getBlockSharedMemDynSizeBytes<AccCpuSerial<TDim, TIdx>>(
69  m_kernelFnObj,
70  blockThreadExtent,
71  threadElemExtent,
72  args...);
73  },
74  m_args);
75 
76 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
77  std::cout << __func__ << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B"
78  << std::endl;
79 # endif
80 
81  AccCpuSerial<TDim, TIdx> acc(
82  *static_cast<WorkDivMembers<TDim, TIdx> const*>(this),
83  blockSharedMemDynSizeBytes);
84 
85  if(blockThreadExtent.prod() != static_cast<TIdx>(1u))
86  {
87  throw std::runtime_error("A block for the serial accelerator can only ever have one single thread!");
88  }
89 
90  // Execute the blocks serially.
91  meta::ndLoopIncIdx(
92  gridBlockExtent,
93  [&](Vec<TDim, TIdx> const& blockThreadIdx)
94  {
95  acc.m_gridBlockIdx = blockThreadIdx;
96 
97  std::apply(m_kernelFnObj, std::tuple_cat(std::tie(acc), m_args));
98 
99  // After a block has been processed, the shared memory has to be deleted.
100  freeSharedVars(acc);
101  });
102  }
103 
104  private:
105  TKernelFnObj m_kernelFnObj;
106  std::tuple<std::decay_t<TArgs>...> m_args;
107  };
