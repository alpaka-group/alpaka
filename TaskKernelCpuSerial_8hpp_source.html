37  //! The CPU serial execution task implementation.
38  template<typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
39  class TaskKernelCpuSerial final : public WorkDivMembers<TDim, TIdx>
40  {
41  public:
42  template<typename TWorkDiv>
43  ALPAKA_FN_HOST TaskKernelCpuSerial(TWorkDiv&& workDiv, TKernelFnObj kernelFnObj, TArgs&&... args)
44  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
45  , m_kernelFnObj(std::move(kernelFnObj))
46  , m_args(std::forward<TArgs>(args)...)
47  {
48  static_assert(
49  Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
50  "The work division and the execution task have to be of the same dimensionality!");
51  }
52 
53  //! Executes the kernel function object.
54  ALPAKA_FN_HOST auto operator()() const -> void
55  {
56  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
57 
58  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(*this);
59  auto const blockThreadExtent = getWorkDiv<Block, Threads>(*this);
60  auto const threadElemExtent = getWorkDiv<Thread, Elems>(*this);
61 
62  // Get the size of the block shared dynamic memory.
63  auto const blockSharedMemDynSizeBytes = std::apply(
64  [&](std::decay_t<TArgs> const&... args)
65  {
66  return getBlockSharedMemDynSizeBytes<AccCpuSerial<TDim, TIdx>>(
67  m_kernelFnObj,
68  blockThreadExtent,
69  threadElemExtent,
70  args...);
71  },
72  m_args);
73 
74 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
75  std::cout << __func__ << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B"
76  << std::endl;
77 # endif
78 
79  AccCpuSerial<TDim, TIdx> acc(
80  *static_cast<WorkDivMembers<TDim, TIdx> const*>(this),
81  blockSharedMemDynSizeBytes);
82 
83  if(blockThreadExtent.prod() != static_cast<TIdx>(1u))
84  {
85  throw std::runtime_error("A block for the serial accelerator can only ever have one single thread!");
86  }
87 
88  // Execute the blocks serially.
89  meta::ndLoopIncIdx(
90  gridBlockExtent,
91  [&](Vec<TDim, TIdx> const& blockThreadIdx)
92  {
93  acc.m_gridBlockIdx = blockThreadIdx;
94 
95  std::apply(m_kernelFnObj, std::tuple_cat(std::tie(acc), m_args));
96 
97  // After a block has been processed, the shared memory has to be deleted.
98  freeSharedVars(acc);
99  });
100  }
101 
102  private:
103  TKernelFnObj m_kernelFnObj;
104  std::tuple<std::decay_t<TArgs>...> m_args;
105  };
