44  typename TDim,

55  ALPAKA_FN_HOST TaskKernelCpuSerial(
56  TWorkDiv && workDiv,
57  TKernelFnObj const & kernelFnObj,
58  TArgs && ... args) :
59  workdiv::WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv)),

78  //-----------------------------------------------------------------------------
79  //! Executes the kernel function object.
80  ALPAKA_FN_HOST auto operator()() const
81  -> void
82  {
83  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
84 
85  auto const gridBlockExtent(
86  workdiv::getWorkDiv<Grid, Blocks>(*this));
87  auto const blockThreadExtent(
88  workdiv::getWorkDiv<Block, Threads>(*this));
89  auto const threadElemExtent(
90  workdiv::getWorkDiv<Thread, Elems>(*this));
91 
92  // Get the size of the block shared dynamic memory.
93  auto const blockSharedMemDynSizeBytes(
94  meta::apply(
95  [&](std::decay_t<TArgs> const & ... args)
96  {
97  return
98  kernel::getBlockSharedMemDynSizeBytes<
99  acc::AccCpuSerial<TDim, TIdx>>(
100  m_kernelFnObj,
101  blockThreadExtent,
102  threadElemExtent,
103  args...);
104  },
105  m_args));
106 
107 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
108  std::cout << __func__
109  << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B" << std::endl;
110 #endif
111  // Bind all arguments except the accelerator.
112  // TODO: With C++14 we could create a perfectly argument forwarding function object within the constructor.
113  auto const boundKernelFnObj(
114  meta::apply(
115  [this](std::decay_t<TArgs> const & ... args)
116  {
117  return
118  std::bind(
119  std::ref(m_kernelFnObj),
120  std::placeholders::_1,
121  std::ref(args)...);
122  },
123  m_args));
124 
125  acc::AccCpuSerial<TDim, TIdx> acc(
126  *static_cast<workdiv::WorkDivMembers<TDim, TIdx> const *>(this),
127  blockSharedMemDynSizeBytes);
128 
129  if(blockThreadExtent.prod() != static_cast<TIdx>(1u))
130  {
131  throw std::runtime_error("A block for the serial accelerator can only ever have one single thread!");
132  }
133 
134  // Execute the blocks serially.
135  meta::ndLoopIncIdx(
136  gridBlockExtent,
137  [&](vec::Vec<TDim, TIdx> const & blockThreadIdx)
138  {
139  acc.m_gridBlockIdx = blockThreadIdx;
140 
141  boundKernelFnObj(
142  acc);
143 
144  // After a block has been processed, the shared memory has to be deleted.
145  block::shared::st::freeMem(acc);
146  });
147  }
148 
149  private:
150  TKernelFnObj m_kernelFnObj;
151  std::tuple<std::decay_t<TArgs>...> m_args;
152  };
153  }
