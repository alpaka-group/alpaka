56  ALPAKA_FN_HOST auto getAllExistingQueues() const
57  -> std::vector<std::shared_ptr<IGenericThreadsQueue<DevOmp5>>>
58  {
59  std::vector<std::shared_ptr<IGenericThreadsQueue<DevOmp5>>> vspQueues;
60 
61  std::lock_guard<std::mutex> lk(m_Mutex);
62  vspQueues.reserve(m_queues.size());
63 
64  for(auto it = m_queues.begin(); it != m_queues.end();)
65  {
66  auto spQueue(it->lock());
67  if(spQueue)
68  {
69  vspQueues.emplace_back(std::move(spQueue));
70  ++it;
71  }
72  else
73  {
74  it = m_queues.erase(it);
75  }
76  }
77  return vspQueues;
78  }

83  {
