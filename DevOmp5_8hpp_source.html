59  ALPAKA_FN_HOST auto getAllExistingQueues() const
60  -> std::vector<std::shared_ptr<IGenericThreadsQueue<DevOmp5>>>
61  {
62  std::vector<std::shared_ptr<IGenericThreadsQueue<DevOmp5>>> vspQueues;
63 
64  std::lock_guard<std::mutex> lk(m_Mutex);
65  vspQueues.reserve(std::size(m_queues));
66 
67  for(auto it = std::begin(m_queues); it != std::end(m_queues);)
68  {
69  auto spQueue(it->lock());
70  if(spQueue)
71  {
72  vspQueues.emplace_back(std::move(spQueue));
73  ++it;
74  }
75  else
76  {
77  it = m_queues.erase(it);
78  }
79  }
80  return vspQueues;
81  }

86  {
87  std::lock_guard<std::mutex> lk(m_Mutex);

89  // Register this queue on the device.
90  m_queues.push_back(spQueue);

98  //! Create and/or return staticlly mapped device pointer of host address.
99  template<typename TElem, typename TExtent>
100  ALPAKA_FN_HOST auto mapStatic(TElem* pHost, TExtent const& extent) -> TElem*
101  {
102  const std::size_t sizeB = extent.prod() * sizeof(TElem);
103  auto m = m_staticMemMap.find(pHost);
104  if(m != m_staticMemMap.end())
105  {
106  if(sizeB != m->second.second)
107  {
108  std::ostringstream os;
109  os << "Statically mapped size cannot change: Static size is " << m->second.second
110  << " requested size is " << sizeB << '.';
111  throw std::runtime_error(os.str());
112  }
113  return reinterpret_cast<TElem*>(m->second.first);
114  }
115 
116  void* pDev = omp_target_alloc(sizeB, getNativeHandle());
117  if(!pDev)
118  return nullptr;
119 
120  /*! Associating pointers for good measure. Not actually
121  * required as long a not `target enter data` is done */
122  omp_target_associate_ptr(pHost, pDev, sizeB, 0u, getNativeHandle());
123 
124  m_staticMemMap[pHost] = std::make_pair(pDev, sizeB);
125  return reinterpret_cast<TElem*>(pDev);
126  }
127

129  std::mutex mutable m_Mutex;
130  std::vector<std::weak_ptr<IGenericThreadsQueue<DevOmp5>>> mutable m_queues;
131  const int m_iDevice;
132 
133  std::map<void*, std::pair<void*, std::size_t>> m_staticMemMap;
134  };
