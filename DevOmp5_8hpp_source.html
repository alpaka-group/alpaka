68  //! Create and/or return staticlly mapped device pointer of host address.
69  template<typename TElem, typename TExtent>
70  ALPAKA_FN_HOST auto mapStatic(TElem* pHost, TExtent const& extent) -> TElem*
71  {
72  const std::size_t sizeB = extent.prod() * sizeof(TElem);
73  auto m = m_staticMemMap.find(pHost);
74  if(m != m_staticMemMap.end())
75  {
76  if(sizeB != m->second.second)
77  {
78  std::ostringstream os;
79  os << "Statically mapped size cannot change: Static size is " << m->second.second
80  << " requested size is " << sizeB << '.';
81  throw std::runtime_error(os.str());
82  }
83  return reinterpret_cast<TElem*>(m->second.first);
84  }
85 
86  void* pDev = omp_target_alloc(sizeB, getNativeHandle());
87  if(!pDev)
88  return nullptr;
89 
90  /*! Associating pointers for good measure. Not actually
91  * required as long a not `target enter data` is done */
92  omp_target_associate_ptr(pHost, pDev, sizeB, 0u, getNativeHandle());
93 
94  m_staticMemMap[pHost] = std::make_pair(pDev, sizeB);
95  return reinterpret_cast<TElem*>(pDev);
96  }
97 
98  private:
99  const int m_iDevice;
100 
101  std::map<void*, std::pair<void*, std::size_t>> m_staticMemMap;
102  };
