30  typename TDim,

32  typename TExtent>
33  struct TaskSetCpuBase
34  {

36  using DstSize = Idx<TView>;
37  using Elem = alpaka::Elem<TView>;
38 
39  static_assert(
40  !std::is_const<TView>::value,
41  "The destination view can not be const!");
42 
43  static_assert(
44  Dim<TView>::value == Dim<TExtent>::value,
45  "The destination view and the extent are required to have the same dimensionality!");
46  static_assert(

48  "The destination view and the input TDim are required to have the same dimensionality!");

50  static_assert(

52  "The view and the extent are required to have compatible idx type!");
53 
54  //-----------------------------------------------------------------------------
55  TaskSetCpuBase(
56  TView & view,
57  std::uint8_t const & byte,
58  TExtent const & extent) :
59  m_byte(byte),
60  m_extent(extent::getExtentVec(extent)),
61  m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem))),
62 #if (!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
63  m_dstExtent(extent::getExtentVec(view)),
64 #endif

66  m_dstMemNative(reinterpret_cast<std::uint8_t *>(getPtrNative(view)))
67  {

69  ALPAKA_ASSERT(m_extentWidthBytes <= m_dstPitchBytes[TDim::value - 1u]);
70  }

98  //! The CPU device ND memory set task.
99  template<
100  typename TDim,
101  typename TView,
102  typename TExtent>
103  struct TaskSetCpu : public TaskSetCpuBase<TDim, TView, TExtent>
104  {
105  using DimMin1 = DimInt<TDim::value - 1u>;
106  using typename TaskSetCpuBase<TDim, TView, TExtent>::ExtentSize;
107  using typename TaskSetCpuBase<TDim, TView, TExtent>::DstSize;
108 
109  //-----------------------------------------------------------------------------
110  using TaskSetCpuBase<TDim, TView, TExtent>::TaskSetCpuBase;
111 
112  //-----------------------------------------------------------------------------
113  ALPAKA_FN_HOST auto operator()() const
114  -> void
115  {
116  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
117 
118 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
119  this->printDebug();
120 #endif
121  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
122  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
123  // [z, y, x] -> [y, x] because the z pitch (the full idx of the buffer) is not required.
124  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
125 
126  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
127  {
128  meta::ndLoopIncIdx(
129  extentWithoutInnermost,
130  [&](Vec<DimMin1, ExtentSize> const & idx)
131  {
132 
133  std::memset(
134  reinterpret_cast<void *>(this->m_dstMemNative + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
135  this->m_byte,
136  static_cast<std::size_t>(this->m_extentWidthBytes));
137  });
138  }
139  }
140  };
