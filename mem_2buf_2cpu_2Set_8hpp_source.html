29  struct TaskSetCpuBase

34 
35  static_assert(!std::is_const<TView>::value, "The destination view can not be const!");
36 
37  static_assert(
38  Dim<TView>::value == Dim<TExtent>::value,
39  "The destination view and the extent are required to have the same dimensionality!");
40  static_assert(
41  Dim<TView>::value == TDim::value,
42  "The destination view and the input TDim are required to have the same dimensionality!");
43 
44  static_assert(
45  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
46  "The view and the extent are required to have compatible idx type!");

48  TaskSetCpuBase(TView& view, std::uint8_t const& byte, TExtent const& extent)
49  : m_byte(byte)
50  , m_extent(extent::getExtentVec(extent))
51  , m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem)))
52  ,
53 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
54  m_dstExtent(extent::getExtentVec(view))
55  ,
56 #endif
57  m_dstPitchBytes(getPitchBytesVec(view))
58  , m_dstMemNative(reinterpret_cast<std::uint8_t*>(getPtrNative(view)))
59  {
60  ALPAKA_ASSERT((castVec<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
61  ALPAKA_ASSERT(m_extentWidthBytes <= m_dstPitchBytes[TDim::value - 1u]);

83  //! The CPU device ND memory set task.
84  template<typename TDim, typename TView, typename TExtent>
85  struct TaskSetCpu : public TaskSetCpuBase<TDim, TView, TExtent>
86  {
87  using DimMin1 = DimInt<TDim::value - 1u>;
88  using typename TaskSetCpuBase<TDim, TView, TExtent>::ExtentSize;
89  using typename TaskSetCpuBase<TDim, TView, TExtent>::DstSize;
90 
91  using TaskSetCpuBase<TDim, TView, TExtent>::TaskSetCpuBase;
92 
93  ALPAKA_FN_HOST auto operator()() const -> void
94  {
95  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
96 
97 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
98  this->printDebug();
99 #endif
100  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one
101  // iteration.
102  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
103  // [z, y, x] -> [y, x] because the z pitch (the full idx of the buffer) is not required.
104  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
105 
106  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
107  {
108  meta::ndLoopIncIdx(extentWithoutInnermost, [&](Vec<DimMin1, ExtentSize> const& idx) {
109  std::memset(
110  reinterpret_cast<void*>(
111  this->m_dstMemNative
112  + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost)
113  .foldrAll(std::plus<DstSize>())),
114  this->m_byte,
115  static_cast<std::size_t>(this->m_extentWidthBytes));
116  });
117  }
118  }
119  };
