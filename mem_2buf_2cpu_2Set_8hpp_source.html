35  //! The CPU device ND memory set task base.
36  template<
37  typename TDim,
38  typename TView,
39  typename TExtent>
40  struct TaskSetCpuBase
41  {
42  using ExtentSize = Idx<TExtent>;
43  using DstSize = Idx<TView>;
44  using Elem = alpaka::Elem<TView>;
45 
46  static_assert(
47  !std::is_const<TView>::value,
48  "The destination view can not be const!");
49 
50  static_assert(
51  Dim<TView>::value == Dim<TExtent>::value,
52  "The destination view and the extent are required to have the same dimensionality!");
53  static_assert(
54  Dim<TView>::value == TDim::value,
55  "The destination view and the input TDim are required to have the same dimensionality!");
56 
57  static_assert(
58  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
59  "The view and the extent are required to have compatible idx type!");
60 
61  //-----------------------------------------------------------------------------
62  TaskSetCpuBase(
63  TView & view,
64  std::uint8_t const & byte,
65  TExtent const & extent) :
66  m_byte(byte),
67  m_extent(extent::getExtentVec(extent)),
68  m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem))),
69 #if (!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
70  m_dstExtent(extent::getExtentVec(view)),
71 #endif
72  m_dstPitchBytes(view::getPitchBytesVec(view)),
73  m_dstMemNative(reinterpret_cast<std::uint8_t *>(view::getPtrNative(view)))
74  {
75  ALPAKA_ASSERT((cast<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
76  ALPAKA_ASSERT(m_extentWidthBytes <= m_dstPitchBytes[TDim::value - 1u]);
77  }

105  //! The CPU device ND memory set task.
106  template<
107  typename TDim,
108  typename TView,
109  typename TExtent>
110  struct TaskSetCpu : public TaskSetCpuBase<TDim, TView, TExtent>
111  {
112  using DimMin1 = DimInt<TDim::value - 1u>;
113  using typename TaskSetCpuBase<TDim, TView, TExtent>::ExtentSize;
114  using typename TaskSetCpuBase<TDim, TView, TExtent>::DstSize;
115 
116  //-----------------------------------------------------------------------------
117  using TaskSetCpuBase<TDim, TView, TExtent>::TaskSetCpuBase;
118 
119  //-----------------------------------------------------------------------------
120  ALPAKA_FN_HOST auto operator()() const
121  -> void
122  {
123  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
124 
125 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
126  this->printDebug();
127 #endif
128  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
129  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
130  // [z, y, x] -> [y, x] because the z pitch (the full idx of the buffer) is not required.
131  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
132 
133  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
134  {
135  meta::ndLoopIncIdx(
136  extentWithoutInnermost,
137  [&](Vec<DimMin1, ExtentSize> const & idx)
138  {
139 
140  memset(
141  reinterpret_cast<void *>(this->m_dstMemNative + (cast<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
142  this->m_byte,
143  static_cast<std::size_t>(this->m_extentWidthBytes));
144  });
145  }
146  }
147  };
