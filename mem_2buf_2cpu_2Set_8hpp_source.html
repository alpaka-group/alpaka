34 
35  static_assert(!std::is_const_v<TView>, "The destination view can not be const!");
36 
37  static_assert(
38  Dim<TView>::value == Dim<TExtent>::value,
39  "The destination view and the extent are required to have the same dimensionality!");
40  static_assert(
41  Dim<TView>::value == TDim::value,
42  "The destination view and the input TDim are required to have the same dimensionality!");
43 
44  static_assert(

46  "The view and the extent are required to have compatible idx type!");
47 
48  TaskSetCpuBase(TView& view, std::uint8_t const& byte, TExtent const& extent)
49  : m_byte(byte)
50  , m_extent(getExtentVec(extent))
51  , m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem)))
52 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
53  , m_dstExtent(getExtentVec(view))
54 #endif

56  , m_dstMemNative(reinterpret_cast<std::uint8_t*>(getPtrNative(view)))
57  {

60  }

81  //! The CPU device ND memory set task.
82  template<typename TDim, typename TView, typename TExtent>
83  struct TaskSetCpu : public TaskSetCpuBase<TDim, TView, TExtent>
84  {
85  using DimMin1 = DimInt<TDim::value - 1u>;
86  using typename TaskSetCpuBase<TDim, TView, TExtent>::ExtentSize;
87  using typename TaskSetCpuBase<TDim, TView, TExtent>::DstSize;
88 
89  using TaskSetCpuBase<TDim, TView, TExtent>::TaskSetCpuBase;
90 
91  ALPAKA_FN_HOST auto operator()() const -> void
92  {
93  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
94 
95 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
96  this->printDebug();
97 #endif
98  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one
99  // iteration.
100  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
101  // [z, y, x] -> [y, x] because the z pitch (the full idx of the buffer) is not required.
102  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
103 
104  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
105  {
106  meta::ndLoopIncIdx(
107  extentWithoutInnermost,
108  [&](Vec<DimMin1, ExtentSize> const& idx)
109  {
110  std::memset(
111  reinterpret_cast<void*>(
112  this->m_dstMemNative
113  + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost)
114  .foldrAll(std::plus<DstSize>())),
115  this->m_byte,
116  static_cast<std::size_t>(this->m_extentWidthBytes));
117  });
118  }
119  }
120  };
