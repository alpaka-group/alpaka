212  //! The ViewSubView native pointer get trait specialization.
213  template<typename TElem, typename TDim, typename TDev, typename TIdx>
214  struct GetPtrNative<ViewSubView<TDev, TElem, TDim, TIdx>>
215  {
216  private:
217  using IdxSequence = std::make_index_sequence<TDim::value>;
218 
219  public:
220  //-----------------------------------------------------------------------------
221  ALPAKA_FN_HOST static auto getPtrNative(ViewSubView<TDev, TElem, TDim, TIdx> const& view) -> TElem const*
222  {
223  // \TODO: pre-calculate this pointer for faster execution.
224  return reinterpret_cast<TElem const*>(
225  reinterpret_cast<std::uint8_t const*>(alpaka::getPtrNative(view.m_viewParentView))
226  + pitchedOffsetBytes(view, IdxSequence()));
227  }
228  //-----------------------------------------------------------------------------
229  ALPAKA_FN_HOST static auto getPtrNative(ViewSubView<TDev, TElem, TDim, TIdx>& view) -> TElem*
230  {
231  // \TODO: pre-calculate this pointer for faster execution.
232  return reinterpret_cast<TElem*>(
233  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view.m_viewParentView))
234  + pitchedOffsetBytes(view, IdxSequence()));
235  }
236 
237  private:
238  //-----------------------------------------------------------------------------
239  //! For a 3D vector this calculates:
240  //!
241  //! getOffset<0u>(view) * getPitchBytes<1u>(view)
242  //! + getOffset<1u>(view) * getPitchBytes<2u>(view)
243  //! + getOffset<2u>(view) * getPitchBytes<3u>(view)
244  //! while getPitchBytes<3u>(view) is equivalent to sizeof(TElem)
245  template<typename TView, std::size_t... TIndices>
246  ALPAKA_FN_HOST static auto pitchedOffsetBytes(TView const& view, std::index_sequence<TIndices...> const&)
247  -> TIdx
248  {
249  return meta::foldr(std::plus<TIdx>(), pitchedOffsetBytesDim<TIndices>(view)...);
250  }
251  //-----------------------------------------------------------------------------
252  template<std::size_t Tidx, typename TView>
253  ALPAKA_FN_HOST static auto pitchedOffsetBytesDim(TView const& view) -> TIdx
254  {
255  return getOffset<Tidx>(view) * getPitchBytes<Tidx + 1u>(view);
256  }
257  };
