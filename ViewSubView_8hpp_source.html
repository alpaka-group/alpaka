203  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view.m_viewParentView))
204  + pitchedOffsetBytes(view, IdxSequence()));
205  }

207  private:
208  //! For a 3D vector this calculates:
209  //!
210  //! getOffsets(view)[0] * getPitchBytes<1u>(view)
211  //! + getOffsets(view)[1] * getPitchBytes<2u>(view)
212  //! + getOffsets(view)[2] * getPitchBytes<3u>(view)
213  //! while getPitchBytes<3u>(view) is equivalent to sizeof(TElem)
214  template<typename TView, std::size_t... TIndices>
215  ALPAKA_FN_HOST static auto pitchedOffsetBytes(TView const& view, std::index_sequence<TIndices...> const&)
216  -> TIdx
217  {
218  return meta::foldr(std::plus<TIdx>(), pitchedOffsetBytesDim<TIndices>(view)..., TIdx{0});
219  }
220 
221  template<std::size_t Tidx, typename TView>
222  ALPAKA_FN_HOST static auto pitchedOffsetBytesDim(TView const& view) -> TIdx
223  {
224  return getOffsets(view)[Tidx] * getPitchBytes<Tidx + 1u>(view);
225  }
