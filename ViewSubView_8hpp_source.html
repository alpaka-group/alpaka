210  private:
211  //! For a 3D vector this calculates:
212  //!
213  //! getOffset<0u>(view) * getPitchBytes<1u>(view)
214  //! + getOffset<1u>(view) * getPitchBytes<2u>(view)
215  //! + getOffset<2u>(view) * getPitchBytes<3u>(view)
216  //! while getPitchBytes<3u>(view) is equivalent to sizeof(TElem)
217  template<typename TView, std::size_t... TIndices>
218  ALPAKA_FN_HOST static auto pitchedOffsetBytes(TView const& view, std::index_sequence<TIndices...> const&)
219  -> TIdx
220  {
221  return meta::foldr(std::plus<TIdx>(), pitchedOffsetBytesDim<TIndices>(view)..., TIdx{0});
