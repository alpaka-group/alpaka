308  public:
309  //-----------------------------------------------------------------------------
310  ALPAKA_FN_HOST static auto getPtrNative(
311  mem::view::ViewSubView<TDev, TElem, TDim, TIdx> const & view)

321  ALPAKA_FN_HOST static auto getPtrNative(
322  mem::view::ViewSubView<TDev, TElem, TDim, TIdx> & view)
323  -> TElem *
324  {
325  // \TODO: pre-calculate this pointer for faster execution.
326  return
327  reinterpret_cast<TElem *>(
328  reinterpret_cast<std::uint8_t *>(mem::view::getPtrNative(view.m_viewParentView))
329  + pitchedOffsetBytes(view, IdxSequence()));
330  }
331 
332  private:
333  //-----------------------------------------------------------------------------
334  //! For a 3D vector this calculates:
335  //!
336  //! offset::getOffset<0u>(view) * mem::view::getPitchBytes<1u>(view)
337  //! + offset::getOffset<1u>(view) * mem::view::getPitchBytes<2u>(view)
338  //! + offset::getOffset<2u>(view) * mem::view::getPitchBytes<3u>(view)
339  //! while mem::view::getPitchBytes<3u>(view) is equivalent to sizeof(TElem)
340  template<
341  typename TView,
342  std::size_t... TIndices>
343  ALPAKA_FN_HOST static auto pitchedOffsetBytes(
344  TView const & view,
345  std::index_sequence<TIndices...> const &)
346  -> TIdx
347  {
348  return
349  meta::foldr(
350  std::plus<TIdx>(),
351  pitchedOffsetBytesDim<TIndices>(view)...);
352  }
353  //-----------------------------------------------------------------------------
354  template<
355  std::size_t Tidx,
356  typename TView>
357  ALPAKA_FN_HOST static auto pitchedOffsetBytesDim(
358  TView const & view)
359  -> TIdx
360  {
361  return
362  offset::getOffset<Tidx>(view)
363  * mem::view::getPitchBytes<Tidx + 1u>(view);
364  }
