94  ALPAKA_FN_HOST TaskCopyOaccBase(

96  TViewSrc const& viewSrc,
97  TExtent const& extent,
98  DevOacc const& dev,
99  TCopyPred copyPred)
100  : m_dev(dev)
101  , m_extent(getExtentVec(extent))
102  , m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem)))
103  , m_dstPitchBytes(getPitchBytesVec(viewDst))
104  , m_srcPitchBytes(getPitchBytesVec(viewSrc))
105  ,
106 # if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
107  m_dstExtent(getExtentVec(viewDst))
108  , m_srcExtent(getExtentVec(viewSrc))
109  ,
110 # endif
111  m_dstMemNative(reinterpret_cast<std::uint8_t*>(getPtrNative(viewDst)))
112  , m_srcMemNative(reinterpret_cast<std::uint8_t const*>(getPtrNative(viewSrc)))
113  , m_copyPred(copyPred)
114  {
115  ALPAKA_ASSERT((castVec<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
116  ALPAKA_ASSERT((castVec<SrcSize>(m_extent) <= m_srcExtent).foldrAll(std::logical_or<bool>()));
117  ALPAKA_ASSERT(static_cast<DstSize>(m_extentWidthBytes) <= m_dstPitchBytes[TDim::value - 1u]);
118  ALPAKA_ASSERT(static_cast<SrcSize>(m_extentWidthBytes) <= m_srcPitchBytes[TDim::value - 1u]);
119  }
120 
121 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
122  ALPAKA_FN_HOST auto printDebug() const -> void
123  {
124  std::cout << __func__ << " dev: " << m_dev.getNativeHandle() << " ew: " << m_extent
125  << " dw: " << m_dstExtent << " dptr: " << static_cast<const void*>(m_dstMemNative)
126  << " sw: " << m_srcExtent << " sptr: " << static_cast<const void*>(m_srcMemNative)

128  }
129 # endif
130  const DevOacc m_dev;
131  Vec<TDim, ExtentSize> m_extent;
132  ExtentSize const m_extentWidthBytes;
133  Vec<TDim, DstSize> m_dstPitchBytes;

144  //! The OpenACC Nd device memory copy task.
145  //!
146  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent, typename TCopyPred>
147  struct TaskCopyOacc : public TaskCopyOaccBase<TDim, TViewDst, TViewSrc, TExtent, TCopyPred>
148  {
149  using DimMin1 = DimInt<TDim::value - 1u>;
150  using typename TaskCopyOaccBase<TDim, TViewDst, TViewSrc, TExtent, TCopyPred>::ExtentSize;
151  using typename TaskCopyOaccBase<TDim, TViewDst, TViewSrc, TExtent, TCopyPred>::DstSize;
152  using typename TaskCopyOaccBase<TDim, TViewDst, TViewSrc, TExtent, TCopyPred>::SrcSize;
153 
154  using TaskCopyOaccBase<TDim, TViewDst, TViewSrc, TExtent, TCopyPred>::TaskCopyOaccBase;
155 
156  ALPAKA_FN_HOST auto operator()() const -> void
157  {
158  ALPAKA_DEBUG_FULL_LOG_SCOPE;
159 
160 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
161  this->printDebug();
162 # endif
163  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
164  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
165  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
166  Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_srcPitchBytes));
167 
168  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
169  {
170  this->m_dev.makeCurrent();
171  meta::ndLoopIncIdx(
172  extentWithoutInnermost,
173  [&](Vec<DimMin1, ExtentSize> const& idx)
174  {
175  this->m_copyPred(
176  reinterpret_cast<void*>(
177  this->m_dstMemNative
178  + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost).sum()),
179  const_cast<void*>(reinterpret_cast<const void*>(
180  this->m_srcMemNative
181  + (castVec<SrcSize>(idx) * srcPitchBytesWithoutOutmost).sum())),
182  static_cast<std::size_t>(this->m_extentWidthBytes));
183  });
184  }
185  }
186  };
