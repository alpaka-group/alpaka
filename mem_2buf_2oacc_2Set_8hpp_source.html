41  //! The OpenACC device memory set trait specialization.
42  template<typename TDim>
43  struct CreateTaskMemset<TDim, DevOacc>
44  {
45  template<typename TExtent, typename TViewFwd>
46  ALPAKA_FN_HOST static auto createTaskMemset(TViewFwd&& view, std::uint8_t const& byte, TExtent const& extent)
47  {
48  using TView = std::remove_reference_t<TViewFwd>;
49  using Idx = typename trait::IdxType<TExtent>::type;
50  auto pitch = getPitchBytesVec(view);
51  auto byteExtent = getExtentVec(extent);
52  byteExtent[TDim::value - 1] *= static_cast<Idx>(sizeof(Elem<TView>));
53  constexpr auto lastDim = TDim::value - 1;
54 
55  if(pitch[0] <= 0)
56  return createTaskKernel<AccOacc<TDim, Idx>>(
57  WorkDivMembers<TDim, Idx>(
58  Vec<TDim, Idx>::zeros(),
59  Vec<TDim, Idx>::zeros(),
60  Vec<TDim, Idx>::zeros()),
61  MemSetKernel(),
62  byte,
63  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view)),
64  byteExtent,
65  pitch); // NOP if size is zero
66 
67 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
68  std::cout << "Set TDim=" << TDim::value << " pitch=" << pitch << " byteExtent=" << byteExtent << std::endl;
69 # endif
70  auto elementsPerThread = Vec<TDim, Idx>::all(static_cast<Idx>(1u));
71  elementsPerThread[lastDim] = 4;
72  // Let alpaka calculate good block and grid sizes given our full problem extent
73  WorkDivMembers<TDim, Idx> const workDiv(getValidWorkDiv<AccOacc<TDim, Idx>>(
74  getDev(view),
75  byteExtent,
76  elementsPerThread,
77  false,
78  alpaka::GridBlockExtentSubDivRestrictions::Unrestricted));
79  return createTaskKernel<AccOacc<TDim, Idx>>(
80  workDiv,
81  MemSetKernel(),
82  byte,
83  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view)),
84  byteExtent,
85  pitch);
86  }
87  };

108  };
