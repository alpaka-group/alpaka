49  {

55 
56  if(pitch[0] <= 0)
57  return createTaskKernel<AccOacc<TDim, Idx>>(

59  Vec<TDim, Idx>::zeros(),
60  Vec<TDim, Idx>::zeros(),
61  Vec<TDim, Idx>::zeros()),
62  MemSetKernel(),
63  byte,
64  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view)),
65  byteExtent,
66  pitch); // NOP if size is zero
67 
68 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
69  std::cout << "Set TDim=" << TDim::value << " pitch=" << pitch << " byteExtent=" << byteExtent
70  << std::endl;
71 # endif
72  auto elementsPerThread = Vec<TDim, Idx>::all(static_cast<Idx>(1u));
73  elementsPerThread[lastDim] = 4;
