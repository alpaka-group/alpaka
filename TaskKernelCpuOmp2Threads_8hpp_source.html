49  //! The CPU OpenMP 2.0 thread accelerator execution task.
50  template<
51  typename TDim,
52  typename TIdx,
53  typename TKernelFnObj,
54  typename... TArgs>
55  class TaskKernelCpuOmp2Threads final :
56  public workdiv::WorkDivMembers<TDim, TIdx>
57  {
58  public:
59  //-----------------------------------------------------------------------------
60  template<
61  typename TWorkDiv>
62  ALPAKA_FN_HOST TaskKernelCpuOmp2Threads(
63  TWorkDiv && workDiv,
64  TKernelFnObj const & kernelFnObj,
65  TArgs && ... args) :
66  workdiv::WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv)),
67  m_kernelFnObj(kernelFnObj),
68  m_args(std::forward<TArgs>(args)...)
69  {
70  static_assert(
71  dim::Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
72  "The work division and the execution task have to be of the same dimensionality!");
73  }
74  //-----------------------------------------------------------------------------
75  TaskKernelCpuOmp2Threads(TaskKernelCpuOmp2Threads const &) = default;
76  //-----------------------------------------------------------------------------
77  TaskKernelCpuOmp2Threads(TaskKernelCpuOmp2Threads &&) = default;

79  auto operator=(TaskKernelCpuOmp2Threads const &) -> TaskKernelCpuOmp2Threads & = default;
80  //-----------------------------------------------------------------------------
81  auto operator=(TaskKernelCpuOmp2Threads &&) -> TaskKernelCpuOmp2Threads & = default;
82  //-----------------------------------------------------------------------------
83  ~TaskKernelCpuOmp2Threads() = default;

85  //-----------------------------------------------------------------------------
86  //! Executes the kernel function object.
87  ALPAKA_FN_HOST auto operator()() const
88  -> void
89  {
90  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
91 
92  auto const gridBlockExtent(
93  workdiv::getWorkDiv<Grid, Blocks>(*this));
94  auto const blockThreadExtent(
95  workdiv::getWorkDiv<Block, Threads>(*this));
96  auto const threadElemExtent(
97  workdiv::getWorkDiv<Thread, Elems>(*this));
98 
99  // Get the size of the block shared dynamic memory.
100  auto const blockSharedMemDynSizeBytes(
101  meta::apply(
102  [&](std::decay_t<TArgs> const & ... args)
103  {
104  return
105  kernel::getBlockSharedMemDynSizeBytes<
106  acc::AccCpuOmp2Threads<TDim, TIdx>>(
107  m_kernelFnObj,
108  blockThreadExtent,
109  threadElemExtent,
110  args...);
111  },
112  m_args));
113 
114 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
115  std::cout << __func__
116  << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B" << std::endl;
117 #endif
118  // Bind all arguments except the accelerator.
119  // TODO: With C++14 we could create a perfectly argument forwarding function object within the constructor.
120  auto const boundKernelFnObj(
121  meta::apply(
122  [this](std::decay_t<TArgs> const & ... args)
123  {
124  return
125  std::bind(
126  std::ref(m_kernelFnObj),
127  std::placeholders::_1,
128  std::ref(args)...);
129  },
130  m_args));
131 
132  acc::AccCpuOmp2Threads<TDim, TIdx> acc(
133  *static_cast<workdiv::WorkDivMembers<TDim, TIdx> const *>(this),
134  blockSharedMemDynSizeBytes);
135 
136  // The number of threads in this block.
137  TIdx const blockThreadCount(blockThreadExtent.prod());
138  int const iBlockThreadCount(static_cast<int>(blockThreadCount));
139  alpaka::ignore_unused(iBlockThreadCount);
140 
141  if(::omp_in_parallel() != 0)
142  {
143  throw std::runtime_error("The OpenMP 2.0 thread backend can not be used within an existing parallel region!");
144  }
145 
146  // Force the environment to use the given number of threads.
147  int const ompIsDynamic(::omp_get_dynamic());
148  ::omp_set_dynamic(0);
149 
150  // Execute the blocks serially.
151  meta::ndLoopIncIdx(
152  gridBlockExtent,
153  [&](vec::Vec<TDim, TIdx> const & gridBlockIdx)
154  {
155  acc.m_gridBlockIdx = gridBlockIdx;
156 
157  // Execute the threads in parallel.
158 
159  // Parallel execution of the threads in a block is required because when syncBlockThreads is called all of them have to be done with their work up to this line.
160  // So we have to spawn one OS thread per thread in a block.
161  // 'omp for' is not useful because it is meant for cases where multiple iterations are executed by one thread but in our case a 1:1 mapping is required.
162  // Therefore we use 'omp parallel' with the specified number of threads in a block.
163  #pragma omp parallel num_threads(iBlockThreadCount)
164  {
165  #pragma omp single nowait
166  {
167  // The OpenMP runtime does not create a parallel region when only one thread is required in the num_threads clause.
168  // In all other cases we expect to be in a parallel region now.
169  if((iBlockThreadCount > 1) && (::omp_in_parallel() == 0))
170  {
171  throw std::runtime_error("The OpenMP 2.0 runtime did not create a parallel region!");
172  }
173 
174  // GCC fails with:
175  // error: redeclaration of const int& iBlockThreadCount
176  // if(numThreads != iBlockThreadCount)
177  // ^
178  // note: const int& iBlockThreadCount previously declared here
179  // #pragma omp parallel num_threads(iBlockThreadCount)
180  // ^
181 #if (!BOOST_COMP_GNUC) || (BOOST_COMP_GNUC >= BOOST_VERSION_NUMBER(6, 0, 0))
182  int const numThreads(::omp_get_num_threads());
183  if(numThreads != iBlockThreadCount)
184  {
185  throw std::runtime_error("The OpenMP 2.0 runtime did not use the number of threads that had been required!");
186  }
187 #endif
188  }
189  boundKernelFnObj(
190  acc);
191 
192  // Wait for all threads to finish before deleting the shared memory.
193  // This is done by default if the omp 'nowait' clause is missing on the omp parallel directive
194  //block::sync::syncBlockThreads(acc);
195  }
196 
197  // After a block has been processed, the shared memory has to be deleted.
198  block::shared::st::freeMem(acc);
199  });
200 
201  // Reset the dynamic thread number setting.
202  ::omp_set_dynamic(ompIsDynamic);
203  }
204 
205  private:
206  TKernelFnObj m_kernelFnObj;
207  std::tuple<std::decay_t<TArgs>...> m_args;
208  };
