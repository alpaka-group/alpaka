47  //! The CPU OpenMP 2.0 thread accelerator execution task.
48  template<typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
49  class TaskKernelCpuOmp2Threads final : public WorkDivMembers<TDim, TIdx>
50  {
51  public:
52  template<typename TWorkDiv>
53  ALPAKA_FN_HOST TaskKernelCpuOmp2Threads(TWorkDiv&& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
54  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
55  , m_kernelFnObj(kernelFnObj)
56  , m_args(std::forward<TArgs>(args)...)
57  {
58  static_assert(
59  Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
60  "The work division and the execution task have to be of the same dimensionality!");
61  }
62 
63  //! Executes the kernel function object.
64  ALPAKA_FN_HOST auto operator()() const -> void
65  {
66  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
67 
68  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(*this);
69  auto const blockThreadExtent = getWorkDiv<Block, Threads>(*this);
70  auto const threadElemExtent = getWorkDiv<Thread, Elems>(*this);
71 
72  // Get the size of the block shared dynamic memory.
73  auto const blockSharedMemDynSizeBytes = meta::apply(
74  [&](ALPAKA_DECAY_T(TArgs) const&... args)
75  {
76  return getBlockSharedMemDynSizeBytes<AccCpuOmp2Threads<TDim, TIdx>>(
77  m_kernelFnObj,
78  blockThreadExtent,
79  threadElemExtent,
80  args...);
81  },
82  m_args);
83 
84 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
85  std::cout << __func__ << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B"
86  << std::endl;
87 # endif
88  // Bind all arguments except the accelerator.
89  // TODO: With C++14 we could create a perfectly argument forwarding function object within the constructor.
90  auto const boundKernelFnObj = meta::apply(
91  [this](ALPAKA_DECAY_T(TArgs) const&... args)
92  { return std::bind(std::ref(m_kernelFnObj), std::placeholders::_1, std::ref(args)...); },
93  m_args);
94 
95  AccCpuOmp2Threads<TDim, TIdx> acc(
96  *static_cast<WorkDivMembers<TDim, TIdx> const*>(this),
97  blockSharedMemDynSizeBytes);
98 
99  // The number of threads in this block.
100  TIdx const blockThreadCount(blockThreadExtent.prod());
101  int const iBlockThreadCount(static_cast<int>(blockThreadCount));
102  alpaka::ignore_unused(iBlockThreadCount);
103 
104  if(::omp_in_parallel() != 0)
105  {
106  throw std::runtime_error(
107  "The OpenMP 2.0 thread backend can not be used within an existing parallel region!");
108  }
109 
110  // Force the environment to use the given number of threads.
111  int const ompIsDynamic(::omp_get_dynamic());
112  ::omp_set_dynamic(0);
113 
114  // Execute the blocks serially.
115  meta::ndLoopIncIdx(
116  gridBlockExtent,
117  [&](Vec<TDim, TIdx> const& gridBlockIdx)
118  {
119  acc.m_gridBlockIdx = gridBlockIdx;
120 
121 // Execute the threads in parallel.
122 
123 // Parallel execution of the threads in a block is required because when syncBlockThreads is called all of them have to
124 // be done with their work up to this line. So we have to spawn one OS thread per thread in a block. 'omp for' is not
125 // useful because it is meant for cases where multiple iterations are executed by one thread but in our case a 1:1
126 // mapping is required. Therefore we use 'omp parallel' with the specified number of threads in a block.
127 # pragma omp parallel num_threads(iBlockThreadCount)
128  {
129  // The guard is for gcc internal compiler error, as discussed in #735
130 # if(!BOOST_COMP_GNUC) || (BOOST_COMP_GNUC >= BOOST_VERSION_NUMBER(8, 1, 0))
131 # pragma omp single nowait
132  {
133  // The OpenMP runtime does not create a parallel region when only one thread is required in
134  // the num_threads clause. In all other cases we expect to be in a parallel region now.
135  if((iBlockThreadCount > 1) && (::omp_in_parallel() == 0))
136  {
137  throw std::runtime_error("The OpenMP 2.0 runtime did not create a parallel region!");
138  }
139 
140  int const numThreads(::omp_get_num_threads());
141  if(numThreads != iBlockThreadCount)
142  {
143  throw std::runtime_error("The OpenMP 2.0 runtime did not use the number of threads "
144  "that had been required!");
145  }
146  }
147 # endif
148  boundKernelFnObj(acc);
149 
150  // Wait for all threads to finish before deleting the shared memory.
151  // This is done by default if the omp 'nowait' clause is missing on the omp parallel directive
152  // syncBlockThreads(acc);
153  }
154 
155  // After a block has been processed, the shared memory has to be deleted.
156  freeSharedVars(acc);
157  });
158 
159  // Reset the dynamic thread number setting.
160  ::omp_set_dynamic(ompIsDynamic);
161  }
162 
163  private:
164  TKernelFnObj m_kernelFnObj;
165  std::tuple<std::decay_t<TArgs>...> m_args;
166  };
