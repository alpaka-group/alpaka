39  //! Queues the given task in the given queue.
40  //!
41  //! Special Handling for events:
42  //! If the event has previously been queued, then this call will overwrite any existing state of the event.
43  //! Any subsequent calls which examine the status of event will only examine the completion of this most recent
44  //! call to enqueue.
45  template<typename TQueue, typename TTask>
46  ALPAKA_FN_HOST auto enqueue(TQueue& queue, TTask&& task) -> void
47  {
48  traits::Enqueue<TQueue, std::decay_t<TTask>>::enqueue(queue, std::forward<TTask>(task));
49  }

51  //! Tests if the queue is empty (all ops in the given queue have been completed).
52  //!
53  //! \warning This function is allowed to return false negatives. An empty queue can reported as
54  //! non empty because the status information are not fully propagated by the used alpaka backend.
55  //! \return true queue is empty else false.

57  ALPAKA_FN_HOST auto empty(TQueue const& queue) -> bool
58  {
