98 /**
99  * This function changes internal state of tinymt32.
100  * Users should not call this function directly.
101  * @param random tinymt internal status
102  */
103 inline static void tinymt32_next_state(tinymt32_t * random) {
104  uint32_t x;
105  uint32_t y;
106 
107  y = random->status[3];
108  x = (random->status[0] & TINYMT32_MASK)
109  ^ random->status[1]
110  ^ random->status[2];
111  x ^= (x << TINYMT32_SH0);
112  y ^= (y >> TINYMT32_SH0) ^ x;
113  random->status[0] = random->status[1];
114  random->status[1] = random->status[2];
115  random->status[2] = x ^ (y << TINYMT32_SH1);
116  random->status[3] = y;
117  int32_t const a = -((int32_t)(y & 1)) & (int32_t)random->mat1;
118  int32_t const b = -((int32_t)(y & 1)) & (int32_t)random->mat2;
119  random->status[1] ^= (uint32_t)a;
120  random->status[2] ^= (uint32_t)b;
121 }

123 /**
124  * This function outputs 32-bit unsigned integer from internal state.
125  * Users should not call this function directly.
126  * @param random tinymt internal status
127  * @return 32-bit unsigned pseudorandom number
128  */
129 inline static uint32_t tinymt32_temper(tinymt32_t * random) {
130  uint32_t t0, t1;
131  t0 = random->status[3];
132 #if defined(LINEARITY_CHECK)
133  t1 = random->status[0]
134  ^ (random->status[2] >> TINYMT32_SH8);
135 #else
136  t1 = random->status[0]
137  + (random->status[2] >> TINYMT32_SH8);
138 #endif
139  t0 ^= t1;
140  if ((t1 & 1) != 0) {
141  t0 ^= random->tmat;
142  }
143  return t0;
144 }

146 /**
147  * This function outputs floating point number from internal state.
148  * Users should not call this function directly.
149  * @param random tinymt internal status
150  * @return floating point number r (1.0 <= r < 2.0)
151  */
152 inline static float tinymt32_temper_conv(tinymt32_t * random) {
153  uint32_t t0, t1;
154  union {
155  uint32_t u;
156  float f;
157  } conv;
158 
159  t0 = random->status[3];
160 #if defined(LINEARITY_CHECK)
161  t1 = random->status[0]
162  ^ (random->status[2] >> TINYMT32_SH8);
163 #else
164  t1 = random->status[0]
165  + (random->status[2] >> TINYMT32_SH8);
166 #endif
167  t0 ^= t1;
168  if ((t1 & 1) != 0) {
169  conv.u = ((t0 ^ random->tmat) >> 9) | UINT32_C(0x3f800000);
170  } else {
171  conv.u = (t0 >> 9) | UINT32_C(0x3f800000);
172  }
173  return conv.f;
174 }

216 /**
217  * This function outputs floating point number from internal state.
218  * This function is implemented using multiplying by (1 / 2^24).
219  * floating point multiplication is faster than using union trick in
220  * my Intel CPU.
221  * @param random tinymt internal status
222  * @return floating point number r (0.0 <= r < 1.0)
223  */
224 inline static float tinymt32_generate_float(tinymt32_t * random) {
225  tinymt32_next_state(random);

317 static void period_certification(tinymt32_t * random) {
318  if ((random->status[0] & TINYMT32_MASK) == 0 &&
319  random->status[1] == 0 &&
320  random->status[2] == 0 &&
321  random->status[3] == 0) {

324  random->status[2] = 'N';

329 /**
330  * This function initializes the internal state array with a 32-bit
331  * unsigned integer seed.
332  * @param random tinymt state vector.
333  * @param seed a 32-bit unsigned integer used as a seed.
334  */
335 void tinymt32_init(tinymt32_t * random, uint32_t seed) {
336  random->status[0] = seed;
337  random->status[1] = random->mat1;
338  random->status[2] = random->mat2;
339  random->status[3] = random->tmat;
340  for (unsigned int i = 1; i < MIN_LOOP; i++) {
341  random->status[i & 3] ^= i + UINT32_C(1812433253)
342  * (random->status[(i - 1) & 3]
343  ^ (random->status[(i - 1) & 3] >> 30));
344  }
345  period_certification(random);
346  for (unsigned int i = 0; i < PRE_LOOP; i++) {
347  tinymt32_next_state(random);
348  }
349 }

351 /**
352  * This function initializes the internal state array,
353  * with an array of 32-bit unsigned integers used as seeds
354  * @param random tinymt state vector.
355  * @param init_key the array of 32-bit integers, used as a seed.
356  * @param key_length the length of init_key.
357  */
358 void tinymt32_init_by_array(tinymt32_t * random, uint32_t init_key[],
359  int key_length) {
360  const unsigned int lag = 1;
361  const unsigned int mid = 1;
362  const unsigned int size = 4;
363  unsigned int i, j;
364  unsigned int count;
365  uint32_t r;
366  uint32_t * st = &random->status[0];
367 
368  st[0] = 0;
369  st[1] = random->mat1;
370  st[2] = random->mat2;
371  st[3] = random->tmat;
372  if (key_length + 1 > MIN_LOOP) {
373  count = (unsigned int)key_length + 1;
374  } else {
375  count = MIN_LOOP;
376  }
377  r = ini_func1(st[0] ^ st[mid % size]
378  ^ st[(size - 1) % size]);
379  st[mid % size] += r;
380  r += (unsigned int)key_length;
381  st[(mid + lag) % size] += r;
382  st[0] = r;
383  count--;
384  for (i = 1, j = 0; (j < count) && (j < (unsigned int)key_length); j++) {
385  r = ini_func1(st[i % size]
386  ^ st[(i + mid) % size]
387  ^ st[(i + size - 1) % size]);
388  st[(i + mid) % size] += r;
389  r += init_key[j] + i;
390  st[(i + mid + lag) % size] += r;
391  st[i % size] = r;
392  i = (i + 1) % size;
393  }
394  for (; j < count; j++) {
395  r = ini_func1(st[i % size]
396  ^ st[(i + mid) % size]
397  ^ st[(i + size - 1) % size]);
398  st[(i + mid) % size] += r;
399  r += i;
400  st[(i + mid + lag) % size] += r;
401  st[i % size] = r;
402  i = (i + 1) % size;
403  }
404  for (j = 0; j < size; j++) {
405  r = ini_func2(st[i % size]
406  + st[(i + mid) % size]
407  + st[(i + size - 1) % size]);
408  st[(i + mid) % size] ^= r;
409  r -= i;
410  st[(i + mid + lag) % size] ^= r;
411  st[i % size] = r;
412  i = (i + 1) % size;
413  }
414  period_certification(random);
415  for (i = 0; i < PRE_LOOP; i++) {
416  tinymt32_next_state(random);
417  }
418 }
