44  auto wait(std::size_t const& enqueueCount, std::unique_lock<std::mutex>& lk) const noexcept -> void
45  {
46  ALPAKA_ASSERT(enqueueCount <= m_enqueueCount);
47 
48  while(enqueueCount > m_LastReadyEnqueueCount)
49  {
50  auto future = m_future;
51  lk.unlock();
52  future.get();
53  lk.lock();
54  }
55  }
56

58 
59  std::mutex mutable m_mutex; //!< The mutex used to synchronize access to the event.
60  std::shared_future<void> m_future; //!< The future signaling the event completion.
61  std::size_t m_enqueueCount = 0u; //!< The number of times this event has been enqueued.
62  std::size_t m_LastReadyEnqueueCount = 0u; //!< The time this event has been ready the last time.
63  //!< Ready means that the event was not waiting within a queue
64  //!< (not enqueued or already completed). If m_enqueueCount ==
65  //!< m_LastReadyEnqueueCount, the event is currently not enqueued
66  };

124  //! The CPU non-blocking device queue enqueue trait specialization.
125  template<typename TDev>
126  struct Enqueue<alpaka::generic::detail::QueueGenericThreadsNonBlockingImpl<TDev>, EventGenericThreads<TDev>>
127  {
128  ALPAKA_FN_HOST static auto enqueue(
129  [[maybe_unused]] alpaka::generic::detail::QueueGenericThreadsNonBlockingImpl<TDev>& queueImpl,
130  EventGenericThreads<TDev>& event) -> void
131  {
132  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
133 
134  // Copy the shared pointer of the event implementation.
135  // This is forwarded to the lambda that is enqueued into the queue to ensure that the event
136  // implementation is alive as long as it is enqueued.
137  auto spEventImpl = event.m_spEventImpl;
138 
139  // Setting the event state and enqueuing it has to be atomic.
140  std::lock_guard<std::mutex> lk(spEventImpl->m_mutex);
141 
142  ++spEventImpl->m_enqueueCount;
143 
144  auto const enqueueCount = spEventImpl->m_enqueueCount;
145 
146  // Enqueue a task that only resets the events flag if it is completed.
147  spEventImpl->m_future = queueImpl.m_workerThread.submit(
148  [spEventImpl, enqueueCount]() mutable
149  {
150  std::unique_lock<std::mutex> lk2(spEventImpl->m_mutex);
151 
152  // Nothing to do if it has been re-enqueued to a later position in the queue.
153  if(enqueueCount == spEventImpl->m_enqueueCount)
154  {
155  spEventImpl->m_LastReadyEnqueueCount = spEventImpl->m_enqueueCount;
156  }
157  spEventImpl
158  .reset(); // avoid keeping the event alive as part of the background thread task's future
159  });
160  }
161  };

227  template<typename TDev>
228  ALPAKA_FN_HOST auto currentThreadWaitForDevice(TDev const& dev) -> void
229  {
230  // Get all the queues on the device at the time of invocation.
231  // All queues added afterwards are ignored.
232  auto vQueues = dev.getAllQueues();
233  // Furthermore there should not even be a chance to enqueue something between getting the queues and
234  // adding our wait events!
235  std::vector<EventGenericThreads<TDev>> vEvents;
236  for(auto&& spQueue : vQueues)
237  {
238  vEvents.emplace_back(dev);
239  spQueue->enqueue(vEvents.back());
240  }
241 
242  // Now wait for all the events.
243  for(auto&& event : vEvents)
244  {
245  wait(event);
246  }
247  }
