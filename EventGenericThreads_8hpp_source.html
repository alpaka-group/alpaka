38  //! The CPU device event implementation.
39  template<
40  typename TDev>
41  class EventGenericThreadsImpl final : public concepts::Implements<ConceptCurrentThreadWaitFor, EventGenericThreadsImpl<TDev>>
42  {
43  public:
44  //-----------------------------------------------------------------------------
45  EventGenericThreadsImpl(
46  TDev const & dev) noexcept :
47  m_dev(dev),
48  m_mutex(),
49  m_enqueueCount(0u),
50  m_LastReadyEnqueueCount(0u)
51  {}
52  //-----------------------------------------------------------------------------
53  EventGenericThreadsImpl(EventGenericThreadsImpl<TDev> const &) = delete;
54  //-----------------------------------------------------------------------------
55  EventGenericThreadsImpl(EventGenericThreadsImpl<TDev> &&) = delete;
56  //-----------------------------------------------------------------------------
57  auto operator=(EventGenericThreadsImpl<TDev> const &) -> EventGenericThreadsImpl<TDev> & = delete;
58  //-----------------------------------------------------------------------------
59  auto operator=(EventGenericThreadsImpl<TDev> &&) -> EventGenericThreadsImpl<TDev> & = delete;
60  //-----------------------------------------------------------------------------
61  ~EventGenericThreadsImpl() noexcept = default;
62 
63  //-----------------------------------------------------------------------------
64  auto isReady() noexcept -> bool
65  {
66  return (m_LastReadyEnqueueCount == m_enqueueCount);
67  }
68 
69  //-----------------------------------------------------------------------------
70  auto wait(std::size_t const & enqueueCount, std::unique_lock<std::mutex>& lk) const noexcept -> void
71  {
72  ALPAKA_ASSERT(enqueueCount <= m_enqueueCount);
73 
74  while(enqueueCount > m_LastReadyEnqueueCount)
75  {
76  auto future = m_future;
77  lk.unlock();
78  future.get();
79  lk.lock();
80  }
81  }
82 
83  public:
84  TDev const m_dev; //!< The device this event is bound to.
85 
86  std::mutex mutable m_mutex; //!< The mutex used to synchronize access to the event.
87  std::shared_future<void> m_future; //!< The future signaling the event completion.
88  std::size_t m_enqueueCount; //!< The number of times this event has been enqueued.
89  std::size_t m_LastReadyEnqueueCount; //!< The time this event has been ready the last time.
90  //!< Ready means that the event was not waiting within a queue (not enqueued or already completed).
91  //!< If m_enqueueCount == m_LastReadyEnqueueCount, the event is currently not enqueued
92  };
93  }
94  }

176  //! The CPU non-blocking device queue enqueue trait specialization.
177  template<
178  typename TDev>
179  struct Enqueue<
180  alpaka::generic::detail::QueueGenericThreadsNonBlockingImpl<TDev>,
181  EventGenericThreads<TDev>>
182  {
183  //-----------------------------------------------------------------------------
184  ALPAKA_FN_HOST static auto enqueue(
185  alpaka::generic::detail::QueueGenericThreadsNonBlockingImpl<TDev> & queueImpl,
186  EventGenericThreads<TDev> & event)
187  -> void
188  {
189 #if (BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
190  alpaka::ignore_unused(queueImpl);
191 #endif
192  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
193 
194  // Copy the shared pointer of the event implementation.
195  // This is forwarded to the lambda that is enqueued into the queue to ensure that the event implementation is alive as long as it is enqueued.
196  auto spEventImpl(event.m_spEventImpl);
197 
198  // Setting the event state and enqueuing it has to be atomic.
199  std::lock_guard<std::mutex> lk(spEventImpl->m_mutex);
200 
201  ++spEventImpl->m_enqueueCount;
202 
203 // Workaround: Clang can not support this when natively compiling device code. See ConcurrentExecPool.hpp.
204 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
205  auto const enqueueCount = spEventImpl->m_enqueueCount;
206 
207  // Enqueue a task that only resets the events flag if it is completed.
208  spEventImpl->m_future = queueImpl.m_workerThread.enqueueTask(
209  [spEventImpl, enqueueCount]()
210  {
211  std::unique_lock<std::mutex> lk2(spEventImpl->m_mutex);
212 
213  // Nothing to do if it has been re-enqueued to a later position in the queue.
214  if(enqueueCount == spEventImpl->m_enqueueCount)
215  {
216  spEventImpl->m_LastReadyEnqueueCount = spEventImpl->m_enqueueCount;
217  }
218  });
219 #endif
220  }
221  };

272 
273  eventImpl.m_future = promise.get_future();
274  }
275 
276  queueImpl.m_bCurrentlyExecutingTask = false;

305  template<typename TDev>
306  ALPAKA_FN_HOST auto currentThreadWaitForDevice(
307  TDev const & dev
308  )
309  ->void
310  {
311  // Get all the queues on the device at the time of invocation.
312  // All queues added afterwards are ignored.
313  auto vQueues(dev.getAllQueues());
314  // Furthermore there should not even be a chance to enqueue something between getting the queues and adding our wait events!
315  std::vector<EventGenericThreads<TDev>> vEvents;
316  for(auto && spQueue : vQueues)
317  {
318  vEvents.emplace_back(dev);
319  spQueue->enqueue(vEvents.back());
320  }
321 
322  // Now wait for all the events.
323  for(auto && event : vEvents)
324  {
325  wait(event);
326  }
327  }
328  }
