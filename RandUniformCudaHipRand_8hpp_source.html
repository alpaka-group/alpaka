181  //! The CUDA/HIP random number float uniform distribution.
182  template<>
183  class UniformReal<
184  float>
185  {
186  public:
187  //-----------------------------------------------------------------------------
188  UniformReal() = default;
189 
190  //-----------------------------------------------------------------------------
191  template<
192  typename TGenerator>
193  __device__ auto operator()(
194  TGenerator & generator)
195  -> float
196  {
197  // (0.f, 1.0f]
198 #ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
199  float const fUniformRand(curand_uniform(&generator.m_State));
200 #else
201  float const fUniformRand(hiprand_uniform(&generator.m_State));
202 #endif
203  // NOTE: (1.0f - curand_uniform) does not work, because curand_uniform seems to return denormalized floats around 0.f.
204  // [0.f, 1.0f)
205  return fUniformRand * static_cast<float>( fUniformRand != 1.0f );
206  }
207  };
