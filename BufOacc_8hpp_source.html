43  //! The OpenACC memory buffer detail.
44  template<typename TElem, typename TDim, typename TIdx>
45  class BufOaccImpl
46  {
47  static_assert(
48  !std::is_const<TElem>::value,
49  "The elem type of the buffer can not be const because the C++ Standard forbids containers of "
50  "const elements!");
51  static_assert(!std::is_const<TIdx>::value, "The idx type of the buffer can not be const!");
52 
53  private:
54  using Elem = TElem;
55  using Dim = TDim;
56  //! Calculate the pitches purely from the extents.
57  template<typename TExtent>
58  ALPAKA_FN_HOST static auto calculatePitchesFromExtents(TExtent const& extent) -> Vec<TDim, TIdx>
59  {
60  Vec<TDim, TIdx> pitchBytes(Vec<TDim, TIdx>::all(0));
61  pitchBytes[TDim::value - 1u] = extent[TDim::value - 1u] * static_cast<TIdx>(sizeof(TElem));
62  for(TIdx i = TDim::value - 1u; i > static_cast<TIdx>(0u); --i)
63  {
64  pitchBytes[i - 1] = extent[i - 1] * pitchBytes[i];
65  }
66  return pitchBytes;
67  }
68 
69  public:
70  //! Constructor
71  template<typename TExtent>
72  ALPAKA_FN_HOST BufOaccImpl(DevOacc const& dev, TElem* const pMem, TExtent const& extent)
73  : m_dev(dev)
74  , m_extentElements(extent::getExtentVecEnd<TDim>(extent))
75  , m_pitchBytes(calculatePitchesFromExtents(m_extentElements))
76  , m_pMem(pMem)
77  {
78  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
79 
80  static_assert(
81  TDim::value == alpaka::Dim<TExtent>::value,
82  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to "
83  "be identical!");
84  static_assert(
85  std::is_same<TIdx, Idx<TExtent>>::value,
86  "The idx type of TExtent and the TIdx template parameter have to be identical!");
87  }
88 
89  public:
90  DevOacc m_dev;
91  Vec<TDim, TIdx> m_extentElements;
92  Vec<TDim, TIdx> m_pitchBytes;
93  TElem* m_pMem;
94 
95  BufOaccImpl(BufOaccImpl&&) = default;
96  BufOaccImpl& operator=(BufOaccImpl&&) = default;
