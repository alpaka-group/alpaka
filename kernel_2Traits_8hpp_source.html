220  //! Creates a kernel execution task.
221  //!
222  //! \tparam TAcc The accelerator type.
223  //! \param workDiv The index domain work division.
224  //! \param kernelFnObj The kernel function object which should be executed.
225  //! \param args,... The kernel invocation arguments.
226  //! \return The kernel execution task.
227 #if BOOST_COMP_CLANG
228 # pragma clang diagnostic pop
229 #endif
230  template<typename TAcc, typename TWorkDiv, typename TKernelFnObj, typename... TArgs>
231  ALPAKA_FN_HOST auto createTaskKernel(TWorkDiv const& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
232  {
233  // check for void return type
234  detail::CheckFnReturnType<TAcc>{}(kernelFnObj, args...);
235 
236  static_assert(
237  Dim<std::decay_t<TWorkDiv>>::value == Dim<TAcc>::value,
238  "The dimensions of TAcc and TWorkDiv have to be identical!");
239  static_assert(
240  std::is_same<Idx<std::decay_t<TWorkDiv>>, Idx<TAcc>>::value,
241  "The idx type of TAcc and the idx type of TWorkDiv have to be identical!");
242 
243 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
244  std::cout << __func__ << " workDiv: " << workDiv << ", kernelFnObj: " << typeid(kernelFnObj).name()
245  << std::endl;
246 #endif
247  return traits::CreateTaskKernel<TAcc, TWorkDiv, TKernelFnObj, TArgs...>::createTaskKernel(
248  workDiv,
249  kernelFnObj,
250  std::forward<TArgs>(args)...);
251  }
