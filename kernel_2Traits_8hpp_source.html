219  //! Creates a kernel execution task.
220  //!
221  //! \tparam TAcc The accelerator type.
222  //! \param workDiv The index domain work division.
223  //! \param kernelFnObj The kernel function object which should be executed.
224  //! \param args,... The kernel invocation arguments.
225  //! \return The kernel execution task.
226 #if BOOST_COMP_CLANG
227 # pragma clang diagnostic pop
228 #endif
229  template<typename TAcc, typename TWorkDiv, typename TKernelFnObj, typename... TArgs>
230  ALPAKA_FN_HOST auto createTaskKernel(TWorkDiv const& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
231  {
232  // check for void return type
233  detail::CheckFnReturnType<TAcc>{}(kernelFnObj, args...);
234 
235 #if BOOST_COMP_NVCC
236  static_assert(
237  std::is_trivially_copyable_v<TKernelFnObj> || __nv_is_extended_device_lambda_closure_type(TKernelFnObj)
238  || __nv_is_extended_host_device_lambda_closure_type(TKernelFnObj),
239  "Kernels must be trivially copyable or an extended CUDA lambda expression!");
240 #else
241  static_assert(std::is_trivially_copyable_v<TKernelFnObj>, "Kernels must be trivially copyable!");
242 #endif
243  static_assert(
244  (std::is_trivially_copyable_v<std::decay_t<TArgs>> && ...),
245  "Kernel arguments must be trivially copyable!");
246  static_assert(
247  Dim<std::decay_t<TWorkDiv>>::value == Dim<TAcc>::value,
248  "The dimensions of TAcc and TWorkDiv have to be identical!");
249  static_assert(
250  std::is_same_v<Idx<std::decay_t<TWorkDiv>>, Idx<TAcc>>,
251  "The idx type of TAcc and the idx type of TWorkDiv have to be identical!");
252 
253 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
254  std::cout << __func__ << " workDiv: " << workDiv
255  << ", kernelFnObj: " << core::demangled<decltype(kernelFnObj)> << std::endl;
256 #endif
257  return trait::CreateTaskKernel<TAcc, TWorkDiv, TKernelFnObj, TArgs...>::createTaskKernel(
258  workDiv,
259  kernelFnObj,
260  std::forward<TArgs>(args)...);
261  }
