61  static auto getDevByIdx(std::size_t const& devIdx)
62  {
63  ALPAKA_DEBUG_FULL_LOG_SCOPE;
64 
65  auto exception_handler = [](sycl::exception_list exceptions)
66  {
67  auto ss_err = std::stringstream{};
68  ss_err << "Caught asynchronous SYCL exception(s):\n";
69  for(std::exception_ptr e : exceptions)
70  {
71  try
72  {
73  std::rethrow_exception(e);
74  }
75  catch(sycl::exception const& err)
76  {
77  ss_err << err.what() << " (" << err.code() << ")\n";
78  }
79  }
80  throw std::runtime_error(ss_err.str());
81  };
82 
83  static auto pf = sycl::platform{typename TPltf::selector{}};
84  static auto devices = pf.get_devices();
85  static auto ctx = sycl::context{devices, exception_handler};
86 
87  if(devIdx >= devices.size())
88  {
89  auto ss_err = std::stringstream{};
90  ss_err << "Unable to return device handle for device " << devIdx << ". There are only "
91  << devices.size() << " SYCL devices!";
92  throw std::runtime_error(ss_err.str());
93  }
94 
95  auto sycl_dev = devices.at(devIdx);
96 
97  // Log this device.
98 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
99  printDeviceProperties(sycl_dev);
100 # elif ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
101  std::cout << __func__ << sycl_dev.get_info<info::device::name>() << '\n';
102 # endif
103  return typename DevType<TPltf>::type{sycl_dev, ctx};
104  }
