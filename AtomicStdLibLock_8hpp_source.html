20  //! The CPU threads accelerator atomic ops.
21  //
22  // Atomics can be used in the grids, blocks and threads hierarchy levels.
23  // Atomics are not guaranteed to be save between devices.
24  //
25  // \tparam THashTableSize size of the hash table to allow concurrency between
26  // atomics to different addresses
27  template<size_t THashTableSize>
28  class AtomicStdLibLock
29  {
30  public:
31  template<typename TAtomic, typename TOp, typename T, typename THierarchy, typename TSfinae>
32  friend struct trait::AtomicOp;

34  static constexpr auto nextPowerOf2(size_t const value, size_t const bit = 0u) -> size_t
35  {
36  return value <= (static_cast<size_t>(1u) << bit) ? (static_cast<size_t>(1u) << bit)
37  : nextPowerOf2(value, bit + 1u);
38  }
39 
40  //! get a hash value of the pointer
41  //
42  // This is no perfect hash, there will be collisions if the size of pointer type
43  // is not a power of two.
44  template<typename TPtr>
45  static auto hash(TPtr const* const ptr) -> size_t
46  {
47  auto const ptrAddr = reinterpret_cast<size_t>(ptr);
48  // using power of two for the next division will increase the performance
49  constexpr size_t typeSizePowerOf2 = nextPowerOf2(sizeof(TPtr));
50  // division removes the stride between indices
51  return (ptrAddr / typeSizePowerOf2);
52  }
53 
54  template<typename TPtr>
55  auto getMutex(TPtr const* const ptr) const -> std::mutex&
56  {
57  //! get the size of the hash table
58  //
59  // The size is at least 1 or THashTableSize rounded up to the next power of 2
60  constexpr size_t hashTableSize = THashTableSize == 0u ? 1u : nextPowerOf2(THashTableSize);
61 
62  size_t const hashedAddr = hash(ptr) & (hashTableSize - 1u);
63 #if BOOST_COMP_CLANG
64 # pragma clang diagnostic push
65 # pragma clang diagnostic ignored "-Wexit-time-destructors"
66 #endif
67  static std::array<
68  std::mutex,
69  hashTableSize>
70  m_mtxAtomic; //!< The mutex protecting access for an atomic operation.
71 #if BOOST_COMP_CLANG
72 # pragma clang diagnostic pop
73 #endif
74  return m_mtxAtomic[hashedAddr];
75  }
76  };
