24  //! The CPU threads accelerator atomic ops.
25  //
26  // Atomics can be used in the grids, blocks and threads hierarchy levels.
27  // Atomics are not guaranteed to be save between devices.

29  // \tparam THashTableSize size of the hash table to allow concurrency between
30  // atomics to different addresses

32  class AtomicStdLibLock

43  static constexpr size_t nextPowerOf2(size_t const value, size_t const bit = 0u)
44  {
45  return value <= (static_cast<size_t>(1u) << bit) ?
46  (static_cast<size_t>(1u) << bit) : nextPowerOf2(value, bit + 1u);
47  }

49  //-----------------------------------------------------------------------------
50  //! get a hash value of the pointer
51  //
52  // This is no perfect hash, there will be collisions if the size of pointer type
53  // is not a power of two.
54  template<typename TPtr>
55  static size_t hash(TPtr const * const ptr)
56  {
57  size_t const ptrAddr = reinterpret_cast< size_t >( ptr );
58  // using power of two for the next division will increase the performance
59  constexpr size_t typeSizePowerOf2 = nextPowerOf2(sizeof(TPtr));
60  // division removes the stride between indices
61  return (ptrAddr / typeSizePowerOf2);
62  }
