53  //! The CPU threads accelerator.
54  //!
55  //! This accelerator allows parallel kernel execution on a CPU device.
56  //! It uses std::thread to implement the parallelism.
57  template<
58  typename TDim,
59  typename TIdx>
60  class AccCpuThreads final :
61  public WorkDivMembers<TDim, TIdx>,
62  public gb::IdxGbRef<TDim, TIdx>,
63  public bt::IdxBtRefThreadIdMap<TDim, TIdx>,
64  public AtomicHierarchy<
65  AtomicStdLibLock<16>, // grid atomics
66  AtomicStdLibLock<16>, // block atomics
67  AtomicStdLibLock<16> // thread atomics
68  >,
69  public math::MathStdLib,
70  public BlockSharedMemDynAlignedAlloc,
71  public BlockSharedMemStMasterSync,
72  public BlockSyncBarrierThread<TIdx>,
73  public IntrinsicCpu,
74  public rand::RandStdLib,
75  public TimeStdLib,
76  public warp::WarpSingleThread,
77  public concepts::Implements<ConceptAcc, AccCpuThreads<TDim, TIdx>>
78  {
79  static_assert(
80  sizeof(TIdx) >= sizeof(int),
81  "Index type is not supported, consider using int or a larger type.");
82 
83  public:
84  // Partial specialization with the correct TDim and TIdx is not allowed.
85  template<typename TDim2, typename TIdx2, typename TKernelFnObj, typename... TArgs>
86  friend class ::alpaka::TaskKernelCpuThreads;
87 
88  private:
89  //-----------------------------------------------------------------------------

91  ALPAKA_FN_HOST AccCpuThreads(TWorkDiv const& workDiv, std::size_t const& blockSharedMemDynSizeBytes)
92  : WorkDivMembers<TDim, TIdx>(workDiv)
93  , gb::IdxGbRef<TDim, TIdx>(m_gridBlockIdx)
94  , bt::IdxBtRefThreadIdMap<TDim, TIdx>(m_threadToIndexMap)

152 # ifdef ALPAKA_CI
153  auto const blockThreadCountMax(static_cast<TIdx>(8));
154 # else
155  // \TODO: Magic number. What is the maximum? Just set a reasonable value? There is a implementation
156  // defined maximum where the creation of a new thread crashes. std::thread::hardware_concurrency can
157  // return 0, so 1 is the default case?
158  auto const blockThreadCountMax(std::max(
159  static_cast<TIdx>(1),
160  alpaka::core::clipCast<TIdx>(std::thread::hardware_concurrency() * 8)));
161 # endif
162  return {// m_multiProcessorCount
163  static_cast<TIdx>(1),
164  // m_gridBlockExtentMax
165  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
166  // m_gridBlockCountMax
167  std::numeric_limits<TIdx>::max(),
168  // m_blockThreadExtentMax
169  Vec<TDim, TIdx>::all(blockThreadCountMax),
170  // m_blockThreadCountMax
171  blockThreadCountMax,
172  // m_threadElemExtentMax

174  // m_threadElemCountMax
