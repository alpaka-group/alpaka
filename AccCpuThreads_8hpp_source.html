48  //! The CPU threads accelerator.

50  //! This accelerator allows parallel kernel execution on a CPU device.

60  AtomicCpu, // grid atomics

62  AtomicCpu // thread atomics
63  >,
64  public math::MathStdLib,

67  public BlockSyncBarrierThread<TIdx>,
