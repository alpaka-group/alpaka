43  //! The CUDA/HIP RT queue implementation.
44  class QueueUniformCudaHipRtImpl final
45  {
46  public:
47  ALPAKA_FN_HOST QueueUniformCudaHipRtImpl(DevUniformCudaHipRt const& dev)
48  : m_dev(dev)
49  , m_UniformCudaHipQueue()
50  {
51  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
52 
53  // Set the current device.
54  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(m_dev.getNativeHandle()));
55 
56  // - [cuda/hip]StreamDefault: Default queue creation flag.
57  // - [cuda/hip]StreamNonBlocking: Specifies that work running in the created queue may run
58  // concurrently with work in queue 0 (the NULL queue),
59  // and that the created queue should perform no implicit synchronization with queue 0.
60  // Create the queue on the current device.
61  // NOTE: [cuda/hip]StreamNonBlocking is required to match the semantic implemented in the alpaka
62  // CPU queue. It would be too much work to implement implicit default queue synchronization on CPU.
63 
64  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(
65  StreamCreateWithFlags)(&m_UniformCudaHipQueue, ALPAKA_API_PREFIX(StreamNonBlocking)));
66  }
67  QueueUniformCudaHipRtImpl(QueueUniformCudaHipRtImpl&&) = default;
68  auto operator=(QueueUniformCudaHipRtImpl&&) -> QueueUniformCudaHipRtImpl& = delete;
69  ALPAKA_FN_HOST ~QueueUniformCudaHipRtImpl()
70  {
71  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
72 
73  // In case the device is still doing work in the queue when cuda/hip-StreamDestroy() is called, the
74  // function will return immediately and the resources associated with queue will be released
75  // automatically once the device has completed all work in queue.
76  // -> No need to synchronize here.
77  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(ALPAKA_API_PREFIX(StreamDestroy)(m_UniformCudaHipQueue));
78  }
79 
80  [[nodiscard]] auto getNativeHandle() const noexcept
81  {
82  return m_UniformCudaHipQueue;
83  }
84 
85  public:
86  DevUniformCudaHipRt const m_dev; //!< The device this queue is bound to.
87  private:
88  ALPAKA_API_PREFIX(Stream_t) m_UniformCudaHipQueue;
89  };

185  //! The CUDA/HIP RT queue enqueue trait specialization.
186  template<bool TBlocking, typename TTask>
187  struct Enqueue<uniform_cuda_hip::detail::QueueUniformCudaHipRt<TBlocking>, TTask>
188  {
189  enum class CallbackState
190  {
191  enqueued,
192  notified,
193  finished,
194  };
195 
196  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
197  {
198  std::mutex m_mutex;
199  std::condition_variable m_event;
200  CallbackState m_state = CallbackState::enqueued;
201  };
202 
203 # if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
204  static void CUDART_CB
205 # else
206  static void HIPRT_CB
207 # endif
208  uniformCudaHipRtCallback(
209  ALPAKA_API_PREFIX(Stream_t) /*queue*/,
210  ALPAKA_API_PREFIX(Error_t) /*status*/,
211  void* arg)
212  {
213  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread
214  // has already finished.
215  const auto spCallbackSynchronizationData
216  = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
217 
218  // Notify the executing thread.
219  {
220  std::unique_lock<std::mutex> lock(spCallbackSynchronizationData->m_mutex);
221  spCallbackSynchronizationData->m_state = CallbackState::notified;
222  }
223  spCallbackSynchronizationData->m_event.notify_one();
224 
225  // Wait for the executing thread to finish the task if it has not already finished.
226  std::unique_lock<std::mutex> lock(spCallbackSynchronizationData->m_mutex);
227  if(spCallbackSynchronizationData->m_state != CallbackState::finished)
228  {
229  spCallbackSynchronizationData->m_event.wait(
230  lock,
231  [&spCallbackSynchronizationData]()
232  { return spCallbackSynchronizationData->m_state == CallbackState::finished; });
233  }
234  }
235 
236  ALPAKA_FN_HOST static auto enqueue(
237  uniform_cuda_hip::detail::QueueUniformCudaHipRt<TBlocking>& queue,
238  TTask const& task) -> void
239  {
240  auto spCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
241  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamAddCallback)(
242  queue.getNativeHandle(),
243  uniformCudaHipRtCallback,
244  spCallbackSynchronizationData.get(),
245  0u));
246 
247  // We start a new std::thread which stores the task to be executed.
248  // This circumvents the limitation that it is not possible to call CUDA/HIP methods within the CUDA/HIP
249  // callback thread. The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
250  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
251  // before it executes the next task in the queue (CUDA/HIP stream).
252  std::thread t(
253  [spCallbackSynchronizationData, task]()
254  {
255  // If the callback has not yet been called, we wait for it.
256  {
257  std::unique_lock<std::mutex> lock(spCallbackSynchronizationData->m_mutex);
258  if(spCallbackSynchronizationData->m_state != CallbackState::notified)
259  {
260  spCallbackSynchronizationData->m_event.wait(
261  lock,
262  [&spCallbackSynchronizationData]()
263  { return spCallbackSynchronizationData->m_state == CallbackState::notified; });
264  }
265 
266  task();
267 
268  // Notify the waiting CUDA/HIP thread.
269  spCallbackSynchronizationData->m_state = CallbackState::finished;
270  }
271  spCallbackSynchronizationData->m_event.notify_one();
272  });
273 
274  if constexpr(TBlocking)
275  {
276  t.join();
277  }
278  else
279  {
280  t.detach();
281  }
282  }
283  };
