35  //! The CUDA/HIP RT queue implementation.
36  template<typename TApi>
37  class QueueUniformCudaHipRtImpl final
38  {
39  public:
40  ALPAKA_FN_HOST QueueUniformCudaHipRtImpl(DevUniformCudaHipRt<TApi> const& dev)
41  : m_dev(dev)
42  , m_UniformCudaHipQueue()
43  {
44  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
45 
46  // Set the current device.
47  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::setDevice(m_dev.getNativeHandle()));
48 
49  // - [cuda/hip]StreamDefault: Default queue creation flag.
50  // - [cuda/hip]StreamNonBlocking: Specifies that work running in the created queue may run
51  // concurrently with work in queue 0 (the NULL queue),
52  // and that the created queue should perform no implicit synchronization with queue 0.
53  // Create the queue on the current device.
54  // NOTE: [cuda/hip]StreamNonBlocking is required to match the semantic implemented in the alpaka
55  // CPU queue. It would be too much work to implement implicit default queue synchronization on CPU.
56 
57  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
58  TApi::streamCreateWithFlags(&m_UniformCudaHipQueue, TApi::streamNonBlocking));
59  }
60  QueueUniformCudaHipRtImpl(QueueUniformCudaHipRtImpl&&) = default;
61  auto operator=(QueueUniformCudaHipRtImpl&&) -> QueueUniformCudaHipRtImpl& = delete;
62  ALPAKA_FN_HOST ~QueueUniformCudaHipRtImpl()
63  {
64  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
65 
66  // Make sure all pending async work is finished before destroying the stream to guarantee determinism.
67  // This would not be necessary for plain CUDA/HIP operations, but we can have host functions in the
68  // stream, which reference this queue instance and its CallbackThread. Make sure they are done.
69  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(TApi::streamSynchronize(m_UniformCudaHipQueue));
70  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(TApi::streamDestroy(m_UniformCudaHipQueue));
71  }
72 
73  [[nodiscard]] auto getNativeHandle() const noexcept
74  {
75  return m_UniformCudaHipQueue;
76  }
77 
78  public:
79  DevUniformCudaHipRt<TApi> const m_dev; //!< The device this queue is bound to.
80  core::CallbackThread m_callbackThread;
81 
82  private:
83  typename TApi::Stream_t m_UniformCudaHipQueue;
84  };

184  //! The CUDA/HIP RT blocking queue enqueue trait specialization.
185  template<typename TApi, bool TBlocking, typename TTask>
186  struct Enqueue<uniform_cuda_hip::detail::QueueUniformCudaHipRt<TApi, TBlocking>, TTask>
187  {
188  using QueueImpl = uniform_cuda_hip::detail::QueueUniformCudaHipRtImpl<TApi>;
189 
190  struct HostFuncData
191  {
192  // We don't need to keep the queue alive, because in it's dtor it will synchronize with the CUDA/HIP
193  // stream and wait until all host functions and the CallbackThread are done. It's actually an error to
194  // copy the queue into the host function. Destroying it here would call CUDA/HIP APIs from the host
195  // function. Passing it further to the Callback thread, would make the Callback thread hold a task
196  // containing the queue with the CallbackThread itself. Destroying the task if no other queue instance
197  // exists will make the CallbackThread join itself and crash.
198  QueueImpl& q;
199  TTask t;
200  };
201 
202  ALPAKA_FN_HOST static void uniformCudaHipRtHostFunc(void* arg)
203  {
204  auto data = std::unique_ptr<HostFuncData>(reinterpret_cast<HostFuncData*>(arg));
205  auto& queue = data->q;
206  auto f = queue.m_callbackThread.submit(
207  [data = std::move(data)]() mutable
208  {
209  data->t();
210  data.reset(); // destroy the task
211  });
212  f.wait();
213  } // destroys the future `f`, destroying the packaged task and the above lambda
214 
215  ALPAKA_FN_HOST static auto enqueue(
216  uniform_cuda_hip::detail::QueueUniformCudaHipRt<TApi, TBlocking>& queue,
217  TTask const& task) -> void
218  {
219  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::launchHostFunc(
220  queue.getNativeHandle(),
221  uniformCudaHipRtHostFunc,
222  new HostFuncData{*queue.m_spQueueImpl, task}));
223  if constexpr(TBlocking)
224  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::streamSynchronize(queue.getNativeHandle()));
225  }
226  };
