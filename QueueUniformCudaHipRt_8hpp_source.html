46  //! The CUDA/HIP RT queue implementation.
47  template<typename TApi>
48  class QueueUniformCudaHipRtImpl final
49  {
50  public:
51  ALPAKA_FN_HOST QueueUniformCudaHipRtImpl(DevUniformCudaHipRt<TApi> const& dev)
52  : m_dev(dev)
53  , m_UniformCudaHipQueue()
54  {
55  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
56 
57  // Set the current device.
58  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::setDevice(m_dev.getNativeHandle()));
59 
60  // - [cuda/hip]StreamDefault: Default queue creation flag.
61  // - [cuda/hip]StreamNonBlocking: Specifies that work running in the created queue may run
62  // concurrently with work in queue 0 (the NULL queue),
63  // and that the created queue should perform no implicit synchronization with queue 0.
64  // Create the queue on the current device.
65  // NOTE: [cuda/hip]StreamNonBlocking is required to match the semantic implemented in the alpaka
66  // CPU queue. It would be too much work to implement implicit default queue synchronization on CPU.
67 
68  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
69  TApi::streamCreateWithFlags(&m_UniformCudaHipQueue, TApi::streamNonBlocking));
70  }
71  QueueUniformCudaHipRtImpl(QueueUniformCudaHipRtImpl&&) = default;
72  auto operator=(QueueUniformCudaHipRtImpl&&) -> QueueUniformCudaHipRtImpl& = delete;
73  ALPAKA_FN_HOST ~QueueUniformCudaHipRtImpl()
74  {
75  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
76 
77  // In case the device is still doing work in the queue when cuda/hip-StreamDestroy() is called, the
78  // function will return immediately and the resources associated with queue will be released
79  // automatically once the device has completed all work in queue.
80  // -> No need to synchronize here.
81  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(TApi::streamDestroy(m_UniformCudaHipQueue));
82  }
83 
84  [[nodiscard]] auto getNativeHandle() const noexcept
85  {
86  return m_UniformCudaHipQueue;
87  }
88 
89  public:
90  DevUniformCudaHipRt<TApi> const m_dev; //!< The device this queue is bound to.
91  core::CallbackThread m_callbackThread;
92 
93  private:
94  typename TApi::Stream_t m_UniformCudaHipQueue;
95  };

195  //! The CUDA/HIP RT blocking queue enqueue trait specialization.
196  template<typename TApi, bool TBlocking, typename TTask>
197  struct Enqueue<uniform_cuda_hip::detail::QueueUniformCudaHipRt<TApi, TBlocking>, TTask>
198  {
199  enum class CallbackState
200  {
201  enqueued,
202  notified,
203  finished,
204  };
205 
206  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
207  {
208  std::mutex m_mutex;
209  std::condition_variable m_event;
210  CallbackState m_state = CallbackState::enqueued;
211  };
212 
213  ALPAKA_FN_HOST static void uniformCudaHipRtHostFunc(void* arg)
214  {
215  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread
216  // has already finished.
217  const auto spCallbackSynchronizationData
218  = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
219 
220  // Notify the executing thread.
221  {
222  std::unique_lock<std::mutex> lock(spCallbackSynchronizationData->m_mutex);
223  spCallbackSynchronizationData->m_state = CallbackState::notified;
224  }
225  spCallbackSynchronizationData->m_event.notify_one();
226 
227  // Wait for the executing thread to finish the task if it has not already finished.
228  std::unique_lock<std::mutex> lock(spCallbackSynchronizationData->m_mutex);
229  if(spCallbackSynchronizationData->m_state != CallbackState::finished)
230  {
231  spCallbackSynchronizationData->m_event.wait(
232  lock,
233  [&spCallbackSynchronizationData]()
234  { return spCallbackSynchronizationData->m_state == CallbackState::finished; });
235  }
236  }
237 
238  ALPAKA_FN_HOST static auto enqueue(
239  uniform_cuda_hip::detail::QueueUniformCudaHipRt<TApi, TBlocking>& queue,
240  TTask const& task) -> void
241  {
242  auto spCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
243  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::launchHostFunc(
244  queue.getNativeHandle(),
245  uniformCudaHipRtHostFunc,
246  spCallbackSynchronizationData.get()));
247 
248  // We start a new std::thread which stores the task to be executed.
249  // This circumvents the limitation that it is not possible to call CUDA/HIP methods within the CUDA/HIP
250  // callback thread. The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
251  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
252  // before it executes the next task in the queue (CUDA/HIP stream).
253  auto f = queue.getCallbackThread().submit(std::packaged_task<void()>(
254  [spCallbackSynchronizationData, task]()
255  {
256  // If the callback has not yet been called, we wait for it.
257  {
258  std::unique_lock<std::mutex> lock(spCallbackSynchronizationData->m_mutex);
259  if(spCallbackSynchronizationData->m_state != CallbackState::notified)
260  {
261  spCallbackSynchronizationData->m_event.wait(
262  lock,
263  [&spCallbackSynchronizationData]()
264  { return spCallbackSynchronizationData->m_state == CallbackState::notified; });
265  }
266 
267  task();
268 
269  // Notify the waiting CUDA/HIP thread.
270  spCallbackSynchronizationData->m_state = CallbackState::finished;
271  }
272  spCallbackSynchronizationData->m_event.notify_one();
273  }));
274 
275  if constexpr(TBlocking)
276  {
277  f.wait();
278  }
279  }
280  };
