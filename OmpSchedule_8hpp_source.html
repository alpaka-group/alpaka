25  //! Representation of OpenMP schedule information: kind and chunk size. This class can be used regardless of
26  //! whether OpenMP is enabled.
27  struct Schedule
28  {
29  //! Schedule kinds corresponding to arguments of OpenMP schedule clause
30  //!
31  //! Kinds also present in omp_sched_t enum have the same integer values.
32  //! It is enum, not enum class, for shorter usage as omp::Schedule::[kind] and to keep interface of 0.6.0.
33  enum Kind
34  {
35  // Corresponds to not setting schedule
36  NoSchedule,
37  Static = 1u,
38  Dynamic = 2u,
39  Guided = 3u,
40  // Auto supported since OpenMP 3.0
41 #if defined _OPENMP && _OPENMP >= 200805
42  Auto = 4u,
43 #endif
44  Runtime = 5u
45  };
46 
47  //! Schedule kind.
48  Kind kind;
49 
50  //! Chunk size. Same as in OpenMP, value 0 corresponds to default chunk size. Using int and not a
51  //! fixed-width type to match OpenMP API.
52  int chunkSize;
53 
54  //! Create a schedule with the given kind and chunk size
55  ALPAKA_FN_HOST constexpr Schedule(Kind myKind = NoSchedule, int myChunkSize = 0)
56  : kind(myKind)
57  , chunkSize(myChunkSize)
58  {
59  }
60  };
61 
62  //! Get the OpenMP schedule that is applied when the runtime schedule is used.
63  //!
64  //! For OpenMP >= 3.0 returns the value of the internal control variable run-sched-var.
65  //! Without OpenMP or with OpenMP < 3.0, returns the default schedule.
66  //!
67  //! \return Schedule object.
68  ALPAKA_FN_HOST inline auto getSchedule()
69  {
70  // Getting a runtime schedule requires OpenMP 3.0 or newer
71 #if defined _OPENMP && _OPENMP >= 200805
72  omp_sched_t ompKind;
73  int chunkSize = 0;
74  omp_get_schedule(&ompKind, &chunkSize);
75  return Schedule{static_cast<Schedule::Kind>(ompKind), chunkSize};
76 #else
77  return Schedule{};
78 #endif
79  }
80 
81  //! Set the OpenMP schedule that is applied when the runtime schedule is used for future parallel regions.
82  //!
83  //! For OpenMP >= 3.0 sets the value of the internal control variable run-sched-var according to the given
84  //! schedule. Without OpenMP or with OpenMP < 3.0, does nothing.
85  //!
86  //! Note that calling from inside a parallel region does not have an immediate effect.
87  ALPAKA_FN_HOST inline void setSchedule(Schedule schedule)
88  {
89  if((schedule.kind != Schedule::NoSchedule) && (schedule.kind != Schedule::Runtime))
90  {
91 #if defined _OPENMP && _OPENMP >= 200805
92  omp_set_schedule(static_cast<omp_sched_t>(schedule.kind), schedule.chunkSize);
93 #endif
94  }
95  }
