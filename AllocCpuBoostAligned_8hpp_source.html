42  //! The CPU boost aligned allocator memory allocation trait specialization.
43  template<
44  typename T,
45  typename TAlignment>
46  struct Alloc<
47  T,
48  AllocCpuBoostAligned<TAlignment>>
49  {
50  //-----------------------------------------------------------------------------
51  ALPAKA_FN_HOST static auto alloc(
52  AllocCpuBoostAligned<TAlignment> const & alloc,
53  std::size_t const & sizeElems)
54  -> T *
55  {
56 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
57  // For CUDA host memory must be aligned to 4 kib to pin it with `cudaHostRegister`,
58  // this was described in older programming guides but was removed later.
59  // From testing with PIConGPU and cuda-memcheck we found out that the alignment is still required.
60  //
61  // For HIP the required alignment is the size of a cache line.
62  // https://rocm-developer-tools.github.io/HIP/group__Memory.html#gab8258f051e1a1f7385f794a15300e674
63  // To avoid issues with HIP(cuda) the alignment will be set also for HIP(clang/hcc)
64  // to 4kib.
65  // @todo evaluate requirements when the HIP ecosystem is more stable
66  constexpr size_t minAlignement = 4096;
67 #else
68  constexpr size_t minAlignement = TAlignment::value;
69 #endif
70  alpaka::ignore_unused(alloc);
71  return
72  reinterpret_cast<T *>(
73  boost::alignment::aligned_alloc(std::max(TAlignment::value, minAlignement), sizeElems * sizeof(T)));
74  }
75  };
