386  //! Not being able to enable peer access does not prevent such device to device memory copies.
387  //! However, those copies may be slower because the memory is copied via the CPU.
388  inline auto enablePeerAccessIfPossible(
389  const int & devSrc,
390  const int & devDst)
391  -> void
392  {
393  ALPAKA_ASSERT(devSrc != devDst);
394 
395 #if BOOST_COMP_CLANG
396  #pragma clang diagnostic push
397  #pragma clang diagnostic ignored "-Wexit-time-destructors"
398 #endif
399  static std::set<std::pair<int, int>> alreadyCheckedPeerAccessDevices;
400 #if BOOST_COMP_CLANG
401  #pragma clang diagnostic pop
402 #endif
403  auto const devicePair = std::make_pair(devSrc, devDst);
404 
405  if(alreadyCheckedPeerAccessDevices.find(devicePair) == alreadyCheckedPeerAccessDevices.end())
406  {
407  alreadyCheckedPeerAccessDevices.insert(devicePair);
408 
409  int canAccessPeer = 0;
410  ALPAKA_CUDA_RT_CHECK(cudaDeviceCanAccessPeer(&canAccessPeer, devSrc, devDst));
411  if(!canAccessPeer) {
412 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
413  std::cout << __func__
414  << " Direct peer access between given GPUs is not possible!"
415  << " src=" << devSrc
416  << " dst=" << devDst
417  << std::endl;
418 #endif
419  return;
420  }
421 
422  ALPAKA_CUDA_RT_CHECK(cudaSetDevice(devSrc));
423  // NOTE: "until access is explicitly disabled using cudaDeviceDisablePeerAccess() or either device is reset using cudaDeviceReset()."
424  // We do not remove a device from the enabled device pairs on cudaDeviceReset.
425  // Note that access granted by this call is unidirectional and that in order to access memory on the current device from peerDevice, a separate symmetric call to cudaDeviceEnablePeerAccess() is required.
426  ALPAKA_CUDA_RT_CHECK(cudaDeviceEnablePeerAccess(devDst, 0));
427  }
428  }
429  }
