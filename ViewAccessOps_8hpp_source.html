91  [[nodiscard]] ALPAKA_FN_HOST auto ptr_at([[maybe_unused]] Vec<DimInt<TDim>, TIdx> index) const -> const_pointer
92  {
93  using Idx = alpaka::Idx<TView>;
94  static_assert(
95  Dim<TView>::value == TDim,
96  "the index type must have the same dimensionality as the Buffer or View");
97  static_assert(
98  std::is_convertible_v<TIdx, Idx>,
99  "the index type must be convertible to the index of the Buffer or View");
100 
101  auto ptr = reinterpret_cast<uintptr_t>(data());
102  if constexpr(TDim > 0)
103  {
104  auto const pitchesInBytes = getPitchBytesVec(*static_cast<TView const*>(this));
105  for(std::size_t i = 0u; i < TDim; i++)
106  {
107  const Idx pitch = i + 1 < TDim ? pitchesInBytes[i + 1] : static_cast<Idx>(sizeof(value_type));
108  ptr += static_cast<uintptr_t>(index[i] * pitch);
109  }
110  }
111  return reinterpret_cast<const_pointer>(ptr);
