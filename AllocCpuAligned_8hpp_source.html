32  //! The CPU boost aligned allocator memory allocation trait specialization.
33  template<typename T, typename TAlignment>
34  struct Malloc<T, AllocCpuAligned<TAlignment>>
35  {
36  ALPAKA_FN_HOST static auto malloc(
37  AllocCpuAligned<TAlignment> const& /* alloc */,
38  std::size_t const& sizeElems) -> T*
39  {
40 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED) || defined(ALPAKA_ACC_GPU_HIP_ENABLED)
41  // For CUDA/HIP host memory must be aligned to 4 kib to pin it with `cudaHostRegister`,
42  // this was described in older programming guides but was removed later.
43  // From testing with PIConGPU and cuda-memcheck we found out that the alignment is still required.
44  //
45  // For HIP the required alignment is the size of a cache line.
46  // https://rocm-developer-tools.github.io/HIP/group__Memory.html#gab8258f051e1a1f7385f794a15300e674
47  // On most x86 systems the page size is 4KiB and on OpenPower 64KiB.
48  // Page size can be tested on the terminal with: `getconf PAGE_SIZE`
49  size_t minAlignement = std::max<size_t>(TAlignment::value, cpu::detail::getPageSize());
50 #else
51  constexpr size_t minAlignement = TAlignment::value;
52 #endif
53  return reinterpret_cast<T*>(core::alignedAlloc(minAlignement, sizeElems * sizeof(T)));
54  }
55  };
