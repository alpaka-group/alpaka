63  //! CUDA driver API error checking with log and exception, ignoring specific error values
64  ALPAKA_FN_HOST inline auto cudaDrvCheck(
65  CUresult const& error,
66  char const* desc,
67  char const* file,
68  int const& line) -> void
69  {
70  if(error == CUDA_SUCCESS)
71  return;
72 
73  char const* cu_err_name = nullptr;
74  char const* cu_err_string = nullptr;
75  CUresult cu_result_name = cuGetErrorName(error, &cu_err_name);
76  CUresult cu_result_string = cuGetErrorString(error, &cu_err_string);
77  std::string sError
78  = std::string(file) + "(" + std::to_string(line) + ") " + std::string(desc) + " : '";
79  if(cu_result_name == CUDA_SUCCESS && cu_result_string == CUDA_SUCCESS)
80  {
81  sError += std::string(cu_err_name) + "': '" + std::string(cu_err_string) + "'!";
82  }
83  else
84  {
85  // cuGetError*() failed, so append corresponding error message
86  if(cu_result_name == CUDA_ERROR_INVALID_VALUE)
87  {
88  sError += " cuGetErrorName: 'Invalid Value'!";
89  }
90  if(cu_result_string == CUDA_ERROR_INVALID_VALUE)
91  {
92  sError += " cuGetErrorString: 'Invalid Value'!";
93  }
94  }
95 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
96  std::cerr << sError << std::endl;
97 # endif
98  ALPAKA_DEBUG_BREAK;
99  throw std::runtime_error(sError);
100  }
101  } // namespace detail
102  } // namespace cuda
103 } // namespace alpaka
