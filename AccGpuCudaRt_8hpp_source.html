134  //! The GPU CUDA accelerator device properties get trait specialization.
135  template<
136  typename TDim,
137  typename TIdx>
138  struct GetAccDevProps<
139  acc::AccGpuCudaRt<TDim, TIdx>>
140  {
141  //-----------------------------------------------------------------------------
142  ALPAKA_FN_HOST static auto getAccDevProps(
143  dev::DevCudaRt const & dev)
144  -> acc::AccDevProps<TDim, TIdx>
145  {
146  // Reading only the necessary attributes with cudaDeviceGetAttribute is faster than reading all with cudaGetDeviceProperties
147  // https://devblogs.nvidia.com/cuda-pro-tip-the-fast-way-to-query-device-properties/
148  int multiProcessorCount = {};
149  ALPAKA_CUDA_RT_CHECK(cudaDeviceGetAttribute(
150  &multiProcessorCount,
151  cudaDevAttrMultiProcessorCount,
152  dev.m_iDevice));
153 
154  int maxGridSize[3] = {};
155  ALPAKA_CUDA_RT_CHECK(cudaDeviceGetAttribute(
156  &maxGridSize[0],
157  cudaDevAttrMaxGridDimX,
158  dev.m_iDevice));
159  ALPAKA_CUDA_RT_CHECK(cudaDeviceGetAttribute(
160  &maxGridSize[1],
161  cudaDevAttrMaxGridDimY,
162  dev.m_iDevice));
163  ALPAKA_CUDA_RT_CHECK(cudaDeviceGetAttribute(
164  &maxGridSize[2],
165  cudaDevAttrMaxGridDimZ,
166  dev.m_iDevice));
167 
168  int maxBlockDim[3] = {};
169  ALPAKA_CUDA_RT_CHECK(cudaDeviceGetAttribute(
170  &maxBlockDim[0],
171  cudaDevAttrMaxBlockDimX,
172  dev.m_iDevice));
173  ALPAKA_CUDA_RT_CHECK(cudaDeviceGetAttribute(
174  &maxBlockDim[1],
175  cudaDevAttrMaxBlockDimY,
176  dev.m_iDevice));
177  ALPAKA_CUDA_RT_CHECK(cudaDeviceGetAttribute(
178  &maxBlockDim[2],
179  cudaDevAttrMaxBlockDimZ,
180  dev.m_iDevice));
181 
182  int maxThreadsPerBlock = {};
183  ALPAKA_CUDA_RT_CHECK(cudaDeviceGetAttribute(
184  &maxThreadsPerBlock,
185  cudaDevAttrMaxThreadsPerBlock,
186  dev.m_iDevice));
187 
188  return {
189  // m_multiProcessorCount
190  alpaka::core::clipCast<TIdx>(multiProcessorCount),
191  // m_gridBlockExtentMax
192  extent::getExtentVecEnd<TDim>(
193  vec::Vec<dim::DimInt<3u>, TIdx>(
194  alpaka::core::clipCast<TIdx>(maxGridSize[2u]),
195  alpaka::core::clipCast<TIdx>(maxGridSize[1u]),
196  alpaka::core::clipCast<TIdx>(maxGridSize[0u]))),
197  // m_gridBlockCountMax
198  std::numeric_limits<TIdx>::max(),
199  // m_blockThreadExtentMax
200  extent::getExtentVecEnd<TDim>(
201  vec::Vec<dim::DimInt<3u>, TIdx>(
202  alpaka::core::clipCast<TIdx>(maxBlockDim[2u]),
203  alpaka::core::clipCast<TIdx>(maxBlockDim[1u]),
204  alpaka::core::clipCast<TIdx>(maxBlockDim[0u]))),
205  // m_blockThreadCountMax
206  alpaka::core::clipCast<TIdx>(maxThreadsPerBlock),
207  // m_threadElemExtentMax
208  vec::Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
209  // m_threadElemCountMax
210  std::numeric_limits<TIdx>::max()};
211  }
212  };
