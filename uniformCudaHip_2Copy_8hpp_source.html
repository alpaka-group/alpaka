52  //! The 1D CUDA/HIP memory copy trait.
53  template<typename TViewDst, typename TViewSrc, typename TExtent>
54  struct TaskCopyUniformCudaHip<DimInt<1>, TViewDst, TViewSrc, TExtent>
55  {
56  using MemcpyKind = ALPAKA_API_PREFIX(MemcpyKind);
57 
58  static_assert(!std::is_const<TViewDst>::value, "The destination view can not be const!");
59 
60  static_assert(
61  Dim<TViewDst>::value == Dim<TViewSrc>::value,
62  "The source and the destination view are required to have the same dimensionality!");
63  static_assert(
64  Dim<TViewDst>::value == Dim<TExtent>::value,
65  "The views and the extent are required to have the same dimensionality!");
66  // TODO: Maybe check for Idx of TViewDst and TViewSrc to have greater or equal range than TExtent.
67  static_assert(
68  std::is_same<Elem<TViewDst>, std::remove_const_t<Elem<TViewSrc>>>::value,
69  "The source and the destination view are required to have the same element type!");
70 
71  using Idx = Idx<TExtent>;
72 
73  ALPAKA_FN_HOST TaskCopyUniformCudaHip(
74  TViewDst& viewDst,
75  TViewSrc const& viewSrc,
76  TExtent const& extent,
77  MemcpyKind const& uniformMemCpyKind,
78  int const& iDstDevice,
79  int const& iSrcDevice)
80  : m_uniformMemCpyKind(uniformMemCpyKind)
81  , m_iDstDevice(iDstDevice)
82  , m_iSrcDevice(iSrcDevice)
83 
84 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
85  , m_extentWidth(extent::getWidth(extent))
86  , m_dstWidth(static_cast<Idx>(extent::getWidth(viewDst)))
87  , m_srcWidth(static_cast<Idx>(extent::getWidth(viewSrc)))
88 # endif
89  , m_extentWidthBytes(extent::getWidth(extent) * static_cast<Idx>(sizeof(Elem<TViewDst>)))
90  , m_dstMemNative(reinterpret_cast<void*>(getPtrNative(viewDst)))
91  , m_srcMemNative(reinterpret_cast<void const*>(getPtrNative(viewSrc)))
92  {
93 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
94  ALPAKA_ASSERT(m_extentWidth <= m_dstWidth);
95  ALPAKA_ASSERT(m_extentWidth <= m_srcWidth);
96 # endif
97  }
98 
99  template<typename TQueue>
100  auto enqueue(TQueue& queue) const -> void
101  {
102 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
103  printDebug();
104 # endif
105  if(m_extentWidthBytes == 0)
106  {
107  return;
108  }
109 
110  auto const& uniformCudaHipMemCpyKind(m_uniformMemCpyKind);
111 
112  // Set the current device.
113  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(m_iDstDevice));
114  // Initiate the memory copy.
115  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(MemcpyAsync)(
116  m_dstMemNative,
117  m_srcMemNative,
118  static_cast<std::size_t>(m_extentWidthBytes),
119  uniformCudaHipMemCpyKind,
120  queue.m_spQueueImpl->m_UniformCudaHipQueue));
121  }
122 
123  private:
124 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
125  ALPAKA_FN_HOST auto printDebug() const -> void
126  {
127  std::cout << __func__ << " ddev: " << m_iDstDevice << " ew: " << m_extentWidth
128  << " ewb: " << m_extentWidthBytes << " dw: " << m_dstWidth << " dptr: " << m_dstMemNative
129  << " sdev: " << m_iSrcDevice << " sw: " << m_srcWidth << " sptr: " << m_srcMemNative
130  << std::endl;
131  }
132 # endif
133 
134  MemcpyKind m_uniformMemCpyKind;
135  int m_iDstDevice;
136  int m_iSrcDevice;
137 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
138  Idx m_extentWidth;
139  Idx m_dstWidth;
140  Idx m_srcWidth;
141 # endif
142  Idx m_extentWidthBytes;
143  void* m_dstMemNative;
144  void const* m_srcMemNative;
145  };

334  template<typename TQueue>
335  auto enqueue(TQueue& queue) const -> void
336  {
337 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
338  printDebug();
339 # endif
340  // This is not only an optimization but also prevents a division by zero.
341  if(m_extentWidthBytes == 0 || m_extentHeight == 0 || m_extentDepth == 0)
342  {
343  return;
344  }
345 
346  // Create the struct describing the copy.
347  ALPAKA_API_PREFIX(Memcpy3DParms)
348  const uniformCudaHipMemCpy3DParms(buildUniformCudaHipMemcpy3DParms());
349  // Set the current device.
350  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(m_iDstDevice));
351 
352  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(
353  Memcpy3DAsync)(&uniformCudaHipMemCpy3DParms, queue.m_spQueueImpl->m_UniformCudaHipQueue));
354  }
355 
356  private:
357  ALPAKA_FN_HOST auto buildUniformCudaHipMemcpy3DParms() const -> ALPAKA_API_PREFIX(Memcpy3DParms)
358  {
359  ALPAKA_DEBUG_FULL_LOG_SCOPE;
360 
361  // Fill CUDA/HIP parameter structure.
362  ALPAKA_API_PREFIX(Memcpy3DParms) memCpy3DParms;
363  memCpy3DParms.srcArray = nullptr; // Either srcArray or srcPtr.
364  memCpy3DParms.srcPos
365  = ALPAKA_PP_CONCAT(make_, ALPAKA_API_PREFIX(Pos))(0, 0, 0); // Optional. Offset in bytes.
366  memCpy3DParms.srcPtr = ALPAKA_PP_CONCAT(make_, ALPAKA_API_PREFIX(PitchedPtr))(
367  const_cast<void*>(m_srcMemNative),
368  static_cast<std::size_t>(m_srcpitchBytesX),
369  static_cast<std::size_t>(m_srcWidth),
370  static_cast<std::size_t>(m_srcPitchBytesY / m_srcpitchBytesX));
371  memCpy3DParms.dstArray = nullptr; // Either dstArray or dstPtr.
372  memCpy3DParms.dstPos
373  = ALPAKA_PP_CONCAT(make_, ALPAKA_API_PREFIX(Pos))(0, 0, 0); // Optional. Offset in bytes.
374  memCpy3DParms.dstPtr = ALPAKA_PP_CONCAT(make_, ALPAKA_API_PREFIX(PitchedPtr))(
375  m_dstMemNative,
376  static_cast<std::size_t>(m_dstpitchBytesX),
377  static_cast<std::size_t>(m_dstWidth),
378  static_cast<std::size_t>(m_dstPitchBytesY / m_dstpitchBytesX));
379  memCpy3DParms.extent = ALPAKA_PP_CONCAT(make_, ALPAKA_API_PREFIX(Extent))(
380  static_cast<std::size_t>(m_extentWidthBytes),
381  static_cast<std::size_t>(m_extentHeight),
382  static_cast<std::size_t>(m_extentDepth));
383 # if defined(ALPAKA_ACC_GPU_HIP_ENABLED) && defined(__HIP_PLATFORM_NVCC__)
384  memCpy3DParms.kind = hipMemcpyKindToCudaMemcpyKind(m_uniformMemCpyKind);
385 # else
386  memCpy3DParms.kind = m_uniformMemCpyKind;
387 # endif
388  return memCpy3DParms;
389  }
