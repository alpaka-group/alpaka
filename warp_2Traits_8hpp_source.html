146  //! Exchange data between threads within a warp.
147  //!
148  //! Effectively executes:
149  //!
150  //! __shared__ int32_t values[warpsize];
151  //! values[threadIdx.x] = value;
152  //! __syncthreads();
153  //! return values[(srcLane + width*floor(threadIdx.x/width))%width];
154  //!
155  //! However, it does not use shared memory.
156  //!
157  //! Notes:
158  //! * The programmer must ensure that all threads calling this
159  //! function (and the srcLane) are executing the same line of code.
160  //! In particular it is not portable to write if(a) {shfl} else {shfl}.
161  //!
162  //! * Commonly used with width = warpsize (the default), (returns values[srcLane])
163  //!
164  //! * Width must be a power of 2.
165  //!
166  //! \tparam TWarp warp implementation type
167  //! \param warp warp implementation
168  //! \param value value to broadcast (only meaningful from threadIdx == srcLane)
169  //! \param srcLane source lane sending value
170  //! \param width number of threads receiving a single value
171  //! \return val from the thread index srcLane.
172  ALPAKA_NO_HOST_ACC_WARNING
173  template<typename TWarp>
174  ALPAKA_FN_ACC auto shfl(TWarp const& warp, std::int32_t value, std::int32_t srcLane, std::int32_t width = 0)
175  {
176  using ImplementationBase = concepts::ImplementationBase<ConceptWarp, TWarp>;
177  return traits::Shfl<ImplementationBase>::shfl(warp, value, srcLane, width ? width : getSize(warp));
178  }
