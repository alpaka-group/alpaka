128  //! Waits for all the other threads to reach the barrier.
129  template<typename TOp>
130  ALPAKA_FN_HOST auto wait(int predicate) -> int
131  {
132  TIdx const generationWhenEnteredTheWait = m_generation;
133  std::unique_lock<std::mutex> lock(m_mtxBarrier);
134 
135  auto const generationMod2 = m_generation % static_cast<TIdx>(2u);
136  if(m_curThreadCount == m_threadCount)
137  {
138  m_result[generationMod2] = TOp::InitialValue;
139  }
140 
141  std::atomic<int>& result(m_result[generationMod2]);
142  bool const predicateBool(predicate != 0);
143 
144  detail::AtomicOp<TOp>()(result, predicateBool);
145 
146  if(--m_curThreadCount == 0)
147  {
148  m_curThreadCount = m_threadCount;
149  ++m_generation;
150  m_cvAllThreadsReachedBarrier.notify_all();
151  }
152  else
153  {
154  m_cvAllThreadsReachedBarrier.wait(
155  lock,
156  [this, generationWhenEnteredTheWait] { return generationWhenEnteredTheWait != m_generation; });
157  }
158  return m_result[generationMod2];
159  }
