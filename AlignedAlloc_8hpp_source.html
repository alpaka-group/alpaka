26  //! Rounds to the next higher power of two (if not already power of two).
27  // Adapted from llvm/ADT/SmallPtrSet.h
28  ALPAKA_FN_INLINE ALPAKA_FN_HOST
29  void* alignedAlloc(size_t alignment, size_t size)
30  {
31 #if BOOST_OS_WINDOWS
32  return _aligned_malloc(size, alignment);
33 #elif BOOST_OS_MACOS
34  void * ptr = nullptr;
35  posix_memalign(&ptr, alignment, size);
36  return ptr;
37 #else
38  // the amount of bytes to allocate must be a multiple of the alignment
39  size_t sizeToAllocate = ((size + alignment - 1u) / alignment) * alignment;
40  return ::aligned_alloc(alignment, sizeToAllocate);
41 #endif
42  }
