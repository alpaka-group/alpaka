386  template<
387  typename TDim,
388  typename TView>
389  struct Print
390  {
391  ALPAKA_FN_HOST static auto print(
392  TView const & view,
393  Elem<TView> const * const ptr,
394  Vec<Dim<TView>, Idx<TView>> const & extent,
395  std::ostream & os,
396  std::string const & elementSeparator,
397  std::string const & rowSeparator,
398  std::string const & rowPrefix,
399  std::string const & rowSuffix)
400  -> void
401  {
402  os << rowPrefix;
403 
404  auto const pitch(view::getPitchBytes<TDim::value+1u>(view));
405  auto const lastIdx(extent[TDim::value]-1u);
406  for(auto i(decltype(lastIdx)(0)); i<=lastIdx ;++i)
407  {
408  Print<
409  DimInt<TDim::value+1u>,
410  TView>
411  ::print(
412  view,
413  reinterpret_cast<Elem<TView> const *>(reinterpret_cast<std::uint8_t const *>(ptr)+i*pitch),
414  extent,
415  os,
416  elementSeparator,
417  rowSeparator,
418  rowPrefix,
419  rowSuffix);
420 
421  // While we are not at the end of a row, add the row separator.
422  if(i != lastIdx)
423  {
424  os << rowSeparator;
425  }
426  }
427 
428  os << rowSuffix;
429  }
430  };
431  //-----------------------------------------------------------------------------
432  template<
433  typename TView>
434  struct Print<
435  DimInt<Dim<TView>::value-1u>,
436  TView>
437  {
438  ALPAKA_FN_HOST static auto print(
439  TView const & view,
440  Elem<TView> const * const ptr,
441  Vec<Dim<TView>, Idx<TView>> const & extent,
442  std::ostream & os,
443  std::string const & elementSeparator,
444  std::string const & rowSeparator,
445  std::string const & rowPrefix,
446  std::string const & rowSuffix)
447  -> void
448  {
449  alpaka::ignore_unused(view);
450  alpaka::ignore_unused(rowSeparator);
451 
452  os << rowPrefix;
453 
454  auto const lastIdx(extent[Dim<TView>::value-1u]-1u);
455  for(auto i(decltype(lastIdx)(0)); i<=lastIdx ;++i)
456  {
457  // Add the current element.
458  os << *(ptr+i);
459 
460  // While we are not at the end of a line, add the element separator.
461  if(i != lastIdx)
462  {
463  os << elementSeparator;
464  }
465  }
466 
467  os << rowSuffix;
468  }
469  };
