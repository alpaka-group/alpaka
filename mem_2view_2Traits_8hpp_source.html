41  //! The pitch in bytes.
42  //! This is the distance in bytes in the linear memory between two consecutive elements in the next higher
43  //! dimension (TIdx-1).
44  //!
45  //! The default implementation uses the extent to calculate the pitch.
46  template<typename TIdx, typename TView, typename TSfinae = void>
47  struct GetPitchBytes
48  {
49  using ViewIdx = Idx<TView>;
50 
51  ALPAKA_FN_HOST static auto getPitchBytes(TView const& view) -> ViewIdx
52  {
53  return getPitchBytesDefault(view);
54  }
55 
56  private:
57  static auto getPitchBytesDefault(TView const& view) -> ViewIdx
58  {
59  constexpr auto idx = TIdx::value;
60  constexpr auto viewDim = Dim<TView>::value;
61  if constexpr(idx < viewDim - 1)
62  return getExtent<idx>(view) * GetPitchBytes<DimInt<idx + 1>, TView>::getPitchBytes(view);
63  else if constexpr(idx == viewDim - 1)
64  return getExtent<viewDim - 1>(view) * static_cast<ViewIdx>(sizeof(Elem<TView>));
65  else
66  return static_cast<ViewIdx>(sizeof(Elem<TView>));
67  ALPAKA_UNREACHABLE({});
68  }
69  };

235  template<typename TDim, typename TView>
236  struct Print
237  {
238  ALPAKA_FN_HOST static auto print(
239  TView const& view,
240  Elem<TView> const* const ptr,
241  Vec<Dim<TView>, Idx<TView>> const& extent,
242  std::ostream& os,
243  std::string const& elementSeparator,
244  std::string const& rowSeparator,
245  std::string const& rowPrefix,
246  std::string const& rowSuffix) -> void
247  {
248  os << rowPrefix;
249 
250  auto const pitch(getPitchBytes<TDim::value + 1u>(view));
251  auto const lastIdx(extent[TDim::value] - 1u);
252  for(auto i(decltype(lastIdx)(0)); i <= lastIdx; ++i)
253  {
254  Print<DimInt<TDim::value + 1u>, TView>::print(
255  view,
256  reinterpret_cast<Elem<TView> const*>(reinterpret_cast<std::uint8_t const*>(ptr) + i * pitch),
257  extent,
258  os,
259  elementSeparator,
260  rowSeparator,
261  rowPrefix,
262  rowSuffix);
263 
264  // While we are not at the end of a row, add the row separator.
265  if(i != lastIdx)
266  {
267  os << rowSeparator;
268  }
269  }
270 
271  os << rowSuffix;
272  }
273  };

275  struct Print<DimInt<Dim<TView>::value - 1u>, TView>
276  {
277  ALPAKA_FN_HOST static auto print(
278  TView const& /* view */,
279  Elem<TView> const* const ptr,
280  Vec<Dim<TView>, Idx<TView>> const& extent,
281  std::ostream& os,
282  std::string const& elementSeparator,
283  std::string const& /* rowSeparator */,
284  std::string const& rowPrefix,
285  std::string const& rowSuffix) -> void
286  {
287  os << rowPrefix;
288 
289  auto const lastIdx(extent[Dim<TView>::value - 1u] - 1u);
290  for(auto i(decltype(lastIdx)(0)); i <= lastIdx; ++i)
291  {
292  // Add the current element.
293  os << *(ptr + i);
294 
295  // While we are not at the end of a line, add the element separator.
296  if(i != lastIdx)
297  {
298  os << elementSeparator;
299  }
300  }
301 
302  os << rowSuffix;
303  }

343  //! Calculate the pitches purely from the extents.
344  template<typename TElem, typename TDim, typename TIdx>
345  ALPAKA_FN_HOST inline auto calculatePitchesFromExtents(Vec<TDim, TIdx> const& extent)
346  {
347  Vec<TDim, TIdx> pitchBytes(Vec<TDim, TIdx>::all(0));
348  if constexpr(TDim::value > 0)
349  {
350  pitchBytes[TDim::value - 1u] = extent[TDim::value - 1u] * static_cast<TIdx>(sizeof(TElem));
351  for(TIdx i = TDim::value - 1u; i > static_cast<TIdx>(0u); --i)
352  {
353  pitchBytes[i - 1] = extent[i - 1] * pitchBytes[i];
354  }
355  }
356  return pitchBytes;
357  }
358  } // namespace detail
