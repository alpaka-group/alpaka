390  template<
391  typename TDim,
392  typename TView>
393  struct Print
394  {
395  ALPAKA_FN_HOST static auto print(
396  TView const & view,
397  elem::Elem<TView> const * const ptr,
398  vec::Vec<dim::Dim<TView>, idx::Idx<TView>> const & extent,
399  std::ostream & os,
400  std::string const & elementSeparator,
401  std::string const & rowSeparator,
402  std::string const & rowPrefix,
403  std::string const & rowSuffix)
404  -> void
405  {
406  os << rowPrefix;
407 
408  auto const pitch(view::getPitchBytes<TDim::value+1u>(view));
409  auto const lastIdx(extent[TDim::value]-1u);
410  for(auto i(decltype(lastIdx)(0)); i<=lastIdx ;++i)
411  {
412  Print<
413  dim::DimInt<TDim::value+1u>,
414  TView>
415  ::print(
416  view,
417  reinterpret_cast<elem::Elem<TView> const *>(reinterpret_cast<std::uint8_t const *>(ptr)+i*pitch),
418  extent,
419  os,
420  elementSeparator,
421  rowSeparator,
422  rowPrefix,
423  rowSuffix);
424 
425  // While we are not at the end of a row, add the row separator.
426  if(i != lastIdx)
427  {
428  os << rowSeparator;
429  }
430  }
431 
432  os << rowSuffix;
433  }
434  };
435  //-----------------------------------------------------------------------------
436  template<
437  typename TView>
438  struct Print<
439  dim::DimInt<dim::Dim<TView>::value-1u>,
440  TView>
441  {
442  ALPAKA_FN_HOST static auto print(
443  TView const & view,
444  elem::Elem<TView> const * const ptr,
445  vec::Vec<dim::Dim<TView>, idx::Idx<TView>> const & extent,
446  std::ostream & os,
447  std::string const & elementSeparator,
448  std::string const & rowSeparator,
449  std::string const & rowPrefix,
450  std::string const & rowSuffix)
451  -> void
452  {
453  alpaka::ignore_unused(view);
454  alpaka::ignore_unused(rowSeparator);
455 
456  os << rowPrefix;
457 
458  auto const lastIdx(extent[dim::Dim<TView>::value-1u]-1u);
459  for(auto i(decltype(lastIdx)(0)); i<=lastIdx ;++i)
460  {
461  // Add the current element.
462  os << *(ptr+i);
463 
464  // While we are not at the end of a line, add the element separator.
465  if(i != lastIdx)
466  {
467  os << elementSeparator;
468  }
469  }
470 
471  os << rowSuffix;
472  }
473  };
