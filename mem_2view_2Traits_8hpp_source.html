40  //! The pitch in bytes.
41  //! This is the distance in bytes in the linear memory between two consecutive elements in the next higher
42  //! dimension (TIdx-1).
43  //!
44  //! The default implementation uses the extent to calculate the pitch.
45  template<typename TIdx, typename TView, typename TSfinae = void>
46  struct GetPitchBytes
47  {
48  using ViewIdx = Idx<TView>;
49 
50  ALPAKA_FN_HOST static auto getPitchBytes(TView const& view) -> ViewIdx
51  {
52  return getPitchBytesDefault(view);
53  }
54 
55  private:
56  static auto getPitchBytesDefault(TView const& view) -> ViewIdx
57  {
58  constexpr auto idx = TIdx::value;
59  constexpr auto viewDim = Dim<TView>::value;
60  if constexpr(idx < viewDim - 1)
61  return getExtent<idx>(view) * GetPitchBytes<DimInt<idx + 1>, TView>::getPitchBytes(view);
62  else if constexpr(idx == viewDim - 1)
63  return getExtent<viewDim - 1>(view) * static_cast<ViewIdx>(sizeof(Elem<TView>));
64  else
65  return static_cast<ViewIdx>(sizeof(Elem<TView>));
66  ALPAKA_UNREACHABLE({});
67  }
68  };

192  //!
193  //! \param viewDst The destination memory view.
194  //! \param viewSrc The source memory view.
195  //! \param extent The extent of the view to copy.
196  template<typename TExtent, typename TViewSrc, typename TViewDstFwd>
197  ALPAKA_FN_HOST auto createTaskMemcpy(TViewDstFwd&& viewDst, TViewSrc const& viewSrc, TExtent const& extent)
198  {
199  using TViewDst = std::remove_reference_t<TViewDstFwd>;
200  using SrcElem = Elem<TViewSrc>;
201  using DstElem = Elem<TViewDst>;
202  using ExtentIdx = Idx<TExtent>;
203  using DstIdx = Idx<TViewDst>;
204  using SrcIdx = Idx<TViewSrc>;
205 
206  static_assert(!std::is_const_v<TViewDst>, "The destination view must not be const!");
207  static_assert(!std::is_const_v<DstElem>, "The destination view's element type must not be const!");
208  static_assert(

213  "The destination view and the extent must have the same dimensionality!");

219  "The destination view and the extent are required to have compatible index types!");
220  static_assert(
221  meta::IsIntegralSuperset<SrcIdx, ExtentIdx>::value,
222  "The source view and the extent are required to have compatible index types!");
223 
224  return trait::CreateTaskMemcpy<Dim<TViewDst>, Dev<TViewDst>, Dev<TViewSrc>>::createTaskMemcpy(
225  std::forward<TViewDstFwd>(viewDst),
226  viewSrc,
227  extent);
228  }

258  template<typename TDim, typename TView>
259  struct Print
260  {
261  ALPAKA_FN_HOST static auto print(
262  TView const& view,
263  Elem<TView> const* const ptr,
264  Vec<Dim<TView>, Idx<TView>> const& extent,
265  std::ostream& os,
266  std::string const& elementSeparator,
267  std::string const& rowSeparator,
268  std::string const& rowPrefix,
269  std::string const& rowSuffix) -> void
270  {
271  os << rowPrefix;
272 
273  auto const pitch(getPitchBytes<TDim::value + 1u>(view));
274  auto const lastIdx(extent[TDim::value] - 1u);
275  for(auto i(decltype(lastIdx)(0)); i <= lastIdx; ++i)
276  {
277  Print<DimInt<TDim::value + 1u>, TView>::print(
278  view,
279  reinterpret_cast<Elem<TView> const*>(reinterpret_cast<std::uint8_t const*>(ptr) + i * pitch),
280  extent,
281  os,
282  elementSeparator,
283  rowSeparator,
284  rowPrefix,
285  rowSuffix);
286 
287  // While we are not at the end of a row, add the row separator.
288  if(i != lastIdx)
289  {
290  os << rowSeparator;
291  }
292  }
293 
294  os << rowSuffix;
295  }
296  };
297  template<typename TView>
298  struct Print<DimInt<Dim<TView>::value - 1u>, TView>
299  {
300  ALPAKA_FN_HOST static auto print(
301  TView const& /* view */,
302  Elem<TView> const* const ptr,
303  Vec<Dim<TView>, Idx<TView>> const& extent,
304  std::ostream& os,
305  std::string const& elementSeparator,
306  std::string const& /* rowSeparator */,
307  std::string const& rowPrefix,
308  std::string const& rowSuffix) -> void
309  {
310  os << rowPrefix;
311 
312  auto const lastIdx(extent[Dim<TView>::value - 1u] - 1u);
313  for(auto i(decltype(lastIdx)(0)); i <= lastIdx; ++i)
314  {
315  // Add the current element.
316  os << *(ptr + i);
317 
318  // While we are not at the end of a line, add the element separator.
319  if(i != lastIdx)
320  {
321  os << elementSeparator;
322  }
323  }
324 
325  os << rowSuffix;
326  }
327  };
328  } // namespace detail

365  //! Calculate the pitches purely from the extents.
366  template<typename TElem, typename TDim, typename TIdx>
367  ALPAKA_FN_HOST inline auto calculatePitchesFromExtents(Vec<TDim, TIdx> const& extent)
368  {
369  Vec<TDim, TIdx> pitchBytes(Vec<TDim, TIdx>::all(0));
370  if constexpr(TDim::value > 0)
371  {
372  pitchBytes[TDim::value - 1u] = extent[TDim::value - 1u] * static_cast<TIdx>(sizeof(TElem));
373  for(TIdx i = TDim::value - 1u; i > static_cast<TIdx>(0u); --i)
374  {
375  pitchBytes[i - 1] = extent[i - 1] * pitchBytes[i];
376  }
377  }
378  return pitchBytes;
379  }
380  } // namespace detail

532  ALPAKA_FN_HOST auto makeExtents(TView const& view, std::index_sequence<Is...>)

537  template<typename TView, std::size_t... Is>
538  ALPAKA_FN_HOST auto makeStrides(TView const& view, std::index_sequence<Is...>)
539  {
540  constexpr auto fastestIndexPitch = static_cast<Idx<TView>>(sizeof(Elem<TView>));
541  [[maybe_unused]] constexpr auto dim = Dim<TView>::value;
542  // alpaka pitches are right-shifted by 1. We skip getPitchBytes<0> (the size in bytes of the entire
543  // buffer) and append the element size last
544  return std::array<Idx<TView>, dim>{
545  (Is < dim - 1 ? getPitchBytes<Is + 1>(view) : fastestIndexPitch)...};
546  }

551  struct GetMdSpan
552  {
553  ALPAKA_FN_HOST static auto getMdSpan(TView& view)
554  {

556  using Element = Elem<TView>;

560  layout_stride::mapping<decltype(extents)> m{extents, strides};

562  ptr,
563  m};
564  }
