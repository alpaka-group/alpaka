33  //! Calculate the pitches purely from the extents.
34  template<typename TElem, typename TDim, typename TIdx>
35  ALPAKA_FN_HOST_ACC inline constexpr auto calculatePitchesFromExtents(Vec<TDim, TIdx> const& extent)
36  {
37  Vec<TDim, TIdx> pitchBytes{};
38  constexpr auto dim = TIdx{TDim::value};
39  if constexpr(dim > 0)
40  pitchBytes.back() = static_cast<TIdx>(sizeof(TElem));
41  if constexpr(dim > 1)
42  for(TIdx i = TDim::value - 1; i > 0; i--)
43  pitchBytes[i - 1] = extent[i] * pitchBytes[i];
44  return pitchBytes;
45  }
46  } // namespace detail

59  //! The pitch in bytes.
60  //! This is the distance in bytes in the linear memory between two consecutive elements in the next higher
61  //! dimension (TIdx-1).
62  //!
63  //! The default implementation uses the extent to calculate the pitch.
64  template<typename TIdx, typename TView, typename TSfinae = void>
65  struct [[deprecated("Use GetPitchesInBytes instead")]] GetPitchBytes
66  {
67  using ViewIdx = Idx<TView>;
68 
69  ALPAKA_FN_HOST static auto getPitchBytes(TView const& view) -> ViewIdx
70  {
71  return getPitchBytesDefault(view);
72  }
73 
74  private:
75  static auto getPitchBytesDefault(TView const& view) -> ViewIdx
76  {
77  constexpr auto idx = TIdx::value;
78  constexpr auto viewDim = Dim<TView>::value;
79  if constexpr(idx < viewDim - 1)
80  {
81 #if BOOST_COMP_CLANG || BOOST_COMP_GNUC
82 # pragma GCC diagnostic push
83 # pragma GCC diagnostic ignored "-Wdeprecated-declarations"
84 #endif
85  return getExtents(view)[idx] * GetPitchBytes<DimInt<idx + 1>, TView>::getPitchBytes(view);
86 #if BOOST_COMP_CLANG || BOOST_COMP_GNUC
87 # pragma GCC diagnostic pop
88 #endif
89  }
90  else if constexpr(idx == viewDim - 1)
91  return getExtents(view)[viewDim - 1] * static_cast<ViewIdx>(sizeof(Elem<TView>));
92  else
93  return static_cast<ViewIdx>(sizeof(Elem<TView>));
94  ALPAKA_UNREACHABLE({});
95  }
96  };

317  template<typename TDim, typename TView>
318  struct Print
319  {
320  ALPAKA_FN_HOST static auto print(
321  TView const& view,
322  Elem<TView> const* const ptr,
323  Vec<Dim<TView>, Idx<TView>> const& extent,
324  std::ostream& os,
325  std::string const& elementSeparator,
326  std::string const& rowSeparator,
327  std::string const& rowPrefix,
328  std::string const& rowSuffix) -> void
329  {
330  os << rowPrefix;
331 
332  auto const pitch = getPitchesInBytes(view)[TDim::value + 1];
333  auto const lastIdx(extent[TDim::value] - 1u);
334  for(auto i(decltype(lastIdx)(0)); i <= lastIdx; ++i)
335  {
336  Print<DimInt<TDim::value + 1u>, TView>::print(
337  view,
338  reinterpret_cast<Elem<TView> const*>(reinterpret_cast<std::uint8_t const*>(ptr) + i * pitch),
339  extent,
340  os,
341  elementSeparator,
342  rowSeparator,
343  rowPrefix,
344  rowSuffix);
345 
346  // While we are not at the end of a row, add the row separator.
347  if(i != lastIdx)
348  {
349  os << rowSeparator;
350  }
351  }
352 
353  os << rowSuffix;
354  }
355  };
356 
357  template<typename TView>
358  struct Print<DimInt<Dim<TView>::value - 1u>, TView>
359  {
360  ALPAKA_FN_HOST static auto print(
361  TView const& /* view */,
362  Elem<TView> const* const ptr,
363  Vec<Dim<TView>, Idx<TView>> const& extent,
364  std::ostream& os,
365  std::string const& elementSeparator,
366  std::string const& /* rowSeparator */,
367  std::string const& rowPrefix,
368  std::string const& rowSuffix) -> void
369  {
370  os << rowPrefix;
371 
372  auto const lastIdx(extent[Dim<TView>::value - 1u] - 1u);
373  for(auto i(decltype(lastIdx)(0)); i <= lastIdx; ++i)
374  {
375  // Add the current element.
376  os << *(ptr + i);
377 
378  // While we are not at the end of a line, add the element separator.
379  if(i != lastIdx)
380  {
381  os << elementSeparator;
382  }
383  }
384 
385  os << rowSuffix;
386  }
387  };
