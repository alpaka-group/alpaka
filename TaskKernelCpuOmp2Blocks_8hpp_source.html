599  //! Executor of parallel OpenMP loop
600  //!
601  //! Performs dispatch based on schedule kind and forwards to the corresponding ParallelForImpl.
602  //! The default implementation is for the kernels that do not set schedule in any way, compile-time dispatch.
603  //!
604  //! \tparam TKernel The kernel type.
605  //! \tparam TSchedule The schedule type (not necessarily omp::Schedule).
606  template<typename TKernel, typename TSchedule, typename TSfinae = void>
607  struct ParallelFor
608  {
609  //! Run parallel OpenMP loop
610  //!
611  //! \tparam TLoopBody The loop body functor type.
612  //! \tparam TIdx The index type.
613  //!
614  //! \param kernel The kernel instance reference
615  //! \param loopBody The loop body functor instance, takes iteration index as input.
616  //! \param numIterations The number of loop iterations.
617  //! \param schedule The schedule object.
618  template<typename TLoopBody, typename TIdx>
619  ALPAKA_FN_HOST void operator()(
620  TKernel const& kernel,
621  TLoopBody&& loopBody,
622  TIdx const numIterations,
623  TSchedule const& schedule)
624  {
625  // Forward to ParallelForImpl that performs dispatch by by chunk size
626  ParallelForImpl<TKernel, TSchedule, omp::Schedule::NoSchedule>{}(
627  kernel,
628  std::forward<TLoopBody>(loopBody),
629  numIterations,
630  schedule);
631  }

788  //! The CPU OpenMP 2.0 block accelerator execution task.
789  template<typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
790  class TaskKernelCpuOmp2Blocks final : public WorkDivMembers<TDim, TIdx>
791  {
792  public:
793  template<typename TWorkDiv>
794  ALPAKA_FN_HOST TaskKernelCpuOmp2Blocks(TWorkDiv&& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
795  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
796  , m_kernelFnObj(kernelFnObj)
797  , m_args(std::forward<TArgs>(args)...)
798  {
799  static_assert(
800  Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
801  "The work division and the execution task have to be of the same dimensionality!");
802  }
803 
804  //! Executes the kernel function object.
805  ALPAKA_FN_HOST auto operator()() const -> void
806  {
807  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
808 
809  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(*this);
810  auto const blockThreadExtent = getWorkDiv<Block, Threads>(*this);
811  auto const threadElemExtent = getWorkDiv<Thread, Elems>(*this);
812 
813  // Get the size of the block shared dynamic memory.
814  auto const blockSharedMemDynSizeBytes = std::apply(
815  [&](ALPAKA_DECAY_T(TArgs) const&... args)
816  {
817  return getBlockSharedMemDynSizeBytes<AccCpuOmp2Blocks<TDim, TIdx>>(
818  m_kernelFnObj,
819  blockThreadExtent,
820  threadElemExtent,
821  args...);
822  },
823  m_args);
824 
825 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
826  std::cout << __func__ << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B"
827  << std::endl;
828 # endif
829 
830  // The number of blocks in the grid.
831  TIdx const numBlocksInGrid(gridBlockExtent.prod());
832  if(blockThreadExtent.prod() != static_cast<TIdx>(1u))
833  {
834  throw std::runtime_error("Only one thread per block allowed in the OpenMP 2.0 block accelerator!");
835  }
836 
837  // Get the OpenMP schedule information for the given kernel and parameter types
838  auto const schedule = std::apply(
839  [&](ALPAKA_DECAY_T(TArgs) const&... args) {
840  return getOmpSchedule<AccCpuOmp2Blocks<TDim, TIdx>>(
841  m_kernelFnObj,
842  blockThreadExtent,
843  threadElemExtent,
844  args...);
845  },
846  m_args);
847 
848  if(::omp_in_parallel() != 0)
849  {
850 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
851  std::cout << __func__ << " already within a parallel region." << std::endl;
852 # endif
853  parallelFn(blockSharedMemDynSizeBytes, numBlocksInGrid, gridBlockExtent, schedule);
854  }
855  else
856  {
857 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
858  std::cout << __func__ << " opening new parallel region." << std::endl;
859 # endif
860 # pragma omp parallel
861  parallelFn(blockSharedMemDynSizeBytes, numBlocksInGrid, gridBlockExtent, schedule);
862  }
863  }
864 
865  private:
866  template<typename TSchedule>
867  ALPAKA_FN_HOST auto parallelFn(
868  std::size_t const& blockSharedMemDynSizeBytes,
869  TIdx const& numBlocksInGrid,
870  Vec<TDim, TIdx> const& gridBlockExtent,
871  TSchedule const& schedule) const -> void
872  {
873 # pragma omp single nowait
874  {
875  // The OpenMP runtime does not create a parallel region when either:
876  // * only one thread is required in the num_threads clause
877  // * or only one thread is available
878  // In all other cases we expect to be in a parallel region now.
879  if((numBlocksInGrid > 1) && (::omp_get_max_threads() > 1) && (::omp_in_parallel() == 0))
880  {
881  throw std::runtime_error("The OpenMP 2.0 runtime did not create a parallel region!");
882  }
883 
884 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
885  std::cout << __func__ << " omp_get_num_threads: " << ::omp_get_num_threads() << std::endl;
886 # endif
887  }
888 
889  AccCpuOmp2Blocks<TDim, TIdx> acc(
890  *static_cast<WorkDivMembers<TDim, TIdx> const*>(this),
891  blockSharedMemDynSizeBytes);
892 
893  // Body of the OpenMP parallel loop to be executed.
894  // Index type is auto since we have a difference for OpenMP 2.0 and later ones
895  auto loopBody = [&](auto currentIndex)
896  {
897 # if _OPENMP < 200805
898  auto const i_tidx = static_cast<TIdx>(currentIndex); // for issue #840
899  auto const index = Vec<DimInt<1u>, TIdx>(i_tidx); // for issue #840
900 # else
901  auto const index = Vec<DimInt<1u>, TIdx>(currentIndex); // for issue #840
902 # endif
903  acc.m_gridBlockIdx = mapIdx<TDim::value>(index, gridBlockExtent);
904 
905  std::apply(m_kernelFnObj, std::tuple_cat(std::tie(acc), m_args));
906 
907  // After a block has been processed, the shared memory has to be deleted.
908  freeSharedVars(acc);
909  };
910 
911  detail::parallelFor(m_kernelFnObj, loopBody, numBlocksInGrid, schedule);
912  }
913 
914  TKernelFnObj m_kernelFnObj;
915  std::tuple<std::decay_t<TArgs>...> m_args;
916  };
