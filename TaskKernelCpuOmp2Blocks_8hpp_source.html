48  //! The CPU OpenMP 2.0 block accelerator execution task.
49  template<
50  typename TDim,
51  typename TIdx,
52  typename TKernelFnObj,
53  typename... TArgs>
54  class TaskKernelCpuOmp2Blocks final :
55  public WorkDivMembers<TDim, TIdx>
56  {
57  public:
58  //-----------------------------------------------------------------------------
59  template<
60  typename TWorkDiv>
61  ALPAKA_FN_HOST TaskKernelCpuOmp2Blocks(
62  TWorkDiv && workDiv,
63  TKernelFnObj const & kernelFnObj,
64  TArgs && ... args) :

85  //-----------------------------------------------------------------------------
86  //! Executes the kernel function object.
87  ALPAKA_FN_HOST auto operator()() const
88  -> void
89  {
90  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
91 
92  auto const gridBlockExtent(
93  getWorkDiv<Grid, Blocks>(*this));
94  auto const blockThreadExtent(
95  getWorkDiv<Block, Threads>(*this));
96  auto const threadElemExtent(
97  getWorkDiv<Thread, Elems>(*this));
98 
99  // Get the size of the block shared dynamic memory.
100  auto const blockSharedMemDynSizeBytes(
101  meta::apply(
102  [&](ALPAKA_DECAY_T(TArgs) const & ... args)
103  {
104  return
105  getBlockSharedMemDynSizeBytes<
106  AccCpuOmp2Blocks<TDim, TIdx>>(
107  m_kernelFnObj,
108  blockThreadExtent,
109  threadElemExtent,
110  args...);
111  },
112  m_args));
113 
114 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
115  std::cout << __func__
116  << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B" << std::endl;
117 #endif
118  // Bind all arguments except the accelerator.
119  // TODO: With C++14 we could create a perfectly argument forwarding function object within the constructor.
120  auto const boundKernelFnObj(
121  meta::apply(
122  [this](ALPAKA_DECAY_T(TArgs) const & ... args)
123  {
124  return
125  std::bind(
126  std::ref(m_kernelFnObj),
127  std::placeholders::_1,
128  std::ref(args)...);
129  },
130  m_args));
131 
132  // The number of blocks in the grid.
133  TIdx const numBlocksInGrid(gridBlockExtent.prod());
134  if(blockThreadExtent.prod() != static_cast<TIdx>(1u))
135  {
136  throw std::runtime_error("Only one thread per block allowed in the OpenMP 2.0 block accelerator!");
137  }
138 
139  if(::omp_in_parallel() != 0)
140  {
141 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
142  std::cout << __func__ << " already within a parallel region." << std::endl;
143 #endif
144  parallelFn(
145  boundKernelFnObj,
146  blockSharedMemDynSizeBytes,
147  numBlocksInGrid,
148  gridBlockExtent);
149  }
150  else
151  {
152 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
153  std::cout << __func__ << " opening new parallel region." << std::endl;
154 #endif
155  #pragma omp parallel
156  parallelFn(
157  boundKernelFnObj,
158  blockSharedMemDynSizeBytes,
159  numBlocksInGrid,
160  gridBlockExtent);
161  }
162  }
163 
164  private:
165  template<
166  typename FnObj>
167  ALPAKA_FN_HOST auto parallelFn(
168  FnObj const & boundKernelFnObj,
169  std::size_t const & blockSharedMemDynSizeBytes,
170  TIdx const & numBlocksInGrid,
171  Vec<TDim, TIdx> const & gridBlockExtent) const
172  -> void
173  {
174  #pragma omp single nowait
175  {
176  // The OpenMP runtime does not create a parallel region when either:
177  // * only one thread is required in the num_threads clause
178  // * or only one thread is available
179  // In all other cases we expect to be in a parallel region now.
180  if((numBlocksInGrid > 1) && (::omp_get_max_threads() > 1) && (::omp_in_parallel() == 0))
181  {
182  throw std::runtime_error("The OpenMP 2.0 runtime did not create a parallel region!");
183  }
184 
185 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
186  std::cout << __func__ << " omp_get_num_threads: " << ::omp_get_num_threads() << std::endl;
187 #endif
188  }
189 
190  AccCpuOmp2Blocks<TDim, TIdx> acc(
191  *static_cast<WorkDivMembers<TDim, TIdx> const *>(this),
192  blockSharedMemDynSizeBytes);
193 
194  // NOTE: schedule(static) does not improve performance.
195 #if _OPENMP < 200805 // For OpenMP < 3.0 you have to declare the loop index (a signed integer) outside of the loop header.
196  std::intmax_t iNumBlocksInGrid(static_cast<std::intmax_t>(numBlocksInGrid));
197  std::intmax_t i;
198  #pragma omp for nowait schedule(guided)
199  for(i = 0; i < iNumBlocksInGrid; ++i)
200 #else
201  #pragma omp for nowait schedule(guided)
202  for(TIdx i = 0; i < numBlocksInGrid; ++i)
203 #endif
204  {
205 #if _OPENMP < 200805
206  auto const i_tidx = static_cast<TIdx>(i); // for issue #840
207  auto const index = Vec<DimInt<1u>, TIdx>( i_tidx ); // for issue #840
208 #else
209  auto const index = Vec<DimInt<1u>, TIdx>( i ); // for issue #840
210 #endif
211  acc.m_gridBlockIdx = mapIdx<TDim::value>(index,
212  gridBlockExtent);
213 
214  boundKernelFnObj(
215  acc);
216 
217  // After a block has been processed, the shared memory has to be deleted.
218  freeMem(acc);
219  }
220  }
221 
222  TKernelFnObj m_kernelFnObj;
223  std::tuple<std::decay_t<TArgs>...> m_args;
224  };
