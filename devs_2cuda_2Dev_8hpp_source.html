109  //-----------------------------------------------------------------------------
110  //! \return The number of devices available.
111  //-----------------------------------------------------------------------------
112  ALPAKA_FCT_HOST static auto getDevCount()
113  -> std::size_t
114  {
115  int iNumDevices(0);
116  ALPAKA_CUDA_RT_CHECK(cudaGetDeviceCount(&iNumDevices));
117 
118  return static_cast<std::size_t>(iNumDevices);
119  }
120  //-----------------------------------------------------------------------------
121  //! \return The number of devices available.
122  //-----------------------------------------------------------------------------
123  ALPAKA_FCT_HOST static auto getDevByIdx(
124  std::size_t const & uiIdx)
125  -> DevCuda
126  {
127  DevCuda dev;
128 
129  std::size_t const uiNumDevices(getDevCount());
130  if(uiIdx >= uiNumDevices)
131  {
132  std::stringstream ssErr;
133  ssErr << "Unable to return device handle for device " << uiIdx << " because there are only " << uiNumDevices << " CUDA devices!";
134  throw std::runtime_error(ssErr.str());
135  }
136 
137  // Initialize the cuda runtime.
138  init();
139 
140  // Try all devices if the given one is unusable.
141  for(std::size_t iDeviceOffset(0); iDeviceOffset < uiNumDevices; ++iDeviceOffset)
142  {
143  std::size_t const iDevice((uiIdx + iDeviceOffset) % uiNumDevices);
144 
145  if(isDevUsable(iDevice))
146  {
147  dev.m_iDevice = static_cast<int>(iDevice);
148 
149  // Log this device.
150 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
151  cudaDeviceProp devProp;
152  ALPAKA_CUDA_RT_CHECK(cudaGetDeviceProperties(&devProp, dev.m_iDevice));
153 #endif
154 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
155  printDeviceProperties(devProp);
156 #elif ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
157  std::cout << devProp.name << std::endl;
158 #endif
159  return dev;
160  }
161  }
162 
163  // If we came until here, none of the devices was usable.
164  std::stringstream ssErr;
165  ssErr << "Unable to return device handle for device " << uiIdx << " because none of the " << uiNumDevices << " CUDA devices is usable!";
166  throw std::runtime_error(ssErr.str());
167  }

169  private:
170  //-----------------------------------------------------------------------------
171  //! \return If the device is usable.
172  //-----------------------------------------------------------------------------
173  ALPAKA_FCT_HOST static auto isDevUsable(
174  std::size_t iDevice)
175  -> bool
176  {
177  cudaError rc(cudaSetDevice(static_cast<int>(iDevice)));
178 
179  // Create a dummy stream to check if the device is already used by an other process.
180  // \TODO: Check if this workaround is needed!
181  // Since NVIDIA changed something in the runtime cudaSetDevice never returns an error if another process already uses the selected device and gpu compute mode is set "process exclusive".
182  if(rc == cudaSuccess)
183  {
184  cudaStream_t stream;
185  rc = cudaStreamCreate(&stream);
186  }
187 
188  if(rc == cudaSuccess)
189  {
190  return true;
191  }
192  else
193  {
194  ALPAKA_CUDA_RT_CHECK(rc);
195  // Reset the Error state.
196  cudaGetLastError();
197 
198  return false;
199  }
200  }
201  //-----------------------------------------------------------------------------
202  //! Initializes the cuda runtime.
203  //-----------------------------------------------------------------------------
204  ALPAKA_FCT_HOST static auto init()
205  -> void
206  {
207  static bool s_bInitialized = false;
208 
209  if(!s_bInitialized)
210  {
211  s_bInitialized = true;
212 
213  // - cudaDeviceScheduleSpin:
214  // Instruct CUDA to actively spin when waiting for results from the device.
215  // This can decrease latency when waiting for the device, but may lower the performance of CPU threads if they are performing work in parallel with the CUDA thread.
216  // - cudaDeviceMapHost:
217  // This flag must be set in order to allocate pinned host memory that is accessible to the device.
218  // If this flag is not set, cudaHostGetDevicePointer() will always return a failure code.
219  ALPAKA_CUDA_RT_CHECK(cudaSetDeviceFlags(
220  cudaDeviceScheduleSpin | cudaDeviceMapHost));
221  }
222  }
