158  //! The SYCL vectors' extent get trait specialization.
159  template<typename TExtent>
160  struct GetExtent<
161  DimInt<Dim<TExtent>::value>,
162  TExtent,
163  std::enable_if_t<experimental::IsSyclBuiltInType<TExtent>::value>>
164  {
165  static auto getExtent(TExtent const& extent)
166  {
167  if constexpr(std::is_scalar_v<TExtent>)
168  return extent;
169  else
170  {
171  // Creates a SYCL vector with one element from a multidimensional vector. The element is a reference
172  // to the requested dimension's vector element. Then return the element's value.
173  return extent.template swizzle<DimInt<Dim<TExtent>::value>::value>();
174  }
175  }
176  };
