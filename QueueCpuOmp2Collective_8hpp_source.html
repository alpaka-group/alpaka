73  //! The CPU collective device queue.
74  //
75  // @attention Queue can only be used together with the accelerator AccCpuOmp2Blocks.
76  //
77  // This queue is an example for a user provided queue and the behavior is strongly coupled
78  // to the user workflows.
79  //
80  // Within a OpenMP parallel region kernel will be performed collectively.
81  // All other operations will be performed from one thread (it is not defined which thread).
82  //
83  // Outside of a OpenMP parallel region the queue behaves like QueueCpuBlocking.
84  class QueueCpuOmp2Collective final
85  : public concepts::Implements<ConceptCurrentThreadWaitFor, QueueCpuOmp2Collective>
86  {
87  public:
88  QueueCpuOmp2Collective(DevCpu const& dev)
89  : m_spQueueImpl(std::make_shared<cpu::detail::QueueCpuOmp2CollectiveImpl>(dev))
90  , m_spBlockingQueue(std::make_shared<QueueCpuBlocking>(dev))
91  {
92  dev.registerQueue(m_spQueueImpl);
93  }
94  auto operator==(QueueCpuOmp2Collective const& rhs) const -> bool
95  {
96  return m_spQueueImpl == rhs.m_spQueueImpl && m_spBlockingQueue == rhs.m_spBlockingQueue;
97  }
98  auto operator!=(QueueCpuOmp2Collective const& rhs) const -> bool
99  {
100  return !((*this) == rhs);
101  }
102 
103  public:
104  std::shared_ptr<cpu::detail::QueueCpuOmp2CollectiveImpl> m_spQueueImpl;
105  std::shared_ptr<QueueCpuBlocking> m_spBlockingQueue;
106  };

133  //! The CPU blocking device queue enqueue trait specialization.
134  //! This default implementation for all tasks directly invokes the function call operator of the task.
135  template<typename TTask>
136  struct Enqueue<QueueCpuOmp2Collective, TTask>
137  {
138  ALPAKA_FN_HOST static auto enqueue(QueueCpuOmp2Collective& queue, TTask const& task) -> void
139  {
140  if(::omp_in_parallel() != 0)
141  {
142  // wait for all tasks en-queued before the parallel region
143  while(!empty(*queue.m_spBlockingQueue))
144  {
145  }
146  queue.m_spQueueImpl->m_uCurrentlyExecutingTask += 1u;
147 
148 # pragma omp single nowait
149  task();
150 
151  queue.m_spQueueImpl->m_uCurrentlyExecutingTask -= 1u;
152  }
153  else
154  {
155  std::lock_guard<std::mutex> lk(queue.m_spQueueImpl->m_mutex);
156  alpaka::enqueue(*queue.m_spBlockingQueue, task);
157  }
158  }
159  };
