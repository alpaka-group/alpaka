40  //! The CPU memory buffer.
41  template<
42  typename TElem,
43  typename TDim,
44  typename TIdx>
45  class BufCpuImpl final :
46  public AllocCpuAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>
47  {
48  static_assert(
49  !std::is_const<TElem>::value,
50  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const elements!");
51  static_assert(
52  !std::is_const<TIdx>::value,
53  "The idx type of the buffer can not be const!");
54  public:
55  //-----------------------------------------------------------------------------
56  template<
57  typename TExtent>
58  ALPAKA_FN_HOST BufCpuImpl(
59  DevCpu const & dev,
60  TExtent const & extent) :
61  AllocCpuAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>(),
62  m_dev(dev),
63  m_extentElements(extent::getExtentVecEnd<TDim>(extent)),
64  m_pMem(alpaka::malloc<TElem>(*this, static_cast<std::size_t>(computeElementCount(extent)))),
65  m_pitchBytes(static_cast<TIdx>(extent::getWidth(extent) * static_cast<TIdx>(sizeof(TElem))))
66 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
67  ,m_bPinned(false)
68 #endif
69  {
70  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
71 
72  static_assert(
73  TDim::value == Dim<TExtent>::value,
74  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be identical!");
75  static_assert(
76  std::is_same<TIdx, Idx<TExtent>>::value,
77  "The idx type of TExtent and the TIdx template parameter have to be identical!");
78 
79 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
80  std::cout << __func__
81  << " e: " << m_extentElements
82  << " ptr: " << static_cast<void *>(m_pMem)
83  << " pitch: " << m_pitchBytes
84  << std::endl;
85 #endif
86  }
87  //-----------------------------------------------------------------------------
88  BufCpuImpl(BufCpuImpl const &) = delete;
89  //-----------------------------------------------------------------------------
90  BufCpuImpl(BufCpuImpl &&) = default;
91  //-----------------------------------------------------------------------------
92  auto operator=(BufCpuImpl const &) -> BufCpuImpl & = delete;
93  //-----------------------------------------------------------------------------
94  auto operator=(BufCpuImpl &&) -> BufCpuImpl & = default;
95  //-----------------------------------------------------------------------------
96  ALPAKA_FN_HOST ~BufCpuImpl()
97  {
98  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
99 
100 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
101  // Unpin this memory if it is currently pinned.
102  unpin(*this);
103 #endif
104  // NOTE: m_pMem is allowed to be a nullptr here.
105  alpaka::free(*this, m_pMem);
106  }
107 
108  private:
109  //-----------------------------------------------------------------------------

120  }

412  //! The BufCpu memory pinning trait specialization.
413  template<
414  typename TElem,
415  typename TDim,
416  typename TIdx>
417  struct Pin<
418  BufCpu<TElem, TDim, TIdx>>
419  {
420  //-----------------------------------------------------------------------------
421  ALPAKA_FN_HOST static auto pin(
422  BufCpu<TElem, TDim, TIdx> & buf)
423  -> void
424  {
425  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
426 
427  if(!isPinned(buf))
428  {
429 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
430  if(buf.m_spBufCpuImpl->m_extentElements.prod() != 0)
431  {
432  // - cudaHostRegisterDefault:
433  // See http://cgi.cs.indiana.edu/~nhusted/dokuwiki/doku.php?id=programming:cudaperformance1
434  // - cudaHostRegisterPortable:
435  // The memory returned by this call will be considered as pinned memory by all CUDA contexts, not just the one that performed the allocation.
436  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_IGNORE(
437  ALPAKA_API_PREFIX(HostRegister)(
438  const_cast<void *>(reinterpret_cast<void const *>(getPtrNative(buf))),
439  extent::getExtentProduct(buf) * sizeof(Elem<BufCpu<TElem, TDim, TIdx>>),
440  ALPAKA_API_PREFIX(HostRegisterDefault)),
441  ALPAKA_API_PREFIX(ErrorHostMemoryAlreadyRegistered));
442 
443  buf.m_spBufCpuImpl->m_bPinned = true;
444  }
445 #else
446  static_assert(
447  meta::DependentFalseType<TElem>::value,
448  "Memory pinning of BufCpu is not implemented when CUDA or HIP is not enabled!");
449 #endif
450  }
451  }
452  };
