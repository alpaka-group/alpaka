42  //! The CPU memory buffer.
43  template<typename TElem, typename TDim, typename TIdx>
44  class BufCpuImpl final
45  {
46  static_assert(
47  !std::is_const_v<TElem>,
48  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const "
49  "elements!");
50  static_assert(!std::is_const_v<TIdx>, "The idx type of the buffer can not be const!");
51 
52  public:
53  template<typename TExtent>
54  ALPAKA_FN_HOST BufCpuImpl(
55  DevCpu dev,
56  TElem* pMem,
57  std::function<void(TElem*)> deleter,
58  TExtent const& extent) noexcept
59  : m_dev(std::move(dev))
60  , m_extentElements(getExtentVecEnd<TDim>(extent))
61  , m_pMem(pMem)
62  , m_deleter(std::move(deleter))
63  , m_bPinned(false)
64  {
65  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
66 
67  static_assert(
68  TDim::value == Dim<TExtent>::value,
69  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be "
70  "identical!");
71  static_assert(
72  std::is_same_v<TIdx, Idx<TExtent>>,
73  "The idx type of TExtent and the TIdx template parameter have to be identical!");
74 
75 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
76  std::cout << __func__ << " e: " << m_extentElements << " ptr: " << static_cast<void*>(m_pMem)
77  << std::endl;
78 #endif
79  }
80  BufCpuImpl(BufCpuImpl&&) = delete;
81  auto operator=(BufCpuImpl&&) -> BufCpuImpl& = delete;
82  ALPAKA_FN_HOST ~BufCpuImpl()
83  {
84  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
85 
86  // Unpin this memory if it is currently pinned.
87  if(m_bPinned)
88  {
89  try
90  {
91  unpin(*this); // May throw std::runtime_error
92  }
93  catch(std::runtime_error const& err)
94  {
95  std::cerr << "Caught runtime error while unpinning in ~BufCpuImpl(): " << err.what()
96  << std::endl;
97  }
98  }
99  // NOTE: m_pMem is allowed to be a nullptr here.
100  m_deleter(m_pMem);
101  }
102 
103  public:
104  DevCpu const m_dev;
105  Vec<TDim, TIdx> const m_extentElements;
106  TElem* const m_pMem;
107  std::function<void(TElem*)> m_deleter;
108  bool m_bPinned;
109  };

215  //! The BufCpu memory allocation trait specialization.
216  template<typename TElem, typename TDim, typename TIdx>
217  struct BufAlloc<TElem, TDim, TIdx, DevCpu>
218  {
219  template<typename TExtent>
220  ALPAKA_FN_HOST static auto allocBuf(DevCpu const& dev, TExtent const& extent) -> BufCpu<TElem, TDim, TIdx>
221  {
222  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
223 
224  // If ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT is defined, positive, and a power of 2, use it as the
225  // default alignment for host memory allocations. Otherwise, the alignment is chosen to enable optimal
226  // performance dependant on the target architecture.
227 #if defined(ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT)
228  static_assert(
229  ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT > 0
230  && ((ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT & (ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT - 1)) == 0),
231  "If defined, ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT must be a power of 2.");
232  constexpr std::size_t alignment = static_cast<std::size_t>(ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT);
233 #else
234  constexpr std::size_t alignment = core::vectorization::defaultAlignment;
235 #endif
236  // alpaka::AllocCpuAligned is stateless
237  using Allocator = AllocCpuAligned<std::integral_constant<std::size_t, alignment>>;
238  static_assert(std::is_empty_v<Allocator>, "AllocCpuAligned is expected to be stateless");
239  auto* memPtr = alpaka::malloc<TElem>(Allocator{}, static_cast<std::size_t>(getExtentProduct(extent)));
240  auto deleter = [](TElem* ptr) { alpaka::free(Allocator{}, ptr); };
241 
242  return BufCpu<TElem, TDim, TIdx>(dev, memPtr, std::move(deleter), extent);
243  }

274  auto* memPtr = alpaka::malloc<TElem>(Allocator{}, static_cast<std::size_t>(getExtentProduct(extent)));
275  auto deleter = [queue = std::move(queue)](TElem* ptr) mutable

338  //! The BufCpu memory pinning trait specialization.
339  template<typename TElem, typename TDim, typename TIdx>
340  struct Pin<BufCpu<TElem, TDim, TIdx>>
341  {
342  ALPAKA_FN_HOST static auto pin(BufCpu<TElem, TDim, TIdx>& buf) -> void
343  {
344  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
345 
346  if(!isPinned(buf))
347  {
348  if(buf.m_spBufCpuImpl->m_extentElements.prod() != 0)
349  {
350  // - cudaHostRegisterDefault:
351  // See http://cgi.cs.indiana.edu/~nhusted/dokuwiki/doku.php?id=programming:cudaperformance1
352  // - cudaHostRegisterPortable:
353  // The memory returned by this call will be considered as pinned memory by all CUDA contexts,
354  // not just the one that performed the allocation.
355 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
356  {
357  using TApi = ApiCudaRt;
358  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_IGNORE(
359  TApi::hostRegister(
360  const_cast<void*>(reinterpret_cast<void const*>(getPtrNative(buf))),
361  getExtentProduct(buf) * sizeof(Elem<BufCpu<TElem, TDim, TIdx>>),
362  TApi::hostRegisterDefault),
363  TApi::errorHostMemoryAlreadyRegistered);
364  }
365 #endif
366 #if defined(ALPAKA_ACC_GPU_HIP_ENABLED)
367  {
368  using TApi = ApiHipRt;
369  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_IGNORE(
370  TApi::hostRegister(
371  const_cast<void*>(reinterpret_cast<void const*>(getPtrNative(buf))),
372  getExtentProduct(buf) * sizeof(Elem<BufCpu<TElem, TDim, TIdx>>),
373  TApi::hostRegisterDefault),
374  TApi::errorHostMemoryAlreadyRegistered);
375  }
376 #endif
377  buf.m_spBufCpuImpl->m_bPinned = true;
378  }
379  }
380  }
381  };
