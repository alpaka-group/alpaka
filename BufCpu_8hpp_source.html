200  //! The BufCpu memory allocation trait specialization.
201  template<typename TElem, typename TDim, typename TIdx>
202  struct BufAlloc<TElem, TDim, TIdx, DevCpu>
203  {
204  template<typename TExtent>
205  ALPAKA_FN_HOST static auto allocBuf(DevCpu const& dev, TExtent const& extent) -> BufCpu<TElem, TDim, TIdx>
206  {
207  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
208 
209  // If ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT is defined, positive, and a power of 2, use it as the
210  // default alignment for host memory allocations. Otherwise, the alignment is chosen to enable optimal
211  // performance dependant on the target architecture.
212 #if defined(ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT)
213  static_assert(
214  ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT > 0
215  && ((ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT & (ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT - 1)) == 0),
216  "If defined, ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT must be a power of 2.");
217  constexpr std::size_t alignment = static_cast<std::size_t>(ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT);
218 #else
219  constexpr std::size_t alignment = core::vectorization::defaultAlignment;
220 #endif
221  // alpaka::AllocCpuAligned is stateless
222  using Allocator = AllocCpuAligned<std::integral_constant<std::size_t, alignment>>;
223  static_assert(std::is_empty_v<Allocator>, "AllocCpuAligned is expected to be stateless");
224  auto* memPtr = alpaka::malloc<TElem>(Allocator{}, static_cast<std::size_t>(getExtentProduct(extent)));
225  auto deleter = [](TElem* ptr) { alpaka::free(Allocator{}, ptr); };
226 
227  return BufCpu<TElem, TDim, TIdx>(dev, memPtr, std::move(deleter), extent);
228  }

230  //! The BufCpu stream-ordered memory allocation trait specialization.

240  std::is_same_v<Dev<TQueue>, DevCpu>,

242  DevCpu const& dev = getDev(queue);
243 
244  // If ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT is defined, positive, and a power of 2, use it as the
245  // default alignment for host memory allocations. Otherwise, the alignment is chosen to enable optimal
246  // performance dependant on the target architecture.
247 #if defined(ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT)
248  static_assert(
249  ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT > 0
250  && ((ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT & (ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT - 1)) == 0),
251  "If defined, ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT must be a power of 2.");
252  constexpr std::size_t alignment = static_cast<std::size_t>(ALPAKA_DEFAULT_HOST_MEMORY_ALIGNMENT);
253 #else
254  constexpr std::size_t alignment = core::vectorization::defaultAlignment;
255 #endif
256  // alpaka::AllocCpuAligned is stateless
257  using Allocator = AllocCpuAligned<std::integral_constant<std::size_t, alignment>>;
258  static_assert(std::is_empty_v<Allocator>, "AllocCpuAligned is expected to be stateless");
259  auto* memPtr = alpaka::malloc<TElem>(Allocator{}, static_cast<std::size_t>(getExtentProduct(extent)));
260  auto deleter = [queue = std::move(queue)](TElem* ptr) mutable
261  {
262  alpaka::enqueue(
263  queue,
264  [ptr]()
265  {
266  // free the memory
267  alpaka::free(Allocator{}, ptr);
268  });
269  };
270 
271  return BufCpu<TElem, TDim, TIdx>(dev, memPtr, std::move(deleter), extent);
272  }
