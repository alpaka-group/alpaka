36  //! The CPU memory buffer.
37  template<typename TElem, typename TDim, typename TIdx>
38  class BufCpuImpl final
39  : public AllocCpuAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>
40  {
41  static_assert(
42  !std::is_const<TElem>::value,
43  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const "
44  "elements!");
45  static_assert(!std::is_const<TIdx>::value, "The idx type of the buffer can not be const!");
46 
47  public:
48  template<typename TExtent>
49  ALPAKA_FN_HOST BufCpuImpl(DevCpu const& dev, TExtent const& extent)
50  : AllocCpuAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>()
51  , m_dev(dev)
52  , m_extentElements(extent::getExtentVecEnd<TDim>(extent))
53  , m_pMem(alpaka::malloc<TElem>(*this, static_cast<std::size_t>(computeElementCount(extent))))
54  , m_pitchBytes(static_cast<TIdx>(extent::getWidth(extent) * static_cast<TIdx>(sizeof(TElem))))
55 #if(defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
56  , m_bPinned(false)
57 #endif
58  {
59  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
60 
61  static_assert(
62  TDim::value == Dim<TExtent>::value,
63  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be "
64  "identical!");
65  static_assert(
66  std::is_same<TIdx, Idx<TExtent>>::value,
67  "The idx type of TExtent and the TIdx template parameter have to be identical!");
68 
69 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
70  std::cout << __func__ << " e: " << m_extentElements << " ptr: " << static_cast<void*>(m_pMem)
71  << " pitch: " << m_pitchBytes << std::endl;
72 #endif
73  }
74  BufCpuImpl(BufCpuImpl&&) = default;
75  auto operator=(BufCpuImpl&&) -> BufCpuImpl& = default;
76  ALPAKA_FN_HOST ~BufCpuImpl()
77  {
78  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
79 
80 #if(defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
81  // Unpin this memory if it is currently pinned.
82  unpin(*this);
83 #endif
84  // NOTE: m_pMem is allowed to be a nullptr here.
85  alpaka::free(*this, m_pMem);
86  }
87 
88  private:
89  //! \return The number of elements to allocate.
90  template<typename TExtent>
91  ALPAKA_FN_HOST static auto computeElementCount(TExtent const& extent) -> TIdx
92  {
93  auto const extentElementCount = extent::getExtentProduct(extent);
94 
95  return extentElementCount;
96  }
97 
98  public:
99  DevCpu const m_dev;
100  Vec<TDim, TIdx> const m_extentElements;
101  TElem* const m_pMem;
102  TIdx const m_pitchBytes;
103 #if(defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
104  bool m_bPinned;
105 #endif
106  };

269  //! The BufCpu memory pinning trait specialization.
270  template<typename TElem, typename TDim, typename TIdx>
271  struct Pin<BufCpu<TElem, TDim, TIdx>>
272  {
273  ALPAKA_FN_HOST static auto pin(BufCpu<TElem, TDim, TIdx>& buf) -> void
274  {
275  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
276 
277  if(!isPinned(buf))
278  {
279 #if(defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
280  if(buf.m_spBufCpuImpl->m_extentElements.prod() != 0)
281  {
282  // - cudaHostRegisterDefault:
283  // See http://cgi.cs.indiana.edu/~nhusted/dokuwiki/doku.php?id=programming:cudaperformance1
284  // - cudaHostRegisterPortable:
285  // The memory returned by this call will be considered as pinned memory by all CUDA contexts,
286  // not just the one that performed the allocation.
287  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_IGNORE(
288  ALPAKA_API_PREFIX(HostRegister)(
289  const_cast<void*>(reinterpret_cast<void const*>(getPtrNative(buf))),
290  extent::getExtentProduct(buf) * sizeof(Elem<BufCpu<TElem, TDim, TIdx>>),
291  ALPAKA_API_PREFIX(HostRegisterDefault)),
292  ALPAKA_API_PREFIX(ErrorHostMemoryAlreadyRegistered));
293 
294  buf.m_spBufCpuImpl->m_bPinned = true;
295  }
296 #else
297  static_assert(
298  meta::DependentFalseType<TElem>::value,
299  "Memory pinning of BufCpu is not implemented when CUDA or HIP is not enabled!");
300 #endif
301  }
302  }
303  };
