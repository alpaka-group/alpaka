125  /// Helper struct providing [] subscript access to CUDA vector types
126  template<typename TScalar, unsigned N>
127  struct CudaVectorArrayWrapper;
128 
129  template<typename TScalar>
130  struct CudaVectorArrayWrapper<TScalar, 4> : public detail::CudaVectorArrayTypeTraits<TScalar, 4>::type
131  {
132  using value_type = TScalar;
133  constexpr static unsigned size = 4;
134  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE CudaVectorArrayWrapper(std::initializer_list<TScalar> init)
135  {
136  auto it = std::begin(init);
137  this->x = *it++;
138  this->y = *it++;
139  this->z = *it++;
140  this->w = *it++;
141  }
142  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE constexpr value_type& operator[](int const k) noexcept
143  {
144  assert(k >= 0 && k < 4);
145  return k == 0 ? this->x : (k == 1 ? this->y : (k == 2 ? this->z : this->w));
146  }
147 
148  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE constexpr const value_type& operator[](int const k) const noexcept
149  {
150  assert(k >= 0 && k < 4);
151  return k == 0 ? this->x : (k == 1 ? this->y : (k == 2 ? this->z : this->w));
152  }
153  };
