128  /// Helper struct providing [] subscript access to CUDA vector types
129  template<typename TScalar, unsigned N>
130  struct CudaVectorArrayWrapper;
131 
132  template<typename TScalar>
133  struct CudaVectorArrayWrapper<TScalar, 4> : public detail::CudaVectorArrayTypeTraits<TScalar, 4>::type
134  {
135  using value_type = TScalar;
136  constexpr static unsigned size = 4;
137  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE CudaVectorArrayWrapper(std::initializer_list<TScalar> init)
138  {
139  auto it = std::begin(init);
140  this->x = *it++;
141  this->y = *it++;
142  this->z = *it++;
143  this->w = *it++;
144  }
145  template<class Other>
146  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE CudaVectorArrayWrapper(Other const& o)
147  {
148  static_assert(std::tuple_size_v<Other> == size, "Can only convert between vectors of same size.");
149  static_assert(
150  std::is_same_v<typename Other::value_type, value_type>,
151  "Can only convert between vectors of same element type.");
152  this->x = o[0];
153  this->y = o[1];
154  this->z = o[2];
155  this->w = o[3];
156  }
157  ALPAKA_FN_HOST_ACC constexpr operator std::array<value_type, size>() const
158  {
159  std::array<value_type, size> ret;
160  ret[0] = this->x;
161  ret[1] = this->y;
162  ret[2] = this->z;
163  ret[3] = this->w;
164  return ret;
165  }
166 
167  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE constexpr value_type& operator[](int const k) noexcept
168  {
169  assert(k >= 0 && k < 4);
170  return k == 0 ? this->x : (k == 1 ? this->y : (k == 2 ? this->z : this->w));
171  }
172 
173  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE constexpr value_type const& operator[](int const k) const noexcept
174  {
175  assert(k >= 0 && k < 4);
176  return k == 0 ? this->x : (k == 1 ? this->y : (k == 2 ? this->z : this->w));
177  }
178  };
