136  /// Helper struct providing [] subscript access to CUDA vector types
137  template<typename TScalar, unsigned N>
138  struct CudaVectorArrayWrapper;
139 
140  template<typename TScalar>
141  struct CudaVectorArrayWrapper<TScalar, 4> : public detail::CudaVectorArrayTypeTraits<TScalar, 4>::type
142  {
143  using value_type = TScalar;
144  constexpr static unsigned size = 4;
145  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE CudaVectorArrayWrapper(std::initializer_list<TScalar> init)
146  {
147  auto it = std::begin(init);
148  this->x = *it++;
149  this->y = *it++;
150  this->z = *it++;
151  this->w = *it++;
152  }
153  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE CXX17_CONSTEXPR value_type& operator[](int const k) noexcept
154  {
155  assert(k >= 0 && k < 4);
156  return k == 0 ? this->x : (k == 1 ? this->y : (k == 2 ? this->z : this->w));
157  }
158 
159  ALPAKA_FN_HOST_ACC ALPAKA_FN_INLINE constexpr const value_type& operator[](int const k) const noexcept
160  {
161  assert(k >= 0 && k < 4);
162  return k == 0 ? this->x : (k == 1 ? this->y : (k == 2 ? this->z : this->w));
163  }
164  };
