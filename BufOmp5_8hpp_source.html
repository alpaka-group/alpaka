42  class BufOmp5Impl

44  static_assert(

46  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const "

48  static_assert(!std::is_const_v<TIdx>, "The idx type of the buffer can not be const!");
49 
50  public:
51  //! Constructor

54  : m_dev(dev)
55  , m_extentElements(getExtentVecEnd<TDim>(extent))
56  , m_pMem(pMem)
57  {
58  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
59 
60  static_assert(
61  TDim::value == alpaka::Dim<TExtent>::value,
62  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be "
63  "identical!");
64  static_assert(
65  std::is_same<TIdx, Idx<TExtent>>::value,
66  "The idx type of TExtent and the TIdx template parameter have to be identical!");
67  }
68 
69  public:
70  DevOmp5 m_dev;
71  Vec<TDim, TIdx> m_extentElements;
72  TElem* m_pMem;
73 
74  BufOmp5Impl(BufOmp5Impl&&) = delete;
75  auto operator=(BufOmp5Impl&&) -> BufOmp5Impl& = delete;
76  ~BufOmp5Impl()

199  //! The BufOmp5 1D memory allocation trait specialization.
200  template<typename TElem, typename TIdx>
201  struct BufAlloc<TElem, DimInt<1u>, TIdx, DevOmp5>
202  {
203  template<typename TExtent>
204  ALPAKA_FN_HOST static auto allocBuf(DevOmp5 const& dev, TExtent const& extent)
205  -> BufOmp5<TElem, DimInt<1u>, TIdx>
206  {
207  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
208 
209  auto const width(getWidth(extent));
210  auto const widthBytes(width * static_cast<TIdx>(sizeof(TElem)));
211 
212  void* memPtr = omp_target_alloc(static_cast<std::size_t>(widthBytes), dev.getNativeHandle());
213 
214 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
215  std::cout << __func__ << " ew: " << width << " ewb: " << widthBytes << " ptr: " << memPtr
216  << " device: " << dev.getNativeHandle() << std::endl;
217 # endif
218  return BufOmp5<TElem, DimInt<1u>, TIdx>(dev, reinterpret_cast<TElem*>(memPtr), extent);
219  }
220  };
