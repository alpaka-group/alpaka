39  //! The OMP5 memory buffer detail.
40  template<typename TElem, typename TDim, typename TIdx>
41  class BufOmp5Impl
42  {
43  static_assert(
44  !std::is_const<TElem>::value,
45  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const "
46  "elements!");
47  static_assert(!std::is_const<TIdx>::value, "The idx type of the buffer can not be const!");
48 
49  private:
50  using Elem = TElem;
51  using Dim = TDim;
52  //! Calculate the pitches purely from the extents.
53  template<typename TExtent>
54  ALPAKA_FN_HOST static auto calculatePitchesFromExtents(TExtent const& extent) -> Vec<TDim, TIdx>
55  {
56  Vec<TDim, TIdx> pitchBytes(Vec<TDim, TIdx>::all(0));
57  pitchBytes[TDim::value - 1u] = extent[TDim::value - 1u] * static_cast<TIdx>(sizeof(TElem));
58  for(TIdx i = TDim::value - 1u; i > static_cast<TIdx>(0u); --i)
59  {
60  pitchBytes[i - 1] = extent[i - 1] * pitchBytes[i];
61  }
62  return pitchBytes;
63  }
64 
65  public:
66  //! Constructor
67  template<typename TExtent>
68  ALPAKA_FN_HOST BufOmp5Impl(DevOmp5 const& dev, TElem* const pMem, TExtent const& extent)
69  : m_dev(dev)
70  , m_extentElements(extent::getExtentVecEnd<TDim>(extent))
71  , m_pitchBytes(calculatePitchesFromExtents(m_extentElements))
72  , m_pMem(pMem)
73  {
74  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
75 
76  static_assert(
77  TDim::value == alpaka::Dim<TExtent>::value,
78  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be "
79  "identical!");
80  static_assert(
81  std::is_same<TIdx, Idx<TExtent>>::value,
82  "The idx type of TExtent and the TIdx template parameter have to be identical!");
83  }
84 
85  public:
86  DevOmp5 m_dev;
87  Vec<TDim, TIdx> m_extentElements;
88  Vec<TDim, TIdx> m_pitchBytes;
89  TElem* m_pMem;
90

92  BufOmp5Impl(BufOmp5Impl&&) = default;

94  BufOmp5Impl& operator=(BufOmp5Impl&&) = default;
