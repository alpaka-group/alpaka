28  /** Most common Philox engine variant, outputs single number
29  *
30  * This is a variant of the Philox engine generator which outputs a single float. The counter size is \f$4
31  * \times 32 = 128\f$ bits. Since the engine returns a single number, the generated result, which has the same
32  * size as the counter, has to be stored between invocations. Additionally a 32 bit pointer is stored. The
33  * total size of the state is 352 bits = 44 bytes.
34  *
35  * Ref.: J. K. Salmon, M. A. Moraes, R. O. Dror and D. E. Shaw, "Parallel random numbers: As easy as 1, 2, 3,"
36  * SC '11: Proceedings of 2011 International Conference for High Performance Computing, Networking, Storage and
37  * Analysis, 2011, pp. 1-12, doi: 10.1145/2063384.2063405.
38  *
39  * @tparam TAcc Accelerator type as defined in alpaka/acc
40  */
41  template<typename TAcc>
42  class Philox4x32x10 : public concepts::Implements<ConceptRand, Philox4x32x10<TAcc>>
43  {
44  public:
45  using EngineParams
46  = engine::PhiloxParams<4, 32, 10>; ///< Philox algorithm: 10 rounds, 4 numbers of size 32.
47  using EngineVariant = engine::PhiloxSingle<TAcc, EngineParams>; ///< Engine outputs a single number
48

86  /** Most common Philox engine variant, outputs a 4-vector of floats
87  *
88  * This is a variant of the Philox engine generator which outputs a vector containing 4 floats. The counter
89  * size is \f$4 \times 32 = 128\f$ bits. Since the engine returns the whole generated vector, it is up to the
90  * user to extract individual floats as they need. The benefit is smaller state size since the state does not
91  * contain the intermediate results. The total size of the state is 192 bits = 24 bytes.
92  *
93  * Ref.: J. K. Salmon, M. A. Moraes, R. O. Dror and D. E. Shaw, "Parallel random numbers: As easy as 1, 2, 3,"
94  * SC '11: Proceedings of 2011 International Conference for High Performance Computing, Networking, Storage and
95  * Analysis, 2011, pp. 1-12, doi: 10.1145/2063384.2063405.
96  *
97  * @tparam TAcc Accelerator type as defined in alpaka/acc
98  */

165  template<typename TResult>
166  class UniformReal<TResult, std::enable_if_t<meta::IsArrayOrVector<TResult>::value>>
167  : public concepts::Implements<ConceptRand, UniformReal<TResult>>
168  {
169  using T = typename TResult::value_type;
170 
171  public:
172  ALPAKA_FN_HOST_ACC UniformReal() : UniformReal(0, 1)
173  {
174  }
175 
176  ALPAKA_FN_HOST_ACC UniformReal(T min, T max) : _min(min), _max(max), _range(_max - _min)
177  {
178  }
179 
180  template<typename TEngine>
181  ALPAKA_FN_HOST_ACC TResult operator()(TEngine& engine)
182  {
183  auto result = engine();
184  T scale = static_cast<T>(1) / engine.max() * _range;
185  TResult ret{
186  static_cast<T>(result[0]) * scale + _min,
187  static_cast<T>(result[1]) * scale + _min,
188  static_cast<T>(result[2]) * scale + _min,
189  static_cast<T>(result[3]) * scale + _min};
190  return ret;
191  }
192 
193  private:
194  const T _min;
195  const T _max;
196  const T _range;
197  };
