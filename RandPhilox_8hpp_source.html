25  /** Most common Philox engine variant, outputs single number
26  *
27  * This is a variant of the Philox engine generator which outputs a single float. The counter size is \f$4
28  * \times 32 = 128\f$ bits. Since the engine returns a single number, the generated result, which has the same
29  * size as the counter, has to be stored between invocations. Additionally a 32 bit pointer is stored. The
30  * total size of the state is 352 bits = 44 bytes.
31  *
32  * Ref.: J. K. Salmon, M. A. Moraes, R. O. Dror and D. E. Shaw, "Parallel random numbers: As easy as 1, 2, 3,"
33  * SC '11: Proceedings of 2011 International Conference for High Performance Computing, Networking, Storage and
34  * Analysis, 2011, pp. 1-12, doi: 10.1145/2063384.2063405.
35  *
36  * @tparam TAcc Accelerator type as defined in alpaka/acc
37  */
38  template<typename TAcc>
39  class Philox4x32x10 : public concepts::Implements<ConceptRand, Philox4x32x10<TAcc>>
40  {
41  public:
42  using EngineParams = engine::PhiloxParams<4, 32, 10>; ///< Philox algorithm: 10 rounds, 4 numbers of size 32.
43  using EngineVariant = engine::PhiloxSingle<TAcc, EngineParams>; ///< Engine outputs a single number
44 
45  /** Initialize a new Philox engine
46  *
47  * @param seed Set the Philox generator key
48  * @param subsequence Select a subsequence of size 2^64
49  * @param offset Skip \a offset numbers form the start of the subsequence
50  */
51  ALPAKA_FN_HOST_ACC Philox4x32x10(
52  std::uint64_t const seed = 0,
53  std::uint64_t const subsequence = 0,
54  std::uint64_t const offset = 0)
55  : engineVariant(seed, subsequence, offset)
56  {
57  }
58 
59  // STL UniformRandomBitGenerator concept
60  // https://en.cppreference.com/w/cpp/named_req/UniformRandomBitGenerator
61  using result_type = std::uint32_t;
62  ALPAKA_FN_HOST_ACC constexpr auto min() -> result_type
63  {
64  return 0;
65  }
66  ALPAKA_FN_HOST_ACC constexpr auto max() -> result_type
67  {
68  return std::numeric_limits<result_type>::max();
69  }
70  ALPAKA_FN_HOST_ACC auto operator()() -> result_type
71  {
72  return engineVariant();
73  }
74 
75  private:
76  EngineVariant engineVariant;
77  };

79  /** Most common Philox engine variant, outputs a 4-vector of floats
80  *
81  * This is a variant of the Philox engine generator which outputs a vector containing 4 floats. The counter
82  * size is \f$4 \times 32 = 128\f$ bits. Since the engine returns the whole generated vector, it is up to the
83  * user to extract individual floats as they need. The benefit is smaller state size since the state does not
84  * contain the intermediate results. The total size of the state is 192 bits = 24 bytes.
85  *
86  * Ref.: J. K. Salmon, M. A. Moraes, R. O. Dror and D. E. Shaw, "Parallel random numbers: As easy as 1, 2, 3,"
87  * SC '11: Proceedings of 2011 International Conference for High Performance Computing, Networking, Storage and
88  * Analysis, 2011, pp. 1-12, doi: 10.1145/2063384.2063405.
89  *
90  * @tparam TAcc Accelerator type as defined in alpaka/acc
91  */

146  /// TEMP: Distributions to be decided on later. The generator should be compatible with STL as of now.
147  template<typename TResult, typename TSfinae = void>
148  class UniformReal : public concepts::Implements<ConceptRand, UniformReal<TResult>>
149  {
150  template<typename TRes, typename TEnable = void>
151  struct ResultType
152  {
153  using type = TRes;
154  };
155 
156  template<typename TRes>
157  struct ResultType<TRes, std::enable_if_t<meta::IsArrayOrVector<TRes>::value>>
158  {
159  using type = typename TRes::value_type;
160  };
161 
162  using T = typename ResultType<TResult>::type;
163  static_assert(std::is_floating_point_v<T>, "Only floating-point types are supported");
164 
165  public:
166  ALPAKA_FN_HOST_ACC UniformReal() : UniformReal(0, 1)
167  {
168  }
169 
170  ALPAKA_FN_HOST_ACC UniformReal(T min, T max) : _min(min), _max(max), _range(_max - _min)
171  {
172  }
173 
174  template<typename TEngine>
175  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> TResult
176  {
177  if constexpr(meta::IsArrayOrVector<TResult>::value)
178  {
179  auto result = engine();
180  T scale = static_cast<T>(1) / engine.max() * _range;
181  TResult ret{
182  static_cast<T>(result[0]) * scale + _min,
183  static_cast<T>(result[1]) * scale + _min,
184  static_cast<T>(result[2]) * scale + _min,
185  static_cast<T>(result[3]) * scale + _min};
186  return ret;
187  }
188  else
189  {
190  // Since it's possible to get a host-only engine here, the call has to go through proxy
191  return static_cast<T>(EngineCallHostAccProxy<TEngine>{}(engine)) / engine.max() * _range + _min;
192  }
193 
194  ALPAKA_UNREACHABLE(TResult{});
195  }
196 
197  private:
198  const T _min;
199  const T _max;
200  const T _range;
201  };
