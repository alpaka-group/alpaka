179  //! The compare and swap function object.
180  struct AtomicCas
181  {
182  //! AtomicCas for non floating point values
183  // \return The old value of addr.
184  ALPAKA_NO_HOST_ACC_WARNING
185  template<typename T, std::enable_if_t<!std::is_floating_point_v<T>, bool> = true>
186  ALPAKA_FN_HOST_ACC auto operator()(T* addr, T const& compare, T const& value) const -> T
187  {
188  auto const old = *addr;
189  auto& ref = *addr;
190 
191 // gcc-7.4.0 assumes for an optimization that a signed overflow does not occur here.
192 // That's fine, so ignore that warning.
193 #if BOOST_COMP_GNUC && (BOOST_COMP_GNUC == BOOST_VERSION_NUMBER(7, 4, 0))
194 # pragma GCC diagnostic push
195 # pragma GCC diagnostic ignored "-Wstrict-overflow"
196 #endif
197  // check if values are bit-wise equal
198  ref = ((old == compare) ? value : old);
199 #if BOOST_COMP_GNUC && (BOOST_COMP_GNUC == BOOST_VERSION_NUMBER(7, 4, 0))
200 # pragma GCC diagnostic pop
201 #endif
202  return old;
203  }
204  //! AtomicCas for floating point values
205  // \return The old value of addr.
206  ALPAKA_NO_HOST_ACC_WARNING
207  template<typename T, std::enable_if_t<std::is_floating_point_v<T>, bool> = true>
208  ALPAKA_FN_HOST_ACC auto operator()(T* addr, T const& compare, T const& value) const -> T
209  {
210  static_assert(sizeof(T) == 4u || sizeof(T) == 8u, "AtomicCas is supporting only 32bit and 64bit values!");
211  // Type to reinterpret too to perform the bit comparison
212  using BitType = std::conditional_t<sizeof(T) == 4u, unsigned int, unsigned long long>;
213 
214  // type used to have a safe way to reinterprete the data into another type
215  // std::variant can not be used because clang8 has issues to compile std::variant
216  struct BitUnion
217  {
218  union
219  {
220  T value;
221  BitType r;
222  };
223  };
224 
225  auto const old = *addr;
226  auto& ref = *addr;
227 
228 // gcc-7.4.0 assumes for an optimization that a signed overflow does not occur here.
229 // That's fine, so ignore that warning.
230 #if BOOST_COMP_GNUC && (BOOST_COMP_GNUC == BOOST_VERSION_NUMBER(7, 4, 0))
231 # pragma GCC diagnostic push
232 # pragma GCC diagnostic ignored "-Wstrict-overflow"
233 #endif
234  BitUnion o{old};
235  BitUnion c{compare};
236 
237  ref = ((o.r == c.r) ? value : old);
238 #if BOOST_COMP_GNUC && (BOOST_COMP_GNUC == BOOST_VERSION_NUMBER(7, 4, 0))
239 # pragma GCC diagnostic pop
240 #endif
241  return old;
242  }
