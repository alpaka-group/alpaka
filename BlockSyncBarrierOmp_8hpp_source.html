117  ALPAKA_FN_ACC static auto syncBlockThreadsPredicate(
118  block::BlockSyncBarrierOmp const & blockSync,
119  int predicate)
120  -> int
121  {
122  // The first thread initializes the value.
123  // There is an implicit barrier at the end of omp single.
124  // NOTE: This code is executed only once for all OpenMP threads.
125  // If multiple blocks with multiple threads are executed in parallel
126  // this reduction is executed only for one block!
127  #pragma omp single
128  {
129  ++blockSync.m_generation;
130  blockSync.m_result[blockSync.m_generation % 2u] = TOp::InitialValue;
131  }
132 
133  auto const generationMod2(blockSync.m_generation % 2u);
134  int& result(blockSync.m_result[generationMod2]);
135  bool const predicateBool(predicate != 0);
136 
137  detail::AtomicOp<TOp>()(result, predicateBool);
138 
139  // Wait for all threads to write their predicate into the vector.
140  // NOTE: This waits for all threads in all blocks.
141  // If multiple blocks are executed in parallel this is not optimal.
142  #pragma omp barrier
143 
144  return blockSync.m_result[generationMod2];
145  }
