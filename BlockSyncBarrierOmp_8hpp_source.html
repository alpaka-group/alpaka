114  ALPAKA_NO_HOST_ACC_WARNING
115  ALPAKA_FN_ACC static auto syncBlockThreadsPredicate(
116  BlockSyncBarrierOmp const & blockSync,
117  int predicate)
118  -> int
119  {
120  // The first thread initializes the value.
121  // There is an implicit barrier at the end of omp single.
122  // NOTE: This code is executed only once for all OpenMP threads.
123  // If multiple blocks with multiple threads are executed in parallel
124  // this reduction is executed only for one block!
125  #pragma omp single
126  {
127  ++blockSync.m_generation;
128  blockSync.m_result[blockSync.m_generation % 2u] = TOp::InitialValue;
129  }
130 
131  auto const generationMod2(blockSync.m_generation % 2u);
132  int& result(blockSync.m_result[generationMod2]);
133  bool const predicateBool(predicate != 0);
134 
135  detail::AtomicOp<TOp>()(result, predicateBool);
136 
137  // Wait for all threads to write their predicate into the vector.
138  // NOTE: This waits for all threads in all blocks.
139  // If multiple blocks are executed in parallel this is not optimal.
140  #pragma omp barrier
141 
142  return blockSync.m_result[generationMod2];
143  }
144  };
