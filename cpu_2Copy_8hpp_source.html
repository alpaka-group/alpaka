35  //! The CPU device memory copy task base.
36  //!
37  //! Copies from CPU memory into CPU memory.
38  template<
39  typename TDim,
40  typename TViewDst,
41  typename TViewSrc,
42  typename TExtent>
43  struct TaskCopyCpuBase
44  {
45  using ExtentSize = Idx<TExtent>;
46  using DstSize = Idx<TViewDst>;
47  using SrcSize = Idx<TViewSrc>;
48  using Elem = elem::Elem<TViewSrc>;
49 
50  static_assert(
51  !std::is_const<TViewDst>::value,
52  "The destination view can not be const!");
53 
54  static_assert(
55  Dim<TViewDst>::value == Dim<TViewSrc>::value,
56  "The source and the destination view are required to have the same dimensionality!");
57  static_assert(
58  Dim<TViewDst>::value == Dim<TExtent>::value,
59  "The views and the extent are required to have the same dimensionality!");
60  static_assert(
61  Dim<TViewDst>::value == TDim::value,
62  "The destination view and the input TDim are required to have the same dimensionality!");
63 
64  static_assert(
65  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
66  "The destination view and the extent are required to have compatible idx type!");
67  static_assert(
68  meta::IsIntegralSuperset<SrcSize, ExtentSize>::value,
69  "The source view and the extent are required to have compatible idx type!");
70 
71  static_assert(
72  std::is_same<elem::Elem<TViewDst>, std::remove_const_t<elem::Elem<TViewSrc>>>::value,
73  "The source and the destination view are required to have the same element type!");
74

78  TViewSrc const & viewSrc,
79  TExtent const & extent) :

82 #if (!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)

132  //! The CPU device ND memory copy task.
133  template<
134  typename TDim,
135  typename TViewDst,
136  typename TViewSrc,
137  typename TExtent>
138  struct TaskCopyCpu : public TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>
139  {
140  using DimMin1 = DimInt<TDim::value - 1u>;
141  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::ExtentSize;
142  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::DstSize;
143  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::SrcSize;
144 
145  //-----------------------------------------------------------------------------
146  using TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::TaskCopyCpuBase;
147 
148  //-----------------------------------------------------------------------------
149  ALPAKA_FN_HOST auto operator()() const
150  -> void
151  {
152  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
153 
154 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
155  this->printDebug();
156 #endif
157  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one iteration.
158  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
159  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
160  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
161  Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_srcPitchBytes));
162 
163  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
164  {
165  meta::ndLoopIncIdx(
166  extentWithoutInnermost,
167  [&](Vec<DimMin1, ExtentSize> const & idx)
168  {
169  std::memcpy(
170  reinterpret_cast<void *>(this->m_dstMemNative + (cast<DstSize>(idx) * dstPitchBytesWithoutOutmost).foldrAll(std::plus<DstSize>())),
171  reinterpret_cast<void const *>(this->m_srcMemNative + (cast<SrcSize>(idx) * srcPitchBytesWithoutOutmost).foldrAll(std::plus<SrcSize>())),
172  static_cast<std::size_t>(this->m_extentWidthBytes));
173  });
174  }
175  }
176  };
