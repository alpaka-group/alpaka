31  //! The CPU device memory copy task base.
32  //!
33  //! Copies from CPU memory into CPU memory.
34  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
35  struct TaskCopyCpuBase
36  {
37  using ExtentSize = Idx<TExtent>;
38  using DstSize = Idx<TViewDst>;
39  using SrcSize = Idx<TViewSrc>;
40  using Elem = alpaka::Elem<TViewSrc>;
41 
42  template<typename TViewFwd>
43  TaskCopyCpuBase(TViewFwd&& viewDst, TViewSrc const& viewSrc, TExtent const& extent)
44  : m_extent(getExtentVec(extent))
45  , m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem)))
46 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
47  , m_dstExtent(getExtentVec(viewDst))
48  , m_srcExtent(getExtentVec(viewSrc))
49 #endif
50  , m_dstPitchBytes(getPitchBytesVec(viewDst))
51  , m_srcPitchBytes(getPitchBytesVec(viewSrc))
52  , m_dstMemNative(reinterpret_cast<std::uint8_t*>(getPtrNative(viewDst)))
53  , m_srcMemNative(reinterpret_cast<std::uint8_t const*>(getPtrNative(viewSrc)))
54  {
55  if constexpr(TDim::value > 0)
56  {
57  ALPAKA_ASSERT((castVec<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
58  ALPAKA_ASSERT((castVec<SrcSize>(m_extent) <= m_srcExtent).foldrAll(std::logical_or<bool>()));
59  ALPAKA_ASSERT(static_cast<DstSize>(m_extentWidthBytes) <= m_dstPitchBytes[TDim::value - 1u]);
60  ALPAKA_ASSERT(static_cast<SrcSize>(m_extentWidthBytes) <= m_srcPitchBytes[TDim::value - 1u]);
61  }
62  }
63 
64 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
65  ALPAKA_FN_HOST auto printDebug() const -> void
66  {
67  std::cout << __func__ << " e: " << m_extent << " ewb: " << this->m_extentWidthBytes
68  << " de: " << m_dstExtent << " dptr: " << reinterpret_cast<void*>(m_dstMemNative)
69  << " dpitchb: " << m_dstPitchBytes << " se: " << m_srcExtent
70  << " sptr: " << reinterpret_cast<void const*>(m_srcMemNative)
71  << " spitchb: " << m_srcPitchBytes << std::endl;
72  }
73 #endif
74 
75  Vec<TDim, ExtentSize> const m_extent;
76  ExtentSize const m_extentWidthBytes;
77 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
78  Vec<TDim, DstSize> const m_dstExtent;
79  Vec<TDim, SrcSize> const m_srcExtent;
80 #endif
81  Vec<TDim, DstSize> const m_dstPitchBytes;
82  Vec<TDim, SrcSize> const m_srcPitchBytes;
83 
84  std::uint8_t* const m_dstMemNative;
85  std::uint8_t const* const m_srcMemNative;
86  };

88  //! The CPU device ND memory copy task.
89  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
90  struct TaskCopyCpu : public TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>
91  {
92  using DimMin1 = DimInt<TDim::value - 1u>;
93  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::ExtentSize;
94  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::DstSize;
95  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::SrcSize;
96 
97  using TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::TaskCopyCpuBase;
98 
99  ALPAKA_FN_HOST auto operator()() const -> void
100  {
101  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
102 
103 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
104  this->printDebug();
105 #endif
106  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one
107  // iteration.
108  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
109  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
110  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
111  Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_srcPitchBytes));
112 
113  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
114  {
115  meta::ndLoopIncIdx(
116  extentWithoutInnermost,
117  [&](Vec<DimMin1, ExtentSize> const& idx)
118  {
119  std::memcpy(
120  reinterpret_cast<void*>(
121  this->m_dstMemNative
122  + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost)
123  .foldrAll(std::plus<DstSize>())),
124  reinterpret_cast<void const*>(
125  this->m_srcMemNative
126  + (castVec<SrcSize>(idx) * srcPitchBytesWithoutOutmost)
127  .foldrAll(std::plus<SrcSize>())),
128  static_cast<std::size_t>(this->m_extentWidthBytes));
129  });
130  }
131  }
132  };
