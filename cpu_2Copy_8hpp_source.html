30  //! The CPU device memory copy task base.
31  //!
32  //! Copies from CPU memory into CPU memory.
33  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
34  struct TaskCopyCpuBase
35  {
36  using ExtentSize = Idx<TExtent>;
37  using DstSize = Idx<TViewDst>;
38  using SrcSize = Idx<TViewSrc>;
39  using Elem = alpaka::Elem<TViewSrc>;
40 
41  static_assert(!std::is_const<TViewDst>::value, "The destination view can not be const!");
42 
43  static_assert(
44  Dim<TViewDst>::value == Dim<TViewSrc>::value,
45  "The source and the destination view are required to have the same dimensionality!");
46  static_assert(
47  Dim<TViewDst>::value == Dim<TExtent>::value,
48  "The views and the extent are required to have the same dimensionality!");
49  static_assert(
50  Dim<TViewDst>::value == TDim::value,
51  "The destination view and the input TDim are required to have the same dimensionality!");
52 
53  static_assert(
54  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
55  "The destination view and the extent are required to have compatible idx type!");
56  static_assert(
57  meta::IsIntegralSuperset<SrcSize, ExtentSize>::value,
58  "The source view and the extent are required to have compatible idx type!");
59 
60  static_assert(
61  std::is_same<alpaka::Elem<TViewDst>, std::remove_const_t<alpaka::Elem<TViewSrc>>>::value,
62  "The source and the destination view are required to have the same element type!");
63 
64  TaskCopyCpuBase(TViewDst& viewDst, TViewSrc const& viewSrc, TExtent const& extent)
65  : m_extent(extent::getExtentVec(extent))
66  , m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem)))
67  ,
68 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
69  m_dstExtent(extent::getExtentVec(viewDst))
70  , m_srcExtent(extent::getExtentVec(viewSrc))
71  ,
72 #endif
73  m_dstPitchBytes(getPitchBytesVec(viewDst))
74  , m_srcPitchBytes(getPitchBytesVec(viewSrc))
75  ,
76 
77  m_dstMemNative(reinterpret_cast<std::uint8_t*>(getPtrNative(viewDst)))
78  , m_srcMemNative(reinterpret_cast<std::uint8_t const*>(getPtrNative(viewSrc)))
79  {
80  ALPAKA_ASSERT((castVec<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
81  ALPAKA_ASSERT((castVec<SrcSize>(m_extent) <= m_srcExtent).foldrAll(std::logical_or<bool>()));
82  ALPAKA_ASSERT(static_cast<DstSize>(m_extentWidthBytes) <= m_dstPitchBytes[TDim::value - 1u]);
83  ALPAKA_ASSERT(static_cast<SrcSize>(m_extentWidthBytes) <= m_srcPitchBytes[TDim::value - 1u]);
84  }
85 
86 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
87  ALPAKA_FN_HOST auto printDebug() const -> void
88  {
89  std::cout << __func__ << " e: " << m_extent << " ewb: " << this->m_extentWidthBytes
90  << " de: " << m_dstExtent << " dptr: " << reinterpret_cast<void*>(m_dstMemNative)
91  << " dpitchb: " << m_dstPitchBytes << " se: " << m_srcExtent
92  << " sptr: " << reinterpret_cast<void const*>(m_srcMemNative)
93  << " spitchb: " << m_srcPitchBytes << std::endl;
94  }
95 #endif
96 
97  Vec<TDim, ExtentSize> const m_extent;
98  ExtentSize const m_extentWidthBytes;
99 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
100  Vec<TDim, DstSize> const m_dstExtent;
101  Vec<TDim, SrcSize> const m_srcExtent;
102 #endif
103  Vec<TDim, DstSize> const m_dstPitchBytes;
104  Vec<TDim, SrcSize> const m_srcPitchBytes;
105 
106  std::uint8_t* const m_dstMemNative;
107  std::uint8_t const* const m_srcMemNative;
108  };

111  //! The CPU device ND memory copy task.
112  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
113  struct TaskCopyCpu : public TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>
114  {
115  using DimMin1 = DimInt<TDim::value - 1u>;
116  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::ExtentSize;
117  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::DstSize;
118  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::SrcSize;
119 
120  using TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::TaskCopyCpuBase;
121 
122  ALPAKA_FN_HOST auto operator()() const -> void
123  {
124  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
125 
126 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
127  this->printDebug();
128 #endif
129  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one
130  // iteration.
131  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
132  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
133  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
134  Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_srcPitchBytes));
135 
136  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
137  {
138  meta::ndLoopIncIdx(
139  extentWithoutInnermost,
140  [&](Vec<DimMin1, ExtentSize> const& idx)
141  {
142  std::memcpy(
143  reinterpret_cast<void*>(
144  this->m_dstMemNative
145  + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost)
146  .foldrAll(std::plus<DstSize>())),
147  reinterpret_cast<void const*>(
148  this->m_srcMemNative
149  + (castVec<SrcSize>(idx) * srcPitchBytesWithoutOutmost)
150  .foldrAll(std::plus<SrcSize>())),
151  static_cast<std::size_t>(this->m_extentWidthBytes));
152  });
153  }
154  }
155  };
