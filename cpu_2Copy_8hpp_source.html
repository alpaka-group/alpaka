31  //! The CPU device memory copy task base.
32  //!
33  //! Copies from CPU memory into CPU memory.
34  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
35  struct TaskCopyCpuBase
36  {
37  using ExtentSize = Idx<TExtent>;
38  using DstSize = Idx<TViewDst>;
39  using SrcSize = Idx<TViewSrc>;
40  using Elem = alpaka::Elem<TViewSrc>;
41 
42  static_assert(!std::is_const_v<TViewDst>, "The destination view can not be const!");
43 
44  static_assert(
45  Dim<TViewDst>::value == Dim<TViewSrc>::value,
46  "The source and the destination view are required to have the same dimensionality!");
47  static_assert(
48  Dim<TViewDst>::value == Dim<TExtent>::value,
49  "The views and the extent are required to have the same dimensionality!");
50  static_assert(
51  Dim<TViewDst>::value == TDim::value,
52  "The destination view and the input TDim are required to have the same dimensionality!");
53 
54  static_assert(
55  meta::IsIntegralSuperset<DstSize, ExtentSize>::value,
56  "The destination view and the extent are required to have compatible idx type!");
57  static_assert(
58  meta::IsIntegralSuperset<SrcSize, ExtentSize>::value,
59  "The source view and the extent are required to have compatible idx type!");
60 
61  static_assert(
62  std::is_same_v<alpaka::Elem<TViewDst>, std::remove_const_t<alpaka::Elem<TViewSrc>>>,
63  "The source and the destination view are required to have the same element type!");
64 
65  TaskCopyCpuBase(TViewDst& viewDst, TViewSrc const& viewSrc, TExtent const& extent)
66  : m_extent(getExtentVec(extent))
67  , m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem)))
68 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
69  , m_dstExtent(getExtentVec(viewDst))
70  , m_srcExtent(getExtentVec(viewSrc))
71 #endif
72  , m_dstPitchBytes(getPitchBytesVec(viewDst))
73  , m_srcPitchBytes(getPitchBytesVec(viewSrc))
74  , m_dstMemNative(reinterpret_cast<std::uint8_t*>(getPtrNative(viewDst)))
75  , m_srcMemNative(reinterpret_cast<std::uint8_t const*>(getPtrNative(viewSrc)))
76  {
77  if constexpr(TDim::value > 0)
78  {
79  ALPAKA_ASSERT((castVec<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
80  ALPAKA_ASSERT((castVec<SrcSize>(m_extent) <= m_srcExtent).foldrAll(std::logical_or<bool>()));
81  ALPAKA_ASSERT(static_cast<DstSize>(m_extentWidthBytes) <= m_dstPitchBytes[TDim::value - 1u]);
82  ALPAKA_ASSERT(static_cast<SrcSize>(m_extentWidthBytes) <= m_srcPitchBytes[TDim::value - 1u]);
83  }
84  }
85 
86 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
87  ALPAKA_FN_HOST auto printDebug() const -> void
88  {
89  std::cout << __func__ << " e: " << m_extent << " ewb: " << this->m_extentWidthBytes
90  << " de: " << m_dstExtent << " dptr: " << reinterpret_cast<void*>(m_dstMemNative)
91  << " dpitchb: " << m_dstPitchBytes << " se: " << m_srcExtent
92  << " sptr: " << reinterpret_cast<void const*>(m_srcMemNative)
93  << " spitchb: " << m_srcPitchBytes << std::endl;
94  }
95 #endif
96 
97  Vec<TDim, ExtentSize> const m_extent;
98  ExtentSize const m_extentWidthBytes;
99 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
100  Vec<TDim, DstSize> const m_dstExtent;
101  Vec<TDim, SrcSize> const m_srcExtent;
102 #endif
103  Vec<TDim, DstSize> const m_dstPitchBytes;
104  Vec<TDim, SrcSize> const m_srcPitchBytes;
105 
106  std::uint8_t* const m_dstMemNative;
107  std::uint8_t const* const m_srcMemNative;
108  };

110  //! The CPU device ND memory copy task.
111  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
112  struct TaskCopyCpu : public TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>
113  {
114  using DimMin1 = DimInt<TDim::value - 1u>;
115  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::ExtentSize;
116  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::DstSize;
117  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::SrcSize;
118 
119  using TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::TaskCopyCpuBase;
120 
121  ALPAKA_FN_HOST auto operator()() const -> void
122  {
123  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
124 
125 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
126  this->printDebug();
127 #endif
128  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one
129  // iteration.
130  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
131  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
132  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
133  Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_srcPitchBytes));
134 
135  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
136  {
137  meta::ndLoopIncIdx(
138  extentWithoutInnermost,
139  [&](Vec<DimMin1, ExtentSize> const& idx)
140  {
141  std::memcpy(
142  reinterpret_cast<void*>(
143  this->m_dstMemNative
144  + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost)
145  .foldrAll(std::plus<DstSize>())),
146  reinterpret_cast<void const*>(
147  this->m_srcMemNative
148  + (castVec<SrcSize>(idx) * srcPitchBytesWithoutOutmost)
149  .foldrAll(std::plus<SrcSize>())),
150  static_cast<std::size_t>(this->m_extentWidthBytes));
151  });
152  }
153  }
154  };
