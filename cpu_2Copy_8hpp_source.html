26  //! The CPU device memory copy task base.
27  //!
28  //! Copies from CPU memory into CPU memory.
29  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
30  struct TaskCopyCpuBase
31  {
32  using ExtentSize = Idx<TExtent>;
33  using DstSize = Idx<TViewDst>;
34  using SrcSize = Idx<TViewSrc>;
35  using Elem = alpaka::Elem<TViewSrc>;
36 
37  template<typename TViewFwd>
38  TaskCopyCpuBase(TViewFwd&& viewDst, TViewSrc const& viewSrc, TExtent const& extent)
39  : m_extent(getExtentVec(extent))
40  , m_extentWidthBytes(m_extent[TDim::value - 1u] * static_cast<ExtentSize>(sizeof(Elem)))
41 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
42  , m_dstExtent(getExtentVec(viewDst))
43  , m_srcExtent(getExtentVec(viewSrc))
44 #endif
45  , m_dstPitchBytes(getPitchBytesVec(viewDst))
46  , m_srcPitchBytes(getPitchBytesVec(viewSrc))
47  , m_dstMemNative(reinterpret_cast<std::uint8_t*>(getPtrNative(viewDst)))
48  , m_srcMemNative(reinterpret_cast<std::uint8_t const*>(getPtrNative(viewSrc)))
49  {
50  if constexpr(TDim::value > 0)
51  {
52  ALPAKA_ASSERT((castVec<DstSize>(m_extent) <= m_dstExtent).foldrAll(std::logical_or<bool>()));
53  ALPAKA_ASSERT((castVec<SrcSize>(m_extent) <= m_srcExtent).foldrAll(std::logical_or<bool>()));
54  ALPAKA_ASSERT(static_cast<DstSize>(m_extentWidthBytes) <= m_dstPitchBytes[TDim::value - 1u]);
55  ALPAKA_ASSERT(static_cast<SrcSize>(m_extentWidthBytes) <= m_srcPitchBytes[TDim::value - 1u]);
56  }
57  }
58 
59 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
60  ALPAKA_FN_HOST auto printDebug() const -> void
61  {
62  std::cout << __func__ << " e: " << m_extent << " ewb: " << this->m_extentWidthBytes
63  << " de: " << m_dstExtent << " dptr: " << reinterpret_cast<void*>(m_dstMemNative)
64  << " dpitchb: " << m_dstPitchBytes << " se: " << m_srcExtent
65  << " sptr: " << reinterpret_cast<void const*>(m_srcMemNative)
66  << " spitchb: " << m_srcPitchBytes << std::endl;
67  }
68 #endif
69 
70  Vec<TDim, ExtentSize> const m_extent;
71  ExtentSize const m_extentWidthBytes;
72 #if(!defined(NDEBUG)) || (ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL)
73  Vec<TDim, DstSize> const m_dstExtent;
74  Vec<TDim, SrcSize> const m_srcExtent;
75 #endif
76  Vec<TDim, DstSize> const m_dstPitchBytes;
77  Vec<TDim, SrcSize> const m_srcPitchBytes;
78 
79  std::uint8_t* const m_dstMemNative;
80  std::uint8_t const* const m_srcMemNative;
81  };

83  //! The CPU device ND memory copy task.
84  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
85  struct TaskCopyCpu : public TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>
86  {
87  using DimMin1 = DimInt<TDim::value - 1u>;
88  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::ExtentSize;
89  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::DstSize;
90  using typename TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::SrcSize;
91 
92  using TaskCopyCpuBase<TDim, TViewDst, TViewSrc, TExtent>::TaskCopyCpuBase;
93 
94  ALPAKA_FN_HOST auto operator()() const -> void
95  {
96  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
97 
98 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
99  this->printDebug();
100 #endif
101  // [z, y, x] -> [z, y] because all elements with the innermost x dimension are handled within one
102  // iteration.
103  Vec<DimMin1, ExtentSize> const extentWithoutInnermost(subVecBegin<DimMin1>(this->m_extent));
104  // [z, y, x] -> [y, x] because the z pitch (the full size of the buffer) is not required.
105  Vec<DimMin1, DstSize> const dstPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_dstPitchBytes));
106  Vec<DimMin1, SrcSize> const srcPitchBytesWithoutOutmost(subVecEnd<DimMin1>(this->m_srcPitchBytes));
107 
108  if(static_cast<std::size_t>(this->m_extent.prod()) != 0u)
109  {
110  meta::ndLoopIncIdx(
111  extentWithoutInnermost,
112  [&](Vec<DimMin1, ExtentSize> const& idx)
113  {
114  std::memcpy(
115  reinterpret_cast<void*>(
116  this->m_dstMemNative
117  + (castVec<DstSize>(idx) * dstPitchBytesWithoutOutmost)
118  .foldrAll(std::plus<DstSize>())),
119  reinterpret_cast<void const*>(
120  this->m_srcMemNative
121  + (castVec<SrcSize>(idx) * srcPitchBytesWithoutOutmost)
122  .foldrAll(std::plus<SrcSize>())),
123  static_cast<std::size_t>(this->m_extentWidthBytes));
124  });
125  }
126  }
127  };
