50  std::size_t,
51  std::thread, // The concurrent execution type.

56  // true.
57  false>; // If the threads should yield.

59  public:

61  : m_dev(std::move(dev))

63  {
64  }
65  QueueGenericThreadsNonBlockingImpl(QueueGenericThreadsNonBlockingImpl<TDev> const&) = delete;

71  ~QueueGenericThreadsNonBlockingImpl() override
72  {
73  m_dev.registerCleanup(
74  [pool = std::weak_ptr<ThreadPool>(m_workerThread)]() noexcept
75  {
