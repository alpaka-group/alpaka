82  //! Create and/or return staticlly mapped device pointer of host address.
83  template<typename TElem, typename TExtent>
84  ALPAKA_FN_HOST auto mapStatic(TElem* pHost, TExtent const& extent) -> TElem*
85  {
86  makeCurrent();
87  void* pDev = acc_deviceptr(pHost);
88  if(!pDev)
89  {
90 # pragma acc enter data create(pHost [0:extent.prod()])
91  pDev = acc_deviceptr(pHost);
92  assert(pDev != nullptr);
93  }
94  return reinterpret_cast<TElem*>(pDev);
95  }
96

98  acc_device_t m_deviceType;
99  int m_iDevice;
100  std::uint32_t* m_gridsLock = nullptr;
101  };
