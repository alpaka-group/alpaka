68  //-----------------------------------------------------------------------------
69  ALPAKA_FN_HOST auto getAllExistingQueues() const -> std::vector<std::shared_ptr<cpu::ICpuQueue>>
70  {
71  std::vector<std::shared_ptr<cpu::ICpuQueue>> vspQueues;
72 
73  std::lock_guard<std::mutex> lk(m_Mutex);
74  vspQueues.reserve(m_queues.size());
75 
76  for(auto it = m_queues.begin(); it != m_queues.end();)
77  {
78  auto spQueue(it->lock());
79  if(spQueue)
80  {
81  vspQueues.emplace_back(std::move(spQueue));
82  ++it;
83  }
84  else
85  {
86  it = m_queues.erase(it);
87  }
88  }
89  return vspQueues;
90  }

95  ALPAKA_FN_HOST auto registerQueue(std::shared_ptr<cpu::ICpuQueue> spQueue) const -> void
96  {
97  std::lock_guard<std::mutex> lk(m_Mutex);
98 
99  // Register this queue on the device.
100  m_queues.push_back(spQueue);
101  }
