46  //! The CPU device implementation.
47  class DevCpuImpl
48  {
49  public:
50  ALPAKA_FN_HOST auto getAllExistingQueues() const -> std::vector<std::shared_ptr<cpu::ICpuQueue>>
51  {
52  std::vector<std::shared_ptr<cpu::ICpuQueue>> vspQueues;
53 
54  std::lock_guard<std::mutex> lk(m_Mutex);
55  vspQueues.reserve(std::size(m_queues));
56 
57  for(auto it = std::begin(m_queues); it != std::end(m_queues);)
58  {
59  auto spQueue(it->lock());
60  if(spQueue)
61  {
62  vspQueues.emplace_back(std::move(spQueue));
63  ++it;
64  }
65  else
66  {
67  it = m_queues.erase(it);
68  }
69  }
70  return vspQueues;
71  }
72 
73  //! Registers the given queue on this device.
74  //! NOTE: Every queue has to be registered for correct functionality of device wait operations!
75  ALPAKA_FN_HOST auto registerQueue(std::shared_ptr<cpu::ICpuQueue> spQueue) const -> void
76  {
77  std::lock_guard<std::mutex> lk(m_Mutex);
78 
79  // Register this queue on the device.
80  m_queues.push_back(spQueue);
81  }
82 
83  private:
84  std::mutex mutable m_Mutex;
85  std::vector<std::weak_ptr<cpu::ICpuQueue>> mutable m_queues;
86  };
