199  //! The standard library max trait specialization.
200  template<typename Tx, typename Ty>
201  struct Max<MaxStdLib, Tx, Ty, std::enable_if_t<std::is_arithmetic_v<Tx> && std::is_arithmetic_v<Ty>>>
202  {
203  ALPAKA_FN_HOST auto operator()(MaxStdLib const& /* max_ctx */, Tx const& x, Ty const& y)
204  {
205  using std::fmax;
206  using std::max;
207 
208  if constexpr(std::is_integral_v<Tx> && std::is_integral_v<Ty>)
209  return max(x, y);
210  else if constexpr(
211  is_decayed_v<
212  Tx,
213  float> || is_decayed_v<Ty, float> || is_decayed_v<Tx, double> || is_decayed_v<Ty, double>)
214  return fmax(x, y);
215  else
216  static_assert(!sizeof(Tx), "Unsupported data type");
217 
218  ALPAKA_UNREACHABLE(std::common_type_t<Tx, Ty>{});
219  }
220  };
