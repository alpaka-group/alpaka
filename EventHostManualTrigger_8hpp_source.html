46  //! Event that can be enqueued into a queue and can be triggered by the Host.
47  template<class TDev = DevCpu>
48  class EventHostManualTriggerCpuImpl
49  {
50  public:
51  //! Constructor.
52  ALPAKA_FN_HOST EventHostManualTriggerCpuImpl(TDev const& dev) noexcept
53  : m_dev(dev)
54  , m_mutex()
55  , m_enqueueCount(0u)
56  , m_bIsReady(true)
57  {
58  }
59  EventHostManualTriggerCpuImpl(EventHostManualTriggerCpuImpl const& other) = delete;
60  auto operator=(EventHostManualTriggerCpuImpl const&) -> EventHostManualTriggerCpuImpl& = delete;
61 
62  void trigger()
63  {
64  {
65  std::unique_lock<std::mutex> lock(m_mutex);
66  m_bIsReady = true;
67  }
68  m_conditionVariable.notify_one();
69  }
70 
71  public:
72  TDev const m_dev; //!< The device this event is bound to.
73 
74  mutable std::mutex m_mutex; //!< The mutex used to synchronize access to the event.
75 
76  mutable std::condition_variable
77  m_conditionVariable; //!< The condition signaling the event completion.
78  std::size_t m_enqueueCount; //!< The number of times this event has been enqueued.
79 
80  bool m_bIsReady; //!< If the event is not waiting within a queue (not enqueued or already
81  //!< completed).
82  };

194  template<typename TDev>
195  struct Enqueue<QueueGenericThreadsNonBlocking<TDev>, test::EventHostManualTriggerCpu<TDev>>
196  {
197  //
198  ALPAKA_FN_HOST static auto enqueue(
199 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
200  QueueGenericThreadsNonBlocking<TDev>& queue,
201 #else
202  QueueGenericThreadsNonBlocking<TDev>&,
203 #endif
204  test::EventHostManualTriggerCpu<TDev>& event) -> void
205  {
206  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
207 
208  // Copy the shared pointer to ensure that the event implementation is alive as long as it is enqueued.
209  auto spEventImpl = event.m_spEventImpl;
210 
211  // Setting the event state and enqueuing it has to be atomic.
212  std::lock_guard<std::mutex> lk(spEventImpl->m_mutex);
213 
214  // The event should not yet be enqueued.
215  ALPAKA_ASSERT(spEventImpl->m_bIsReady);
216 
217  // Set its state to enqueued.
218  spEventImpl->m_bIsReady = false;
219 
220  // Increment the enqueue counter. This is used to skip waits for events that had already been finished
221  // and re-enqueued which would lead to deadlocks.
222  ++spEventImpl->m_enqueueCount;
223 
224  // Workaround: Clang can not support this when natively compiling device code. See
225  // ConcurrentExecPool.hpp.
226 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
227  auto const enqueueCount = spEventImpl->m_enqueueCount;
228 
229  // Enqueue a task that only resets the events flag if it is completed.
230  queue.m_spQueueImpl->m_workerThread.enqueueTask(
231  [spEventImpl, enqueueCount]()
232  {
233  std::unique_lock<std::mutex> lk2(spEventImpl->m_mutex);
234  spEventImpl->m_conditionVariable.wait(
235  lk2,
236  [spEventImpl, enqueueCount]
237  { return (enqueueCount != spEventImpl->m_enqueueCount) || spEventImpl->m_bIsReady; });
238  });
239 #endif
240  }
241  };
