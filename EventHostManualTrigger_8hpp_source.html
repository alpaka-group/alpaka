151  template<typename TDev>
152  struct Enqueue<QueueGenericThreadsNonBlocking<TDev>, test::EventHostManualTriggerCpu<TDev>>
153  {
154  //
155  ALPAKA_FN_HOST static auto enqueue(
156  QueueGenericThreadsNonBlocking<TDev>& queue,
157  test::EventHostManualTriggerCpu<TDev>& event) -> void
158  {
159  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
160 
161  // Copy the shared pointer to ensure that the event implementation is alive as long as it is enqueued.
162  auto spEventImpl = event.m_spEventImpl;
163 
164  // Setting the event state and enqueuing it has to be atomic.
165  std::lock_guard<std::mutex> lk(spEventImpl->m_mutex);
166 
167  // The event should not yet be enqueued.
168  ALPAKA_ASSERT(spEventImpl->m_bIsReady);
169 
170  // Set its state to enqueued.
171  spEventImpl->m_bIsReady = false;
172 
173  // Increment the enqueue counter. This is used to skip waits for events that had already been finished
174  // and re-enqueued which would lead to deadlocks.
175  ++spEventImpl->m_enqueueCount;
176 
177  auto const enqueueCount = spEventImpl->m_enqueueCount;
178 
179  // Enqueue a task that only resets the events flag if it is completed.
180  queue.m_spQueueImpl->m_workerThread.submit(
181  [spEventImpl, enqueueCount]() mutable
182  {
183  std::unique_lock<std::mutex> lk2(spEventImpl->m_mutex);
184  spEventImpl->m_conditionVariable.wait(
185  lk2,
186  [spEventImpl, enqueueCount]
187  { return (enqueueCount != spEventImpl->m_enqueueCount) || spEventImpl->m_bIsReady; });
188  });
189  }
190  };

353  // TODO: Replace with cuStreamWaitValue32 once support for CUDA < 12 is dropped.
354  inline auto streamWaitValue(CUstream stream, CUdeviceptr addr, cuuint32_t value, unsigned int flags)
355  -> CUresult
356  {
357  // NVIDIA introduced a new stream memory ops API with CUDA 11.7 (called v2). The corresponding CUDA
358  // functions were suffixed with `_v2`. With CUDA 12.0 v1 of the API was removed and the `_v2` removed
359  // from the new functions. So CUDA <= 11.6 and CUDA >= 12.0 share the same function signature but
360  // internally do different things.
361 # if(CUDA_VERSION < 11070) || (CUDA_VERSION >= 12000)
362  return cuStreamWaitValue32(stream, addr, value, flags);
363 # else
364  return cuStreamWaitValue32_v2(stream, addr, value, flags);
365 # endif
366  }
