44  //! The CPU fibers accelerator execution task.
45  template<typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
46  class TaskKernelCpuFibers final : public WorkDivMembers<TDim, TIdx>
47  {
48  private:
49  //! The type given to the ConcurrentExecPool for yielding the current fiber.
50  struct FiberPoolYield
51  {
52  //! Yields the current fiber.
53  ALPAKA_FN_HOST static auto yield() -> void
54  {
55  boost::this_fiber::yield();
56  }
57  };
58  // Yielding is not faster for fibers. Therefore we use condition variables.
59  // It is better to wake them up when the conditions are fulfilled because this does not cost as much as for
60  // real threads.
61  using FiberPool = alpaka::core::detail::ConcurrentExecPool<
62  TIdx,
63  boost::fibers::fiber, // The concurrent execution type.
64  boost::fibers::promise, // The promise type.
65  FiberPoolYield, // The type yielding the current concurrent execution.
66  boost::fibers::mutex, // The mutex type to use. Only required if TisYielding is true.
67  boost::fibers::condition_variable, // The condition variable type to use. Only required if TisYielding is
68  // true.
69  false>; // If the threads should yield.
70 
71  public:
72  template<typename TWorkDiv>
73  ALPAKA_FN_HOST TaskKernelCpuFibers(TWorkDiv&& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
74  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
75  , m_kernelFnObj(kernelFnObj)
76  , m_args(std::forward<TArgs>(args)...)
77  {
78  static_assert(
79  Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
80  "The work division and the execution task have to be of the same dimensionality!");
81  }

83  //! Executes the kernel function object.
84  ALPAKA_FN_HOST auto operator()() const -> void
85  {
86  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
87 
88  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(*this);
89  auto const blockThreadExtent = getWorkDiv<Block, Threads>(*this);
90  auto const threadElemExtent = getWorkDiv<Thread, Elems>(*this);
91 
92  // Get the size of the block shared dynamic memory.
93  auto const blockSharedMemDynSizeBytes = meta::apply(
94  [&](ALPAKA_DECAY_T(TArgs) const&... args)
95  {
96  return getBlockSharedMemDynSizeBytes<AccCpuFibers<TDim, TIdx>>(
97  m_kernelFnObj,
98  blockThreadExtent,
99  threadElemExtent,
100  args...);
101  },
102  m_args);
103 
104 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
105  std::cout << __func__ << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B"
106  << std::endl;
107 # endif
108  AccCpuFibers<TDim, TIdx> acc(
109  *static_cast<WorkDivMembers<TDim, TIdx> const*>(this),
110  blockSharedMemDynSizeBytes);
111 
112 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
113  std::cout << __func__
114  << " Fiber stack idx: " << boost::fibers::fixedsize_stack::traits_type::default_size() << " B"
115  << std::endl;
116 # endif
117 
118  auto const blockThreadCount(blockThreadExtent.prod());
119  FiberPool fiberPool(blockThreadCount);
120 
121  auto const boundGridBlockExecHost = meta::apply(
122  [this, &acc, &blockThreadExtent, &fiberPool](ALPAKA_DECAY_T(TArgs) const&... args)
123  {
124  // Bind the kernel and its arguments to the grid block function.
125  return std::bind(
126  &TaskKernelCpuFibers::gridBlockExecHost,
127  std::ref(acc),
128  std::placeholders::_1,
129  std::ref(blockThreadExtent),
130  std::ref(fiberPool),
131  std::ref(m_kernelFnObj),
132  std::ref(args)...);
133  },
134  m_args);
135 
136  // Execute the blocks serially.
137  meta::ndLoopIncIdx(gridBlockExtent, boundGridBlockExecHost);
138  }
139 
140  private:
141  //! The function executed for each grid block.
142  ALPAKA_FN_HOST static auto gridBlockExecHost(
143  AccCpuFibers<TDim, TIdx>& acc,
144  Vec<TDim, TIdx> const& gridBlockIdx,
145  Vec<TDim, TIdx> const& blockThreadExtent,
146  FiberPool& fiberPool,
147  TKernelFnObj const& kernelFnObj,
148  std::decay_t<TArgs> const&... args) -> void
149  {
150  // The futures of the threads in the current block.
151  std::vector<boost::fibers::future<void>> futuresInBlock;
152 
153  // Set the index of the current block
154  acc.m_gridBlockIdx = gridBlockIdx;
155 
156  // Bind the kernel and its arguments to the host block thread execution function.
157  auto boundBlockThreadExecHost(std::bind(
158  &TaskKernelCpuFibers::blockThreadExecHost,
159  std::ref(acc),
160  std::ref(futuresInBlock),
161  std::placeholders::_1,
162  std::ref(fiberPool),
163  std::ref(kernelFnObj),
164  std::ref(args)...));
165  // Execute the block threads in parallel.
166  meta::ndLoopIncIdx(blockThreadExtent, boundBlockThreadExecHost);
167 
168  // Wait for the completion of the block thread kernels.
169  std::for_each(
170  futuresInBlock.begin(),
171  futuresInBlock.end(),
172  [](boost::fibers::future<void>& t) { t.wait(); });
173  // Clean up.
174  futuresInBlock.clear();
175 
176  acc.m_fibersToIndices.clear();
177 
178  // After a block has been processed, the shared memory has to be deleted.
179  freeSharedVars(acc);
180  }
181  //! The function executed for each block thread.
182  ALPAKA_FN_HOST static auto blockThreadExecHost(
183  AccCpuFibers<TDim, TIdx>& acc,
184 # if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
185  std::vector<boost::fibers::future<void>>& futuresInBlock,
186  Vec<TDim, TIdx> const& blockThreadIdx,
187  FiberPool& fiberPool,
188 # else
189  std::vector<boost::fibers::future<void>>&,
190  Vec<TDim, TIdx> const& blockThreadIdx,
191  FiberPool&,
192 # endif
193  TKernelFnObj const& kernelFnObj,
194  std::decay_t<TArgs> const&... args) -> void
195  {
196  // Bind the arguments to the accelerator block thread execution function.
197  // The blockThreadIdx is required to be copied in because the variable will get changed for the next
198  // iteration/thread.
199  auto boundBlockThreadExecAcc([&, blockThreadIdx]()
200  { blockThreadFiberFn(acc, blockThreadIdx, kernelFnObj, args...); });
201  // Add the bound function to the block thread pool.
202 // Workaround: Clang can not support this when natively compiling device code. See ConcurrentExecPool.hpp.
203 # if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
204  futuresInBlock.emplace_back(fiberPool.enqueueTask(boundBlockThreadExecAcc));
205 # else
206  (void) boundBlockThreadExecAcc;
207 # endif
208  }
209  //! The fiber entry point.
210  ALPAKA_FN_HOST static auto blockThreadFiberFn(
211  AccCpuFibers<TDim, TIdx>& acc,
212  Vec<TDim, TIdx> const& blockThreadIdx,
213  TKernelFnObj const& kernelFnObj,
214  std::decay_t<TArgs> const&... args) -> void
215  {
216  // We have to store the fiber data before the kernel is calling any of the methods of this class depending
217  // on them.
218  auto const fiberId(boost::this_fiber::get_id());
219 
220  // Set the master thread id.
221  if(blockThreadIdx.sum() == 0)
222  {
223  acc.m_masterFiberId = fiberId;
224  }
225 
226  // Save the fiber id, and index.
227  acc.m_fibersToIndices.emplace(fiberId, blockThreadIdx);
228 
229  // Sync all threads so that the maps with thread id's are complete and not changed after here.
230  syncBlockThreads(acc);
231 
232  // Execute the kernel itself.
233  kernelFnObj(const_cast<AccCpuFibers<TDim, TIdx> const&>(acc), args...);
234 
235  // We have to sync all fibers here because if a fiber would finish before all fibers have been started, the
236  // new fiber could get a recycled (then duplicate) fiber id!
237  syncBlockThreads(acc);
238  }
239 
240  TKernelFnObj m_kernelFnObj;
241  std::tuple<std::decay_t<TArgs>...> m_args;
242  };
