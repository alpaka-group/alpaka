43  //! The CPU fibers accelerator execution task.
44  template<typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
45  class TaskKernelCpuFibers final : public WorkDivMembers<TDim, TIdx>
46  {
47  private:
48  //! The type given to the ConcurrentExecPool for yielding the current fiber.
49  struct FiberPoolYield
50  {
51  //! Yields the current fiber.
52  ALPAKA_FN_HOST static auto yield() -> void
53  {
54  boost::this_fiber::yield();
55  }
56  };
57  // Yielding is not faster for fibers. Therefore we use condition variables.
58  // It is better to wake them up when the conditions are fulfilled because this does not cost as much as for
59  // real threads.
60  using FiberPool = alpaka::core::detail::ConcurrentExecPool<
61  TIdx,
62  boost::fibers::fiber, // The concurrent execution type.
63  boost::fibers::promise, // The promise type.
64  FiberPoolYield, // The type yielding the current concurrent execution.
65  boost::fibers::mutex, // The mutex type to use. Only required if TisYielding is true.
66  boost::fibers::condition_variable, // The condition variable type to use. Only required if TisYielding is
67  // true.
68  false>; // If the threads should yield.
69 
70  public:
71  template<typename TWorkDiv>
72  ALPAKA_FN_HOST TaskKernelCpuFibers(TWorkDiv&& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
73  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
74  , m_kernelFnObj(kernelFnObj)
75  , m_args(std::forward<TArgs>(args)...)
76  {
77  static_assert(
78  Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
79  "The work division and the execution task have to be of the same dimensionality!");
80  }

82  //! Executes the kernel function object.
83  ALPAKA_FN_HOST auto operator()() const -> void
84  {
85  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
86 
87  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(*this);
88  auto const blockThreadExtent = getWorkDiv<Block, Threads>(*this);
89  auto const threadElemExtent = getWorkDiv<Thread, Elems>(*this);
90 
91  // Get the size of the block shared dynamic memory.
92  auto const blockSharedMemDynSizeBytes = std::apply(
93  [&](ALPAKA_DECAY_T(TArgs) const&... args)
94  {
95  return getBlockSharedMemDynSizeBytes<AccCpuFibers<TDim, TIdx>>(
96  m_kernelFnObj,
97  blockThreadExtent,
98  threadElemExtent,
99  args...);
100  },
101  m_args);
102 
103 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
104  std::cout << __func__ << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B"
105  << std::endl;
106 # endif
107  AccCpuFibers<TDim, TIdx> acc(
108  *static_cast<WorkDivMembers<TDim, TIdx> const*>(this),
109  blockSharedMemDynSizeBytes);
110 
111 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
112  std::cout << __func__
113  << " Fiber stack idx: " << boost::fibers::fixedsize_stack::traits_type::default_size() << " B"
114  << std::endl;
115 # endif
116 
117  auto const blockThreadCount(blockThreadExtent.prod());
118  FiberPool fiberPool(blockThreadCount);
119 
120  auto const boundGridBlockExecHost = std::apply(
121  [this, &acc, &blockThreadExtent, &fiberPool](ALPAKA_DECAY_T(TArgs) const&... args)
122  {
123  // Bind the kernel and its arguments to the grid block function.
124  return std::bind(
125  &TaskKernelCpuFibers::gridBlockExecHost,
126  std::ref(acc),
127  std::placeholders::_1,
128  std::ref(blockThreadExtent),
129  std::ref(fiberPool),
130  std::ref(m_kernelFnObj),
131  std::ref(args)...);
132  },
133  m_args);
134 
135  // Execute the blocks serially.
136  meta::ndLoopIncIdx(gridBlockExtent, boundGridBlockExecHost);
137  }
138 
139  private:
140  //! The function executed for each grid block.
141  ALPAKA_FN_HOST static auto gridBlockExecHost(
142  AccCpuFibers<TDim, TIdx>& acc,
143  Vec<TDim, TIdx> const& gridBlockIdx,
144  Vec<TDim, TIdx> const& blockThreadExtent,
145  FiberPool& fiberPool,
146  TKernelFnObj const& kernelFnObj,
147  std::decay_t<TArgs> const&... args) -> void
148  {
149  // The futures of the threads in the current block.
150  std::vector<boost::fibers::future<void>> futuresInBlock;
151 
152  // Set the index of the current block
153  acc.m_gridBlockIdx = gridBlockIdx;
154 
155  // Bind the kernel and its arguments to the host block thread execution function.
156  auto boundBlockThreadExecHost(std::bind(
157  &TaskKernelCpuFibers::blockThreadExecHost,
158  std::ref(acc),
159  std::ref(futuresInBlock),
160  std::placeholders::_1,
161  std::ref(fiberPool),
162  std::ref(kernelFnObj),
163  std::ref(args)...));
164  // Execute the block threads in parallel.
165  meta::ndLoopIncIdx(blockThreadExtent, boundBlockThreadExecHost);
166 
167  // Wait for the completion of the block thread kernels.
168  for(auto& t : futuresInBlock)
169  t.wait();
170 
171  // Clean up.
172  futuresInBlock.clear();
173 
174  acc.m_fibersToIndices.clear();
175 
176  // After a block has been processed, the shared memory has to be deleted.
177  freeSharedVars(acc);
178  }
179  //! The function executed for each block thread.
180  ALPAKA_FN_HOST static auto blockThreadExecHost(
181  AccCpuFibers<TDim, TIdx>& acc,
182 # if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
183  std::vector<boost::fibers::future<void>>& futuresInBlock,
184  Vec<TDim, TIdx> const& blockThreadIdx,
185  FiberPool& fiberPool,
186 # else
187  std::vector<boost::fibers::future<void>>&,
188  Vec<TDim, TIdx> const& blockThreadIdx,
189  FiberPool&,
190 # endif
191  TKernelFnObj const& kernelFnObj,
192  std::decay_t<TArgs> const&... args) -> void
193  {
194  // Bind the arguments to the accelerator block thread execution function.
195  // The blockThreadIdx is required to be copied in because the variable will get changed for the next
196  // iteration/thread.
197  auto boundBlockThreadExecAcc([&, blockThreadIdx]()
198  { blockThreadFiberFn(acc, blockThreadIdx, kernelFnObj, args...); });
199  // Add the bound function to the block thread pool.
200 // Workaround: Clang can not support this when natively compiling device code. See ConcurrentExecPool.hpp.
201 # if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
202  futuresInBlock.emplace_back(fiberPool.enqueueTask(boundBlockThreadExecAcc));
203 # else
204  (void) boundBlockThreadExecAcc;
205 # endif
206  }
207  //! The fiber entry point.
208  ALPAKA_FN_HOST static auto blockThreadFiberFn(
209  AccCpuFibers<TDim, TIdx>& acc,
210  Vec<TDim, TIdx> const& blockThreadIdx,
211  TKernelFnObj const& kernelFnObj,
212  std::decay_t<TArgs> const&... args) -> void
213  {
214  // We have to store the fiber data before the kernel is calling any of the methods of this class depending
215  // on them.
216  auto const fiberId(boost::this_fiber::get_id());
217 
218  // Set the master thread id.
219  if(blockThreadIdx.sum() == 0)
220  {
221  acc.m_masterFiberId = fiberId;
222  }
223 
224  // Save the fiber id, and index.
225  acc.m_fibersToIndices.emplace(fiberId, blockThreadIdx);
226 
227  // Sync all threads so that the maps with thread id's are complete and not changed after here.
228  syncBlockThreads(acc);
229 
230  // Execute the kernel itself.
231  kernelFnObj(const_cast<AccCpuFibers<TDim, TIdx> const&>(acc), args...);
232 
233  // We have to sync all fibers here because if a fiber would finish before all fibers have been started, the
234  // new fiber could get a recycled (then duplicate) fiber id!
235  syncBlockThreads(acc);
236  }
237 
238  TKernelFnObj m_kernelFnObj;
239  std::tuple<std::decay_t<TArgs>...> m_args;
240  };
