53  //! The CPU fibers accelerator.
54  //!
55  //! This accelerator allows parallel kernel execution on a CPU device.
56  //! It uses boost::fibers to implement the cooperative parallelism.
57  //! By using fibers the shared memory can reside in the closest memory/cache available.
58  //! Furthermore there is no false sharing between neighboring threads as it is the case in real multi-threading.
59  template<
60  typename TDim,
61  typename TIdx>
62  class AccCpuFibers final :
63  public WorkDivMembers<TDim, TIdx>,
64  public gb::IdxGbRef<TDim, TIdx>,
65  public bt::IdxBtRefFiberIdMap<TDim, TIdx>,
66  public AtomicHierarchy<
67  AtomicStdLibLock<16>, // grid atomics
68  AtomicStdLibLock<16>, // block atomics
69  AtomicNoOp // thread atomics
70  >,
71  public math::MathStdLib,
72  public BlockSharedMemDynMember<>,
73  public BlockSharedMemStMemberMasterSync<>,
74  public BlockSyncBarrierFiber<TIdx>,
75  public IntrinsicCpu,
76  public MemFenceCpu,
77  public rand::RandStdLib,
78  public TimeStdLib,
79  public warp::WarpSingleThread,
80  public concepts::Implements<ConceptAcc, AccCpuFibers<TDim, TIdx>>
81  {
82  static_assert(
83  sizeof(TIdx) >= sizeof(int),
84  "Index type is not supported, consider using int or a larger type.");
85 
86  public:
87  // Partial specialization with the correct TDim and TIdx is not allowed.
88  template<typename TDim2, typename TIdx2, typename TKernelFnObj, typename... TArgs>
89  friend class ::alpaka::TaskKernelCpuFibers;
90 
91  private:
92  template<typename TWorkDiv>
93  ALPAKA_FN_HOST AccCpuFibers(TWorkDiv const& workDiv, std::size_t const& blockSharedMemDynSizeBytes)
94  : WorkDivMembers<TDim, TIdx>(workDiv)
95  , gb::IdxGbRef<TDim, TIdx>(m_gridBlockIdx)
96  , bt::IdxBtRefFiberIdMap<TDim, TIdx>(m_fibersToIndices)
97  , AtomicHierarchy<
98  AtomicStdLibLock<16>, // atomics between grids
99  AtomicStdLibLock<16>, // atomics between blocks
100  AtomicNoOp // atomics between threads
101  >()
102  , math::MathStdLib()
103  , BlockSharedMemDynMember<>(blockSharedMemDynSizeBytes)
104  , BlockSharedMemStMemberMasterSync<>(
105  staticMemBegin(),
106  staticMemCapacity(),
107  [this]() { syncBlockThreads(*this); },
108  [this]() noexcept { return (m_masterFiberId == boost::this_fiber::get_id()); })
109  , BlockSyncBarrierFiber<TIdx>(getWorkDiv<Block, Threads>(workDiv).prod())
110  , MemFenceCpu()
111  , rand::RandStdLib()
112  , TimeStdLib()
113  , m_gridBlockIdx(Vec<TDim, TIdx>::zeros())
114  {
115  }
116 
117  private:
118  // getIdx
119  typename bt::IdxBtRefFiberIdMap<TDim, TIdx>::
120  FiberIdToIdxMap mutable m_fibersToIndices; //!< The mapping of fibers id's to indices.
121  Vec<TDim, TIdx> mutable m_gridBlockIdx; //!< The index of the currently executed block.
122 
123  // allocBlockSharedArr
124  boost::fibers::fiber::id mutable m_masterFiberId; //!< The id of the master fiber.
125  };
