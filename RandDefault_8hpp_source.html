55  //! The GPU random number normal distribution.
56  template<typename T>
57  class UniformUint
58  {
59  static_assert(std::is_integral<T>::value, "Return type of UniformUint must be integral.");
60 
61  public:
62  UniformUint() = default;
63 
64  template<typename TEngine>
65  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
66  {
67  using BitsT = typename TEngine::result_type;
68  T ret = 0;
69  constexpr auto N = sizeof(T) / sizeof(BitsT);
70  for(unsigned int a = 0; a < N; ++a)
71  {
72  ret
73  ^= (static_cast<T>(engine())
74  << (sizeof(BitsT) * std::numeric_limits<unsigned char>::digits * a));
75  }
76  return ret;
77  }
78  };

80  //! The GPU random number uniform distribution.
81  template<typename T>
82  class UniformReal
83  {
84  static_assert(
85  std::is_floating_point<T>::value,
86  "Return type of UniformReal must be floating point.");
87 
88  using BitsT = typename detail::BitsType<T>::type;
89 
90  public:
91  UniformReal() = default;
92 
93  template<typename TEngine>
94  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
95  {
96  constexpr BitsT limit = static_cast<BitsT>(1) << std::numeric_limits<T>::digits;
97  const BitsT b = UniformUint<BitsT>()(engine);
98  const auto ret = static_cast<T>(b & (limit - 1)) / limit;
99  return ret;
100  }

103  /*! The GPU random number normal distribution.
104  *
105  * \note
106  * This type contains state and is not thread-safe: To be used
107  * per thread, not shared.
108  *
109  * \note When reproducibility is a concern, each instance of
110  * this class should be used with only on random engine
111  * instance, or two consecutive number should be generated with
112  * each engine used. This is due to the implicit caching of one
113  * Gaussian random number.
114  */
115  template<typename Acc, typename T>
116  class NormalReal
117  {
118  static_assert(
119  std::is_floating_point<T>::value,
120  "Return type of NormalReal must be floating point.");
121 
122  const Acc& m_acc;
123  T m_cache = std::numeric_limits<T>::quiet_NaN();
124 
125  public:
126  /*! \warning Retains a reference to \p acc, thus must not
127  * outlive it.
128  */
129  NormalReal(const Acc& acc) : m_acc(acc)
130  {
131  }
132 
133  NormalReal(const NormalReal& o) = delete;
134 
135  //! The move ctor clears `m_cache` of source.
136  //! \todo This is to be deleted when moving to C++17
137  NormalReal(NormalReal&& o) : m_acc(o.m_acc), m_cache(o.m_cache)
138  {
139  o.m_cache = std::numeric_limits<T>::quiet_NaN();
140  }
141 
142  template<typename TEngine>
143  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
144  {
145  constexpr T sigma = 1., mu = 0.;
146  if(math::isnan(m_acc, m_cache))
147  {
148  UniformReal<T> uni;
149 
150  T u1, u2;
151  do
152  {
153  u1 = uni(engine);
154  u2 = uni(engine);
155  } while(u1 <= std::numeric_limits<T>::epsilon());
156 
157  // compute z0 and z1
158  const T mag = sigma * math::sqrt(m_acc, static_cast<T>(-2.) * math::log(m_acc, u1));
159  constexpr T twoPi = static_cast<T>(2. * M_PI);
160  // getting two normal number out of this, store one for later
161  m_cache = mag * static_cast<T>(math::cos(m_acc, twoPi * u2)) + mu;
162 
163  return mag * static_cast<T>(math::sin(m_acc, twoPi * u2)) + mu;
164  }
165  else
166  {
167  const T ret = m_cache;
168  m_cache = std::numeric_limits<T>::quiet_NaN();
169  return ret;
170  }
171  }
172  };
