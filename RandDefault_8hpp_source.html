46  //! The GPU random number normal distribution.
47  template<typename T>
48  class UniformUint
49  {
50  static_assert(std::is_integral_v<T>, "Return type of UniformUint must be integral.");
51 
52  public:
53  UniformUint() = default;
54 
55  template<typename TEngine>
56  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
57  {
58  using BitsT = typename TEngine::result_type;
59  T ret = 0;
60  constexpr auto N = sizeof(T) / sizeof(BitsT);
61  for(unsigned int a = 0; a < N; ++a)
62  {
63  ret
64  ^= (static_cast<T>(engine())
65  << (sizeof(BitsT) * std::numeric_limits<unsigned char>::digits * a));
66  }
67  return ret;
68  }
69  };

71  //! The GPU random number uniform distribution.
72  template<typename T>
73  class UniformReal
74  {
75  static_assert(std::is_floating_point_v<T>, "Return type of UniformReal must be floating point.");
76 
77  using BitsT = typename detail::BitsType<T>::type;
78 
79  public:
80  UniformReal() = default;
81 
82  template<typename TEngine>
83  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
84  {
85  constexpr BitsT limit = static_cast<BitsT>(1) << std::numeric_limits<T>::digits;
86  const BitsT b = UniformUint<BitsT>()(engine);
87  const auto ret = static_cast<T>(b & (limit - 1)) / limit;
88  return ret;
89  }

92  /*! The GPU random number normal distribution.
93  *
94  * \note
95  * This type contains state and is not thread-safe: To be used
96  * per thread, not shared.
97  *
98  * \note When reproducibility is a concern, each instance of
99  * this class should be used with only on random engine
100  * instance, or two consecutive number should be generated with
101  * each engine used. This is due to the implicit caching of one
102  * Gaussian random number.
103  */
104  template<typename Acc, typename T>
105  class NormalReal
106  {
107  static_assert(std::is_floating_point_v<T>, "Return type of NormalReal must be floating point.");
108 
109  const Acc* m_acc;
110  T m_cache = std::numeric_limits<T>::quiet_NaN();
111 
112  public:
113  /*! \warning Retains a reference to \p acc, thus must not outlive it.
114  */
115  ALPAKA_FN_HOST_ACC constexpr NormalReal(const Acc& acc) : m_acc(&acc)
116  {
117  }
118 
119  // All copy operations (and thus also move since we don't declare those and they fall back to copy) do NOT
120  // copy m_cache. This way we can ensure that the following holds:
121  // NormalReal<Acc> a(acc), b(acc);
122  // Engine<Acc> e(acc);
123  // assert(a(e) != b(e)); // because of two engine invocations
124  // b = a;
125  // assert(a(e) != b(e)); // because of two engine invocations
126 
127  ALPAKA_FN_HOST_ACC constexpr NormalReal(const NormalReal& other) : m_acc(other.m_acc)
128  {
129  }
130 
131  ALPAKA_FN_HOST_ACC constexpr auto operator=(const NormalReal& other) -> NormalReal&
132  {
133  m_acc = other.m_acc;
134  return *this;
135  }
136 
137  template<typename TEngine>
138  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
139  {
140  constexpr T sigma = 1., mu = 0.;
141  if(math::isnan(*m_acc, m_cache))
142  {
143  UniformReal<T> uni;
144 
145  T u1, u2;
146  do
147  {
148  u1 = uni(engine);
149  u2 = uni(engine);
150  } while(u1 <= std::numeric_limits<T>::epsilon());
151 
152  // compute z0 and z1
153  const T mag = sigma * math::sqrt(*m_acc, static_cast<T>(-2.) * math::log(*m_acc, u1));
154  constexpr T twoPi = static_cast<T>(2. * M_PI);
155  // getting two normal number out of this, store one for later
156  m_cache = mag * static_cast<T>(math::cos(*m_acc, twoPi * u2)) + mu;
157 
158  return mag * static_cast<T>(math::sin(*m_acc, twoPi * u2)) + mu;
159  }
160 
161  const T ret = m_cache;
162  m_cache = std::numeric_limits<T>::quiet_NaN();
163  return ret;
164  }
165  };
