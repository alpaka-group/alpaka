41  //! The GPU random number normal distribution.
42  template<typename T>
43  class UniformUint
44  {
45  static_assert(std::is_integral_v<T>, "Return type of UniformUint must be integral.");
46 
47  public:
48  UniformUint() = default;
49 
50  template<typename TEngine>
51  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
52  {
53  using BitsT = typename TEngine::result_type;
54  T ret = 0;
55  constexpr auto N = sizeof(T) / sizeof(BitsT);
56  for(unsigned int a = 0; a < N; ++a)
57  {
58  ret
59  ^= (static_cast<T>(engine())
60  << (sizeof(BitsT) * std::numeric_limits<unsigned char>::digits * a));
61  }
62  return ret;
63  }
64  };

66  //! The GPU random number uniform distribution.
67  template<typename T>
68  class UniformReal
69  {
70  static_assert(std::is_floating_point_v<T>, "Return type of UniformReal must be floating point.");
71 
72  using BitsT = typename detail::BitsType<T>::type;
73 
74  public:
75  UniformReal() = default;
76 
77  template<typename TEngine>
78  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
79  {
80  constexpr BitsT limit = static_cast<BitsT>(1) << std::numeric_limits<T>::digits;
81  const BitsT b = UniformUint<BitsT>()(engine);
82  auto const ret = static_cast<T>(b & (limit - 1)) / limit;
83  return ret;
84  }

87  /*! The GPU random number normal distribution.
88  *
89  * \note
90  * This type contains state and is not thread-safe: To be used
91  * per thread, not shared.
92  *
93  * \note When reproducibility is a concern, each instance of
94  * this class should be used with only on random engine
95  * instance, or two consecutive number should be generated with
96  * each engine used. This is due to the implicit caching of one
97  * Gaussian random number.
98  */
99  template<typename Acc, typename T>
100  class NormalReal
101  {
102  static_assert(std::is_floating_point_v<T>, "Return type of NormalReal must be floating point.");
103 
104  Acc const* m_acc;
105  T m_cache = std::numeric_limits<T>::quiet_NaN();
106 
107  public:
108  /*! \warning Retains a reference to \p acc, thus must not outlive it.
109  */
110  ALPAKA_FN_HOST_ACC constexpr NormalReal(Acc const& acc) : m_acc(&acc)
111  {
112  }
113 
114  // All copy operations (and thus also move since we don't declare those and they fall back to copy) do NOT
115  // copy m_cache. This way we can ensure that the following holds:
116  // NormalReal<Acc> a(acc), b(acc);
117  // Engine<Acc> e(acc);
118  // assert(a(e) != b(e)); // because of two engine invocations
119  // b = a;
120  // assert(a(e) != b(e)); // because of two engine invocations
121 
122  ALPAKA_FN_HOST_ACC constexpr NormalReal(NormalReal const& other) : m_acc(other.m_acc)
123  {
124  }
125 
126  ALPAKA_FN_HOST_ACC constexpr auto operator=(NormalReal const& other) -> NormalReal&
127  {
128  m_acc = other.m_acc;
129  return *this;
130  }
131 
132  template<typename TEngine>
133  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
134  {
135  constexpr auto sigma = T{1};
136  constexpr auto mu = T{0};
137  if(math::isnan(*m_acc, m_cache))
138  {
139  UniformReal<T> uni;
140 
141  T u1, u2;
142  do
143  {
144  u1 = uni(engine);
145  u2 = uni(engine);
146  } while(u1 <= std::numeric_limits<T>::epsilon());
147 
148  // compute z0 and z1
149  const T mag = sigma * math::sqrt(*m_acc, static_cast<T>(-2.) * math::log(*m_acc, u1));
150  constexpr T twoPi = static_cast<T>(2. * math::constants::pi);
151  // getting two normal number out of this, store one for later
152  m_cache = mag * static_cast<T>(math::cos(*m_acc, twoPi * u2)) + mu;
153 
154  return mag * static_cast<T>(math::sin(*m_acc, twoPi * u2)) + mu;
155  }
156 
157  const T ret = m_cache;
158  m_cache = std::numeric_limits<T>::quiet_NaN();
159  return ret;
160  }
161  };
