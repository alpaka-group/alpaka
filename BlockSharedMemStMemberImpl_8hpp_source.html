55  template<typename T>
56  void alloc(std::uint32_t id) const
57  {
58  // Add meta data chunk in front of the user data
59  m_allocdBytes = varChunkEnd<MetaData>(m_allocdBytes);
60  ALPAKA_ASSERT_OFFLOAD(m_allocdBytes <= m_capacity);
61  auto* meta = getLatestVarPtr<MetaData>();
62 
63  // Allocate variable
64  m_allocdBytes = varChunkEnd<T>(m_allocdBytes);
65  ALPAKA_ASSERT_OFFLOAD(m_allocdBytes <= m_capacity);
66 
67  // Update meta data with id and offset for the allocated variable.
68  meta->id = id;
69  meta->offset = m_allocdBytes;
70  }

78  //! Give the pointer to an exiting variable
79  //!
80  //! @tparam T type of the variable
81  //! @param id unique id of the variable
82  //! @return nullptr if variable with id not exists
83  template<typename T>
84  auto getVarPtr(std::uint32_t id) const -> T*
85  {
86  // Offset in bytes to the next unaligned meta data header behind the variable.
87  std::uint32_t off = 0;
88 
89  // Iterate over allocated data only
90  while(off < m_allocdBytes)
91  {
92  // Adjust offset to be aligned
93  std::uint32_t const alignedMetaDataOffset
94  = varChunkEnd<MetaData>(off) - static_cast<std::uint32_t>(sizeof(MetaData));
95  ALPAKA_ASSERT_OFFLOAD(
96  (alignedMetaDataOffset + static_cast<std::uint32_t>(sizeof(MetaData))) <= m_allocdBytes);
97  auto* metaDataPtr = reinterpret_cast<MetaData*>(m_mem + alignedMetaDataOffset);
98  off = metaDataPtr->offset;
99 
100  if(metaDataPtr->id == id)
101  return reinterpret_cast<T*>(&m_mem[off - sizeof(T)]);
102  }
103 
104  // Variable not found.
105  return nullptr;
106  }

120  //! Byte offset to the end of the memory chunk
121  //!
122  //! Calculate bytes required to store a type with a aligned starting address in m_mem.
123  //! Start offset to the origin of the user data chunk can be calculated with `result - sizeof(T)`.
124  //! The padding is always before the origin of the user data chunk and can be zero byte.
125  //!
126  //! \tparam T type should fit into the chunk
127  //! \param byteOffset Current byte offset.
128  //! \result Byte offset to the end of the data chunk, relative to m_mem..
129  template<typename T>
130  auto varChunkEnd(std::uint32_t byteOffset) const -> std::uint32_t
131  {
132  auto const ptr = reinterpret_cast<std::size_t>(m_mem + byteOffset);
133  constexpr size_t align = std::max(TMinDataAlignBytes, alignof(T));
134  std::size_t const newPtrAdress = ((ptr + align - 1u) / align) * align + sizeof(T);
135  return static_cast<uint32_t>(newPtrAdress - reinterpret_cast<std::size_t>(m_mem));
136  }
137 
138  //! Offset in bytes relative to m_mem to next free data area.
139  //! The last aligned before the free area is always a meta data header.
140  mutable std::uint32_t m_allocdBytes = 0u;
141

144  //! Size of padding can be zero if data after padding is already aligned.
145  std::uint8_t* const m_mem;
146 #ifndef NDEBUG
147  const std::uint32_t m_capacity;
148 #endif
