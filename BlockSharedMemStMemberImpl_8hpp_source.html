61  template<typename T>
62  void alloc(uint32_t id) const
63  {
64  // Add meta data chunk in front of the user data
65  m_allocdBytes = varChunkEnd<MetaData>(m_allocdBytes);
66  ALPAKA_ASSERT_OFFLOAD(m_allocdBytes <= m_capacity);
67  MetaData* meta = getLatestVarPtr<MetaData>();
68 
69  // Allocate variable
70  m_allocdBytes = varChunkEnd<T>(m_allocdBytes);
71  ALPAKA_ASSERT_OFFLOAD(m_allocdBytes <= m_capacity);
72 
73  // Update meta data with id and offset for the allocated variable.
74  meta->id = id;
75  meta->offset = m_allocdBytes;
76  }

84  //! Give the pointer to an exiting variable
85  //!
86  //! @tparam T type of the variable
87  //! @param id unique id of the variable
88  //! @return nullptr if variable with id not exists
89  template<typename T>
90  T* getVarPtr(uint32_t id) const
91  {
92  // Offset in bytes to the next unaligned meta data header behind the variable.
93  uint32_t off = 0;
94 
95  // Iterate over allocated data only
96  while(off < m_allocdBytes)
97  {
98  // Adjust offset to be aligned
99  uint32_t const alignedMetaDataOffset
100  = varChunkEnd<MetaData>(off) - static_cast<uint32_t>(sizeof(MetaData));
101  ALPAKA_ASSERT_OFFLOAD(
102  (alignedMetaDataOffset + static_cast<uint32_t>(sizeof(MetaData))) <= m_allocdBytes);
103  MetaData* metaDataPtr = reinterpret_cast<MetaData*>(m_mem + alignedMetaDataOffset);
104  off = metaDataPtr->offset;
105 
106  if(metaDataPtr->id == id)
107  return reinterpret_cast<T*>(&m_mem[off - sizeof(T)]);
108  }
109 
110  // Variable not found.
111  return nullptr;
112  }
113 
114  //! Get last allocated variable.
115  template<typename T>
116  T* getLatestVarPtr() const
117  {
118  return reinterpret_cast<T*>(&m_mem[m_allocdBytes - sizeof(T)]);

126  //! Byte offset to the end of the memory chunk
127  //!
128  //! Calculate bytes required to store a type with a aligned starting address in m_mem.
129  //! Start offset to the origin of the user data chunk can be calculated with `result - sizeof(T)`.
130  //! The padding is always before the origin of the user data chunk and can be zero byte.
131  //!
132  //! \tparam T type should fit into the chunk
133  //! \param byteOffset Current byte offset.
134  //! \result Byte offset to the end of the data chunk, relative to m_mem..
135  template<typename T>
136  std::uint32_t varChunkEnd(uint32_t byteOffset) const
137  {
138  size_t const ptr = reinterpret_cast<size_t>(m_mem + byteOffset);
139  constexpr size_t align = std::max(TMinDataAlignBytes, alignof(T));
140  size_t const newPtrAdress = ((ptr + align - 1u) / align) * align + sizeof(T);
141  return static_cast<uint32_t>(newPtrAdress - reinterpret_cast<size_t>(m_mem));
142  }
143 
144  //! Offset in bytes relative to m_mem to next free data area.
145  //! The last aligned before the free area is always a meta data header.
146  mutable std::uint32_t m_allocdBytes = 0u;
147 
148  //! Memory layout
149  //! |Header|Padding|Variable|Padding|Header|....uninitialized Data ....

151  uint8_t* const m_mem;
152 #ifndef NDEBUG
153  const std::uint32_t m_capacity;
154 #endif
