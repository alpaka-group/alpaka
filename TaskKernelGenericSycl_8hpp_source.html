138  //! The SYCL accelerator execution task.
139  template<typename TAcc, typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
140  class TaskKernelGenericSycl final : public WorkDivMembers<TDim, TIdx>
141  {
142  public:
143  static_assert(TDim::value > 0 && TDim::value <= 3, "Invalid kernel dimensionality");
144 
145  template<typename TWorkDiv>
146  TaskKernelGenericSycl(TWorkDiv&& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
147  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
148  , m_kernelFnObj{kernelFnObj}
149  , m_args{std::forward<TArgs>(args)...}
150  {
151  }
152 
153  auto operator()(sycl::handler& cgh, sycl::buffer<int, 1>& global_fence_buf) const -> void
154  {
155  // Assign placeholder accessors to this command group
156  detail::require(cgh, m_args);
157 
158  auto const work_groups = WorkDivMembers<TDim, TIdx>::m_gridBlockExtent;
159  auto const group_items = WorkDivMembers<TDim, TIdx>::m_blockThreadExtent;
160  auto const item_elements = WorkDivMembers<TDim, TIdx>::m_threadElemExtent;
161 
162  auto const global_size = get_global_size(work_groups, group_items);
163  auto const local_size = get_local_size(group_items);
164 
165  // allocate dynamic shared memory -- needs at least 1 byte to make the Xilinx Runtime happy
166  auto const dyn_shared_mem_bytes = std::max(
167  1ul,
168  core::apply(
169  [&](std::decay_t<TArgs> const&... args) {
170  return getBlockSharedMemDynSizeBytes<TAcc>(m_kernelFnObj, group_items, item_elements, args...);
171  },
172  m_args));
173 
174  auto dyn_shared_accessor = sycl::local_accessor<std::byte>{sycl::range<1>{dyn_shared_mem_bytes}, cgh};
175 
176  // allocate static shared memory -- value comes from the build system
177  constexpr auto st_shared_mem_bytes = std::size_t{ALPAKA_BLOCK_SHARED_DYN_MEMBER_ALLOC_KIB * 1024};
178  auto st_shared_accessor = sycl::local_accessor<std::byte>{sycl::range<1>{st_shared_mem_bytes}, cgh};
179 
180  // register memory fence dummies
181  auto global_fence_dummy = global_fence_buf.get_access(cgh); // Exists once per queue
182  auto local_fence_dummy = sycl::local_accessor<int>{sycl::range<1>{1}, cgh};
183 
184  // copy-by-value so we don't access 'this' on the device
185  auto k_func = m_kernelFnObj;
186  auto k_args = m_args;
187 
188  constexpr std::size_t sub_group_size = trait::warpSize<TKernelFnObj, TAcc>;
189  bool supported = false;
190 
191  if constexpr(sub_group_size == 0)
192  {
193  // no explicit subgroup size requirement
194  LAUNCH_SYCL_KERNEL_WITH_DEFAULT_SUBGROUP_SIZE
195  supported = true;
196  }
197  else
198  {
199 # if(SYCL_SUBGROUP_SIZE == 0)
200  // no explicit SYCL target, assume JIT compilation
201  LAUNCH_SYCL_KERNEL_IF_SUBGROUP_SIZE_IS(sub_group_size)
202  supported = true;
203 # else
204  // check if the kernel should be launched with a subgroup size of 4
205  if constexpr(sub_group_size == 4)
206  {
207 # if(SYCL_SUBGROUP_SIZE & 4)
208  LAUNCH_SYCL_KERNEL_IF_SUBGROUP_SIZE_IS(4)
209  supported = true;
210 # else
211  // empty kernel, required to keep SYCL happy
212  THROW_AND_LAUNCH_EMPTY_SYCL_KERNEL
213 # endif
214  }
215 
216  // check if the kernel should be launched with a subgroup size of 8
217  if constexpr(sub_group_size == 8)
218  {
219 # if(SYCL_SUBGROUP_SIZE & 8)
220  LAUNCH_SYCL_KERNEL_IF_SUBGROUP_SIZE_IS(8)
221  supported = true;
222 # else
223  // empty kernel, required to keep SYCL happy
224  THROW_AND_LAUNCH_EMPTY_SYCL_KERNEL
225 # endif
226  }
227 
228  // check if the kernel should be launched with a subgroup size of 16
229  if constexpr(sub_group_size == 16)
230  {
231 # if(SYCL_SUBGROUP_SIZE & 16)
232  LAUNCH_SYCL_KERNEL_IF_SUBGROUP_SIZE_IS(16)
233  supported = true;
234 # else
235  // empty kernel, required to keep SYCL happy
236  THROW_AND_LAUNCH_EMPTY_SYCL_KERNEL
237 # endif
238  }
239 
240  // check if the kernel should be launched with a subgroup size of 32
241  if constexpr(sub_group_size == 32)
242  {
243 # if(SYCL_SUBGROUP_SIZE & 32)
244  LAUNCH_SYCL_KERNEL_IF_SUBGROUP_SIZE_IS(32)
245  supported = true;
246 # else
247  // empty kernel, required to keep SYCL happy
248  THROW_AND_LAUNCH_EMPTY_SYCL_KERNEL
249 # endif
250  }
251 
252  // check if the kernel should be launched with a subgroup size of 64
253  if constexpr(sub_group_size == 64)
254  {
255 # if(SYCL_SUBGROUP_SIZE & 64)
256  LAUNCH_SYCL_KERNEL_IF_SUBGROUP_SIZE_IS(64)
257  supported = true;
258 # else
259  // empty kernel, required to keep SYCL happy
260  THROW_AND_LAUNCH_EMPTY_SYCL_KERNEL
261 # endif
262  }
263 # endif
264 
265  // this subgroup size is not supported, raise an exception
266  if(not supported)
267  throw sycl::exception(sycl::make_error_code(sycl::errc::kernel_not_supported));
268  }
269  }

276  auto get_global_size(Vec<TDim, TIdx> const& work_groups, Vec<TDim, TIdx> const& group_items) const
277  {
278  if constexpr(TDim::value == 1)
279  return sycl::range<1>{static_cast<std::size_t>(work_groups[0] * group_items[0])};
280  else if constexpr(TDim::value == 2)
281  return sycl::range<2>{
282  static_cast<std::size_t>(work_groups[1] * group_items[1]),
283  static_cast<std::size_t>(work_groups[0] * group_items[0])};
284  else
285  return sycl::range<3>{
286  static_cast<std::size_t>(work_groups[2] * group_items[2]),
287  static_cast<std::size_t>(work_groups[1] * group_items[1]),
288  static_cast<std::size_t>(work_groups[0] * group_items[0])};
289  }
290 
291  auto get_local_size(Vec<TDim, TIdx> const& group_items) const
292  {
293  if constexpr(TDim::value == 1)
294  return sycl::range<1>{static_cast<std::size_t>(group_items[0])};
295  else if constexpr(TDim::value == 2)
296  return sycl::range<2>{
297  static_cast<std::size_t>(group_items[1]),
298  static_cast<std::size_t>(group_items[0])};
299  else
300  return sycl::range<3>{
301  static_cast<std::size_t>(group_items[2]),
302  static_cast<std::size_t>(group_items[1]),
303  static_cast<std::size_t>(group_items[0])};
304  }
