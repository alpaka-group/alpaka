221  //! The CUDA/HIP RT blocking queue enqueue trait specialization.
222  template<
223  typename TTask>
224  struct Enqueue<
225  queue::QueueUniformCudaHipRtBlocking,
226  TTask>
227  {
228  //#############################################################################
229  enum class CallbackState
230  {
231  enqueued,
232  notified,
233  finished,
234  };
235 
236  //#############################################################################
237  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
238  {
239  std::mutex m_mutex;
240  std::condition_variable m_event;
241  CallbackState state = CallbackState::enqueued;
242  };
243 
244  //-----------------------------------------------------------------------------
245 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
246  static void CUDART_CB
247 #else
248  static void HIPRT_CB
249 #endif
250  uniformCudaHipRtCallback(ALPAKA_API_PREFIX(Stream_t) /*queue*/, ALPAKA_API_PREFIX(Error_t) /*status*/, void *arg)
251  {
252  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread has already finished.
253  const auto pCallbackSynchronizationData = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
254 
255  // Notify the executing thread.
256  {
257  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
258  pCallbackSynchronizationData->state = CallbackState::notified;
259  }
260  pCallbackSynchronizationData->m_event.notify_one();
261 
262  // Wait for the executing thread to finish the task if it has not already finished.
263  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
264  if(pCallbackSynchronizationData->state != CallbackState::finished)
265  {
266  pCallbackSynchronizationData->m_event.wait(
267  lock,
268  [pCallbackSynchronizationData](){
269  return pCallbackSynchronizationData->state == CallbackState::finished;
270  }
271  );
272  }
273  }
274 
275  //-----------------------------------------------------------------------------
276  ALPAKA_FN_HOST static auto enqueue(
277  queue::QueueUniformCudaHipRtBlocking & queue,
278  TTask const & task)
279  -> void
280  {
281  auto pCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
282 
283  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamAddCallback)(
284  queue.m_spQueueImpl->m_UniformCudaHipQueue,
285  uniformCudaHipRtCallback,
286  pCallbackSynchronizationData.get(),
287  0u));
288 
289  // We start a new std::thread which stores the task to be executed.
290  // This circumvents the limitation that it is not possible to call CUDA/HIP methods within the CUDA/HIP callback thread.
291  // The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
292  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
293  // before it executes the next task in the queue (CUDA/HIP stream).
294  std::thread t(
295  [pCallbackSynchronizationData, task](){
296  // If the callback has not yet been called, we wait for it.
297  {
298  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
299  if(pCallbackSynchronizationData->state != CallbackState::notified)
300  {
301  pCallbackSynchronizationData->m_event.wait(
302  lock,
303  [pCallbackSynchronizationData](){
304  return pCallbackSynchronizationData->state == CallbackState::notified;
305  }
306  );
307  }
308 
309  task();
310 
311  // Notify the waiting CUDA/HIP thread.
312  pCallbackSynchronizationData->state = CallbackState::finished;
313  }
314  pCallbackSynchronizationData->m_event.notify_one();
315  }
316  );
317 
318  t.join();
319  }
320  };
