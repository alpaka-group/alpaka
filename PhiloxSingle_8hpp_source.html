41  /** Philox engine generating a single number
42  *
43  * This engine's operator() will return a single number. Since the result is the same size as the counter,
44  * and so it contains more than one number, it has to be stored between individual invocations of
45  * operator(). Additionally a pointer has to be stored indicating which part of the result array is to be
46  * returned next.
47  *
48  * @tparam TAcc Accelerator type as defined in alpaka/acc
49  * @tparam TParams Basic parameters for the Philox algorithm
50  */
51  template<typename TAcc, typename TParams>
52  class PhiloxSingle : public traits::PhiloxBaseTraits<TAcc, TParams, PhiloxSingle<TAcc, TParams>>::Base
53  {
54  public:
55  /// Specialization for different TAcc backends
56  using Traits = typename traits::PhiloxBaseTraits<TAcc, TParams, PhiloxSingle<TAcc, TParams>>;
57 
58  using Counter = typename Traits::Counter; ///< Backend-dependent Counter type
59  using Key = typename Traits::Key; ///< Backend-dependent Key type
60  using State = PhiloxStateSingle<Counter, Key>; ///< Backend-dependent State type
61 
62  State state; ///< Internal engine state
63 
64  protected:
65  /** Advance internal counter to the next value
66  *
67  * Advances the full internal counter array, resets the position pointer and stores the intermediate
68  * result to be recalled when the user requests a number.
69  */
70  ALPAKA_FN_HOST_ACC void advanceState()
71  {
72  this->advanceCounter(state.counter);
73  state.result = this->nRounds(state.counter, state.key);
74  state.position = 0;
75  }
76 
77  /** Get the next random number and advance internal state
78  *
79  * The intermediate result stores N = TParams::counterSize numbers. Check if we've already given out
80  * all of them. If so, generate a new intermediate result (this also resets the pointer to the position
81  * of the actual number). Finally, we return the actual number.
82  *
83  * @return The next random number
84  */
85  ALPAKA_FN_HOST_ACC auto nextNumber()
86  {
87  // Element zero will always contain the next valid random number.
88  auto result = state.result[0];
89  state.position++;
90  if(state.position == TParams::counterSize)
91  {
92  advanceState();
93  }
94  else
95  {
96  // Shift state results to allow hard coded access to element zero.
97  // This will avoid high register usage on NVIDIA devices.
98  // \todo Check if this shifting of the result vector is decreasing CPU performance.
99  // If so this optimization for GPUs (mostly NVIDIA) should be moved into
100  // PhiloxBaseCudaArray.
101  state.result[0] = state.result[1];
102  state.result[1] = state.result[2];
103  state.result[2] = state.result[3];
104  }
105 
106  return result;
107  }
108 
109  /// Skips the next \a offset numbers
110  ALPAKA_FN_HOST_ACC void skip(uint64_t offset)
111  {
112  static_assert(TParams::counterSize == 4, "Only counterSize is supported.");
113  state.position = static_cast<decltype(state.position)>(state.position + (offset & 3));
114  offset += state.position < 4 ? 0 : 4;
115  state.position -= state.position < 4 ? 0 : 4u;
116  for(auto numShifts = state.position; numShifts > 0; --numShifts)
117  {
118  // Shift state results to allow hard coded access to element zero.
119  // This will avoid high register usage on NVIDIA devices.
120  state.result[0] = state.result[1];
121  state.result[1] = state.result[2];
122  state.result[2] = state.result[3];
123  }
124  this->skip4(offset / 4);
125  }
126 
127  public:
128  /** Construct a new Philox engine with single-value output
129  *
130  * @param seed Set the Philox generator key
131  * @param subsequence Select a subsequence of size 2^64
132  * @param offset Skip \a offset numbers form the start of the subsequence
133  */
134  ALPAKA_FN_HOST_ACC PhiloxSingle(uint64_t seed = 0, uint64_t subsequence = 0, uint64_t offset = 0)
135  : state{{0, 0, 0, 0}, {low32Bits(seed), high32Bits(seed)}, {0, 0, 0, 0}, 0}
136  {
137  this->skipSubsequence(subsequence);
138  skip(offset);
139  advanceState();
140  }
141 
142  ALPAKA_FN_HOST_ACC PhiloxSingle(PhiloxSingle const& other) : state{other.state}
143  {
144  }
145 
146  /** Get the next random number
147  *
148  * @return The next random number
149  */
150  ALPAKA_FN_HOST_ACC auto operator()()
151  {
152  return nextNumber();
153  }
154  };
