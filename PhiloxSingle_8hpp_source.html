37  /** Philox engine generating a single number
38  *
39  * This engine's operator() will return a single number. Since the result is the same size as the counter,
40  * and so it contains more than one number, it has to be stored between individual invocations of
41  * operator(). Additionally a pointer has to be stored indicating which part of the result array is to be
42  * returned next.
43  *
44  * @tparam TAcc Accelerator type as defined in alpaka/acc
45  * @tparam TParams Basic parameters for the Philox algorithm
46  */
47  template<typename TAcc, typename TParams>
48  class PhiloxSingle : public trait::PhiloxBaseTraits<TAcc, TParams, PhiloxSingle<TAcc, TParams>>::Base
49  {
50  public:
51  /// Specialization for different TAcc backends
52  using Traits = typename trait::PhiloxBaseTraits<TAcc, TParams, PhiloxSingle<TAcc, TParams>>;
53 
54  using Counter = typename Traits::Counter; ///< Backend-dependent Counter type
55  using Key = typename Traits::Key; ///< Backend-dependent Key type
56  using State = PhiloxStateSingle<Counter, Key>; ///< Backend-dependent State type
57 
58  State state; ///< Internal engine state
59 
60  protected:
61  /** Advance internal counter to the next value
62  *
63  * Advances the full internal counter array, resets the position pointer and stores the intermediate
64  * result to be recalled when the user requests a number.
65  */
66  ALPAKA_FN_HOST_ACC void advanceState()
67  {
68  this->advanceCounter(state.counter);
69  state.result = this->nRounds(state.counter, state.key);
70  state.position = 0;
71  }
72 
73  /** Get the next random number and advance internal state
74  *
75  * The intermediate result stores N = TParams::counterSize numbers. Check if we've already given out
76  * all of them. If so, generate a new intermediate result (this also resets the pointer to the position
77  * of the actual number). Finally, we return the actual number.
78  *
79  * @return The next random number
80  */
81  ALPAKA_FN_HOST_ACC auto nextNumber()
82  {
83  // Element zero will always contain the next valid random number.
84  auto result = state.result[0];
85  state.position++;
86  if(state.position == TParams::counterSize)
87  {
88  advanceState();
89  }
90  else
91  {
92  // Shift state results to allow hard coded access to element zero.
93  // This will avoid high register usage on NVIDIA devices.
94  // \todo Check if this shifting of the result vector is decreasing CPU performance.
95  // If so this optimization for GPUs (mostly NVIDIA) should be moved into
96  // PhiloxBaseCudaArray.
97  state.result[0] = state.result[1];
98  state.result[1] = state.result[2];
99  state.result[2] = state.result[3];
100  }
101 
102  return result;
103  }
104 
105  /// Skips the next \a offset numbers
106  ALPAKA_FN_HOST_ACC void skip(uint64_t offset)
107  {
108  static_assert(TParams::counterSize == 4, "Only counterSize is supported.");
109  state.position = static_cast<decltype(state.position)>(state.position + (offset & 3));
110  offset += state.position < 4 ? 0 : 4;
111  state.position -= state.position < 4 ? 0 : 4u;
112  for(auto numShifts = state.position; numShifts > 0; --numShifts)
113  {
114  // Shift state results to allow hard coded access to element zero.
115  // This will avoid high register usage on NVIDIA devices.
116  state.result[0] = state.result[1];
117  state.result[1] = state.result[2];
118  state.result[2] = state.result[3];
119  }
120  this->skip4(offset / 4);
121  }
122 
123  public:
124  /** Construct a new Philox engine with single-value output
125  *
126  * @param seed Set the Philox generator key
127  * @param subsequence Select a subsequence of size 2^64
128  * @param offset Skip \a offset numbers form the start of the subsequence
129  */
130  ALPAKA_FN_HOST_ACC PhiloxSingle(uint64_t seed = 0, uint64_t subsequence = 0, uint64_t offset = 0)
131  : state{{0, 0, 0, 0}, {low32Bits(seed), high32Bits(seed)}, {0, 0, 0, 0}, 0}
132  {
133  this->skipSubsequence(subsequence);
134  skip(offset);
135  advanceState();
136  }
137 
138  /** Get the next random number
139  *
140  * @return The next random number
141  */
142  ALPAKA_FN_HOST_ACC auto operator()()
143  {
144  return nextNumber();
145  }
146  };
