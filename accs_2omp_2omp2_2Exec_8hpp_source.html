145  }

149  int const iNumThreadsInBlock(static_cast<int>(uiNumThreadsInBlock));
150

152  ndLoop(
153  vuiGridBlockExtents,
154  [&](Vec<TDim> const & vuiGridBlockIdx)
155  {
156  this->AccCpuOmp2<TDim>::m_vuiGridBlockIdx = vuiGridBlockIdx;
157 
158  // Execute the threads in parallel.
159 
160  // Force the environment to use the given number of threads.
161  ::omp_set_dynamic(0);
162 
163  // Parallel execution of the threads in a block is required because when syncBlockThreads is called all of them have to be done with their work up to this line.
164  // So we have to spawn one OS thread per thread in a block.
165  // 'omp for' is not useful because it is meant for cases where multiple iterations are executed by one thread but in our case a 1:1 mapping is required.
166  // Therefore we use 'omp parallel' with the specified number of threads in a block.
167  #pragma omp parallel num_threads(iNumThreadsInBlock)
168  {
169 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
170  // GCC 5.1 fails with:
171  // error: redeclaration of ‘const int& iNumThreadsInBlock’
172  // if(iNumThreads != iNumThreadsInBloc
173  // ^
174  // note: ‘const int& iNumThreadsInBlock’ previously declared here
175  // #pragma omp parallel num_threads(iNumThread
176  // ^
177 #if (!BOOST_COMP_GNUC) || (BOOST_COMP_GNUC < BOOST_VERSION_NUMBER(5, 0, 0))
178  // The first thread does some checks in the first block executed.
179  if((::omp_get_thread_num() == 0) && (this->AccCpuOmp2<TDim>::m_vuiGridBlockIdx.sum() == 0u))
180  {
181  int const iNumThreads(::omp_get_num_threads());
182  std::cout << BOOST_CURRENT_FUNCTION << " omp_get_num_threads: " << iNumThreads << std::endl;
183  if(iNumThreads != iNumThreadsInBlock)
184  {
185  throw std::runtime_error("The OpenMP2 runtime did not use the number of threads that had been required!");
186  }
187  }
188 #endif
189 #endif
190  std::forward<TKernelFunctor>(kernelFunctor)(
191  (*static_cast<AccCpuOmp2<TDim> const *>(this)),
192  std::forward<TArgs>(args)...);
193 
194  // Wait for all threads to finish before deleting the shared memory.
195  this->AccCpuOmp2<TDim>::syncBlockThreads();
196  }
197 
198  // After a block has been processed, the shared memory has to be deleted.
199  this->AccCpuOmp2<TDim>::m_vvuiSharedMem.clear();
200  });
201 
202  // After all blocks have been processed, the external shared memory has to be deleted.
203  this->AccCpuOmp2<TDim>::m_vuiExternalSharedMem.reset();
204  }
