60  {
61  static constexpr std::size_t value =
62 #if defined(__AVX512F__) || defined(__MIC__)
63  // addition (AVX512F,KNC): vaddpd / _mm512_add_pd
64  // subtraction (AVX512F,KNC): vsubpd / _mm512_sub_pd
65  // multiplication (AVX512F,KNC): vmulpd / _mm512_mul_pd
66  8u;
67 #elif defined(__AVX__)
68  // addition (AVX): vaddpd / _mm256_add_pd
69  // subtraction (AVX): vsubpd / _mm256_sub_pd
70  // multiplication (AVX): vmulpd / _mm256_mul_pd
71  4u;
72 #elif defined(__SSE2__)
73  // addition (SSE2): addpd / _mm_add_pd
74  // subtraction (SSE2): subpd / _mm_sub_pd
75  // multiplication (SSE2): mulpd / _mm_mul_pd
76  2u;
77 #elif defined(__ARM_NEON__)
78  // No support for double precision vectorization!
79  1u;
80 #elif defined(__ALTIVEC__)
81  2u;
82 #else
83  1u;
84 #endif
85  };
