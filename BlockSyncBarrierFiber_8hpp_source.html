60  template<typename TOp, typename TIdx>
61  struct SyncBlockThreadsPredicate<TOp, BlockSyncBarrierFiber<TIdx>>
62  {
63  ALPAKA_NO_HOST_ACC_WARNING
64  ALPAKA_FN_ACC static auto syncBlockThreadsPredicate(
65  BlockSyncBarrierFiber<TIdx> const& blockSync,
66  int predicate) -> int
67  {
68  if(blockSync.m_curThreadCount == blockSync.m_threadCount)
69  {
70  blockSync.m_curThreadCount = static_cast<TIdx>(0u);
71  ++blockSync.m_generation;
72  }
73 
74  auto const generationMod2(blockSync.m_generation % static_cast<TIdx>(2u));
75 
76  // The first fiber will reset the value to the initial value.
77  if(blockSync.m_curThreadCount == static_cast<TIdx>(0u))
78  {
79  blockSync.m_result[generationMod2] = TOp::InitialValue;
80  }
81 
82  ++blockSync.m_curThreadCount;
83 
84  // We do not have to lock because there is only ever one fiber active per block.
85  blockSync.m_result[generationMod2] = TOp()(blockSync.m_result[generationMod2], predicate);
86 
87  // After all block threads have combined their values ...
88  blockSync.m_barrier.wait();
89 
90  // ... the result can be returned.
91  return blockSync.m_result[generationMod2];
92  }
93  };
