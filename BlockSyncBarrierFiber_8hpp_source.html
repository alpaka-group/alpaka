89  ALPAKA_NO_HOST_ACC_WARNING
90  ALPAKA_FN_ACC static auto syncBlockThreadsPredicate(
91  block::sync::BlockSyncBarrierFiber<TIdx> const & blockSync,
92  int predicate)
93  -> int
94  {
95  if(blockSync.m_curThreadCount == blockSync.m_threadCount)
96  {
97  blockSync.m_curThreadCount = static_cast<TIdx>(0u);
98  ++blockSync.m_generation;
99  }
100 
101  auto const generationMod2(blockSync.m_generation % static_cast<TIdx>(2u));
102 
103  // The first fiber will reset the value to the initial value.
104  if(blockSync.m_curThreadCount == static_cast<TIdx>(0u))
105  {
106  blockSync.m_result[generationMod2] = TOp::InitialValue;
107  }
108 
109  ++blockSync.m_curThreadCount;
110 
111  // We do not have to lock because there is only ever one fiber active per block.
112  blockSync.m_result[generationMod2] = TOp()(blockSync.m_result[generationMod2], predicate);
113 
114  // After all block threads have combined their values ...
115  blockSync.m_barrier.wait();
116 
117  // ... the result can be returned.
118  return blockSync.m_result[generationMod2];
119  }
120  };
121  }
