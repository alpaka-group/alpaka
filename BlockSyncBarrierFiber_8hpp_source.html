55  template<typename TOp, typename TIdx>
56  struct SyncBlockThreadsPredicate<TOp, BlockSyncBarrierFiber<TIdx>>
57  {
58  ALPAKA_NO_HOST_ACC_WARNING
59  ALPAKA_FN_ACC static auto syncBlockThreadsPredicate(
60  BlockSyncBarrierFiber<TIdx> const& blockSync,
61  int predicate) -> int
62  {
63  if(blockSync.m_curThreadCount == blockSync.m_threadCount)
64  {
65  blockSync.m_curThreadCount = static_cast<TIdx>(0u);
66  ++blockSync.m_generation;
67  }
68 
69  auto const generationMod2(blockSync.m_generation % static_cast<TIdx>(2u));
70 
71  // The first fiber will reset the value to the initial value.
72  if(blockSync.m_curThreadCount == static_cast<TIdx>(0u))
73  {
74  blockSync.m_result[generationMod2] = TOp::InitialValue;
75  }
76 
77  ++blockSync.m_curThreadCount;
78 
79  // We do not have to lock because there is only ever one fiber active per block.
80  blockSync.m_result[generationMod2] = TOp()(blockSync.m_result[generationMod2], predicate);
81 
82  // After all block threads have combined their values ...
83  blockSync.m_barrier.wait();
84 
85  // ... the result can be returned.
86  return blockSync.m_result[generationMod2];
87  }
