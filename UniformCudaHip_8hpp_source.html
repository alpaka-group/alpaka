34  //! CUDA/HIP runtime API error checking with log and exception, ignoring specific error values
35  template<typename TApi, bool TThrow>
36  ALPAKA_FN_HOST inline void rtCheck(
37  typename TApi::Error_t const& error,
38  char const* desc,
39  char const* file,
40  int const& line) noexcept(!TThrow)
41  {
42  if(error != TApi::success)
43  {
44  auto const sError = std::string{
45  std::string(file) + "(" + std::to_string(line) + ") " + std::string(desc) + " : '"
46  + TApi::getErrorName(error) + "': '" + std::string(TApi::getErrorString(error)) + "'!"};
47 
48  if constexpr(!TThrow || ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL)
49  std::cerr << sError << std::endl;
50 
51  ALPAKA_DEBUG_BREAK;
52  // reset the last error to allow user side error handling. Using std::ignore to discard unneeded
53  // return values is suggested by the C++ core guidelines.
54  std::ignore = TApi::getLastError();
55 
56  if constexpr(TThrow)
57  throw std::runtime_error(sError);
58  }
59  }

61  //! CUDA/HIP runtime API error checking with log and exception, ignoring specific error values
62  // NOTE: All ignored errors have to be convertible to TApi::Error_t.
63  template<typename TApi, bool TThrow, typename... TErrors>
64  ALPAKA_FN_HOST inline void rtCheckIgnore(
65  typename TApi::Error_t const& error,
66  char const* cmd,
67  char const* file,
68  int const& line,
69  TErrors&&... ignoredErrorCodes) noexcept(!TThrow)
70  {
71  if(error != TApi::success)
72  {
73  std::array<typename TApi::Error_t, sizeof...(ignoredErrorCodes)> const aIgnoredErrorCodes{
74  {ignoredErrorCodes...}};
75 
76  // If the error code is not one of the ignored ones.
77  if(std::find(std::cbegin(aIgnoredErrorCodes), std::cend(aIgnoredErrorCodes), error)
78  == std::cend(aIgnoredErrorCodes))
79  {
80  rtCheck<TApi, TThrow>(error, ("'" + std::string(cmd) + "' returned error ").c_str(), file, line);
81  }
82  else
83  {
84  // reset the last error to avoid propagation to the next CUDA/HIP API call. Using std::ignore
85  // to discard unneeded return values is recommended by the C++ core guidelines.
86  std::ignore = TApi::getLastError();
87  }
88  }
89  }

91  //! CUDA/HIP runtime API last error checking with log and exception.
92  template<typename TApi, bool TThrow>
93  ALPAKA_FN_HOST inline void rtCheckLastError(char const* desc, char const* file, int const& line) noexcept(!TThrow)
94  {
95  typename TApi::Error_t const error(TApi::getLastError());
