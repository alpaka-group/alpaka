56  //! The CUDA/HIP memory buffer.
57  template<
58  typename TElem,
59  typename TDim,
60  typename TIdx>
61  class BufUniformCudaHipRt
62  {
63  static_assert(
64  !std::is_const<TElem>::value,
65  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const elements!");
66  static_assert(
67  !std::is_const<TIdx>::value,
68  "The idx type of the buffer can not be const!");
69  private:
70  using Elem = TElem;
71  using Dim = TDim;
72 
73  public:
74  //-----------------------------------------------------------------------------
75  //! Constructor
76  template<
77  typename TExtent>
78  ALPAKA_FN_HOST BufUniformCudaHipRt(
79  DevUniformCudaHipRt const & dev,
80  TElem * const pMem,
81  TIdx const & pitchBytes,
82  TExtent const & extent) :
83  m_dev(dev),
84  m_extentElements(extent::getExtentVecEnd<TDim>(extent)),
85  m_spMem(
86  pMem,
87  // NOTE: Because the BufUniformCudaHipRt object can be copied and the original object could have been destroyed,
88  // a std::ref(m_dev) or a this pointer can not be bound to the callback because they are not always valid at time of destruction.
89  std::bind(&BufUniformCudaHipRt::freeBuffer, std::placeholders::_1, m_dev)),
90  m_pitchBytes(pitchBytes)
91  {
92  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
93 
94  static_assert(
95  TDim::value == alpaka::Dim<TExtent>::value,
96  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be identical!");
97  static_assert(
98  std::is_same<TIdx, Idx<TExtent>>::value,
99  "The idx type of TExtent and the TIdx template parameter have to be identical!");
100  }
101 
102  private:
103  //-----------------------------------------------------------------------------
104  //! Frees the shared buffer.
105  ALPAKA_FN_HOST static auto freeBuffer(
106  TElem * const memPtr,
107  DevUniformCudaHipRt const & dev)
108  -> void
109  {
110  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
111 
112  // Set the current device.
113  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
114  ALPAKA_API_PREFIX(SetDevice)(
115  dev.m_iDevice));
116  // Free the buffer.
117  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
118  ALPAKA_API_PREFIX(Free)(reinterpret_cast<void *>(memPtr)));
119  }
120 
121  public:
122  DevUniformCudaHipRt m_dev; // NOTE: The device has to be destructed after the memory pointer because it is required for destruction.
123  Vec<TDim, TIdx> m_extentElements;
124  std::shared_ptr<TElem> m_spMem;
125  TIdx m_pitchBytes;
126  };
127  }

323  -> buf::BufUniformCudaHipRt<TElem, DimInt<1u>, TIdx>
324  {
325  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
326 
327  auto const width(extent::getWidth(extent));
328  auto const widthBytes(width * static_cast<TIdx>(sizeof(TElem)));
329

331  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(

335  void * memPtr;
336  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
337  ALPAKA_API_PREFIX(Malloc)(
338  &memPtr,
339  static_cast<std::size_t>(widthBytes)));
340 
341 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
342  std::cout << __func__
343  << " ew: " << width
344  << " ewb: " << widthBytes
345  << " ptr: " << memPtr
346  << std::endl;
347 #endif
348  return
349  buf::BufUniformCudaHipRt<TElem, DimInt<1u>, TIdx>(
350  dev,
351  reinterpret_cast<TElem *>(memPtr),
352  static_cast<TIdx>(widthBytes),
353  extent);
354  }
355  };

383  std::size_t pitchBytes = 0u;
384 #ifdef ALPAKA_ACC_GPU_HIP_ENABLED
385  //FIXME: HIP cannot handle zero-size input (throws Unknown Error)
386  if(width!=0 && height!=0)
387 #endif

389 
390  // Set the current device.
391  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
392  ALPAKA_API_PREFIX(SetDevice)(
393  dev.m_iDevice));
394 
395 
396  // Allocate the buffer on this device.
397  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
398  ALPAKA_API_PREFIX(MallocPitch)(
399  &memPtr,
400  &pitchBytes,
401  static_cast<std::size_t>(widthBytes),
402  static_cast<std::size_t>(height)));
403  ALPAKA_ASSERT(pitchBytes >= static_cast<std::size_t>(widthBytes) || (width * height) == 0);
404  }
405 
406 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
407  std::cout << __func__
408  << " ew: " << width
409  << " eh: " << height
410  << " ewb: " << widthBytes
411  << " ptr: " << memPtr
412  << " pitch: " << pitchBytes
413  << std::endl;
414 #endif
415  return
416  buf::BufUniformCudaHipRt<TElem, DimInt<2u>, TIdx>(
417  dev,
418  reinterpret_cast<TElem *>(memPtr),
419  static_cast<TIdx>(pitchBytes),
420  extent);
421  }

487  dev,
488  reinterpret_cast<TElem *>(pitchedPtrVal.ptr),
489  static_cast<TIdx>(pitchedPtrVal.pitch),

675  buf::BufCpu<TElem, TDim, TIdx> & buf,
676  DevUniformCudaHipRt const & dev)

678  {
679  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
680

682  if(getDev(buf) != dev)
683  {
684  // cuda/hip-HostRegisterMapped:

686  // This feature is available only on GPUs with compute capability greater than or equal to 1.1.
687  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
688  ALPAKA_API_PREFIX(HostRegister)(
689  const_cast<void *>(reinterpret_cast<void const *>(view::getPtrNative(buf))),
690  extent::getExtentProduct(buf) * sizeof(Elem<BufCpu<TElem, TDim, TIdx>>),
691  ALPAKA_API_PREFIX(HostRegisterMapped)));
692  }
693  }
