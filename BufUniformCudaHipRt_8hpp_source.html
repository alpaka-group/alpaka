49  //! The CUDA/HIP memory buffer.
50  template<typename TElem, typename TDim, typename TIdx>
51  class BufUniformCudaHipRt
52  {
53  static_assert(
54  !std::is_const<TElem>::value,
55  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const "
56  "elements!");
57  static_assert(!std::is_const<TIdx>::value, "The idx type of the buffer can not be const!");
58 
59  private:
60  using Elem = TElem;
61  using Dim = TDim;
62 
63  public:
64  //! Constructor
65  template<typename TExtent>
66  ALPAKA_FN_HOST BufUniformCudaHipRt(
67  DevUniformCudaHipRt const& dev,
68  TElem* const pMem,
69  TIdx const& pitchBytes,
70  TExtent const& extent)
71  : m_dev(dev)
72  , m_extentElements(extent::getExtentVecEnd<TDim>(extent))
73  , m_spMem(
74  pMem,
75  // NOTE: Because the BufUniformCudaHipRt object can be copied and the original object could have been
76  // destroyed, a std::ref(m_dev) or a this pointer can not be bound to the callback because they are
77  // not always valid at time of destruction.
78  std::bind(&BufUniformCudaHipRt::freeBuffer, std::placeholders::_1, m_dev))
79  , m_pitchBytes(pitchBytes)
80  {
81  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
82 
83  static_assert(
84  TDim::value == alpaka::Dim<TExtent>::value,
85  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be "
86  "identical!");
87  static_assert(
88  std::is_same<TIdx, Idx<TExtent>>::value,
89  "The idx type of TExtent and the TIdx template parameter have to be identical!");
90  }
91 
92  private:
93  //! Frees the shared buffer.
94  ALPAKA_FN_HOST static auto freeBuffer(TElem* const memPtr, DevUniformCudaHipRt const& dev) -> void
95  {
96  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
97 
98  // Set the current device.
99  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(dev.m_iDevice));
100  // Free the buffer.
101  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(Free)(reinterpret_cast<void*>(memPtr)));
102  }
103 
104  public:
105  DevUniformCudaHipRt m_dev; // NOTE: The device has to be destructed after the memory pointer because it is
106  // required for destruction.
107  Vec<TDim, TIdx> m_extentElements;
108  std::shared_ptr<TElem> m_spMem;
109  TIdx m_pitchBytes;
110  };

444 
445  // If it is already the same device, nothing has to be mapped.

449  // Maps the allocation into the CUDA/HIP address space.The device pointer to the memory may be
