187  //! The CUDA/HIP memory allocation trait specialization.
188  template<typename TApi, typename TElem, typename Dim, typename TIdx>
189  struct BufAlloc<TElem, Dim, TIdx, DevUniformCudaHipRt<TApi>>
190  {
191  template<typename TExtent>
192  ALPAKA_FN_HOST static auto allocBuf(DevUniformCudaHipRt<TApi> const& dev, TExtent const& extent)
193  -> BufUniformCudaHipRt<TApi, TElem, Dim, TIdx>
194  {
195  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
196 
197  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::setDevice(dev.getNativeHandle()));
198 
199  void* memPtr = nullptr;
200  std::size_t pitchBytes = 0u;
201  if(getExtentProduct(extent) != 0)
202  {
203  if constexpr(Dim::value <= 1)
204  {
205  pitchBytes = static_cast<std::size_t>(getWidth(extent)) * sizeof(TElem);
206  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::malloc(&memPtr, pitchBytes));
207  }
208  else if constexpr(Dim::value == 2)
209  {
210  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::mallocPitch(
211  &memPtr,
212  &pitchBytes,
213  static_cast<std::size_t>(getWidth(extent)) * sizeof(TElem),
214  static_cast<std::size_t>(getHeight(extent))));
215  }
216  else if constexpr(Dim::value == 3)
217  {
218  typename TApi::Extent_t const extentVal = TApi::makeExtent(
219  static_cast<std::size_t>(getWidth(extent)) * sizeof(TElem),
220  static_cast<std::size_t>(getHeight(extent)),
221  static_cast<std::size_t>(getDepth(extent)));
222  typename TApi::PitchedPtr_t pitchedPtrVal;
223  pitchedPtrVal.ptr = nullptr;
224  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::malloc3D(&pitchedPtrVal, extentVal));
225  memPtr = pitchedPtrVal.ptr;
226  pitchBytes = pitchedPtrVal.pitch;
227  }
228  }
229 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
230  std::cout << __func__;
231  if constexpr(Dim::value >= 1)
232  std::cout << " ew: " << getWidth(extent);
233  if constexpr(Dim::value >= 2)
234  std::cout << " eh: " << getHeight(extent);
235  if constexpr(Dim::value >= 3)
236  std::cout << " ed: " << getDepth(extent);
237  std::cout << " ptr: " << memPtr << " pitch: " << pitchBytes << std::endl;
238 # endif
239  return {
240  dev,
241  reinterpret_cast<TElem*>(memPtr),
242  [](TElem* ptr) { ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::free(ptr)); },
243  static_cast<TIdx>(pitchBytes),
244  extent};
245  }
246  };

248  //! The CUDA/HIP stream-ordered memory allocation trait specialization.
249  template<typename TApi, typename TElem, typename TDim, typename TIdx>
250  struct AsyncBufAlloc<TElem, TDim, TIdx, DevUniformCudaHipRt<TApi>>
251  {
252 # if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
253  static_assert(
254  std::is_same_v<TApi, ApiCudaRt> && TApi::version >= BOOST_VERSION_NUMBER(11, 2, 0),
255  "Support for stream-ordered memory buffers requires CUDA 11.2 or higher.");
256 # endif
257 # if defined(ALPAKA_ACC_GPU_HIP_ENABLED)

259  !std::is_same_v<TApi, ApiHipRt>,
260  "HIP devices do not support stream-ordered memory buffers.");

266  template<typename TQueue, typename TExtent>
267  ALPAKA_FN_HOST static auto allocAsyncBuf(TQueue queue, [[maybe_unused]] TExtent const& extent)
268  -> BufUniformCudaHipRt<TApi, TElem, TDim, TIdx>
269  {
270  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
271 
272  static_assert(TDim::value == Dim<TExtent>::value, "extent must have the same dimension as the buffer");
273  auto const width = getWidth(extent);
274 
275  auto const& dev = getDev(queue);
276  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::setDevice(dev.getNativeHandle()));
277  void* memPtr;
278  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::mallocAsync(
279  &memPtr,
280  static_cast<std::size_t>(width) * sizeof(TElem),
281  queue.getNativeHandle()));
282 
283 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
284  std::cout << __func__ << " ew: " << width << " ptr: " << memPtr << std::endl;
285 # endif
286  return BufUniformCudaHipRt<TApi, TElem, TDim, TIdx>(
287  dev,
288  reinterpret_cast<TElem*>(memPtr),
289  [queue = std::move(queue)](TElem* ptr)
290  { ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::freeAsync(ptr, queue.getNativeHandle())); },
291  width * static_cast<TIdx>(sizeof(TElem)),
292  extent);
293  }
294  };

316  {
317  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;

320  // accessible to all CUDA/HIP devices.
