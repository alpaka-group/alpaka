182  //! The CUDA/HIP memory allocation trait specialization.
183  template<typename TApi, typename TElem, typename Dim, typename TIdx>
184  struct BufAlloc<TElem, Dim, TIdx, DevUniformCudaHipRt<TApi>>
185  {
186  template<typename TExtent>
187  ALPAKA_FN_HOST static auto allocBuf(DevUniformCudaHipRt<TApi> const& dev, TExtent const& extent)
188  -> BufUniformCudaHipRt<TApi, TElem, Dim, TIdx>
189  {
190  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
191 
192  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::setDevice(dev.getNativeHandle()));
193 
194  void* memPtr = nullptr;
195  std::size_t pitchBytes = 0u;
196  if(getExtentProduct(extent) != 0)
197  {
198  if constexpr(Dim::value <= 1)
199  {
200  pitchBytes = static_cast<std::size_t>(getWidth(extent)) * sizeof(TElem);
201  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::malloc(&memPtr, pitchBytes));
202  }
203  else if constexpr(Dim::value == 2)
204  {
205  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::mallocPitch(
206  &memPtr,
207  &pitchBytes,
208  static_cast<std::size_t>(getWidth(extent)) * sizeof(TElem),
209  static_cast<std::size_t>(getHeight(extent))));
210  }
211  else if constexpr(Dim::value == 3)
212  {
213  typename TApi::Extent_t const extentVal = TApi::makeExtent(
214  static_cast<std::size_t>(getWidth(extent)) * sizeof(TElem),
215  static_cast<std::size_t>(getHeight(extent)),
216  static_cast<std::size_t>(getDepth(extent)));
217  typename TApi::PitchedPtr_t pitchedPtrVal;
218  pitchedPtrVal.ptr = nullptr;
219  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::malloc3D(&pitchedPtrVal, extentVal));
220  memPtr = pitchedPtrVal.ptr;
221  pitchBytes = pitchedPtrVal.pitch;
222  }
223  }
224 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
225  std::cout << __func__;
226  if constexpr(Dim::value >= 1)
227  std::cout << " ew: " << getWidth(extent);
228  if constexpr(Dim::value >= 2)
229  std::cout << " eh: " << getHeight(extent);
230  if constexpr(Dim::value >= 3)
231  std::cout << " ed: " << getDepth(extent);
232  std::cout << " ptr: " << memPtr << " pitch: " << pitchBytes << std::endl;
233 # endif
234  return {
235  dev,
236  reinterpret_cast<TElem*>(memPtr),
237  [](TElem* ptr) { ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(TApi::free(ptr)); },
238  static_cast<TIdx>(pitchBytes),
239  extent};
240  }
241  };

243  //! The CUDA/HIP stream-ordered memory allocation trait specialization.
244  template<typename TApi, typename TElem, typename TDim, typename TIdx>
245  struct AsyncBufAlloc<TElem, TDim, TIdx, DevUniformCudaHipRt<TApi>>
246  {
247 # if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
248  static_assert(
249  std::is_same_v<TApi, ApiCudaRt> && TApi::version >= BOOST_VERSION_NUMBER(11, 2, 0),
250  "Support for stream-ordered memory buffers requires CUDA 11.2 or higher.");
251 # endif
252 # if defined(ALPAKA_ACC_GPU_HIP_ENABLED)
253  static_assert(
254  !std::is_same_v<TApi, ApiHipRt>,
255  "HIP devices do not support stream-ordered memory buffers.");
256 # endif
257  static_assert(
258  TDim::value <= 1,
259  "CUDA/HIP devices support only one-dimensional stream-ordered memory buffers.");
260 
261  template<typename TQueue, typename TExtent>
262  ALPAKA_FN_HOST static auto allocAsyncBuf(TQueue queue, [[maybe_unused]] TExtent const& extent)
263  -> BufUniformCudaHipRt<TApi, TElem, TDim, TIdx>
264  {
265  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
266 
267  static_assert(TDim::value == Dim<TExtent>::value, "extent must have the same dimension as the buffer");
268  auto const width = getWidth(extent);
269 
270  auto const& dev = getDev(queue);
271  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::setDevice(dev.getNativeHandle()));
272  void* memPtr = nullptr;
273  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::mallocAsync(
274  &memPtr,
275  static_cast<std::size_t>(width) * sizeof(TElem),
276  queue.getNativeHandle()));
277 
278 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
279  std::cout << __func__ << " ew: " << width << " ptr: " << memPtr << std::endl;
280 # endif
281  return BufUniformCudaHipRt<TApi, TElem, TDim, TIdx>(
282  dev,
283  reinterpret_cast<TElem*>(memPtr),
284  [queue = std::move(queue)](TElem* ptr)
285  { ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(TApi::freeAsync(ptr, queue.getNativeHandle())); },
286  width * static_cast<TIdx>(sizeof(TElem)),
287  extent);
288  }
289  };
