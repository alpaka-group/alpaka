74  struct TaskSetUniformCudaHip<DimInt<1>, TView, TExtent>
75  : public TaskSetUniformCudaHipBase<DimInt<1>, TView, TExtent>
76  {
77  TaskSetUniformCudaHip(TView& view, std::uint8_t const& byte, TExtent const& extent)
78  : TaskSetUniformCudaHipBase<DimInt<1>, TView, TExtent>(view, byte, extent)
79  {
80  }
81 
82  template<typename TQueue>
83  auto enqueue(TQueue& queue) const -> void
84  {
85  static_assert(
86  Dim<TView>::value == 1u,
87  "The destination buffer is required to be 1-dimensional for this specialization!");
88  static_assert(
89  Dim<TView>::value == Dim<TExtent>::value,
90  "The destination buffer and the extent are required to have the same dimensionality!");
91 
92  using Idx = Idx<TExtent>;
93 
94  auto& view(this->m_view);
95  auto const& extent(this->m_extent);
96 
97  auto const extentWidth(extent::getWidth(extent));
98 
99  if(extentWidth == 0)
100  {
101  return;
102  }
103 
104  auto const extentWidthBytes(extentWidth * static_cast<Idx>(sizeof(Elem<TView>)));
105 # if !defined(NDEBUG)
106  auto const dstWidth(extent::getWidth(view));
107 # endif
108  auto const dstNativePtr(reinterpret_cast<void*>(getPtrNative(view)));
109  ALPAKA_ASSERT(extentWidth <= dstWidth);
110 
111  // Set the current device.
112  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(this->m_iDevice));
113  // Initiate the memory set.
114  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(MemsetAsync)(
115  dstNativePtr,
116  static_cast<int>(this->m_byte),
117  static_cast<size_t>(extentWidthBytes),
118  queue.m_spQueueImpl->m_UniformCudaHipQueue));
119  }
120  };

179  struct TaskSetUniformCudaHip<DimInt<3>, TView, TExtent>
180  : public TaskSetUniformCudaHipBase<DimInt<3>, TView, TExtent>

182  TaskSetUniformCudaHip(TView& view, std::uint8_t const& byte, TExtent const& extent)
183  : TaskSetUniformCudaHipBase<DimInt<3>, TView, TExtent>(view, byte, extent)
184  {

187  template<typename TQueue>
188  auto enqueue(TQueue& queue) const -> void
189  {
190  static_assert(
191  Dim<TView>::value == 3u,
192  "The destination buffer is required to be 3-dimensional for this specialization!");
193  static_assert(
194  Dim<TView>::value == Dim<TExtent>::value,
195  "The destination buffer and the extent are required to have the same dimensionality!");
196 
197  using Elem = alpaka::Elem<TView>;
198  using Idx = Idx<TExtent>;
199 
200  auto& view(this->m_view);
201  auto const& extent(this->m_extent);
202 
203  auto const extentWidth(extent::getWidth(extent));
204  auto const extentHeight(extent::getHeight(extent));
205  auto const extentDepth(extent::getDepth(extent));
206 
207  // This is not only an optimization but also prevents a division by zero.
208  if(extentWidth == 0 || extentHeight == 0 || extentDepth == 0)
209  {
210  return;
211  }
212 
213  auto const dstWidth(extent::getWidth(view));
214 # if !defined(NDEBUG)
215  auto const dstHeight(extent::getHeight(view));
216  auto const dstDepth(extent::getDepth(view));
217 # endif
218  auto const dstPitchBytesX(getPitchBytes<Dim<TView>::value - 1u>(view));
219  auto const dstPitchBytesY(getPitchBytes<Dim<TView>::value - (2u % Dim<TView>::value)>(view));
220  auto const dstNativePtr(reinterpret_cast<void*>(getPtrNative(view)));
221  ALPAKA_ASSERT(extentWidth <= dstWidth);
222  ALPAKA_ASSERT(extentHeight <= dstHeight);
223  ALPAKA_ASSERT(extentDepth <= dstDepth);
224 
225  // Fill CUDA parameter structures.
226  ALPAKA_API_PREFIX(PitchedPtr)
227  const pitchedPtrVal(ALPAKA_PP_CONCAT(make_, ALPAKA_API_PREFIX(PitchedPtr))(
228  dstNativePtr,
229  static_cast<size_t>(dstPitchBytesX),
230  static_cast<size_t>(dstWidth * static_cast<Idx>(sizeof(Elem))),
231  static_cast<size_t>(dstPitchBytesY / dstPitchBytesX)));
232 
233  ALPAKA_API_PREFIX(Extent)
234  const extentVal(ALPAKA_PP_CONCAT(make_, ALPAKA_API_PREFIX(Extent))(
235  static_cast<size_t>(extentWidth * static_cast<Idx>(sizeof(Elem))),
236  static_cast<size_t>(extentHeight),
237  static_cast<size_t>(extentDepth)));
238 
239  // Set the current device.
240  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(this->m_iDevice));
241  // Initiate the memory set.
242  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(Memset3DAsync)(
243  pitchedPtrVal,
244  static_cast<int>(this->m_byte),
245  extentVal,
246  queue.m_spQueueImpl->m_UniformCudaHipQueue));
247  }
