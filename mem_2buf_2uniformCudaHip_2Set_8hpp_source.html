101  DimInt<1>,
102  TView,
103  TExtent>
104  : public TaskSetUniformCudaHipBase<DimInt<1>, TView, TExtent>
105  {
106  //-----------------------------------------------------------------------------
107  TaskSetUniformCudaHip(
108  TView & view,
109  std::uint8_t const & byte,
110  TExtent const & extent) :
111  TaskSetUniformCudaHipBase<DimInt<1>, TView, TExtent>(view, byte, extent)
112  {
113  }
114 
115  //-----------------------------------------------------------------------------
116  template<
117  typename TQueue
118  >
119  auto enqueue(TQueue & queue) const
120  -> void
121  {
122  static_assert(
123  Dim<TView>::value == 1u,
124  "The destination buffer is required to be 1-dimensional for this specialization!");
125  static_assert(
126  Dim<TView>::value == Dim<TExtent>::value,
127  "The destination buffer and the extent are required to have the same dimensionality!");
128 
129  using Idx = Idx<TExtent>;
130 
131  auto & view(this->m_view);
132  auto const & extent(this->m_extent);
133 
134  auto const extentWidth(extent::getWidth(extent));
135 
136  if(extentWidth == 0)
137  {
138  return;
139  }
140 
141  auto const extentWidthBytes(extentWidth * static_cast<Idx>(sizeof(Elem<TView>)));
142 #if !defined(NDEBUG)
143  auto const dstWidth(extent::getWidth(view));
144 #endif
145  auto const dstNativePtr(reinterpret_cast<void *>(view::getPtrNative(view)));
146  ALPAKA_ASSERT(extentWidth <= dstWidth);
147 
148  // Set the current device.
149  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
150  ALPAKA_API_PREFIX(SetDevice)(
151  this->m_iDevice));
152  // Initiate the memory set.
153  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
154  ALPAKA_API_PREFIX(MemsetAsync)(
155  dstNativePtr,
156  static_cast<int>(this->m_byte),
157  static_cast<size_t>(extentWidthBytes),
158  queue.m_spQueueImpl->m_UniformCudaHipQueue));
159  }
160  };

244  {
245  //-----------------------------------------------------------------------------
246  TaskSetUniformCudaHip(
247  TView & view,
248  std::uint8_t const & byte,
249  TExtent const & extent) :
250  TaskSetUniformCudaHipBase<DimInt<3>, TView, TExtent>(view, byte, extent)
251  {
252  }
253 
254  //-----------------------------------------------------------------------------
255  template<
256  typename TQueue
257  >
258  auto enqueue(TQueue & queue) const
259  -> void
260  {
261  static_assert(
262  Dim<TView>::value == 3u,
263  "The destination buffer is required to be 3-dimensional for this specialization!");
264  static_assert(
265  Dim<TView>::value == Dim<TExtent>::value,
266  "The destination buffer and the extent are required to have the same dimensionality!");
267 
268  using Elem = alpaka::Elem<TView>;
269  using Idx = Idx<TExtent>;
270 
271  auto & view(this->m_view);
272  auto const & extent(this->m_extent);
273 
274  auto const extentWidth(extent::getWidth(extent));
275  auto const extentHeight(extent::getHeight(extent));
276  auto const extentDepth(extent::getDepth(extent));
277 
278  // This is not only an optimization but also prevents a division by zero.
279  if(extentWidth == 0 || extentHeight == 0 || extentDepth == 0)
280  {
281  return;
282  }
283 
284  auto const dstWidth(extent::getWidth(view));
285 #if !defined(NDEBUG)
286  auto const dstHeight(extent::getHeight(view));
287  auto const dstDepth(extent::getDepth(view));
288 #endif
289  auto const dstPitchBytesX(view::getPitchBytes<Dim<TView>::value - 1u>(view));
290  auto const dstPitchBytesY(view::getPitchBytes<Dim<TView>::value - (2u % Dim<TView>::value)>(view));
291  auto const dstNativePtr(reinterpret_cast<void *>(view::getPtrNative(view)));
292  ALPAKA_ASSERT(extentWidth <= dstWidth);
293  ALPAKA_ASSERT(extentHeight <= dstHeight);
294  ALPAKA_ASSERT(extentDepth <= dstDepth);
295 
296  // Fill CUDA parameter structures.
297  ALPAKA_API_PREFIX(PitchedPtr) const pitchedPtrVal(
298  ALPAKA_PP_CONCAT(make_,ALPAKA_API_PREFIX(PitchedPtr))(
299  dstNativePtr,
300  static_cast<size_t>(dstPitchBytesX),
301  static_cast<size_t>(dstWidth * static_cast<Idx>(sizeof(Elem))),
302  static_cast<size_t>(dstPitchBytesY / dstPitchBytesX)));
303 
304  ALPAKA_API_PREFIX(Extent) const extentVal(
305  ALPAKA_PP_CONCAT(make_,ALPAKA_API_PREFIX(Extent))(
306  static_cast<size_t>(extentWidth * static_cast<Idx>(sizeof(Elem))),
307  static_cast<size_t>(extentHeight),
308  static_cast<size_t>(extentDepth)));
309 
310  // Set the current device.
311  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
312  ALPAKA_API_PREFIX(SetDevice)(
313  this->m_iDevice));
314  // Initiate the memory set.
315  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
316  ALPAKA_API_PREFIX(Memset3DAsync)(
317  pitchedPtrVal,
318  static_cast<int>(this->m_byte),
319  extentVal,
320  queue.m_spQueueImpl->m_UniformCudaHipQueue));
321  }
322  };
