105  {

107 
108  auto& view = this->m_view;
109  auto const& extent = this->m_extent;
110 
111  auto const extentWidth = getWidth(extent);
112  ALPAKA_ASSERT(extentWidth <= getWidth(view));
113 
114  if(extentWidth == 0)

119  // Initiate the memory set.
120  auto const extentWidthBytes = extentWidth * static_cast<Idx>(sizeof(Elem<TView>));
121  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::memsetAsync(
122  getPtrNative(view),
123  static_cast<int>(this->m_byte),

125  queue.getNativeHandle()));

143  template<typename TQueue>
144  auto enqueue(TQueue& queue) const -> void
145  {
146  using Idx = Idx<TExtent>;
147 
148  auto& view = this->m_view;
149  auto const& extent = this->m_extent;
150 
151  auto const extentWidth = getWidth(extent);
152  auto const extentHeight = getHeight(extent);
153 
154  if(extentWidth == 0 || extentHeight == 0)
155  {
156  return;
157  }
158 
159  auto const extentWidthBytes = extentWidth * static_cast<Idx>(sizeof(Elem<TView>));
160 
161 # if !defined(NDEBUG)
162  auto const dstWidth = getWidth(view);
163  auto const dstHeight = getHeight(view);
164 # endif
165  auto const dstPitchBytesX = getPitchBytes<Dim<TView>::value - 1u>(view);
166  auto const dstNativePtr = reinterpret_cast<void*>(getPtrNative(view));
167  ALPAKA_ASSERT(extentWidth <= dstWidth);
168  ALPAKA_ASSERT(extentHeight <= dstHeight);
169 
170  // Initiate the memory set.
171  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::memset2DAsync(
172  dstNativePtr,
173  static_cast<size_t>(dstPitchBytesX),
174  static_cast<int>(this->m_byte),
175  static_cast<size_t>(extentWidthBytes),
176  static_cast<size_t>(extentHeight),
177  queue.getNativeHandle()));
178  }

181  //! The 3D CUDA/HIP memory set task.
182  template<typename TApi, typename TView, typename TExtent>
183  struct TaskSetUniformCudaHip<TApi, DimInt<3u>, TView, TExtent>
184  : public TaskSetUniformCudaHipBase<TApi, DimInt<3u>, TView, TExtent>
185  {
186  template<typename TViewFwd>
187  TaskSetUniformCudaHip(TViewFwd&& view, std::uint8_t const& byte, TExtent const& extent)
188  : TaskSetUniformCudaHipBase<TApi, DimInt<3u>, TView, TExtent>(
189  std::forward<TViewFwd>(view),
190  byte,
191  extent)
192  {
193  }

195  template<typename TQueue>
196  auto enqueue(TQueue& queue) const -> void
197  {
198  using Elem = alpaka::Elem<TView>;
199  using Idx = Idx<TExtent>;
200 
201  auto& view = this->m_view;
202  auto const& extent = this->m_extent;
203 
204  auto const extentWidth = getWidth(extent);
205  auto const extentHeight = getHeight(extent);
206  auto const extentDepth = getDepth(extent);
207 
208  // This is not only an optimization but also prevents a division by zero.
209  if(extentWidth == 0 || extentHeight == 0 || extentDepth == 0)
210  {
211  return;
212  }
213 
214  auto const dstWidth = getWidth(view);
215 # if !defined(NDEBUG)
216  auto const dstHeight = getHeight(view);
217  auto const dstDepth = getDepth(view);
218 # endif
219  auto const dstPitchBytesX = getPitchBytes<Dim<TView>::value - 1u>(view);
220  auto const dstPitchBytesY = getPitchBytes<Dim<TView>::value - (2u % Dim<TView>::value)>(view);
221  auto const dstNativePtr = reinterpret_cast<void*>(getPtrNative(view));
222  ALPAKA_ASSERT(extentWidth <= dstWidth);
223  ALPAKA_ASSERT(extentHeight <= dstHeight);
224  ALPAKA_ASSERT(extentDepth <= dstDepth);
225 
226  // Fill CUDA parameter structures.
227  typename TApi::PitchedPtr_t const pitchedPtrVal = TApi::makePitchedPtr(
228  dstNativePtr,
229  static_cast<size_t>(dstPitchBytesX),
230  static_cast<size_t>(dstWidth * static_cast<Idx>(sizeof(Elem))),
231  static_cast<size_t>(dstPitchBytesY / dstPitchBytesX));
232 
233  typename TApi::Extent_t const extentVal = TApi::makeExtent(
234  static_cast<size_t>(extentWidth * static_cast<Idx>(sizeof(Elem))),
235  static_cast<size_t>(extentHeight),
236  static_cast<size_t>(extentDepth));
237 
238  // Initiate the memory set.
239  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::memset3DAsync(
240  pitchedPtrVal,
241  static_cast<int>(this->m_byte),
242  extentVal,
243  queue.getNativeHandle()));
244  }
245  };
