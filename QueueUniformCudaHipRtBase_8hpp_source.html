45  //! The CUDA/HIP RT blocking queue implementation.
46  class QueueUniformCudaHipRtImpl final
47  {
48  public:
49  ALPAKA_FN_HOST QueueUniformCudaHipRtImpl(DevUniformCudaHipRt const& dev)
50  : m_dev(dev)
51  , m_UniformCudaHipQueue()
52  {
53  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
54 
55  // Set the current device.
56  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(m_dev.m_iDevice));
57 
58  // - [cuda/hip]StreamDefault: Default queue creation flag.
59  // - [cuda/hip]StreamNonBlocking: Specifies that work running in the created queue may run
60  // concurrently with work in queue 0 (the NULL queue),
61  // and that the created queue should perform no implicit synchronization with queue 0.
62  // Create the queue on the current device.
63  // NOTE: [cuda/hip]StreamNonBlocking is required to match the semantic implemented in the alpaka
64  // CPU queue. It would be too much work to implement implicit default queue synchronization on CPU.
65 
66  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(
67  StreamCreateWithFlags)(&m_UniformCudaHipQueue, ALPAKA_API_PREFIX(StreamNonBlocking)));
68  }
69  QueueUniformCudaHipRtImpl(QueueUniformCudaHipRtImpl&&) = default;
70  auto operator=(QueueUniformCudaHipRtImpl&&) -> QueueUniformCudaHipRtImpl& = delete;
71  ALPAKA_FN_HOST ~QueueUniformCudaHipRtImpl()
72  {
73  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
74 
75  // In case the device is still doing work in the queue when cuda/hip-StreamDestroy() is called, the
76  // function will return immediately and the resources associated with queue will be released
77  // automatically once the device has completed all work in queue.
78  // -> No need to synchronize here.
79  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamDestroy)(m_UniformCudaHipQueue));
80  }
81 
82  public:
83  DevUniformCudaHipRt const m_dev; //!< The device this queue is bound to.
84  ALPAKA_API_PREFIX(Stream_t) m_UniformCudaHipQueue;
85  };
