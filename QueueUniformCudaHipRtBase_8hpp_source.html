48  //! The CUDA/HIP RT blocking queue implementation.
49  class QueueUniformCudaHipRtImpl final
50  {
51  public:
52  //-----------------------------------------------------------------------------
53  ALPAKA_FN_HOST QueueUniformCudaHipRtImpl(
54  dev::DevUniformCudaHipRt const & dev) :
55  m_dev(dev),
56  m_UniformCudaHipQueue()
57  {
58  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
59 
60  // Set the current device.
61  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
62  ALPAKA_API_PREFIX(SetDevice)(
63  m_dev.m_iDevice));
64 
65  // - [cuda/hip]StreamDefault: Default queue creation flag.
66  // - [cuda/hip]StreamNonBlocking: Specifies that work running in the created queue may run concurrently with work in queue 0 (the NULL queue),
67  // and that the created queue should perform no implicit synchronization with queue 0.
68  // Create the queue on the current device.
69  // NOTE: [cuda/hip]StreamNonBlocking is required to match the semantic implemented in the alpaka CPU queue.
70  // It would be too much work to implement implicit default queue synchronization on CPU.
71 
72  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
73  ALPAKA_API_PREFIX(StreamCreateWithFlags)(
74  &m_UniformCudaHipQueue,
75  ALPAKA_API_PREFIX(StreamNonBlocking)));
76 
77  }
78  //-----------------------------------------------------------------------------
79  QueueUniformCudaHipRtImpl(QueueUniformCudaHipRtImpl const &) = delete;
80  //-----------------------------------------------------------------------------
81  QueueUniformCudaHipRtImpl(QueueUniformCudaHipRtImpl &&) = default;
82  //-----------------------------------------------------------------------------
83  auto operator=(QueueUniformCudaHipRtImpl const &) -> QueueUniformCudaHipRtImpl & = delete;
84  //-----------------------------------------------------------------------------
85  auto operator=(QueueUniformCudaHipRtImpl &&) -> QueueUniformCudaHipRtImpl & = delete;
86  //-----------------------------------------------------------------------------
87  ALPAKA_FN_HOST ~QueueUniformCudaHipRtImpl()
88  {
89  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
90 
91  // Set the current device. \TODO: Is setting the current device before [cuda/hip]StreamDestroy required?
92 
93  // In case the device is still doing work in the queue when [cuda/hip]StreamDestroy() is called, the function will return immediately
94  // and the resources associated with queue will be released automatically once the device has completed all work in queue.
95  // -> No need to synchronize here.
96 
97  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
98  ALPAKA_API_PREFIX(SetDevice)(
99  m_dev.m_iDevice));
100  // In case the device is still doing work in the queue when cuda/hip-StreamDestroy() is called, the function will return immediately
101  // and the resources associated with queue will be released automatically once the device has completed all work in queue.
102  // -> No need to synchronize here.
103  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
104  ALPAKA_API_PREFIX(StreamDestroy)(
105  m_UniformCudaHipQueue));
106 
107  }
108 
109  public:
110  dev::DevUniformCudaHipRt const m_dev; //!< The device this queue is bound to.
111  ALPAKA_API_PREFIX(Stream_t) m_UniformCudaHipQueue;
112  };
