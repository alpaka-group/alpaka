45  //! An accessor is an abstraction for accessing memory objects such as views and buffers.
46  //! @tparam TMemoryHandle A handle to a memory object.
47  //! @tparam TElem The type of the element stored by the memory object. Values and references to this type are
48  //! returned on access.
49  //! @tparam TBufferIdx The integral type used for indexing and index computations.
50  //! @tparam TDim The dimensionality of the accessed data.
51  //! @tparam TAccessModes Either a single access tag type or a `std::tuple` containing multiple access tag
52  //! types.
53  template<typename TMemoryHandle, typename TElem, typename TBufferIdx, std::size_t TDim, typename TAccessModes>
54  struct Accessor;

113  //! Creates an accessor for the given memory object using the specified access modes. Memory objects are e.g.
114  //! alpaka views and buffers.
115  template<
116  typename... TAccessModes,
117  typename TMemoryObject,
118  typename = std::enable_if_t<!internal::IsAccessor<std::decay_t<TMemoryObject>>::value>>
119  ALPAKA_FN_HOST_ACC auto accessWith(TMemoryObject&& memoryObject)
120  {
121  return traits::BuildAccessor<std::decay_t<TMemoryObject>>::template buildAccessor<TAccessModes...>(
