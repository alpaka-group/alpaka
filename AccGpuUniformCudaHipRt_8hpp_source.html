124  //! The GPU CUDA accelerator device properties get trait specialization.
125  template<typename TDim, typename TIdx>
126  struct GetAccDevProps<AccGpuUniformCudaHipRt<TDim, TIdx>>
127  {
128  ALPAKA_FN_HOST static auto getAccDevProps(DevUniformCudaHipRt const& dev) -> AccDevProps<TDim, TIdx>
129  {
130 # ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
131  // Reading only the necessary attributes with cudaDeviceGetAttribute is faster than reading all with
132  // cuda https://devblogs.nvidia.com/cuda-pro-tip-the-fast-way-to-query-device-properties/
133  int multiProcessorCount = {};
134  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
135  cudaDeviceGetAttribute(&multiProcessorCount, cudaDevAttrMultiProcessorCount, dev.m_iDevice));
136 
137  int maxGridSize[3] = {};
138  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
139  cudaDeviceGetAttribute(&maxGridSize[0], cudaDevAttrMaxGridDimX, dev.m_iDevice));
140  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
141  cudaDeviceGetAttribute(&maxGridSize[1], cudaDevAttrMaxGridDimY, dev.m_iDevice));
142  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
143  cudaDeviceGetAttribute(&maxGridSize[2], cudaDevAttrMaxGridDimZ, dev.m_iDevice));
144 
145  int maxBlockDim[3] = {};
146  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
147  cudaDeviceGetAttribute(&maxBlockDim[0], cudaDevAttrMaxBlockDimX, dev.m_iDevice));
148  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
149  cudaDeviceGetAttribute(&maxBlockDim[1], cudaDevAttrMaxBlockDimY, dev.m_iDevice));
150  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
151  cudaDeviceGetAttribute(&maxBlockDim[2], cudaDevAttrMaxBlockDimZ, dev.m_iDevice));
152 
153  int maxThreadsPerBlock = {};
154  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
155  cudaDeviceGetAttribute(&maxThreadsPerBlock, cudaDevAttrMaxThreadsPerBlock, dev.m_iDevice));
156 
157  int sharedMemSizeBytes = {};
158  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
159  cudaDeviceGetAttribute(&sharedMemSizeBytes, cudaDevAttrMaxSharedMemoryPerBlock, dev.m_iDevice));
160 
161  return {// m_multiProcessorCount
162  alpaka::core::clipCast<TIdx>(multiProcessorCount),
163  // m_gridBlockExtentMax
164  extent::getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
165  alpaka::core::clipCast<TIdx>(maxGridSize[2u]),
166  alpaka::core::clipCast<TIdx>(maxGridSize[1u]),
167  alpaka::core::clipCast<TIdx>(maxGridSize[0u]))),
168  // m_gridBlockCountMax
169  std::numeric_limits<TIdx>::max(),
170  // m_blockThreadExtentMax
171  extent::getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
172  alpaka::core::clipCast<TIdx>(maxBlockDim[2u]),
173  alpaka::core::clipCast<TIdx>(maxBlockDim[1u]),
174  alpaka::core::clipCast<TIdx>(maxBlockDim[0u]))),
175  // m_blockThreadCountMax
176  alpaka::core::clipCast<TIdx>(maxThreadsPerBlock),
177  // m_threadElemExtentMax
178  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
179  // m_threadElemCountMax
180  std::numeric_limits<TIdx>::max(),
181  // m_sharedMemSizeBytes
182  static_cast<size_t>(sharedMemSizeBytes)};
183 
184 # else
185  hipDeviceProp_t hipDevProp;
186  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(hipGetDeviceProperties(&hipDevProp, dev.m_iDevice));
187 
188  return {// m_multiProcessorCount
189  alpaka::core::clipCast<TIdx>(hipDevProp.multiProcessorCount),
190  // m_gridBlockExtentMax
191  extent::getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
192  alpaka::core::clipCast<TIdx>(hipDevProp.maxGridSize[2u]),
193  alpaka::core::clipCast<TIdx>(hipDevProp.maxGridSize[1u]),
194  alpaka::core::clipCast<TIdx>(hipDevProp.maxGridSize[0u]))),
195  // m_gridBlockCountMax
196  std::numeric_limits<TIdx>::max(),
197  // m_blockThreadExtentMax
198  extent::getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
199  alpaka::core::clipCast<TIdx>(hipDevProp.maxThreadsDim[2u]),
200  alpaka::core::clipCast<TIdx>(hipDevProp.maxThreadsDim[1u]),
201  alpaka::core::clipCast<TIdx>(hipDevProp.maxThreadsDim[0u]))),
202  // m_blockThreadCountMax
203  alpaka::core::clipCast<TIdx>(hipDevProp.maxThreadsPerBlock),
204  // m_threadElemExtentMax
205  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
206  // m_threadElemCountMax
207  std::numeric_limits<TIdx>::max(),
208  // m_sharedMemSizeBytes
209  static_cast<size_t>(hipDevProp.sharedMemPerBlock)};
210 # endif
211  }
