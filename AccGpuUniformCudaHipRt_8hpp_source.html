121  // Reading only the necessary attributes with cudaDeviceGetAttribute is faster than reading all with

133  dev.getNativeHandle()));

144  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::deviceGetAttribute(

147  dev.getNativeHandle()));

149  &maxBlockDim[1],

154  TApi::deviceAttributeMaxBlockDimZ,
155  dev.getNativeHandle()));
156 
157  int maxThreadsPerBlock = {};
158  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::deviceGetAttribute(
159  &maxThreadsPerBlock,
160  TApi::deviceAttributeMaxThreadsPerBlock,
161  dev.getNativeHandle()));
162 
163  int sharedMemSizeBytes = {};
164  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::deviceGetAttribute(

166  TApi::deviceAttributeMaxSharedMemoryPerBlock,
167  dev.getNativeHandle()));
168 
169  return {// m_multiProcessorCount
170  alpaka::core::clipCast<TIdx>(multiProcessorCount),
171  // m_gridBlockExtentMax
172  getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
173  alpaka::core::clipCast<TIdx>(maxGridSize[2u]),
174  alpaka::core::clipCast<TIdx>(maxGridSize[1u]),
175  alpaka::core::clipCast<TIdx>(maxGridSize[0u]))),
176  // m_gridBlockCountMax
177  std::numeric_limits<TIdx>::max(),
178  // m_blockThreadExtentMax
179  getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
180  alpaka::core::clipCast<TIdx>(maxBlockDim[2u]),
181  alpaka::core::clipCast<TIdx>(maxBlockDim[1u]),
182  alpaka::core::clipCast<TIdx>(maxBlockDim[0u]))),
183  // m_blockThreadCountMax
184  alpaka::core::clipCast<TIdx>(maxThreadsPerBlock),
185  // m_threadElemExtentMax
186  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
187  // m_threadElemCountMax
188  std::numeric_limits<TIdx>::max(),
189  // m_sharedMemSizeBytes
190  static_cast<size_t>(sharedMemSizeBytes)};
191 
192 # else
193  typename TApi::DeviceProp_t properties;
194  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(TApi::getDeviceProperties(&properties, dev.getNativeHandle()));
195 
196  return {// m_multiProcessorCount

198  // m_gridBlockExtentMax

208  alpaka::core::clipCast<TIdx>(properties.maxThreadsDim[1u]),
