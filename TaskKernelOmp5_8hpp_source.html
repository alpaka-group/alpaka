47  //! The OpenMP 5.0 accelerator execution task.
48  template<typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
49  class TaskKernelOmp5 final : public WorkDivMembers<TDim, TIdx>
50  {
51  public:
52  template<typename TWorkDiv>
53  ALPAKA_FN_HOST TaskKernelOmp5(TWorkDiv&& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
54  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
55  , m_kernelFnObj(kernelFnObj)
56  , m_args(std::forward<TArgs>(args)...)
57  {
58  static_assert(
59  Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
60  "The work division and the execution task have to be of the same dimensionality!");
61  }
62 
63  //! Executes the kernel function object.
64  ALPAKA_FN_HOST auto operator()(const DevOmp5& dev) const -> void
65  {
66  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
67 
68  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(*this);
69  auto const blockThreadExtent = getWorkDiv<Block, Threads>(*this);
70  auto const threadElemExtent = getWorkDiv<Thread, Elems>(*this);
71 
72 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
73  std::cout << "m_gridBlockExtent=" << this->m_gridBlockExtent << "\tgridBlockExtent=" << gridBlockExtent
74  << std::endl;
75  std::cout << "m_blockThreadExtent=" << this->m_blockThreadExtent
76  << "\tblockThreadExtent=" << blockThreadExtent << std::endl;
77  std::cout << "m_threadElemExtent=" << this->m_threadElemExtent << "\tthreadElemExtent=" << threadElemExtent
78  << std::endl;
79 # endif
80 
81  // Get the size of the block shared dynamic memory.
82  auto const blockSharedMemDynSizeBytes = core::apply(
83  [&](ALPAKA_DECAY_T(TArgs) const&... args)
84  {
85  return getBlockSharedMemDynSizeBytes<AccOmp5<TDim, TIdx>>(
86  m_kernelFnObj,
87  blockThreadExtent,
88  threadElemExtent,
89  args...);
90  },
91  m_args);
92 
93 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
94  std::cout << __func__ << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B"
95  << std::endl;
96 # endif
97  // We have to make sure, that the OpenMP runtime keeps enough threads for executing a block in parallel.
98  TIdx const maxOmpThreadCount(static_cast<TIdx>(::omp_get_max_threads()));
99  // The number of blocks in the grid.
100  TIdx const gridBlockCount(gridBlockExtent.prod());
101  // The number of threads in a block.
102  TIdx const blockThreadCount(blockThreadExtent.prod());
103 
104  if(gridBlockCount == 0 || blockThreadCount == 0)
105  { //! empty grid is a NOP
106  return;
107  }
108 
109 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
110  if(maxOmpThreadCount < blockThreadExtent.prod())
111  {
112  std::cout
113  << "Warning: TaskKernelOmp5: maxOmpThreadCount smaller than blockThreadCount requested by caller:"
114  << maxOmpThreadCount << " < " << blockThreadExtent.prod() << std::endl;
115  }
116 # endif
117  // make sure there is at least on team
118  TIdx const teamCount(std::max(
119  std::min(static_cast<TIdx>(maxOmpThreadCount / blockThreadCount), gridBlockCount),
120  static_cast<TIdx>(1u)));
121 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
122  std::cout << "threadElemCount=" << threadElemExtent[0u] << std::endl;
123  std::cout << "teamCount=" << teamCount << "\tgridBlockCount=" << gridBlockCount << std::endl;
124 # endif
125 
126  if(::omp_in_parallel() != 0)
127  {
128  throw std::runtime_error("The OpenMP 5.0 backend can not be used within an existing parallel region!");
129  }
130 
131  // Force the environment to use the given number of threads.
132  int const ompIsDynamic(::omp_get_dynamic());
133  ::omp_set_dynamic(0);
134 
135  // `When an if(scalar-expression) evaluates to false, the structured block is executed on the host.`
136  auto argsD = m_args;
137  auto kernelFnObj = m_kernelFnObj;
138  const auto iDevice = dev.getNativeHandle();
139 # pragma omp target device(iDevice)
140  {
141 # pragma omp teams distribute num_teams(teamCount) // thread_limit(blockThreadCount)
142  for(TIdx t = 0u; t < gridBlockCount; ++t)
143  {
144 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL || defined ALPAKA_CI
145  // The first team does some checks ...
146  if(t == 0)
147  {
148  int const iNumTeams(::omp_get_num_teams());
149  printf("%s omp_get_num_teams: %d\n", __func__, iNumTeams);
150  printf("threadElemCount_dev %d\n", int(threadElemExtent[0u]));
151  }
152 # endif
153  AccOmp5<TDim, TIdx>
154  acc(gridBlockExtent, blockThreadExtent, threadElemExtent, t, blockSharedMemDynSizeBytes);
155 
156  // Execute the threads in parallel.
157 
158  // Parallel execution of the threads in a block is required because when syncBlockThreads is called
159  // all of them have to be done with their work up to this line. So we have to spawn one OS thread
160  // per thread in a block. 'omp for' is not useful because it is meant for cases where multiple
161  // iterations are executed by one thread but in our case a 1:1 mapping is required. Therefore we
162  // use 'omp parallel' with the specified number of threads in a block.
163 # ifndef __ibmxl_vrm__
164 // setting num_threads to any value leads XL to run only one thread per team
165 # pragma omp parallel num_threads(blockThreadCount)
166 # else
167 # pragma omp parallel
168 # endif
169  {
170 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL || defined ALPAKA_CI
171  // The first thread does some checks in the first block executed.
172  if((::omp_get_thread_num() == 0) && (t == 0))
173  {
174  int const numThreads = ::omp_get_num_threads();
175  printf("%s omp_get_num_threads: %d\n", __func__, numThreads);
176  if(numThreads != static_cast<int>(blockThreadCount))
177  {
178  printf("ERROR: The OpenMP runtime did not use the number of threads that had been "
179  "requested!\n");
180  }
181  }
182 # endif
183  core::apply(
184  [kernelFnObj, &acc](typename std::decay<TArgs>::type const&... args)
185  { kernelFnObj(acc, args...); },
186  argsD);
187 
188  // Wait for all threads to finish before deleting the shared memory.
189  // This is done by default if the omp 'nowait' clause is missing
190  // syncBlockThreads(acc);
191  }
192  }
193  }
194 
195  // Reset the dynamic thread number setting.
196  ::omp_set_dynamic(ompIsDynamic);
197  }
198 
199  private:
200  TKernelFnObj m_kernelFnObj;
201  core::Tuple<std::decay_t<TArgs>...> m_args;
202  };
