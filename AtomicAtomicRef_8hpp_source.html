45  template<typename T>
46  void isSupportedByAtomicAtomicRef()
47  {
48  static_assert(
49  std::is_trivially_copyable_v<T> && detail::atomic_ref<T>::required_alignment <= alignof(T),
50  "Type not supported by AtomicAtomicRef, please recompile defining "

85  {
86  isSupportedByAtomicAtomicRef<T>();
87  detail::atomic_ref<T> ref(*addr);
88  T old = ref;

210  //! The CPU accelerators AtomicCas.
211  template<typename T, typename THierarchy>
212  struct AtomicOp<AtomicCas, AtomicAtomicRef, T, THierarchy>
213  {
214  ALPAKA_FN_HOST static auto atomicOp(
215  AtomicAtomicRef const&,
216  T* const addr,
217  T const& compare,
218  T const& value) -> T
219  {
220  isSupportedByAtomicAtomicRef<T>();
221  detail::atomic_ref<T> ref(*addr);
222  T old = ref;
223  T result;
224  do
225  {
226  result = ((old == compare) ? value : old);
227  } while(!ref.compare_exchange_weak(old, result));
228  return old;
229  }
