31  template<typename T>
32  void isSupportedByAtomicAtomicRef()
33  {
34  static_assert(
35  std::is_trivially_copyable_v<T> && boost::atomic_ref<T>::required_alignment <= alignof(T),
36  "Type not supported by AtomicAtomicRef, please recompile defining "
37  "ALPAKA_DISABLE_ATOMIC_ATOMICREF.");
38  }

66  //! The CPU accelerators AtomicMin.
67  template<typename T, typename THierarchy>
68  struct AtomicOp<AtomicMin, AtomicAtomicRef, T, THierarchy>
69  {
70  ALPAKA_FN_HOST static auto atomicOp(AtomicAtomicRef const&, T* const addr, T const& value) -> T
71  {
72  isSupportedByAtomicAtomicRef<T>();
73  boost::atomic_ref<T> ref(*addr);
74  T old = ref;
75  T result = old;
76  result = std::min(result, value);
77  while(!ref.compare_exchange_weak(old, result))
78  {
79  result = old;
80  result = std::min(result, value);
81  }
82  return old;
83  }

196  //! The CPU accelerators AtomicCas.
197  template<typename T, typename THierarchy>
198  struct AtomicOp<AtomicCas, AtomicAtomicRef, T, THierarchy>
199  {
200  ALPAKA_FN_HOST static auto atomicOp(
201  AtomicAtomicRef const&,
202  T* const addr,
203  T const& compare,
204  T const& value) -> T
205  {
206  isSupportedByAtomicAtomicRef<T>();
207  boost::atomic_ref<T> ref(*addr);
208  T old = ref;
209  T result;
210  do
211  {
212  result = ((old == compare) ? value : old);
213  } while(!ref.compare_exchange_weak(old, result));
214  return old;
215  }
