123  struct VerifyBytesSetKernel
124  {
125  ALPAKA_NO_HOST_ACC_WARNING
126  template<typename TAcc, typename TIter>
127  ALPAKA_FN_ACC void operator()(
128  TAcc const& acc,
129  bool* success,
130  TIter const& begin,
131  TIter const& end,
132  std::uint8_t const& byte) const
133  {
134  alpaka::ignore_unused(acc);
135 
136  constexpr auto elemSizeInByte = sizeof(decltype(*begin));
137  for(auto it = begin; it != end; ++it)
138  {
139  auto const& elem = *it;
140  auto const pBytes = reinterpret_cast<std::uint8_t const*>(&elem);
141  for(std::size_t i = 0u; i < elemSizeInByte; ++i)
142  {
143  ALPAKA_CHECK(*success, pBytes[i] == byte);
144  }
145  }
146  }

217  //! Fills the given view with increasing values starting at 0.
218  template<typename TView, typename TQueue>
219  ALPAKA_FN_HOST auto iotaFillView(TQueue& queue, TView& view) -> void
220  {
221  using DevHost = alpaka::DevCpu;
222  using PltfHost = alpaka::Pltf<DevHost>;
223 
224  using Elem = alpaka::Elem<TView>;
225 
226  DevHost const devHost = alpaka::getDevByIdx<PltfHost>(0);
227 
228  auto const extent = alpaka::extent::getExtentVec(view);
229 
230  // Init buf with increasing values
231  std::vector<Elem> v(static_cast<std::size_t>(extent.prod()), static_cast<Elem>(0));
232  std::iota(std::begin(v), std::end(v), static_cast<Elem>(0));
233  auto plainBuf = alpaka::createView(devHost, v, extent);
234 
235  // Copy the generated content into the given view.
236  alpaka::memcpy(queue, view, plainBuf);
237 
238  alpaka::wait(queue);
239  }
