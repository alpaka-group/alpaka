113  //! Compares element-wise that all bytes are set to the same value.
114  struct VerifyBytesSetKernel
115  {
116  ALPAKA_NO_HOST_ACC_WARNING
117  template<typename TAcc, typename TIter>
118  ALPAKA_FN_ACC void operator()(
119  TAcc const& acc [[maybe_unused]], // used by SYCL back-end
120  bool* success,
121  TIter const& begin,
122  TIter const& end,
123  std::uint8_t const& byte) const
124  {
125  constexpr auto elemSizeInByte = sizeof(decltype(*begin));
126  for(auto it = begin; it != end; ++it)
127  {
128  auto const& elem = *it;
129  auto const pBytes = reinterpret_cast<std::uint8_t const*>(&elem);
130  for(std::size_t i = 0u; i < elemSizeInByte; ++i)
131  {
132  ALPAKA_CHECK(*success, pBytes[i] == byte);
133  }
134  }
135  }
136  };

200  //! Fills the given view with increasing values starting at 0.
201  template<typename TView, typename TQueue>
202  ALPAKA_FN_HOST auto iotaFillView(TQueue& queue, TView& view) -> void
203  {
204  using Elem = Elem<TView>;
205 
206  auto const platformHost = alpaka::PlatformCpu{};
207  auto const devHost = alpaka::getDevByIdx(platformHost, 0);
208 
209  auto const extent = getExtents(view);
210 
211  // Init buf with increasing values
212  std::vector<Elem> v(static_cast<std::size_t>(extent.prod()), static_cast<Elem>(0));
213  std::iota(std::begin(v), std::end(v), static_cast<Elem>(0));
214  auto plainBuf = createView(devHost, v, extent);
215 
216  // Copy the generated content into the given view.
217  memcpy(queue, view, plainBuf);
218 
219  wait(queue);
220  }
