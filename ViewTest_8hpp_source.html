118  //! Compares element-wise that all bytes are set to the same value.
119  struct VerifyBytesSetKernel
120  {
121  ALPAKA_NO_HOST_ACC_WARNING
122  template<typename TAcc, typename TIter>
123  ALPAKA_FN_ACC void operator()(
124  TAcc const& acc [[maybe_unused]], // used by SYCL back-end
125  bool* success,
126  TIter const& begin,
127  TIter const& end,
128  std::uint8_t const& byte) const
129  {
130  constexpr auto elemSizeInByte = sizeof(decltype(*begin));
131  for(auto it = begin; it != end; ++it)
132  {
133  auto const& elem = *it;
134  auto const pBytes = reinterpret_cast<std::uint8_t const*>(&elem);
135  for(std::size_t i = 0u; i < elemSizeInByte; ++i)
136  {
137  ALPAKA_CHECK(*success, pBytes[i] == byte);
138  }
139  }
140  }
141  };

205  //! Fills the given view with increasing values starting at 0.
206  template<typename TView, typename TQueue>
207  ALPAKA_FN_HOST auto iotaFillView(TQueue& queue, TView& view) -> void
208  {
209  using DevHost = DevCpu;
210  using PltfHost = Pltf<DevHost>;
211 
212  using Elem = Elem<TView>;
213 
214  DevHost const devHost = getDevByIdx<PltfHost>(0);
215 
216  auto const extent = getExtentVec(view);
217 
218  // Init buf with increasing values
219  std::vector<Elem> v(static_cast<std::size_t>(extent.prod()), static_cast<Elem>(0));
220  std::iota(std::begin(v), std::end(v), static_cast<Elem>(0));
221  auto plainBuf = createView(devHost, v, extent);
222 
223  // Copy the generated content into the given view.
224  memcpy(queue, view, plainBuf);
225 
226  wait(queue);
227  }
