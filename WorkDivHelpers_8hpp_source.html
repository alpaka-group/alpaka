35  //! Finds the largest divisor where divident % divisor == 0
36  //! \param dividend The dividend.
37  //! \param maxDivisor The maximum divisor.
38  //! \return The biggest number that satisfies the following conditions:
39  //! 1) dividend%ret==0
40  //! 2) ret<=maxDivisor
41  template<typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
42  ALPAKA_FN_HOST auto nextDivisorLowerOrEqual(T const& dividend, T const& maxDivisor) -> T
43  {
44  core::assertValueUnsigned(dividend);
45  core::assertValueUnsigned(maxDivisor);
46  ALPAKA_ASSERT(dividend >= maxDivisor);
47 
48  T divisor = maxDivisor;
49  while(dividend % divisor != 0)
50  --divisor;
51  return divisor;
52  }
53  //! \param val The value to find divisors of.
54  //! \param maxDivisor The maximum.
55  //! \return A list of all divisors less then or equal to the given maximum.
56  template<typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
57  ALPAKA_FN_HOST auto allDivisorsLessOrEqual(T const& val, T const& maxDivisor) -> std::set<T>
58  {
59  std::set<T> divisorSet;
60 
61  core::assertValueUnsigned(val);
62  core::assertValueUnsigned(maxDivisor);
63  ALPAKA_ASSERT(maxDivisor <= val);
64 
65  for(T i(1); i <= std::min(val, maxDivisor); ++i)
66  {
67  if(val % i == 0)
68  {
69  divisorSet.insert(static_cast<T>(val / i));
70  }
71  }
72 
73  return divisorSet;
74  }

77  //! \tparam TDim The dimensionality of the accelerator device properties.
78  //! \tparam TIdx The idx type of the accelerator device properties.
79  //! \param accDevProps The maxima for the work division.
80  //! \return If the accelerator device properties are valid.
81  template<typename TDim, typename TIdx>
82  ALPAKA_FN_HOST auto isValidAccDevProps(AccDevProps<TDim, TIdx> const& accDevProps) -> bool
83  {
84  // Check that the maximum counts are greater or equal 1.
85  if((accDevProps.m_gridBlockCountMax < 1) || (accDevProps.m_blockThreadCountMax < 1)
86  || (accDevProps.m_threadElemCountMax < 1))
87  {
88  return false;
89  }
90 
91  // Store the maxima allowed for extents of grid, blocks and threads.
92  auto const gridBlockExtentMax = subVecEnd<TDim>(accDevProps.m_gridBlockExtentMax);
93  auto const blockThreadExtentMax = subVecEnd<TDim>(accDevProps.m_blockThreadExtentMax);
94  auto const threadElemExtentMax = subVecEnd<TDim>(accDevProps.m_threadElemExtentMax);
95 
96  // Check that the extents for all dimensions are correct.
97  for(typename TDim::value_type i(0); i < TDim::value; ++i)
98  {
99  // Check that the maximum extents are greater or equal 1.
100  if((gridBlockExtentMax[i] < 1) || (blockThreadExtentMax[i] < 1) || (threadElemExtentMax[i] < 1))
101  {
102  return false;
103  }
104  }
105 
106  return true;
107  }

109  //! Subdivides the given grid thread extent into blocks restricted by the maxima allowed.
110  //! 1. The the maxima block, thread and element extent and counts
111  //! 2. The requirement of the block thread extent to divide the grid thread extent without remainder
112  //! 3. The requirement of the block extent.
113  //!
114  //! \param gridElemExtent
115  //! The full extent of elements in the grid.
116  //! \param threadElemExtent
117  //! the number of elements computed per thread.
118  //! \param accDevProps
119  //! The maxima for the work division.
120  //! \param blockThreadMustDivideGridThreadExtent
121  //! If this is true, the grid thread extent will be multiples of the corresponding block thread extent.
122  //! NOTE: If this is true and gridThreadExtent is prime (or otherwise bad chosen) in a dimension, the block
123  //! thread extent will be one in this dimension.
124  //! \param gridBlockExtentSubDivRestrictions
125  //! The grid block extent subdivision restrictions.
126  template<typename TDim, typename TIdx>
127  ALPAKA_FN_HOST auto subDivideGridElems(
128  Vec<TDim, TIdx> const& gridElemExtent,
129  Vec<TDim, TIdx> const& threadElemExtent,
130  AccDevProps<TDim, TIdx> const& accDevProps,
131  bool blockThreadMustDivideGridThreadExtent = true,
132  GridBlockExtentSubDivRestrictions gridBlockExtentSubDivRestrictions
133  = GridBlockExtentSubDivRestrictions::Unrestricted) -> WorkDivMembers<TDim, TIdx>
134  {
135  using Vec = Vec<TDim, TIdx>;
136  using DimLoopInd = typename TDim::value_type;
137 
138  for(DimLoopInd i(0); i < TDim::value; ++i)
139  {
140  ALPAKA_ASSERT(gridElemExtent[i] >= 1);
141  ALPAKA_ASSERT(threadElemExtent[i] >= 1);
142  ALPAKA_ASSERT(threadElemExtent[i] <= accDevProps.m_threadElemExtentMax[i]);
143  }
144  ALPAKA_ASSERT(threadElemExtent.prod() <= accDevProps.m_threadElemCountMax);
145  ALPAKA_ASSERT(isValidAccDevProps(accDevProps));
146 
147  // Handle threadElemExtent and compute gridThreadExtent. Afterwards, only the blockThreadExtent has to be
148  // optimized.
149  auto const clippedThreadElemExtent = elementwise_min(threadElemExtent, gridElemExtent);
150  auto const gridThreadExtent = [&]
151  {
152  Vec r;
153  for(DimLoopInd i(0u); i < TDim::value; ++i)
154  r[i] = core::divCeil(gridElemExtent[i], clippedThreadElemExtent[i]);
155  return r;
156  }();
157 
158  ///////////////////////////////////////////////////////////////////
159  // Try to calculate an optimal blockThreadExtent.
160 
161  // Restrict the max block thread extent from the maximum possible to the grid thread extent.
162  // This removes dimensions not required in the grid thread extent.
163  // This has to be done before the blockThreadCountMax clipping to get the maximum correctly.
164  auto blockThreadExtent = elementwise_min(accDevProps.m_blockThreadExtentMax, gridThreadExtent);
165 
166  // For equal block thread extent, restrict it to its minimum component.
167  // For example (512, 256, 1024) will get (256, 256, 256).
168  if(gridBlockExtentSubDivRestrictions == GridBlockExtentSubDivRestrictions::EqualExtent)
169  blockThreadExtent = Vec::all(blockThreadExtent.min());
170 
171  // Make the blockThreadExtent product smaller or equal to the accelerator's limit.
172  auto const& blockThreadCountMax = accDevProps.m_blockThreadCountMax;
173  if(blockThreadExtent.prod() > blockThreadCountMax)
174  {
175  switch(gridBlockExtentSubDivRestrictions)
176  {
177  case GridBlockExtentSubDivRestrictions::EqualExtent:
178  blockThreadExtent = Vec::all(core::nthRootFloor(blockThreadCountMax, TIdx{TDim::value}));
179  break;
180  case GridBlockExtentSubDivRestrictions::CloseToEqualExtent:
181  // Very primitive clipping. Just halve the largest value until it fits.
182  while(blockThreadExtent.prod() > blockThreadCountMax)
183  blockThreadExtent[blockThreadExtent.maxElem()] /= TIdx{2};
184  break;
185  case GridBlockExtentSubDivRestrictions::Unrestricted:
186  // Very primitive clipping. Just halve the smallest value (which is not 1) until it fits.
187  while(blockThreadExtent.prod() > blockThreadCountMax)
188  {
189  auto const it = std::min_element(
190  blockThreadExtent.begin(),
191  blockThreadExtent.end() - 1, //! \todo why omit the last element?
192  [](TIdx const& a, TIdx const& b)
193  {
194  if(a == TIdx{1})
195  return false;
196  if(b == TIdx{1})
197  return true;
198  return a < b;
199  });
200  *it /= TIdx{2};
201  }
202  break;
203  }
204  }
205 
206  // Make the block thread extent divide the grid thread extent.
207  if(blockThreadMustDivideGridThreadExtent)
208  {
209  switch(gridBlockExtentSubDivRestrictions)
210  {
211  case GridBlockExtentSubDivRestrictions::EqualExtent:
212  {
213  // For equal size block extent we have to compute the gcd of all grid thread extent that is less
214  // then the current maximal block thread extent. For this we compute the divisors of all grid
215  // thread extent less then the current maximal block thread extent.
216  std::array<std::set<TIdx>, TDim::value> gridThreadExtentDivisors;
217  for(DimLoopInd i(0u); i < TDim::value; ++i)
218  {
219  gridThreadExtentDivisors[i]
220  = detail::allDivisorsLessOrEqual(gridThreadExtent[i], blockThreadExtent[i]);
221  }
222  // The maximal common divisor of all block thread extent is the optimal solution.
223  std::set<TIdx> intersects[2u];
224  for(DimLoopInd i(1u); i < TDim::value; ++i)
225  {
226  intersects[(i - 1u) % 2u] = gridThreadExtentDivisors[0];
227  intersects[(i) % 2u].clear();
228  set_intersection(
229  std::begin(intersects[(i - 1u) % 2u]),
230  std::end(intersects[(i - 1u) % 2u]),
231  std::begin(gridThreadExtentDivisors[i]),
232  std::end(gridThreadExtentDivisors[i]),
233  std::inserter(intersects[i % 2], std::begin(intersects[i % 2u])));
234  }
235  TIdx const maxCommonDivisor = *(--std::end(intersects[(TDim::value - 1) % 2u]));
236  blockThreadExtent = Vec::all(maxCommonDivisor);
237  break;
238  }
239  case GridBlockExtentSubDivRestrictions::CloseToEqualExtent:
240  [[fallthrough]];
241  case GridBlockExtentSubDivRestrictions::Unrestricted:
242  for(DimLoopInd i(0u); i < TDim::value; ++i)
243  blockThreadExtent[i] = detail::nextDivisorLowerOrEqual(gridThreadExtent[i], blockThreadExtent[i]);
244  break;
245  }
246  }
247 
248  // grid blocks extent = grid thread / block thread extent. quotient is rounded up.
249  auto const gridBlockExtent = [&]
250  {
251  Vec r;
252  for(DimLoopInd i = 0; i < TDim::value; ++i)
253  r[i] = core::divCeil(gridThreadExtent[i], blockThreadExtent[i]);
254  return r;
255  }();
256 
257  return WorkDivMembers<TDim, TIdx>(gridBlockExtent, blockThreadExtent, clippedThreadElemExtent);
258  }

260  //! \tparam TAcc The accelerator for which this work division has to be valid.
261  //! \tparam TGridElemExtent The type of the grid element extent.
262  //! \tparam TThreadElemExtent The type of the thread element extent.
263  //! \tparam TDev The type of the device.
264  //! \param dev
265  //! The device the work division should be valid for.
266  //! \param gridElemExtent
267  //! The full extent of elements in the grid.
268  //! \param threadElemExtents
269  //! the number of elements computed per thread.
270  //! \param blockThreadMustDivideGridThreadExtent
271  //! If this is true, the grid thread extent will be multiples of the corresponding block thread extent.
272  //! NOTE: If this is true and gridThreadExtent is prime (or otherwise bad chosen) in a dimension, the block
273  //! thread extent will be one in this dimension.
274  //! \param gridBlockExtentSubDivRestrictions
275  //! The grid block extent subdivision restrictions.
276  //! \return The work division.
277  template<
278  typename TAcc,
279  typename TDev,
280  typename TGridElemExtent = Vec<Dim<TAcc>, Idx<TAcc>>,
281  typename TThreadElemExtent = Vec<Dim<TAcc>, Idx<TAcc>>>
282  ALPAKA_FN_HOST auto getValidWorkDiv(
283  [[maybe_unused]] TDev const& dev,
284  [[maybe_unused]] TGridElemExtent const& gridElemExtent = Vec<Dim<TAcc>, Idx<TAcc>>::ones(),
285  [[maybe_unused]] TThreadElemExtent const& threadElemExtents = Vec<Dim<TAcc>, Idx<TAcc>>::ones(),
286  [[maybe_unused]] bool blockThreadMustDivideGridThreadExtent = true,
287  [[maybe_unused]] GridBlockExtentSubDivRestrictions gridBlockExtentSubDivRestrictions
288  = GridBlockExtentSubDivRestrictions::Unrestricted)
289  -> WorkDivMembers<Dim<TGridElemExtent>, Idx<TGridElemExtent>>
290  {
291  static_assert(
292  Dim<TGridElemExtent>::value == Dim<TAcc>::value,
293  "The dimension of TAcc and the dimension of TGridElemExtent have to be identical!");
294  static_assert(
295  Dim<TThreadElemExtent>::value == Dim<TAcc>::value,
296  "The dimension of TAcc and the dimension of TThreadElemExtent have to be identical!");
297  static_assert(
298  std::is_same_v<Idx<TGridElemExtent>, Idx<TAcc>>,
299  "The idx type of TAcc and the idx type of TGridElemExtent have to be identical!");
300  static_assert(
301  std::is_same_v<Idx<TThreadElemExtent>, Idx<TAcc>>,
302  "The idx type of TAcc and the idx type of TThreadElemExtent have to be identical!");
303 
304  if constexpr(Dim<TGridElemExtent>::value == 0)
305  {
306  auto const zero = Vec<DimInt<0>, Idx<TAcc>>{};
307  ALPAKA_ASSERT(gridElemExtent == zero);
308  ALPAKA_ASSERT(threadElemExtents == zero);
309  return WorkDivMembers<DimInt<0>, Idx<TAcc>>{zero, zero, zero};
310  }
311  else
312  return subDivideGridElems(
313  getExtentVec(gridElemExtent),
314  getExtentVec(threadElemExtents),
315  getAccDevProps<TAcc>(dev),
316  blockThreadMustDivideGridThreadExtent,
317  gridBlockExtentSubDivRestrictions);
318  using V [[maybe_unused]] = Vec<Dim<TGridElemExtent>, Idx<TGridElemExtent>>;
319  ALPAKA_UNREACHABLE(WorkDivMembers<Dim<TGridElemExtent>, Idx<TGridElemExtent>>{V{}, V{}, V{}});
320  }

322  //! \tparam TDim The dimensionality of the accelerator device properties.
323  //! \tparam TIdx The idx type of the accelerator device properties.
324  //! \tparam TWorkDiv The type of the work division.
325  //! \param accDevProps The maxima for the work division.
326  //! \param workDiv The work division to test for validity.
327  //! \return If the work division is valid for the given accelerator device properties.
328  template<typename TDim, typename TIdx, typename TWorkDiv>
329  ALPAKA_FN_HOST auto isValidWorkDiv(AccDevProps<TDim, TIdx> const& accDevProps, TWorkDiv const& workDiv) -> bool
330  {
331  // Get the extents of grid, blocks and threads of the work division to check.
332  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(workDiv);
333  auto const blockThreadExtent = getWorkDiv<Block, Threads>(workDiv);
334  auto const threadElemExtent = getWorkDiv<Block, Threads>(workDiv);
335 
336  // Check that the maximal counts are satisfied.
337  if(accDevProps.m_gridBlockCountMax < gridBlockExtent.prod())
338  {
339  return false;
340  }
341  if(accDevProps.m_blockThreadCountMax < blockThreadExtent.prod())
342  {
343  return false;
344  }
345  if(accDevProps.m_threadElemCountMax < threadElemExtent.prod())
346  {
347  return false;
348  }
349 
350  // Check that the extents for all dimensions are correct.
351  if constexpr(Dim<TWorkDiv>::value > 0)
352  {
353  // Store the maxima allowed for extents of grid, blocks and threads.
354  auto const gridBlockExtentMax = subVecEnd<Dim<TWorkDiv>>(accDevProps.m_gridBlockExtentMax);
355  auto const blockThreadExtentMax = subVecEnd<Dim<TWorkDiv>>(accDevProps.m_blockThreadExtentMax);
356  auto const threadElemExtentMax = subVecEnd<Dim<TWorkDiv>>(accDevProps.m_threadElemExtentMax);
357 
358  for(typename Dim<TWorkDiv>::value_type i(0); i < Dim<TWorkDiv>::value; ++i)
359  {
360  // No extent is allowed to be zero or greater then the allowed maximum.
361  if((gridBlockExtent[i] < 1) || (blockThreadExtent[i] < 1) || (threadElemExtent[i] < 1)
362  || (gridBlockExtentMax[i] < gridBlockExtent[i]) || (blockThreadExtentMax[i] < blockThreadExtent[i])
363  || (threadElemExtentMax[i] < threadElemExtent[i]))
364  {
365  return false;
366  }
367  }
368  }
369 
370  return true;
371  }
372  //! \tparam TAcc The accelerator to test the validity on.
373  //! \param dev The device to test the work division for validity on.
374  //! \param workDiv The work division to test for validity.
375  //! \return If the work division is valid on this accelerator.
376  template<typename TAcc, typename TDev, typename TWorkDiv>
377  ALPAKA_FN_HOST auto isValidWorkDiv(TDev const& dev, TWorkDiv const& workDiv) -> bool
378  {
379  return isValidWorkDiv(getAccDevProps<TAcc>(dev), workDiv);
380  }
