156  //! The CPU blocking device queue enqueue trait specialization.
157  //! This default implementation for all tasks directly invokes the function call operator of the task.
158  template<typename TDev, typename TTask>
159  struct Enqueue<QueueGenericThreadsBlocking<TDev>, TTask>
160  {
161  //-----------------------------------------------------------------------------
162  ALPAKA_FN_HOST static auto enqueue(QueueGenericThreadsBlocking<TDev>& queue, TTask const& task) -> void
163  {
164  std::lock_guard<std::mutex> lk(queue.m_spQueueImpl->m_mutex);
165 
166  queue.m_spQueueImpl->m_bCurrentlyExecutingTask = true;
167 
168  task();
169 
170  queue.m_spQueueImpl->m_bCurrentlyExecutingTask = false;
171  }
172  };

178  //-----------------------------------------------------------------------------
179  ALPAKA_FN_HOST static auto empty(QueueGenericThreadsBlocking<TDev> const& queue) -> bool
