161  //! ConcurrentExecPool using yield or a condition variable to wait for new work.
162  //!
163  //! \tparam TConcurrentExec The type of concurrent executor (for example std::thread).
164  //! \tparam TPromise The promise type returned by the task.
165  //! \tparam TYield The type is required to have a static method "void yield()" to yield the current thread
166  //! if there is no work.
167  //! \tparam TMutex The mutex type used for locking threads.
168  //! \tparam TCondVar The condition variable type used to make the threads wait if there is no work.
169  //! \tparam TisYielding Boolean value if the threads should yield instead of wait for a condition variable.
170  template<
171  typename TIdx,
172  typename TConcurrentExec,
173  template<typename TFnObjReturn>
174  typename TPromise,
175  typename TYield,
176  typename TMutex = void,
177  typename TCondVar = void,
178  bool TisYielding = true>
179  struct ConcurrentExecPool final
180  : std::conditional_t<TisYielding, Empty, ConcurrentExecPoolMutexAndCond<TMutex, TCondVar>>
181  {
182  //! Creates a concurrent executors pool with a specific number of concurrent executors and a maximum
183  //! number of queued tasks.
184  //!
185  //! \param concurrentExecutionCount
186  //! The guaranteed number of concurrent executors used in the pool.
187  //! This is also the maximum number of tasks worked on concurrently.
188  ConcurrentExecPool(TIdx concurrentExecutionCount)
189  {
190  if(concurrentExecutionCount < 1)
191  {
192  throw std::invalid_argument(
193  "The argument 'concurrentExecutionCount' has to be greate or equal to one!");
194  }
195 
196  m_vConcurrentExecs.reserve(static_cast<std::size_t>(concurrentExecutionCount));
197 
198  // Create all concurrent executors.
199  for(TIdx i = 0; i < concurrentExecutionCount; ++i)
200  m_vConcurrentExecs.emplace_back([this]() { concurrentExecFn(); });
201  }
202 
203  ConcurrentExecPool(ConcurrentExecPool const&) = delete;
204  auto operator=(ConcurrentExecPool const&) -> ConcurrentExecPool& = delete;
205 
206  //! Completes any currently running task normally.
207  //! Signals a std::runtime_error exception to any other tasks that was not able to run.
208  ~ConcurrentExecPool()
209  {
210  // Signal that concurrent executors should not perform any new work
211  if constexpr(TisYielding)
212  m_bShutdownFlag.store(true);
213  else
214  {
215  {
216  std::unique_lock<TMutex> lock(this->m_mtxWakeup);
217  m_bShutdownFlag = true;
218  }
219  this->m_cvWakeup.notify_all();
220  }
221 
222  joinAllConcurrentExecs();
223 
224  // Signal to each incomplete task that it will not complete due to pool destruction.
225  while(auto task = popTask())
226  {
227  auto const except
228  = std::runtime_error("Could not perform task before ConcurrentExecPool destruction");
229  task->setException(std::make_exception_ptr(except));
230  }
231  }
232 
233  //! Runs the given function on one of the pool in First In First Out (FIFO) order.
234  //!
235  //! \tparam TFnObj The function type.
236  //! \param task Function object to be called on the pool.
237  //! Takes an arbitrary number of arguments and arbitrary return type.
238  //! \tparam TArgs The argument types pack.
239  //! \param args Arguments for task, cannot be moved.
240  //! If such parameters must be used, use a lambda and capture via move then move the
241  //! lambda.
242  //!
243  //! \return Signals when the task has completed with either success or an exception.
244  //! Also results in an exception if the pool is destroyed before execution has begun.
245  template<typename TFnObj, typename... TArgs>
246  auto enqueueTask(TFnObj&& task, TArgs&&... args)
247  {
248  auto boundTask = [=]() { return task(args...); };
249  auto decrementNumActiveTasks = [this]() { --m_numActiveTasks; };
250 
251  auto extendedTask = [boundTask, decrementNumActiveTasks]()
252  { return invokeBothReturnFirst(std::move(boundTask), std::move(decrementNumActiveTasks)); };
253 
254  using TaskPackage = TaskPkg<TPromise, decltype(extendedTask)>;
255  auto pTaskPackage = new TaskPackage(std::move(extendedTask));
256  std::shared_ptr<ITaskPkg> upTaskPackage(pTaskPackage);
257 
258  auto future = pTaskPackage->m_Promise.get_future();
259 
260  ++m_numActiveTasks;
261  if constexpr(TisYielding)
262  m_qTasks.push(std::move(upTaskPackage));
263  else
264  {
265  {
266  std::lock_guard<TMutex> lock(this->m_mtxWakeup);
267  m_qTasks.push(std::move(upTaskPackage));
268  }
269 
270  this->m_cvWakeup.notify_one();
271  }
272 
273  return future;
274  }

277  [[nodiscard]] auto getConcurrentExecutionCount() const -> TIdx

279  return std::size(m_vConcurrentExecs);

283  [[nodiscard]] auto isIdle() const -> bool

288  void detach(std::shared_ptr<ConcurrentExecPool>&& self)
289  {
290  m_self = std::move(self);
291  if constexpr(TisYielding)
292  m_bDetachedFlag = true;
293  else
294  {
295  std::lock_guard<TMutex> lock(this->m_mtxWakeup);
296  m_bDetachedFlag = true;
297  // we need to notify during the lock, because setting m_bDetachedFlag to true, allows another
298  // thread to delete this and thus destroy m_cvWakeup.
299  this->m_cvWakeup.notify_one();
300  }
301  }
302 
303  auto takeDetachHandle() -> std::shared_ptr<ConcurrentExecPool>
304  {
305  if(m_bDetachedFlag.exchange(false))
306  return std::move(m_self);
307  else
308  return nullptr;
309  }
310 
311  private:
312  //! The function the concurrent executors are executing.
313  void concurrentExecFn()
314  {
315  // Checks whether pool is being destroyed, if so, stop running.
316  while(!m_bShutdownFlag.load(std::memory_order_relaxed))
317  {
318  if constexpr(TisYielding)
319  {
320  if(auto task = popTask())
321  task->runTask();
322  else
323  {
324  if(takeDetachHandle())
325  return; // Pool was detached and is idle, stop and delete
326  TYield::yield();
327  }
328  }
329  else
330  {
331  if(auto task = popTask())
332  task->runTask();
333 
334  std::unique_lock<TMutex> lock(this->m_mtxWakeup);
335  if(m_qTasks.empty())
336  {
337  auto self = takeDetachHandle();
338  if(self)
339  {
340  // Pool was detached and is idle, stop and delete
341  lock.unlock(); // TODO(bgruber): I guess we unlock here so the mutex is not locked when
342  // the dtor of self runs, which also tries to lock?
343  return;
344  }
345 
346  // If the shutdown flag has been set since the last check, return now.
347  if(m_bShutdownFlag)
348  return;
349 
350  this->m_cvWakeup.wait(
351  lock,
352  [this] { return !m_qTasks.empty() || m_bShutdownFlag || m_bDetachedFlag; });
353  }
354  }
355  }
356  }
357 
358  //! Joins all concurrent executors.
359  void joinAllConcurrentExecs()
360  {
361  for(auto&& concurrentExec : m_vConcurrentExecs)
362  {
363  if(isThisThread(concurrentExec))
364  concurrentExec.detach();
365  else
366  concurrentExec.join();
367  }
368  }
369 
370  //! Pops a task from the queue.
371  auto popTask() -> std::shared_ptr<ITaskPkg>
372  {
373  std::shared_ptr<ITaskPkg> out;
374  if(m_qTasks.pop(out))
375  return out;
376  else
377  return nullptr;
378  }
379 
380  private:
381  std::vector<TConcurrentExec> m_vConcurrentExecs;
382  ThreadSafeQueue<std::shared_ptr<ITaskPkg>> m_qTasks;
383  std::atomic<std::uint32_t> m_numActiveTasks = 0u;
384  std::atomic<bool> m_bShutdownFlag = false;
385  std::atomic<bool> m_bDetachedFlag = false;
386  std::shared_ptr<ConcurrentExecPool> m_self = nullptr;
387  };
