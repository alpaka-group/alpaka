54  {
55  std::lock_guard<std::mutex> lk(m_mutex);
56

175  //! ConcurrentExecPool using yield or a condition variable to wait for new work.
176  //!
177  //! \tparam TConcurrentExec The type of concurrent executor (for example std::thread).
178  //! \tparam TPromise The promise type returned by the task.
179  //! \tparam TYield The type is required to have a static method "void yield()" to yield the current thread
180  //! if there is no work.
181  //! \tparam TMutex The mutex type used for locking threads.
182  //! \tparam TCondVar The condition variable type used to make the threads wait if there is no work.
183  //! \tparam TisYielding Boolean value if the threads should yield instead of wait for a condition variable.
184  template<
185  typename TIdx,
186  typename TConcurrentExec,
187  template<typename TFnObjReturn>
188  typename TPromise,
189  typename TYield,
190  typename TMutex = void,
191  typename TCondVar = void,
192  bool TisYielding = true>
193  struct ConcurrentExecPool final
194  : std::conditional_t<TisYielding, Empty, ConcurrentExecPoolMutexAndCond<TMutex, TCondVar>>
195  {
196  //! Creates a concurrent executors pool with a specific number of concurrent executors and a maximum
197  //! number of queued tasks.
198  //!
199  //! \param concurrentExecutionCount
200  //! The guaranteed number of concurrent executors used in the pool.
201  //! This is also the maximum number of tasks worked on concurrently.
202  ConcurrentExecPool(TIdx concurrentExecutionCount)
203  {
204  if(concurrentExecutionCount < 1)
205  {
206  throw std::invalid_argument(
207  "The argument 'concurrentExecutionCount' has to be greate or equal to one!");
208  }
209 
210  m_vConcurrentExecs.reserve(static_cast<std::size_t>(concurrentExecutionCount));
211 
212  // Create all concurrent executors.
213  for(TIdx i = 0; i < concurrentExecutionCount; ++i)
214  m_vConcurrentExecs.emplace_back([this]() { concurrentExecFn(); });
215  }
216 
217  ConcurrentExecPool(ConcurrentExecPool const&) = delete;
218  auto operator=(ConcurrentExecPool const&) -> ConcurrentExecPool& = delete;
219 
220  //! Completes any currently running task normally.
221  //! Signals a std::runtime_error exception to any other tasks that was not able to run.
222  ~ConcurrentExecPool()
223  {
224  // Signal that concurrent executors should not perform any new work
225  if constexpr(TisYielding)
226  m_bShutdownFlag.store(true);
227  else
228  {
229  {
230  std::unique_lock<TMutex> lock(this->m_mtxWakeup);
231  m_bShutdownFlag = true;
232  }
233  this->m_cvWakeup.notify_all();
234  }
235 
236  joinAllConcurrentExecs();
237 
238  // Signal to each incomplete task that it will not complete due to pool destruction.
239  while(auto task = popTask())
240  {
241  auto const except
242  = std::runtime_error("Could not perform task before ConcurrentExecPool destruction");
243 // Workaround: Clang can not support this when natively compiling device code. See ConcurrentExecPool.hpp.
244 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
245  task->setException(std::make_exception_ptr(except));
246 #endif
247  }
248  }

250  //! Runs the given function on one of the pool in First In First Out (FIFO) order.
251  //!
252  //! \tparam TFnObj The function type.
253  //! \param task Function object to be called on the pool.
254  //! Takes an arbitrary number of arguments and arbitrary return type.
255  //! \tparam TArgs The argument types pack.
256  //! \param args Arguments for task, cannot be moved.
257  //! If such parameters must be used, use a lambda and capture via move then move the
258  //! lambda.
259  //!
260  //! \return Signals when the task has completed with either success or an exception.
261  //! Also results in an exception if the pool is destroyed before execution has begun.
262  template<typename TFnObj, typename... TArgs>
263  auto enqueueTask(TFnObj&& task, TArgs&&... args)
264  {
265  auto boundTask = [=]() { return task(args...); };
266  auto decrementNumActiveTasks = [this]() { --m_numActiveTasks; };
267 
268  auto extendedTask = [boundTask, decrementNumActiveTasks]()
269  { return invokeBothReturnFirst(std::move(boundTask), std::move(decrementNumActiveTasks)); };
270 
271  using TaskPackage = TaskPkg<TPromise, decltype(extendedTask)>;
272  auto pTaskPackage = new TaskPackage(std::move(extendedTask));
273  std::shared_ptr<ITaskPkg> upTaskPackage(pTaskPackage);
274 
275  auto future = pTaskPackage->m_Promise.get_future();
276 
277  ++m_numActiveTasks;
278  if constexpr(TisYielding)
279  m_qTasks.push(std::move(upTaskPackage));
280  else
281  {
282  {
283  std::lock_guard<TMutex> lock(this->m_mtxWakeup);
284  m_qTasks.push(std::move(upTaskPackage));
285  }
286 
287  this->m_cvWakeup.notify_one();
288  }
289 
290  return future;
291  }

305  void detach(std::shared_ptr<ConcurrentExecPool>&& self)
306  {
307  m_self = std::move(self);
308  if constexpr(TisYielding)
309  m_bDetachedFlag = true;
310  else
311  {
312  std::lock_guard<TMutex> lock(this->m_mtxWakeup);
313  m_bDetachedFlag = true;
314  // we need to notify during the lock, because setting m_bDetachedFlag to true, allows another
315  // thread to delete this and thus destroy m_cvWakeup.
316  this->m_cvWakeup.notify_one();
317  }
318  }
319 
320  auto takeDetachHandle() -> std::shared_ptr<ConcurrentExecPool>
321  {
322  if(m_bDetachedFlag.exchange(false))
323  return std::move(m_self);
324  else
325  return nullptr;
326  }
327 
328  private:
329  //! The function the concurrent executors are executing.
330  void concurrentExecFn()
331  {
332  // Checks whether pool is being destroyed, if so, stop running.
333  while(!m_bShutdownFlag.load(std::memory_order_relaxed))
334  {
335  if constexpr(TisYielding)
336  {
337  if(auto task = popTask())
338  task->runTask();
339  else
340  {
341  if(takeDetachHandle())
342  return; // Pool was detached and is idle, stop and delete
343  TYield::yield();
344  }
345  }
346  else
347  {
348  if(auto task = popTask())
349  task->runTask();
350 
351  std::unique_lock<TMutex> lock(this->m_mtxWakeup);
352  if(m_qTasks.empty())
353  {
354  auto self = takeDetachHandle();
355  if(self)
356  {
357  // Pool was detached and is idle, stop and delete
358  lock.unlock(); // TODO(bgruber): I guess we unlock here so the mutex is not locked when
359  // the dtor of self runs, which also tries to lock?
360  return;
361  }
362 
363  // If the shutdown flag has been set since the last check, return now.
364  if(m_bShutdownFlag)
365  return;
366 
367  this->m_cvWakeup.wait(
368  lock,
369  [this] { return !m_qTasks.empty() || m_bShutdownFlag || m_bDetachedFlag; });
370  }
371  }
372  }
373  }
374 
375  //! Joins all concurrent executors.
376  void joinAllConcurrentExecs()
377  {
378  for(auto&& concurrentExec : m_vConcurrentExecs)
379  {
380  if(isThisThread(concurrentExec))
381  concurrentExec.detach();
382  else
383  concurrentExec.join();
384  }
385  }

387  //! Pops a task from the queue.
388  auto popTask() -> std::shared_ptr<ITaskPkg>
389  {
390  std::shared_ptr<ITaskPkg> out;
391  if(m_qTasks.pop(out))
392  return out;
393  else
394  return nullptr;
395  }
396 
397  private:
398  std::vector<TConcurrentExec> m_vConcurrentExecs;
399  ThreadSafeQueue<std::shared_ptr<ITaskPkg>> m_qTasks;
400  std::atomic<std::uint32_t> m_numActiveTasks = 0u;
401  std::atomic<bool> m_bShutdownFlag = false;
402  std::atomic<bool> m_bDetachedFlag = false;
403  std::shared_ptr<ConcurrentExecPool> m_self = nullptr;
