48

212  //! ConcurrentExecPool using yield.
213  //!
214  //! \tparam TConcurrentExec The type of concurrent executor (for example std::thread).
215  //! \tparam TPromise The promise type returned by the task.
216  //! \tparam TYield The type is required to have a static method "void yield()" to yield the current thread
217  //! if there is no work. \tparam TMutex Unused. The mutex type used for locking threads. \tparam TCondVar
218  //! Unused. The condition variable type used to make the threads wait if there is no work. \tparam
219  //! TisYielding Boolean value if the threads should yield instead of wait for a condition variable.
220  template<
221  typename TIdx,
222  typename TConcurrentExec,
223  template<typename TFnObjReturn>

227  typename TCondVar = void,
228  bool TisYielding = true>

231  public:
232  //! Creates a concurrent executor pool with a specific number of concurrent executors and a maximum
233  //! number of queued tasks.
234  //!
235  //! \param concurrentExecutionCount
236  //! The guaranteed number of concurrent executors used in the pool.
237  //! This is also the maximum number of tasks worked on concurrently.
238  ConcurrentExecPool(TIdx concurrentExecutionCount)
239  : m_vConcurrentExecs()
240  , m_qTasks()
241  , m_numActiveTasks(0u)
242  , m_bShutdownFlag(false)
243  {
244  if(concurrentExecutionCount < 1)
245  {
246  throw std::invalid_argument(

248  }
249

251

253  for(TIdx concurrentExec(0u); concurrentExec < concurrentExecutionCount; ++concurrentExec)
254  {

284  //! Runs the given function on one of the pool in First In First Out (FIFO) order.
285  //!
286  //! \tparam TFnObj The function type.
287  //! \param task Function object to be called on the pool.
288  //! Takes an arbitrary number of arguments and arbitrary return type.
289  //! \tparam TArgs The argument types pack.
290  //! \param args Arguments for task, cannot be moved.
291  //! If such parameters must be used, use a lambda and capture via move then move the
292  //! lambda.
293  //!
294  //! \return Signals when the task has completed with either success or an exception.
295  //! Also results in an exception if the pool is destroyed before execution has begun.
296  template<typename TFnObj, typename... TArgs>
297  auto enqueueTask(TFnObj&& task, TArgs&&... args)
298  {
299  auto boundTask = [=]() { return task(args...); };
300  auto decrementNumActiveTasks = [this]() { --m_numActiveTasks; };
301 
302  auto extendedTask = [boundTask, decrementNumActiveTasks]()
303  { return invokeBothReturnFirst(std::move(boundTask), std::move(decrementNumActiveTasks)); };
304 
305  using TaskPackage = TaskPkg<TPromise, decltype(extendedTask)>;
306  auto pTaskPackage = new TaskPackage(std::move(extendedTask));
307  std::shared_ptr<ITaskPkg> upTaskPackage(pTaskPackage);
308 
309  auto future = pTaskPackage->m_Promise.get_future();
310 
311  ++m_numActiveTasks;
312  m_qTasks.push(std::move(upTaskPackage));
313 
314  return future;
315  }
316  //! \return The number of concurrent executors available.
317  auto getConcurrentExecutionCount() const -> TIdx
318  {
319  return std::size(m_vConcurrentExecs);
320  }
321  //! \return If the thread pool is idle.
322  [[nodiscard]] auto isIdle() const -> bool
323  {
324  return m_numActiveTasks == 0u;
325  }
326 
327  private:
328  //! The function the concurrent executors are executing.
329  void concurrentExecFn()
330  {
331  // Checks whether pool is being destroyed, if so, stop running.
332  while(!m_bShutdownFlag.load(std::memory_order_relaxed))
333  {
334  auto currentTaskPackage = std::shared_ptr<ITaskPkg>{nullptr};
335 
336  // Use popTask so we only ever have one reference to the ITaskPkg
337  if(popTask(currentTaskPackage))
338  {
339  currentTaskPackage->runTask();
340  }
341  else
342  {
343  TYield::yield();
344  }
345  }
346  }
347 
348  //! Joins all concurrent executors.
349  void joinAllConcurrentExecs()
350  {
351  for(auto&& concurrentExec : m_vConcurrentExecs)
352  {
353  concurrentExec.join();
354  }
355  }
356  //! Pops a task from the queue.

364  }
