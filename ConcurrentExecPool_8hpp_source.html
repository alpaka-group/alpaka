49

197  template<typename TFnObj0, typename TFnObj1>
198  auto invokeBothReturnFirst(TFnObj0&& fn0, TFnObj1&& fn1)
199  {
200  if constexpr(!std::is_same_v<void, decltype(std::declval<TFnObj0>()())>)
201  {
202  auto ret = fn0();
203  fn1();
204  return ret;
205  }

208  fn0();
209  fn1();

283  //! Runs the given function on one of the pool in First In First Out (FIFO) order.
284  //!
285  //! \tparam TFnObj The function type.
286  //! \param task Function object to be called on the pool.
287  //! Takes an arbitrary number of arguments and arbitrary return type.
288  //! \tparam TArgs The argument types pack.
289  //! \param args Arguments for task, cannot be moved.
290  //! If such parameters must be used, use a lambda and capture via move then move the
291  //! lambda.
292  //!
293  //! \return Signals when the task has completed with either success or an exception.
294  //! Also results in an exception if the pool is destroyed before execution has begun.
295  template<typename TFnObj, typename... TArgs>
296  auto enqueueTask(TFnObj&& task, TArgs&&... args)
297  {
298  auto boundTask = [=]() { return task(args...); };
299  auto decrementNumActiveTasks = [this]() { --m_numActiveTasks; };
300 
301  auto extendedTask = [boundTask, decrementNumActiveTasks]()
302  { return invokeBothReturnFirst(std::move(boundTask), std::move(decrementNumActiveTasks)); };
303 
304  using TaskPackage = TaskPkg<TPromise, decltype(extendedTask)>;
305  auto pTaskPackage = new TaskPackage(std::move(extendedTask));
306  std::shared_ptr<ITaskPkg> upTaskPackage(pTaskPackage);
307 
308  auto future = pTaskPackage->m_Promise.get_future();
309 
310  ++m_numActiveTasks;
311  m_qTasks.push(std::move(upTaskPackage));
312 
313  return future;
314  }
315  //! \return The number of concurrent executors available.
316  auto getConcurrentExecutionCount() const -> TIdx
317  {
318  return std::size(m_vConcurrentExecs);
319  }
320  //! \return If the thread pool is idle.
321  [[nodiscard]] auto isIdle() const -> bool
322  {
323  return m_numActiveTasks == 0u;
324  }

340  private:
341  //! The function the concurrent executors are executing.
342  void concurrentExecFn()
343  {
344  // Checks whether pool is being destroyed, if so, stop running.
345  while(!m_bShutdownFlag.load(std::memory_order_relaxed))
346  {
347  auto currentTaskPackage = std::shared_ptr<ITaskPkg>{nullptr};
348 
349  // Use popTask so we only ever have one reference to the ITaskPkg
350  if(popTask(currentTaskPackage))
351  {
352  currentTaskPackage->runTask();
353  }
354  else
355  {
356  auto self = takeDetachHandle();
357  if(self)
358  {
359  // Pool was detached and is idle, stop and delete
360  return;
361  }
362  TYield::yield();
363  }
364  }

542  private:
543  //! The function the concurrent executors are executing.
544  void concurrentExecFn()
545  {
546  // Checks whether pool is being destroyed, if so, stop running (lazy check without mutex).
547  while(!m_bShutdownFlag)
548  {
549  // enter anonymous scope to limit lifetime of `currentTaskPackage`: destroy before entering wait
550  {
551  auto currentTaskPackage = std::shared_ptr<ITaskPkg>{nullptr};
552 
553  // Use popTask so we only ever have one reference to the ITaskPkg
554  if(popTask(currentTaskPackage))
555  {
556  currentTaskPackage->runTask();
557  }
558  }
559  {
560  std::unique_lock<TMutex> lock(m_mtxWakeup);
561  if(m_qTasks.empty())
562  {
563  auto self = takeDetachHandle();
564  if(self)
565  {
566  // Pool was detached and is idle, stop and delete
567  lock.unlock();
568  return;
569  }
570 
571  // If the shutdown flag has been set since the last check, return now.
572  if(m_bShutdownFlag)
573  {
574  return;
575  }
576 
577  m_cvWakeup.wait(
578  lock,
579  [this]() { return ((!m_qTasks.empty()) || m_bShutdownFlag || m_bDetachedFlag); });
580  }
581  }
582  }
