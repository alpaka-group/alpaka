39  //! The CPU device event implementation.
40  class EventCpuImpl final : public concepts::Implements<wait::ConceptCurrentThreadWaitFor, EventCpuImpl>
41  {
42  public:
43  //-----------------------------------------------------------------------------
44  EventCpuImpl(
45  dev::DevCpu const & dev) noexcept :
46  m_dev(dev),
47  m_mutex(),
48  m_enqueueCount(0u),
49  m_LastReadyEnqueueCount(0u)
50  {}
51  //-----------------------------------------------------------------------------
52  EventCpuImpl(EventCpuImpl const &) = delete;
53  //-----------------------------------------------------------------------------
54  EventCpuImpl(EventCpuImpl &&) = delete;
55  //-----------------------------------------------------------------------------
56  auto operator=(EventCpuImpl const &) -> EventCpuImpl & = delete;
57  //-----------------------------------------------------------------------------
58  auto operator=(EventCpuImpl &&) -> EventCpuImpl & = delete;
59  //-----------------------------------------------------------------------------
60  ~EventCpuImpl() noexcept = default;
61 
62  //-----------------------------------------------------------------------------
63  auto isReady() noexcept -> bool
64  {
65  return (m_LastReadyEnqueueCount == m_enqueueCount);
66  }
67 
68  //-----------------------------------------------------------------------------
69  auto wait(std::size_t const & enqueueCount, std::unique_lock<std::mutex>& lk) const noexcept -> void
70  {
71  ALPAKA_ASSERT(enqueueCount <= m_enqueueCount);
72 
73  while(enqueueCount > m_LastReadyEnqueueCount)
74  {
75  auto future = m_future;
76  lk.unlock();
77  future.get();
78  lk.lock();
79  }
80  }
81 
82  public:
83  dev::DevCpu const m_dev; //!< The device this event is bound to.
84 
85  std::mutex mutable m_mutex; //!< The mutex used to synchronize access to the event.
86  std::shared_future<void> m_future; //!< The future signaling the event completion.
87  std::size_t m_enqueueCount; //!< The number of times this event has been enqueued.
88  std::size_t m_LastReadyEnqueueCount; //!< The time this event has been ready the last time.
89  //!< Ready means that the event was not waiting within a queue (not enqueued or already completed).
90  //!< If m_enqueueCount == m_LastReadyEnqueueCount, the event is currently not enqueued
91  };
92  }
93  }

185  //! The CPU non-blocking device queue enqueue trait specialization.
186  template<>
187  struct Enqueue<
188  queue::cpu::detail::QueueCpuNonBlockingImpl,
189  event::EventCpu>
190  {
191  //-----------------------------------------------------------------------------
192  ALPAKA_FN_HOST static auto enqueue(
193 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
194  queue::cpu::detail::QueueCpuNonBlockingImpl & queueImpl,
195 #else
196  queue::cpu::detail::QueueCpuNonBlockingImpl &,
197 #endif
198  event::EventCpu & event)
199  -> void
200  {
201  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
202 
203  // Copy the shared pointer of the event implementation.
204  // This is forwarded to the lambda that is enqueued into the queue to ensure that the event implementation is alive as long as it is enqueued.
205  auto spEventImpl(event.m_spEventImpl);
206 
207  // Setting the event state and enqueuing it has to be atomic.
208  std::lock_guard<std::mutex> lk(spEventImpl->m_mutex);
209 
210  ++spEventImpl->m_enqueueCount;
211 
212 // Workaround: Clang can not support this when natively compiling device code. See ConcurrentExecPool.hpp.
213 #if !(BOOST_COMP_CLANG_CUDA && BOOST_ARCH_PTX)
214  auto const enqueueCount = spEventImpl->m_enqueueCount;
215 
216  // Enqueue a task that only resets the events flag if it is completed.
217  spEventImpl->m_future = queueImpl.m_workerThread.enqueueTask(
218  [spEventImpl, enqueueCount]()
219  {
220  std::unique_lock<std::mutex> lk2(spEventImpl->m_mutex);
221 
222  // Nothing to do if it has been re-enqueued to a later position in the queue.
223  if(enqueueCount == spEventImpl->m_enqueueCount)
224  {
225  spEventImpl->m_LastReadyEnqueueCount = spEventImpl->m_enqueueCount;
226  }
227  });
228 #endif
229  }
230  };

278  eventImpl.m_LastReadyEnqueueCount = eventImpl.m_enqueueCount;
279 
280  eventImpl.m_future = promise.get_future();
281  }
282 
283  queueImpl.m_bCurrentlyExecutingTask = false;
284  }
