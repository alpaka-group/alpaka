131  struct GetAccDevProps<
132  TAcc<TDim, TIdx>,
133  std::enable_if_t<std::is_base_of_v<experimental::AccGenericSycl<TDim, TIdx>, TAcc<TDim, TIdx>>>>

135  static auto getAccDevProps(typename DevType<TAcc<TDim, TIdx>>::type const& dev) -> AccDevProps<TDim, TIdx>
136  {
137  auto const device = dev.m_impl->get_device();
138  auto max_threads_dim = device.template get_info<sycl::info::device::max_work_item_sizes>();
139  return {// m_multiProcessorCount
140  alpaka::core::clipCast<TIdx>(device.template get_info<sycl::info::device::max_compute_units>()),
141  // m_gridBlockExtentMax
142  getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
143  // WARNING: There is no SYCL way to determine these values
144  std::numeric_limits<TIdx>::max(),
145  std::numeric_limits<TIdx>::max(),
146  std::numeric_limits<TIdx>::max())),
147  // m_gridBlockCountMax
148  std::numeric_limits<TIdx>::max(),
149  // m_blockThreadExtentMax
150  getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
151  alpaka::core::clipCast<TIdx>(max_threads_dim[2u]),
152  alpaka::core::clipCast<TIdx>(max_threads_dim[1u]),
153  alpaka::core::clipCast<TIdx>(max_threads_dim[0u]))),
154  // m_blockThreadCountMax
155  alpaka::core::clipCast<TIdx>(device.template get_info<sycl::info::device::max_work_group_size>()),
156  // m_threadElemExtentMax

158  // m_threadElemCountMax
