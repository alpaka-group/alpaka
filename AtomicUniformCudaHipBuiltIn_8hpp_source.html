140  //-----------------------------------------------------------------------------
141  //! The GPU CUDA/HIP accelerator atomic operation.
142  template<typename THierarchy>
143  struct AtomicOp<AtomicAdd, AtomicUniformCudaHipBuiltIn, double, THierarchy>
144  {
145  //-----------------------------------------------------------------------------
146  __device__ static auto atomicOp(
147  AtomicUniformCudaHipBuiltIn const&,
148  double* const addr,
149  double const& value) -> double
150  {
151 # if BOOST_ARCH_PTX >= BOOST_VERSION_NUMBER(6, 0, 0)
152  return ::atomicAdd(addr, value);
153 # else
154  // Code from: http://docs.nvidia.com/cuda/cuda-c-programming-guide/#atomic-functions
155 
156  unsigned long long int* address_as_ull(reinterpret_cast<unsigned long long int*>(addr));
157  unsigned long long int old(*address_as_ull);
158  unsigned long long int assumed;
159  do
160  {
161  assumed = old;
162  old = ::atomicCAS(
163  address_as_ull,
164  assumed,
165  static_cast<unsigned long long>(
166  __double_as_longlong(value + __longlong_as_double(static_cast<long long>(assumed)))));
167  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
168  } while(assumed != old);
169  return __longlong_as_double(static_cast<long long>(old));
170 # endif
171  }
172  };
