175  //-----------------------------------------------------------------------------
176  //! The GPU CUDA/HIP accelerator atomic operation.
177  template<
178  typename THierarchy>
179  struct AtomicOp<
180  AtomicAdd,
181  AtomicUniformCudaHipBuiltIn,
182  double,
183  THierarchy>
184  {
185  //-----------------------------------------------------------------------------
186  __device__ static auto atomicOp(
187  AtomicUniformCudaHipBuiltIn const &,
188  double * const addr,
189  double const & value)
190  -> double
191  {
192 #if BOOST_ARCH_PTX >= BOOST_VERSION_NUMBER(6, 0, 0)
193  return ::atomicAdd(addr, value);
194 #else
195  // Code from: http://docs.nvidia.com/cuda/cuda-c-programming-guide/#atomic-functions
196 
197  unsigned long long int * address_as_ull(reinterpret_cast<unsigned long long int *>(addr));
198  unsigned long long int old(*address_as_ull);
199  unsigned long long int assumed;
200  do
201  {
202  assumed = old;
203  old = ::atomicCAS(
204  address_as_ull,
205  assumed,
206  static_cast<unsigned long long>(__double_as_longlong(value + __longlong_as_double(static_cast<long long>(assumed)))));
207  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
208  }
209  while(assumed != old);
210  return __longlong_as_double(static_cast<long long>(old));
211 #endif
212  }
213  };
