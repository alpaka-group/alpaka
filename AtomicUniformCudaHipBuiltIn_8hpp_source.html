121  //! The GPU CUDA/HIP accelerator atomic operation.
122  template<typename THierarchy>
123  struct AtomicOp<AtomicAdd, AtomicUniformCudaHipBuiltIn, double, THierarchy>
124  {
125  __device__ static auto atomicOp(
126  AtomicUniformCudaHipBuiltIn const&,
127  double* const addr,
128  double const& value) -> double
129  {
130 # if BOOST_ARCH_PTX >= BOOST_VERSION_NUMBER(6, 0, 0)
131  return ::atomicAdd(addr, value);
132 # else
133  // Code from: http://docs.nvidia.com/cuda/cuda-c-programming-guide/#atomic-functions
134 
135  unsigned long long int* address_as_ull(reinterpret_cast<unsigned long long int*>(addr));
136  unsigned long long int old(*address_as_ull);
137  unsigned long long int assumed;
138  do
139  {
140  assumed = old;
141  old = ::atomicCAS(
142  address_as_ull,
143  assumed,
144  static_cast<unsigned long long>(
145  __double_as_longlong(value + __longlong_as_double(static_cast<long long>(assumed)))));
146  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
147  } while(assumed != old);
148  return __longlong_as_double(static_cast<long long>(old));
149 # endif
150  }
151  };
