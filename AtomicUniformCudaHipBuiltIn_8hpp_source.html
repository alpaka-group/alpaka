118  //! The GPU CUDA/HIP accelerator atomic operation.
119  template<typename THierarchy>
120  struct AtomicOp<AtomicAdd, AtomicUniformCudaHipBuiltIn, double, THierarchy>
121  {
122  __device__ static auto atomicOp(
123  AtomicUniformCudaHipBuiltIn const&,
124  double* const addr,
125  double const& value) -> double
126  {
127 # if BOOST_ARCH_PTX >= BOOST_VERSION_NUMBER(6, 0, 0)
128  return ::atomicAdd(addr, value);
129 # else
130  // Code from: http://docs.nvidia.com/cuda/cuda-c-programming-guide/#atomic-functions
131 
132  unsigned long long int* address_as_ull(reinterpret_cast<unsigned long long int*>(addr));
133  unsigned long long int old(*address_as_ull);
134  unsigned long long int assumed;
135  do
136  {
137  assumed = old;
138  old = ::atomicCAS(
139  address_as_ull,
140  assumed,
141  static_cast<unsigned long long>(
142  __double_as_longlong(value + __longlong_as_double(static_cast<long long>(assumed)))));
143  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
144  } while(assumed != old);
145  return __longlong_as_double(static_cast<long long>(old));
146 # endif
147  }
148  };
