60  //! Emulate atomic
61  //
62  // The default implementation will emulate all atomic functions with atomicCAS.
63  template<
64  typename TOp,
65  typename TAtomic,
66  typename T,
67  typename THierarchy,
68  typename TSfinae = void,
69  typename TDefer = void>
70  struct EmulateAtomic : private EmulationBase
71  {
72  public:
73  __device__ static auto atomic(
74  alpaka::AtomicUniformCudaHipBuiltIn const& ctx,
75  T* const addr,
76  T const& value) -> T
77  {
78  auto* const addressAsIntegralType = reinterpretAddress(addr);
79  using EmulatedType = ALPAKA_DECAY_T(decltype(*addressAsIntegralType));
80 
81  // Emulating atomics with atomicCAS is mentioned in the programming guide too.
82  // http://docs.nvidia.com/cuda/cuda-c-programming-guide/#atomic-functions
83 # if BOOST_LANG_HIP
84 # if __has_builtin(__hip_atomic_load)
85  EmulatedType old{
86  __hip_atomic_load(addressAsIntegralType, __ATOMIC_RELAXED, __HIP_MEMORY_SCOPE_AGENT)};
87 # else
88  EmulatedType old{__atomic_load_n(addressAsIntegralType, __ATOMIC_RELAXED)};
89 # endif
90 # else
91  EmulatedType old{*addressAsIntegralType};
92 # endif
93  EmulatedType assumed;
94  do
95  {
96  assumed = old;
97  T v = *(reinterpret_cast<T*>(&assumed));
98  TOp{}(&v, value);
99  using Cas = alpaka::trait::
100  AtomicOp<alpaka::AtomicCas, alpaka::AtomicUniformCudaHipBuiltIn, EmulatedType, THierarchy>;
101  old = Cas::atomicOp(ctx, addressAsIntegralType, assumed, reinterpretValue(v));
102  // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN)
103  } while(assumed != old);
104  return *(reinterpret_cast<T*>(&old));
105  }
106  };

109  template<typename T, typename THierarchy>
110  struct EmulateAtomic<alpaka::AtomicCas, alpaka::AtomicUniformCudaHipBuiltIn, T, THierarchy>
111  : private EmulationBase
112  {
113  __device__ static auto atomic(
114  alpaka::AtomicUniformCudaHipBuiltIn const& ctx,
115  T* const addr,
116  T const& compare,
117  T const& value) -> T
118  {
119  auto* const addressAsIntegralType = reinterpretAddress(addr);
120  using EmulatedType = ALPAKA_DECAY_T(decltype(*addressAsIntegralType));
121  EmulatedType reinterpretedCompare = reinterpretValue(compare);
122  EmulatedType reinterpretedValue = reinterpretValue(value);
123 
124  auto old = alpaka::trait::
125  AtomicOp<alpaka::AtomicCas, alpaka::AtomicUniformCudaHipBuiltIn, EmulatedType, THierarchy>::
126  atomicOp(ctx, addressAsIntegralType, reinterpretedCompare, reinterpretedValue);
127 
128  return *(reinterpret_cast<T*>(&old));
129  }

241  //! Generic atomic implementation
242  //
243  // - unsigned long int will be redirected to unsigned long long int or unsigned int implementation depending if
244  // unsigned long int is a 64 or 32bit data type.
245  // - Atomics which are not available as builtin atomic will be emulated.
246  template<typename TOp, typename T, typename THierarchy>
247  struct AtomicOp<TOp, AtomicUniformCudaHipBuiltIn, T, THierarchy>
248  {
249  __device__ static auto atomicOp(
250  AtomicUniformCudaHipBuiltIn const& ctx,
251  [[maybe_unused]] T* const addr,
252  [[maybe_unused]] T const& value) -> T
253  {
254  static_assert(
255  sizeof(T) == 4u || sizeof(T) == 8u,
256  "atomicOp<TOp, AtomicUniformCudaHipBuiltIn, T>(atomic, addr, value) is not supported! Only 64 and "
257  "32bit atomics are supported.");
258 
259  if constexpr(::AlpakaBuiltInAtomic<TOp, T, THierarchy>::value)
260  return ::AlpakaBuiltInAtomic<TOp, T, THierarchy>::atomic(addr, value);
261 
262  else if constexpr(std::is_same_v<unsigned long int, T>)
263  {
264  if constexpr(sizeof(T) == 4u && ::AlpakaBuiltInAtomic<TOp, unsigned int, THierarchy>::value)
265  return ::AlpakaBuiltInAtomic<TOp, unsigned int, THierarchy>::atomic(
266  reinterpret_cast<unsigned int*>(addr),
267  static_cast<unsigned int>(value));
268  else if constexpr(
269  sizeof(T) == 8u
270  && ::AlpakaBuiltInAtomic<TOp, unsigned long long int, THierarchy>::value) // LP64
271  {
272  return ::AlpakaBuiltInAtomic<TOp, unsigned long long int, THierarchy>::atomic(
273  reinterpret_cast<unsigned long long int*>(addr),
274  static_cast<unsigned long long int>(value));
275  }
276  }
277 
278  return detail::EmulateAtomic<TOp, AtomicUniformCudaHipBuiltIn, T, THierarchy>::atomic(
279  ctx,
280  addr,
281  value);
282  }
283  };
