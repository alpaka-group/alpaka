205  private:
206  //-----------------------------------------------------------------------------
207  //! The function executed for each grid block.
208  //-----------------------------------------------------------------------------
209  ALPAKA_FN_HOST static auto gridBlockExecHost(
210  acc::AccCpuThreads<TDim, TSize> & acc,
211  Vec<TDim, TSize> const & gridBlockIdx,
212  Vec<TDim, TSize> const & blockThreadExtent,
213  ThreadPool & threadPool,
214  TKernelFnObj const & kernelFnObj,
215  TArgs const & ... args)
216  -> void
217  {
218  // The futures of the threads in the current block.
219  std::vector<std::future<void>> futuresInBlock;
220 
221  // Set the index of the current block
222  acc.m_gridBlockIdx = gridBlockIdx;
223 
224  // Bind the kernel and its arguments to the host block thread execution function.
225  auto boundBlockThreadExecHost(std::bind(
226  &ExecCpuThreads<TDim, TSize, TKernelFnObj, TArgs...>::blockThreadExecHost,
227  std::ref(acc),
228  std::ref(futuresInBlock),
229  std::placeholders::_1,
230  std::ref(threadPool),
231  std::ref(kernelFnObj),
232  std::ref(args)...));
233  // Execute the block threads in parallel.
234  core::ndLoopIncIdx(
235  blockThreadExtent,
236  boundBlockThreadExecHost);
237 
238  // Wait for the completion of the block thread kernels.
239  std::for_each(
240  futuresInBlock.begin(),
241  futuresInBlock.end(),
242  [](std::future<void> & t)
243  {
244  t.wait();
245  }
246  );
247  // Clean up.
248  futuresInBlock.clear();
249 
250  acc.m_threadToIndexMap.clear();
251  acc.m_threadToBarrierMap.clear();
252 
253  // After a block has been processed, the shared memory has to be deleted.
254  block::shared::freeMem(acc);
255  }
256  //-----------------------------------------------------------------------------
257  //! The function executed for each block thread on the host.
258  //-----------------------------------------------------------------------------
259  ALPAKA_FN_HOST static auto blockThreadExecHost(
260  acc::AccCpuThreads<TDim, TSize> & acc,
261  std::vector<std::future<void>> & futuresInBlock,
262  Vec<TDim, TSize> const & blockThreadIdx,
263  ThreadPool & threadPool,
264  TKernelFnObj const & kernelFnObj,
265  TArgs const & ... args)
266  -> void
267  {
268  // Bind the arguments to the accelerator block thread execution function.
269  // The blockThreadIdx is required to be copied in because the variable will get changed for the next iteration/thread.
270  auto boundBlockThreadExecAcc(
271  [&, blockThreadIdx]()
272  {
273  blockThreadExecAcc(
274  acc,
275  blockThreadIdx,
276  kernelFnObj,
277  args...);
278  });
279  // Add the bound function to the block thread pool.
280  futuresInBlock.emplace_back(
281  threadPool.enqueueTask(
282  boundBlockThreadExecAcc));
283  }
284  //-----------------------------------------------------------------------------
285  //! The thread entry point on the accelerator.
286  //-----------------------------------------------------------------------------
287  ALPAKA_FN_HOST static auto blockThreadExecAcc(
288  acc::AccCpuThreads<TDim, TSize> & acc,
289  Vec<TDim, TSize> const & blockThreadIdx,
290  TKernelFnObj const & kernelFnObj,
291  TArgs const & ... args)
292  -> void
293  {
294  // We have to store the thread data before the kernel is calling any of the methods of this class depending on them.
295  auto const threadId(std::this_thread::get_id());
296 
297  // Set the master thread id.
298  if(blockThreadIdx.sum() == 0)
299  {
300  acc.m_idMasterThread = threadId;
301  }
302 
303  // We can not use the default syncBlockThreads here because it searches inside m_threadToBarrierMap for the thread id.
304  // Concurrently searching while others use emplace is unsafe!
305  typename std::map<std::thread::id, TSize>::iterator itThreadToBarrier;
306 
307  {
308  // The insertion of elements has to be done one thread at a time.
309  std::lock_guard<std::mutex> lock(acc.m_mtxMapInsert);
310 
311  // Save the thread id, and index.
312  acc.m_threadToIndexMap.emplace(threadId, blockThreadIdx);
313  itThreadToBarrier = acc.m_threadToBarrierMap.emplace(threadId, 0).first;
314  }
315 
316  // Sync all threads so that the maps with thread id's are complete and not changed after here.
317  acc.syncBlockThreads(itThreadToBarrier);
318 
319  // Execute the kernel itself.
320  kernelFnObj(
321  const_cast<acc::AccCpuThreads<TDim, TSize> const &>(acc),
322  args...);
323 
324  // We have to sync all threads here because if a thread would finish before all threads have been started,
325  // a new thread could get the recycled (then duplicate) thread id!
326  acc.syncBlockThreads(itThreadToBarrier);
327  }
328 
329  TKernelFnObj m_kernelFnObj;
330  std::tuple<TArgs...> m_args;
331  };
