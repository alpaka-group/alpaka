81  template<typename TDim, typename TIdx>
82  struct SyncBlockThreads<CtxBlockOacc<TDim, TIdx>>
83  {
84  //! Execute op with single thread (any idx, last thread to
85  //! arrive at barrier executes) syncing before and after
86  template<typename TOp>
87  ALPAKA_FN_HOST static auto masterOpBlockThreads(CtxBlockOacc<TDim, TIdx> const& acc, TOp&& op) -> void
88  {
89  auto const slot = (acc.m_generation & 1) << 1;
90  int const workerNum = static_cast<int>(getWorkDiv<Block, Threads>(acc).prod());
91  int sum;
92  // Workaround to use an array in an atomic capture rather than
93  // using the data member m_syncCounter array directly.
94  // The change is semantically equivalent.
95  // However, this should work per the OpenACC standard, but appears to be compiler
96  // issue causing a runtime error. The error was seen the 20.7 release
97  // of the NVIDIA HPC Compiler but may be corrected in future releases.
98  int* m_syncCounter = acc.m_syncCounter;
99 # pragma acc atomic capture
100  {
101  ++m_syncCounter[slot];
102  sum = m_syncCounter[slot];
103  }
104  if(sum == workerNum)
105  {
106  ++acc.m_generation;
107  int const nextSlot = (acc.m_generation & 1) << 1;
108  m_syncCounter[nextSlot] = 0;
109  m_syncCounter[nextSlot + 1] = 0;
110  op();
111  }
112  while(sum < workerNum)
113  {
114 # pragma acc atomic read
115  sum = m_syncCounter[slot];
116  }
117 # pragma acc atomic capture
118  {
119  ++m_syncCounter[slot + 1];
120  sum = m_syncCounter[slot + 1];
121  }
122  while(sum < workerNum)
123  {
124 # pragma acc atomic read
125  sum = m_syncCounter[slot + 1];
126  }
127  }
128 
129  ALPAKA_FN_HOST static auto syncBlockThreads(CtxBlockOacc<TDim, TIdx> const& acc) -> void
130  {
131  masterOpBlockThreads<>(acc, []() {});
132  }
133  };

174  ALPAKA_NO_HOST_ACC_WARNING

176  CtxBlockOacc<TDim, TIdx> const& blockSync,
177  int predicate) -> int
178  {
179  // implicit snyc
180  SyncBlockThreads<CtxBlockOacc<TDim, TIdx>>::masterOpBlockThreads(
181  blockSync,

183

187  oacc::detail::AtomicOp<TOp>()(result, predicateBool);
