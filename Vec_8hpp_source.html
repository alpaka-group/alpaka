154  //! A n-dimensional vector.
155  //#############################################################################
156  template<
157  typename TDim,
158  typename TSize>
159  class Vec final
160  {
161  public:
162  using Dim = TDim;
163  static constexpr auto s_uiDim = TDim::value;
164  using Val = TSize;
165 
166  private:
167  //! A sequence of integers from 0 to dim-1.
168  //! This can be used to write compile time indexing algorithms.
169  using IdxSequence = meta::MakeIntegerSequence<std::size_t, TDim::value>;
170 
171  public:
172  //-----------------------------------------------------------------------------
173  // The default constructor is only available when the vector is zero-dimensional.
174  //-----------------------------------------------------------------------------
175  ALPAKA_NO_HOST_ACC_WARNING
176  template<
177  bool B = (TDim::value == 0u),
178  typename = typename std::enable_if<B>::type>
179  ALPAKA_FN_HOST_ACC Vec() :
180  m_data{static_cast<TSize>(0u)}
181  {}
182 
183 
184  //-----------------------------------------------------------------------------
185  //! Value constructor.
186  //! This constructor is only available if the number of parameters matches the vector size.
187  //-----------------------------------------------------------------------------
188  ALPAKA_NO_HOST_ACC_WARNING
189  template<
190  typename TArg0,
191  typename... TArgs,
192  typename = typename std::enable_if<
193  // There have to be dim arguments.
194  (sizeof...(TArgs)+1 == TDim::value)
195  &&
196  (std::is_same<TSize, typename std::decay<TArg0>::type>::value)
197  >::type>
198  ALPAKA_FN_HOST_ACC Vec(
199  TArg0 && arg0,
200  TArgs && ... args) :
201  m_data{std::forward<TArg0>(arg0), std::forward<TArgs>(args)...}
202  {}

392  //-----------------------------------------------------------------------------
393  // Equality comparison operator.
394  //-----------------------------------------------------------------------------
395  ALPAKA_NO_HOST_ACC_WARNING
396  ALPAKA_FN_HOST_ACC auto operator==(
397  Vec const & rhs) const
398  -> bool
399  {
400  for(typename TDim::value_type i(0); i < TDim::value; ++i)
401  {
402  if((*this)[i] != rhs[i])
403  {
404  return false;
405  }
406  }
407  return true;
408  }
409  //-----------------------------------------------------------------------------
410  // Inequality comparison operator.
411  //-----------------------------------------------------------------------------
412  ALPAKA_NO_HOST_ACC_WARNING
413  ALPAKA_FN_HOST_ACC auto operator!=(
414  Vec const & rhs) const
415  -> bool
416  {
417  return !((*this) == rhs);
418  }

499  //! \return The min of all values.
500  //-----------------------------------------------------------------------------
501  ALPAKA_NO_HOST_ACC_WARNING
502  ALPAKA_FN_HOST_ACC auto min() const

505  return foldrAll(

512  //! \return The max of all values.

520  {

522  });
523  }
