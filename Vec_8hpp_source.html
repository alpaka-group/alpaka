38  //! Single value constructor helper.
39  ALPAKA_NO_HOST_ACC_WARNING
40  template<
41  typename TDim,
42  template<std::size_t>
43  class TTFnObj,
44  typename... TArgs,
45  typename TIdxSize,
46  TIdxSize... TIndices>
47  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFnArbitrary(
48  std::integer_sequence<TIdxSize, TIndices...> const& indices,
49  TArgs&&... args)
50  {
51  alpaka::ignore_unused(indices);
52 
53  return Vec<TDim, decltype(TTFnObj<0>::create(std::forward<TArgs>(args)...))>(
54  (TTFnObj<TIndices>::create(std::forward<TArgs>(args)...))...);
55  }
56  //! Creator using func<idx>(args...) to initialize all values of the vector.
57  //! The idx is in the range [0, TDim].
58  ALPAKA_NO_HOST_ACC_WARNING
59  template<typename TDim, template<std::size_t> class TTFnObj, typename... TArgs>
60  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFn(TArgs&&... args)
61  {
62  using IdxSequence = std::make_integer_sequence<typename TDim::value_type, TDim::value>;
63  return createVecFromIndexedFnArbitrary<TDim, TTFnObj>(IdxSequence(), std::forward<TArgs>(args)...);
64  }
65  //! Creator using func<idx>(args...) to initialize all values of the vector.
66  //! The idx is in the range [TIdxOffset, TIdxOffset + TDim].
67  ALPAKA_NO_HOST_ACC_WARNING
68  template<typename TDim, template<std::size_t> class TTFnObj, typename TIdxOffset, typename... TArgs>
69  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFnOffset(TArgs&&... args)
70  {
71  using IdxSubSequenceSigned = meta::MakeIntegerSequenceOffset<std::intmax_t, TIdxOffset::value, TDim::value>;
72  using IdxSubSequence = meta::ConvertIntegerSequence<typename TIdxOffset::value_type, IdxSubSequenceSigned>;
73  return createVecFromIndexedFnArbitrary<TDim, TTFnObj>(IdxSubSequence(), std::forward<TArgs>(args)...);
74  }

76  //! A n-dimensional vector.
77  template<typename TDim, typename TVal>
78  class Vec final
79  {
80  public:
81  static_assert(TDim::value >= 0u, "Invalid dimensionality");
82 
83  using Dim = TDim;
84  using Val = TVal;
85 
86  private:
87  //! A sequence of integers from 0 to dim-1.
88  //! This can be used to write compile time indexing algorithms.
89  using IdxSequence = std::make_integer_sequence<std::size_t, TDim::value>;
90 
91  public:
92  // The default constructor is only available when the vector is zero-dimensional.
93  ALPAKA_NO_HOST_ACC_WARNING
94  template<bool B = (TDim::value == 0u), typename = std::enable_if_t<B>>
95  ALPAKA_FN_HOST_ACC Vec() : m_data{static_cast<TVal>(0u)}
96  {
97  }
98 
99 
100  //! Value constructor.
101  //! This constructor is only available if the number of parameters matches the vector idx.
102  ALPAKA_NO_HOST_ACC_WARNING
103  template<
104  typename TArg0,
105  typename... TArgs,
106  typename = std::enable_if_t<
107  // There have to be dim arguments.
108  (sizeof...(TArgs) + 1 == TDim::value) && (std::is_same<TVal, std::decay_t<TArg0>>::value)>>
109  ALPAKA_FN_HOST_ACC Vec(TArg0&& arg0, TArgs&&... args)
110  : m_data{std::forward<TArg0>(arg0), std::forward<TArgs>(args)...}
111  {
112  }
113 
114  ALPAKA_NO_HOST_ACC_WARNING
115  ALPAKA_FN_HOST_ACC
116  Vec(Vec const&) = default;
117  ALPAKA_NO_HOST_ACC_WARNING
118  ALPAKA_FN_HOST_ACC
119  Vec(Vec&&) noexcept = default;
120  ALPAKA_NO_HOST_ACC_WARNING
121  ALPAKA_FN_HOST_ACC
122  auto operator=(Vec const&) -> Vec& = default;
123  ALPAKA_NO_HOST_ACC_WARNING
124  ALPAKA_FN_HOST_ACC
125  auto operator=(Vec&&) noexcept -> Vec& = default;
126  ALPAKA_NO_HOST_ACC_WARNING
127  ALPAKA_FN_HOST_ACC ~Vec() = default;
128 
129  private:
130  //! A function object that returns the given value for each index.
131  template<std::size_t Tidx>
132  struct CreateSingleVal
133  {
134  ALPAKA_NO_HOST_ACC_WARNING
135  ALPAKA_FN_HOST_ACC static auto create(TVal const& val) -> TVal
136  {
137  return val;
138  }
139  };
140 
141  public:
142  //! \brief Single value constructor.
143  //!
144  //! Creates a vector with all values set to val.
145  //! \param val The initial value.
146  ALPAKA_NO_HOST_ACC_WARNING
147  ALPAKA_FN_HOST_ACC static auto all(TVal const& val) -> Vec<TDim, TVal>
148  {
149  return createVecFromIndexedFn<TDim, CreateSingleVal>(val);
150  }
151  //! Zero value constructor.
152  ALPAKA_NO_HOST_ACC_WARNING
153  ALPAKA_FN_HOST_ACC static auto zeros() -> Vec<TDim, TVal>
154  {
155  return all(static_cast<TVal>(0));
156  }
157  //! One value constructor.
158  ALPAKA_NO_HOST_ACC_WARNING
159  ALPAKA_FN_HOST_ACC static auto ones() -> Vec<TDim, TVal>
160  {
161  return all(static_cast<TVal>(1));
162  }
163 
164  //! Value reference accessor at the given non-unsigned integer index.
165  //! \return A reference to the value at the given index.
166  ALPAKA_NO_HOST_ACC_WARNING
167  template<typename TIdx, typename = std::enable_if_t<std::is_integral<TIdx>::value>>
168  ALPAKA_FN_HOST_ACC auto operator[](TIdx const iIdx) -> TVal&
169  {
170  core::assertValueUnsigned(iIdx);
171  auto const idx = static_cast<typename TDim::value_type>(iIdx);
172  core::assertGreaterThan<TDim>(idx);
173  return m_data[idx];
174  }
175 
176  //! Value accessor at the given non-unsigned integer index.
177  //! \return The value at the given index.
178  ALPAKA_NO_HOST_ACC_WARNING
179  template<typename TIdx, typename = std::enable_if_t<std::is_integral<TIdx>::value>>
180  ALPAKA_FN_HOST_ACC auto operator[](TIdx const iIdx) const -> TVal
181  {
182  core::assertValueUnsigned(iIdx);
183  auto const idx = static_cast<typename TDim::value_type>(iIdx);
184  core::assertGreaterThan<TDim>(idx);
185  return m_data[idx];
186  }
187 
188  ALPAKA_NO_HOST_ACC_WARNING
189  ALPAKA_FN_HOST_ACC auto operator==(Vec const& rhs) const -> bool
190  {
191  for(typename TDim::value_type i(0); i < TDim::value; ++i)
192  {
193  if((*this)[i] != rhs[i])
194  {
195  return false;
196  }
197  }
198  return true;
199  }
200  ALPAKA_NO_HOST_ACC_WARNING
201  ALPAKA_FN_HOST_ACC auto operator!=(Vec const& rhs) const -> bool
202  {
203  return !((*this) == rhs);
204  }
205  ALPAKA_NO_HOST_ACC_WARNING
206  template<typename TFnObj, std::size_t... TIndices>
207  ALPAKA_FN_HOST_ACC auto foldrByIndices(
208  TFnObj const& f,
209  std::integer_sequence<std::size_t, TIndices...> const& indices) const
210  {
211  alpaka::ignore_unused(indices);
212 
213  return meta::foldr(f, ((*this)[TIndices])...);
214  }
215  ALPAKA_NO_HOST_ACC_WARNING
216  template<typename TFnObj>
217  ALPAKA_FN_HOST_ACC auto foldrAll(TFnObj const& f) const
218  {
219  return foldrByIndices(f, IdxSequence());
220  }
221 // suppress strange warning produced by nvcc+MSVC in release mode
222 #if BOOST_COMP_MSVC || defined(BOOST_COMP_MSVC_EMULATED)
223 # pragma warning(push)
224 # pragma warning(disable : 4702) // unreachable code
225 #endif
226  //! \return The product of all values.
227  ALPAKA_NO_HOST_ACC_WARNING
228  ALPAKA_FN_HOST_ACC auto prod() const -> TVal
229  {
230  return foldrAll(std::multiplies<TVal>());
231  }
232 #if BOOST_COMP_MSVC || defined(BOOST_COMP_MSVC_EMULATED)
233 # pragma warning(pop)
234 #endif
235  //! \return The sum of all values.
236  ALPAKA_NO_HOST_ACC_WARNING
237  ALPAKA_FN_HOST_ACC auto sum() const -> TVal
238  {
239  return foldrAll(std::plus<TVal>());
240  }
241  //! \return The min of all values.
242  ALPAKA_NO_HOST_ACC_WARNING
243  ALPAKA_FN_HOST_ACC auto min() const -> TVal
244  {
245  return foldrAll(meta::min<TVal>());
246  }
247  //! \return The max of all values.
248  ALPAKA_NO_HOST_ACC_WARNING
249  ALPAKA_FN_HOST_ACC auto max() const -> TVal
250  {
251  return foldrAll(meta::max<TVal>());
252  }
253  //! \return The index of the minimal element.
254  ALPAKA_FN_HOST auto minElem() const -> typename TDim::value_type
255  {
256  return static_cast<typename TDim::value_type>(
257  std::distance(std::begin(m_data), std::min_element(std::begin(m_data), std::end(m_data))));
258  }
259  //! \return The index of the maximal element.
260  ALPAKA_FN_HOST auto maxElem() const -> typename TDim::value_type
261  {
262  return static_cast<typename TDim::value_type>(
263  std::distance(std::begin(m_data), std::max_element(std::begin(m_data), std::end(m_data))));
264  }
265 
266  private:
267  // Zero sized arrays are not allowed, therefore zero-dimensional vectors have one member.
268  TVal m_data[TDim::value == 0u ? 1u : TDim::value];
269  };
