42  //! Single value constructor helper.
43  ALPAKA_NO_HOST_ACC_WARNING
44  template<
45  typename TDim,
46  template<std::size_t> class TTFnObj,
47  typename... TArgs,
48  typename TIdxSize,
49  TIdxSize... TIndices>
50  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFnArbitrary(
51  std::integer_sequence<TIdxSize, TIndices...> const & indices,
52  TArgs && ... args)
53  {
54  alpaka::ignore_unused(indices);
55 
56  return Vec<TDim, decltype(TTFnObj<0>::create(std::forward<TArgs>(args)...))>(
57  (TTFnObj<TIndices>::create(std::forward<TArgs>(args)...))...);
58  }

60  //! Creator using func<idx>(args...) to initialize all values of the vector.
61  //! The idx is in the range [0, TDim].

63  template<
64  typename TDim,
65  template<std::size_t> class TTFnObj,
66  typename... TArgs>
67  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFn(
68  TArgs && ... args)
69  {
70  using IdxSequence = std::make_integer_sequence<typename TDim::value_type, TDim::value>;
71  return
72  createVecFromIndexedFnArbitrary<
73  TDim,
74  TTFnObj>(
75  IdxSequence(),
76  std::forward<TArgs>(args)...);
77  }

79  //! Creator using func<idx>(args...) to initialize all values of the vector.
80  //! The idx is in the range [TIdxOffset, TIdxOffset + TDim].
81  ALPAKA_NO_HOST_ACC_WARNING
82  template<
83  typename TDim,
84  template<std::size_t> class TTFnObj,
85  typename TIdxOffset,
86  typename... TArgs>
87  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFnOffset(
88  TArgs && ... args)
89  {
90  using IdxSubSequenceSigned = meta::MakeIntegerSequenceOffset<std::intmax_t, TIdxOffset::value, TDim::value>;
91  using IdxSubSequence = meta::ConvertIntegerSequence<typename TIdxOffset::value_type, IdxSubSequenceSigned>;

101  //! A n-dimensional vector.
102  template<
103  typename TDim,
104  typename TVal>
105  class Vec final
106  {
107  public:
108  static_assert(TDim::value >= 0u, "Invalid dimensionality");
109 
110  using Dim = TDim;
111  using Val = TVal;
112 
113  private:
114  //! A sequence of integers from 0 to dim-1.
115  //! This can be used to write compile time indexing algorithms.
116  using IdxSequence = std::make_integer_sequence<std::size_t, TDim::value>;
117 
118  public:
119  //-----------------------------------------------------------------------------
120  // The default constructor is only available when the vector is zero-dimensional.
121  ALPAKA_NO_HOST_ACC_WARNING
122  template<
123  bool B = (TDim::value == 0u),
124  typename = std::enable_if_t<B>>
125  ALPAKA_FN_HOST_ACC Vec() :
126  m_data{static_cast<TVal>(0u)}
127  {}
128 
129 
130  //-----------------------------------------------------------------------------
131  //! Value constructor.
132  //! This constructor is only available if the number of parameters matches the vector idx.
133  ALPAKA_NO_HOST_ACC_WARNING
134  template<
135  typename TArg0,
136  typename... TArgs,
137  typename = std::enable_if_t<
138  // There have to be dim arguments.
139  (sizeof...(TArgs)+1 == TDim::value)
140  &&
141  (std::is_same<TVal, std::decay_t<TArg0>>::value)
142  >>
143  ALPAKA_FN_HOST_ACC Vec(
144  TArg0 && arg0,
145  TArgs && ... args) :
146  m_data{std::forward<TArg0>(arg0), std::forward<TArgs>(args)...}
147  {}

162  ALPAKA_NO_HOST_ACC_WARNING
163  ALPAKA_FN_HOST_ACC
164  auto operator=(Vec &&) noexcept -> Vec & = default;

213  ALPAKA_FN_HOST_ACC static auto ones()

215  {

217  }

219  //-----------------------------------------------------------------------------
220  //! Value reference accessor at the given non-unsigned integer index.
221  //! \return A reference to the value at the given index.
222  ALPAKA_NO_HOST_ACC_WARNING
223  template<
224  typename TIdx,
225  typename = std::enable_if_t<
226  std::is_integral<TIdx>::value>>
227  ALPAKA_FN_HOST_ACC auto operator[](

239  //! \return The value at the given index.

241  template<

243  typename = std::enable_if_t<
244  std::is_integral<TIdx>::value>>
245  ALPAKA_FN_HOST_ACC auto operator[](

251  core::assertGreaterThan<TDim>(idx);

255  //-----------------------------------------------------------------------------
256  ALPAKA_NO_HOST_ACC_WARNING
257  ALPAKA_FN_HOST_ACC auto operator==(
258  Vec const & rhs) const
259  -> bool
260  {
261  for(typename TDim::value_type i(0); i < TDim::value; ++i)
262  {
263  if((*this)[i] != rhs[i])
264  {
265  return false;
266  }
267  }
268  return true;
269  }
270  //-----------------------------------------------------------------------------
271  ALPAKA_NO_HOST_ACC_WARNING
272  ALPAKA_FN_HOST_ACC auto operator!=(
273  Vec const & rhs) const
274  -> bool
275  {
276  return !((*this) == rhs);
277  }
278  //-----------------------------------------------------------------------------
279  ALPAKA_NO_HOST_ACC_WARNING
280  template<
281  typename TFnObj,
282  std::size_t... TIndices>
283  ALPAKA_FN_HOST_ACC auto foldrByIndices(
284  TFnObj const & f,
285  std::integer_sequence<std::size_t, TIndices...> const & indices) const
286  {
287  alpaka::ignore_unused(indices);
288 
289  return
290  meta::foldr(
291  f,
292  ((*this)[TIndices])...);
293  }
294  //-----------------------------------------------------------------------------
295  ALPAKA_NO_HOST_ACC_WARNING
296  template<
297  typename TFnObj>
298  ALPAKA_FN_HOST_ACC auto foldrAll(
299  TFnObj const & f) const
300  {
301  return
302  foldrByIndices(
303  f,
304  IdxSequence());
305  }
306 // suppress strange warning produced by nvcc+MSVC in release mode
307 #if BOOST_COMP_MSVC || defined(BOOST_COMP_MSVC_EMULATED)
308 #pragma warning(push)
309 #pragma warning(disable: 4702) // unreachable code
310 #endif
311  //-----------------------------------------------------------------------------
312  //! \return The product of all values.
313  ALPAKA_NO_HOST_ACC_WARNING
314  ALPAKA_FN_HOST_ACC auto prod() const
315  -> TVal
316  {
317  return foldrAll(std::multiplies<TVal>());
318  }
319 #if BOOST_COMP_MSVC || defined(BOOST_COMP_MSVC_EMULATED)
320 #pragma warning(pop)
321 #endif
322  //-----------------------------------------------------------------------------
323  //! \return The sum of all values.
324  ALPAKA_NO_HOST_ACC_WARNING
325  ALPAKA_FN_HOST_ACC auto sum() const
326  -> TVal
327  {
328  return foldrAll(std::plus<TVal>());
329  }
330  //-----------------------------------------------------------------------------
331  //! \return The min of all values.
332  ALPAKA_NO_HOST_ACC_WARNING
333  ALPAKA_FN_HOST_ACC auto min() const
334  -> TVal
335  {
336  return foldrAll(meta::min<TVal>());
337  }
338  //-----------------------------------------------------------------------------
339  //! \return The max of all values.
340  ALPAKA_NO_HOST_ACC_WARNING
341  ALPAKA_FN_HOST_ACC auto max() const
342  -> TVal
343  {
344  return foldrAll(meta::max<TVal>());
345  }
346  //-----------------------------------------------------------------------------
347  //! \return The index of the minimal element.
348  ALPAKA_FN_HOST auto minElem() const
349  -> typename TDim::value_type
350  {
351  return
352  static_cast<typename TDim::value_type>(
353  std::distance(
354  std::begin(m_data),
355  std::min_element(
356  std::begin(m_data),
357  std::end(m_data))));
358  }
359  //-----------------------------------------------------------------------------
360  //! \return The index of the maximal element.
361  ALPAKA_FN_HOST auto maxElem() const
362  -> typename TDim::value_type
363  {
364  return
365  static_cast<typename TDim::value_type>(
366  std::distance(
367  std::begin(m_data),
368  std::max_element(
369  std::begin(m_data),
370  std::end(m_data))));
371  }
372 
373  private:
374  // Zero sized arrays are not allowed, therefore zero-dimensional vectors have one member.
375  TVal m_data[TDim::value == 0u ? 1u : TDim::value];
376  };
