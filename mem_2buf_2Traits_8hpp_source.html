133  //! If supported, allocates stream-ordered memory on the given queue and the associated device.
134  //! Otherwise, allocates regular memory on the device associated to the queue.
135  //! Please note that stream-ordered and regular memory have different semantics:
136  //! this function is provided for convenience in the cases where the difference is not relevant,
137  //! and the stream-ordered memory is only used as a performance optimisation.
138  //!
139  //! \tparam TElem The element type of the returned buffer.
140  //! \tparam TIdx The linear index type of the buffer.
141  //! \tparam TExtent The extent type of the buffer.
142  //! \tparam TQueue The type of queue used to order the buffer allocation.
143  //! \param queue The queue used to order the buffer allocation.
144  //! \param extent The extent of the buffer.
145  //! \return The newly allocated buffer.
146  template<typename TElem, typename TIdx, typename TExtent, typename TQueue>
147  ALPAKA_FN_HOST auto allocAsyncBufIfSupported(TQueue queue, TExtent const& extent = TExtent())
148  {
149  if constexpr(hasAsyncBufSupport<alpaka::Dev<TQueue>, Dim<TExtent>>)
150  {
151  return allocAsyncBuf<TElem, TIdx>(queue, extent);
152  }
153  else
154  {
155  return allocBuf<TElem, TIdx>(getDev(queue), extent);
156  }
157 
158  ALPAKA_UNREACHABLE(allocBuf<TElem, TIdx>(getDev(queue), extent));
159  }
