109  //! If supported, allocates stream-ordered memory on the given queue and the associated device.
110  //! Otherwise, allocates regular memory on the device associated to the queue.
111  //! Please note that stream-ordered and regular memory have different semantics:
112  //! this function is provided for convenience in the cases where the difference is not relevant,
113  //! and the stream-ordered memory is only used as a performance optimisation.
114  //!
115  //! \tparam TElem The element type of the returned buffer.
116  //! \tparam TIdx The linear index type of the buffer.
117  //! \tparam TExtent The extent type of the buffer.
118  //! \tparam TQueue The type of queue used to order the buffer allocation.
119  //! \param queue The queue used to order the buffer allocation.
120  //! \param extent The extent of the buffer.
121  //! \return The newly allocated buffer.
122  template<typename TElem, typename TIdx, typename TExtent, typename TQueue>
123  ALPAKA_FN_HOST auto allocAsyncBufIfSupported(TQueue queue, TExtent const& extent = TExtent())
124  {
125  if constexpr(hasAsyncBufSupport<alpaka::Dev<TQueue>, Dim<TExtent>>)
126  {
127  return allocAsyncBuf<TElem, TIdx>(queue, extent);
128  }
129  else
130  {
131  return allocBuf<TElem, TIdx>(getDev(queue), extent);
132  }
133 
134  ALPAKA_UNREACHABLE(allocBuf<TElem, TIdx>(getDev(queue), extent));
135  }
