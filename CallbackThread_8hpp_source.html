55  auto submit(Task task) -> std::future<void>
56  {
57  // We do not use the future of std::packed_task because the future will keep the task alive
58  // and we can not control the moment the future is set.
59  auto tp = std::make_pair(std::move(task), std::promise<void>{});
60  auto f = tp.second.get_future();
61  {
62  std::unique_lock<std::mutex> lock{m_mutex};
63  m_tasks.emplace(std::move(tp));
64  if(!m_thread.joinable())
65  startWorkerThread();
66  m_cond.notify_one();
67  }
68 
69  return f;
70  }
