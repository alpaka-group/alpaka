82  // Host function helper
83  // Encapsulates the different function signatures used by cudaStreamAddCallback and cudaLaunchHostFn, and the
84  // different calling conventions used by CUDA (__stdcall on Win32) and HIP (standard).
85  struct HostFnAdaptor
86  {
87  HostFn_t func_;
88  void* data_;
89 
90  static void CUDART_CB hostFunction(void* data)
91  {
92  auto ptr = reinterpret_cast<HostFnAdaptor*>(data);
93  ptr->func_(ptr->data_);
94  delete ptr;
95  }
96 
97  static void CUDART_CB streamCallback(Stream_t, Error_t, void* data)
98  {
99  auto ptr = reinterpret_cast<HostFnAdaptor*>(data);
100  ptr->func_(ptr->data_);
101  delete ptr;
102  }
103  };
