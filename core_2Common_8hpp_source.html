15 //! All functions that can be used on an accelerator have to be attributed with ALPAKA_FN_ACC or ALPAKA_FN_HOST_ACC.
16 //!
17 //! Usage:
18 //! ALPAKA_FN_ACC
19 //! auto add(std::int32_t a, std::int32_t b)
20 //! -> std::int32_t;
21 #if BOOST_LANG_CUDA || BOOST_LANG_HIP
22 # if defined(ALPAKA_ACC_GPU_CUDA_ONLY_MODE) || defined(ALPAKA_ACC_GPU_HIP_ONLY_MODE)
23 # define ALPAKA_FN_ACC __device__
24 # else
25 # define ALPAKA_FN_ACC __device__ __host__
26 # endif
27 # define ALPAKA_FN_HOST_ACC __device__ __host__
28 # define ALPAKA_FN_HOST __host__
29 #else
30 # define ALPAKA_FN_ACC
31 # define ALPAKA_FN_HOST_ACC
32 # define ALPAKA_FN_HOST
33 #endif

35 //! All functions marked with ALPAKA_FN_ACC or ALPAKA_FN_HOST_ACC that are exported to / imported from different
36 //! translation units have to be attributed with ALPAKA_FN_EXTERN. Note that this needs to be applied to both the
37 //! declaration and the definition.
38 //!
39 //! Usage:
40 //! ALPAKA_FN_ACC ALPAKA_FN_EXTERN auto add(std::int32_t a, std::int32_t b) -> std::int32_t;
41 //!
42 //! Warning: If this is used together with the SYCL back-end make sure that your SYCL runtime supports generic
43 //! address spaces. Otherwise it is forbidden to use pointers as parameter or return type for functions marked
44 //! with ALPAKA_FN_EXTERN.
45 #ifdef ALPAKA_ACC_SYCL_ENABLED
46 /*
47  This is required by the SYCL standard, section 5.10.1 "SYCL functions and member functions linkage":
48 
49  The default behavior in SYCL applications is that all the definitions and declarations of the functions and member
50  functions are available to the SYCL compiler, in the same translation unit. When this is not the case, all the
51  symbols that need to be exported to a SYCL library or from a C++ library to a SYCL application need to be defined
52  using the macro: SYCL_EXTERNAL.
53 */
54 # define ALPAKA_FN_EXTERN SYCL_EXTERNAL
55 #else
56 # define ALPAKA_FN_EXTERN
57 #endif

87 //! This macro defines a variable lying in global accelerator device memory.
88 //!
89 //! Example:
90 //! ALPAKA_STATIC_ACC_MEM_GLOBAL int i;
91 //!
92 //! Those variables behave like ordinary variables when used in file-scope.
93 //! They have external linkage (are accessible from other compilation units).
94 //! If you want to access it from a different compilation unit, you have to declare it as extern:
95 //! extern ALPAKA_STATIC_ACC_MEM_GLOBAL int i;
96 //! Like ordinary variables, only one definition is allowed (ODR)
97 //! Failure to do so might lead to linker errors.
98 //!
99 //! In contrast to ordinary variables, you can not define such variables
100 //! as static compilation unit local variables with internal linkage
101 //! because this is forbidden by CUDA.
102 //!
103 //! \attention It is not allowed to initialize the variable together with the declaration.
104 //! To initialize the variable alpaka::createStaticDevMemView and alpaka::memcpy must be used.
105 //! \code{.cpp}
106 //! ALPAKA_STATIC_ACC_MEM_GLOBAL int foo;
107 //!
108 //! void initFoo() {
109 //! auto extent = alpaka::Vec<alpaka::DimInt<1u>, size_t>{1};
110 //! auto viewFoo = alpaka::createStaticDevMemView(&foo, device, extent);
111 //! int initialValue = 42;
112 //! alpaka::ViewPlainPtr<DevHost, int, alpaka::DimInt<1u>, size_t> bufHost(&initialValue, devHost, extent);
113 //! alpaka::memcpy(queue, viewGlobalMemUninitialized, bufHost, extent);
114 //! }
115 //! \endcode
116 #if((BOOST_LANG_CUDA && BOOST_COMP_CLANG_CUDA) || (BOOST_LANG_CUDA && BOOST_COMP_NVCC && BOOST_ARCH_PTX) \
117  || BOOST_LANG_HIP)
118 # define ALPAKA_STATIC_ACC_MEM_GLOBAL __device__
119 #elif defined(ALPAKA_ACC_SYCL_ENABLED)
120 # define ALPAKA_STATIC_ACC_MEM_GLOBAL _Pragma("GCC error \"The SYCL backend does not support global device variables.\""))
121 #else
122 # define ALPAKA_STATIC_ACC_MEM_GLOBAL
123 #endif

163 //! This macro disables memory optimizations for annotated device memory.
164 //!
165 //! Example:
166 //! ALPAKA_DEVICE_VOLATILE float* ptr;
167 //!
168 //! This is useful for pointers, (shared) variables and shared memory which are used in combination with

172  || (BOOST_LANG_HIP && defined(__HIP_DEVICE_COMPILE__) && __HIP_DEVICE_COMPILE__ == 1)
173 # define ALPAKA_DEVICE_VOLATILE volatile
174 #else
175 # define ALPAKA_DEVICE_VOLATILE
