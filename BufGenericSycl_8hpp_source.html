145  //! The SYCL memory allocation trait specialization.
146  template<typename TElem, typename TDim, typename TIdx, typename TPltf>
147  struct BufAlloc<TElem, TDim, TIdx, DevGenericSycl<TPltf>>
148  {
149  template<typename TExtent>
150  static auto allocBuf(DevGenericSycl<TPltf> const& dev, TExtent const& ext)
151  -> BufGenericSycl<TElem, TDim, TIdx, DevGenericSycl<TPltf>>
152  {
153  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
154 
155  if constexpr(TDim::value == 0 || TDim::value == 1)
156  {
157  auto const width = getWidth(ext);
158 
159 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
160  auto const widthBytes = width * static_cast<TIdx>(sizeof(TElem));
161  std::cout << __func__ << " ew: " << width << " ewb: " << widthBytes << '\n';
162 # endif
163 
164  auto const range = sycl::range<1>{width};
165  return {dev, sycl::buffer<TElem, 1>{range}, ext};
166  }
167  else if constexpr(TDim::value == 2)
168  {
169  auto const width = getWidth(ext);
170  auto const height = getHeight(ext);
171 
172 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
173  auto const widthBytes = width * static_cast<TIdx>(sizeof(TElem));
174  std::cout << __func__ << " ew: " << width << " eh: " << height << " ewb: " << widthBytes
175  << " pitch: " << widthBytes << '\n';
176 # endif
177 
178  auto const range = sycl::range<2>{width, height};
179  return {dev, sycl::buffer<TElem, 2>{range}, ext};
180  }
181  else if constexpr(TDim::value == 3)
182  {
183  auto const width = getWidth(ext);
184  auto const height = getHeight(ext);
185  auto const depth = getDepth(ext);
186 
187 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
188  auto const widthBytes = width * static_cast<TIdx>(sizeof(TElem));
189  std::cout << __func__ << " ew: " << width << " eh: " << height << " ed: " << depth
190  << " ewb: " << widthBytes << " pitch: " << widthBytes << '\n';
191 # endif
192 
193  auto const range = sycl::range<3>{width, height, depth};
194  return {dev, sycl::buffer<TElem, 3>{range}, ext};
195  }
196  }
197  };
