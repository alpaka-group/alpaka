61  template<typename TKernelFnObj, typename... TArgs>
62  auto operator()(TKernelFnObj const& kernelFnObj, TArgs&&... args) -> bool
63  {
64  // Allocate the result value
65  auto bufAccResult = allocBuf<bool, Idx>(m_devAcc, static_cast<Idx>(1u));
66  memset(m_queue, bufAccResult, static_cast<std::uint8_t>(true));
67 
68  exec<Acc>(m_queue, m_workDiv, kernelFnObj, getPtrNative(bufAccResult), std::forward<TArgs>(args)...);
69 
70  // Copy the result value to the host

72  memcpy(m_queue, bufHostResult, bufAccResult);

74

76 
77  return result;
78  }
