127  typename TTask>
128  struct Enqueue<
129  QueueUniformCudaHipRtNonBlocking,

132  //#############################################################################
133  enum class CallbackState
134  {
135  enqueued,
136  notified,
137  finished,
138  };
139 
140  //#############################################################################
141  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
142  {
143  std::mutex m_mutex;
144  std::condition_variable m_event;
145  CallbackState state = CallbackState::enqueued;
146  };
147 
148  //-----------------------------------------------------------------------------
149 #if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
150  static void CUDART_CB
151 #else
152  static void HIPRT_CB
153 #endif
154  uniformCudaHipRtCallback(ALPAKA_API_PREFIX(Stream_t) /*queue*/, ALPAKA_API_PREFIX(Error_t) /*status*/, void *arg)
155  {
156  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread has already finished.
157  const auto pCallbackSynchronizationData = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
158 
159  // Notify the executing thread.
160  {
161  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
162  pCallbackSynchronizationData->state = CallbackState::notified;
163  }
164  pCallbackSynchronizationData->m_event.notify_one();
165 
166  // Wait for the executing thread to finish the task if it has not already finished.
167  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
168  if(pCallbackSynchronizationData->state != CallbackState::finished)
169  {
170  pCallbackSynchronizationData->m_event.wait(
171  lock,
172  [pCallbackSynchronizationData](){
173  return pCallbackSynchronizationData->state == CallbackState::finished;
174  }
175  );
176  }
177  }
178 
179  //-----------------------------------------------------------------------------
180  ALPAKA_FN_HOST static auto enqueue(
181  QueueUniformCudaHipRtNonBlocking & queue,
182  TTask const & task)
183  -> void
184  {
185 #if BOOST_COMP_HIP
186  // NOTE: hip callbacks are not blocking the stream.
187  // @todo remove this assert when hipStreamAddCallback is fixed
188  static_assert(
189  meta::DependentFalseType<TTask>::value,
190  "Callbacks are not supported for HIP-clang");
191 #endif
192 
193  auto pCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
194  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamAddCallback)(
195  queue.m_spQueueImpl->m_UniformCudaHipQueue,
196  uniformCudaHipRtCallback,
197  pCallbackSynchronizationData.get(),
198  0u));
199 
200  // We start a new std::thread which stores the task to be executed.
201  // This circumvents the limitation that it is not possible to call CUDA methods within the CUDA/HIP callback thread.
202  // The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
203  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
204  // before it executes the next task in the queue (CUDA/HIP stream).
205  std::thread t(
206  [pCallbackSynchronizationData, task](){
207  // If the callback has not yet been called, we wait for it.
208  {
209  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
210  if(pCallbackSynchronizationData->state != CallbackState::notified)
211  {
212  pCallbackSynchronizationData->m_event.wait(
213  lock,
214  [pCallbackSynchronizationData](){
215  return pCallbackSynchronizationData->state == CallbackState::notified;
216  }
217  );
218  }
219 
220  task();
221 
222  // Notify the waiting CUDA thread.
223  pCallbackSynchronizationData->state = CallbackState::finished;
224  }
225  pCallbackSynchronizationData->m_event.notify_one();
226  }
227  );
228 
229  t.detach();
230  }
231  };
232  }
233 }
