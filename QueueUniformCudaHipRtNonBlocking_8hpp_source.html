90  //! The CUDA/HIP RT sync queue enqueue trait specialization.
91  template<typename TTask>
92  struct Enqueue<QueueUniformCudaHipRtNonBlocking, TTask>
93  {
94  enum class CallbackState
95  {
96  enqueued,
97  notified,
98  finished,
99  };
100 
101  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
102  {
103  std::mutex m_mutex;
104  std::condition_variable m_event;
105  CallbackState state = CallbackState::enqueued;
106  };
107 
108 # if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
109  static void CUDART_CB
110 # else
111  static void HIPRT_CB
112 # endif
113  uniformCudaHipRtCallback(
114  ALPAKA_API_PREFIX(Stream_t) /*queue*/,
115  ALPAKA_API_PREFIX(Error_t) /*status*/,
116  void* arg)
117  {
118  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread
119  // has already finished.
120  const auto pCallbackSynchronizationData
121  = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
122 
123  // Notify the executing thread.
124  {
125  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
126  pCallbackSynchronizationData->state = CallbackState::notified;
127  }
128  pCallbackSynchronizationData->m_event.notify_one();
129 
130  // Wait for the executing thread to finish the task if it has not already finished.
131  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
132  if(pCallbackSynchronizationData->state != CallbackState::finished)
133  {
134  pCallbackSynchronizationData->m_event.wait(
135  lock,
136  [pCallbackSynchronizationData]()
137  { return pCallbackSynchronizationData->state == CallbackState::finished; });
138  }
139  }
140 
141  ALPAKA_FN_HOST static auto enqueue(QueueUniformCudaHipRtNonBlocking& queue, TTask const& task) -> void
142  {
143  auto pCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
144  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamAddCallback)(
145  queue.m_spQueueImpl->m_UniformCudaHipQueue,
146  uniformCudaHipRtCallback,
147  pCallbackSynchronizationData.get(),
148  0u));
149 
150  // We start a new std::thread which stores the task to be executed.
151  // This circumvents the limitation that it is not possible to call CUDA methods within the CUDA/HIP
152  // callback thread. The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
153  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
154  // before it executes the next task in the queue (CUDA/HIP stream).
155  std::thread t(
156  [pCallbackSynchronizationData, task]()
157  {
158  // If the callback has not yet been called, we wait for it.
159  {
160  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
161  if(pCallbackSynchronizationData->state != CallbackState::notified)
162  {
163  pCallbackSynchronizationData->m_event.wait(
164  lock,
165  [pCallbackSynchronizationData]()
166  { return pCallbackSynchronizationData->state == CallbackState::notified; });
167  }
168 
169  task();
170 
171  // Notify the waiting CUDA thread.
172  pCallbackSynchronizationData->state = CallbackState::finished;
173  }
174  pCallbackSynchronizationData->m_event.notify_one();
175  });
176 
177  t.detach();
178  }
179  };
