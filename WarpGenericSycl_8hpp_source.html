90  template<typename TDim>
91  struct Shfl<experimental::warp::WarpGenericSycl<TDim>>
92  {
93  template<typename T>
94  static auto shfl(
95  experimental::warp::WarpGenericSycl<TDim> const& warp,
96  T value,
97  std::int32_t srcLane,
98  std::int32_t width)
99  {
100  /* If width < srcLane the sub-group needs to be split into assumed subdivisions. The first item of each
101  subdivision has the assumed index 0. The srcLane index is relative to the subdivisions.
102 
103  Example: If we assume a sub-group size of 32 and a width of 16 we will receive two subdivisions:
104  The first starts at sub-group index 0 and the second at sub-group index 16. For srcLane = 4 the
105  first subdivision will access the value at sub-group index 4 and the second at sub-group index 20. */
106  auto const actual_group = warp.m_item.get_sub_group();
107  auto const actual_item_id = actual_group.get_local_linear_id();
108 
109  auto const assumed_group_id = actual_item_id / width;
110  auto const assumed_item_id = actual_item_id % width;
111 
112  auto const assumed_src_id = static_cast<std::size_t>(srcLane % width);
113  auto const actual_src_id = assumed_src_id + assumed_group_id * width;
114 
115  auto const src = sycl::id<1>{actual_src_id};
116 
117  return sycl::select_from_group(actual_group, value, src);
118  }
119  };
