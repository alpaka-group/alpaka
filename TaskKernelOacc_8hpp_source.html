46  //! The OpenACC accelerator execution task.
47  template<typename TDim, typename TIdx, typename TKernelFnObj, typename... TArgs>
48  class TaskKernelOacc final : public WorkDivMembers<TDim, TIdx>
49  {
50  public:
51  template<typename TWorkDiv>
52  ALPAKA_FN_HOST TaskKernelOacc(TWorkDiv&& workDiv, TKernelFnObj const& kernelFnObj, TArgs&&... args)
53  : WorkDivMembers<TDim, TIdx>(std::forward<TWorkDiv>(workDiv))
54  , m_kernelFnObj(kernelFnObj)
55  , m_args(std::forward<TArgs>(args)...)
56  {
57  static_assert(
58  Dim<std::decay_t<TWorkDiv>>::value == TDim::value,
59  "The work division and the execution task have to be of the same dimensionality!");
60  }
61 
62  //! Executes the kernel function object.
63  ALPAKA_FN_HOST auto operator()(const DevOacc& dev) const -> void
64  {
65  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
66 
67  auto const gridBlockExtent = getWorkDiv<Grid, Blocks>(*this);
68  auto const blockThreadExtent = getWorkDiv<Block, Threads>(*this);
69  auto const threadElemExtent = getWorkDiv<Thread, Elems>(*this);
70 
71 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
72  std::cout << "m_gridBlockExtent=" << this->m_gridBlockExtent << "\tgridBlockExtent=" << gridBlockExtent
73  << std::endl;
74  std::cout << "m_blockThreadExtent=" << this->m_blockThreadExtent
75  << "\tblockThreadExtent=" << blockThreadExtent << std::endl;
76  std::cout << "m_threadElemExtent=" << this->m_threadElemExtent << "\tthreadElemExtent=" << threadElemExtent
77  << std::endl;
78 # endif
79 
80  // Get the size of the block shared dynamic memory.
81  auto const blockSharedMemDynSizeBytes = core::apply(
82  [&](ALPAKA_DECAY_T(TArgs) const&... args)
83  {
84  return getBlockSharedMemDynSizeBytes<AccOacc<TDim, TIdx>>(
85  m_kernelFnObj,
86  blockThreadExtent,
87  threadElemExtent,
88  args...);
89  },
90  m_args);
91 
92 # if ALPAKA_DEBUG > ALPAKA_DEBUG_MINIMAL
93  std::cout << __func__ << " blockSharedMemDynSizeBytes: " << blockSharedMemDynSizeBytes << " B"
94  << std::endl;
95 # endif
96  // The number of blocks in the grid.
97  TIdx const gridBlockCount(gridBlockExtent.prod());
98  // The number of threads in a block.
99  TIdx const blockThreadCount(blockThreadExtent.prod());
100 
101  if(gridBlockCount == 0 || blockThreadCount == 0)
102  { //! empty grid is a NOP
103  return;
104  }
105 
106 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_MINIMAL
107  std::cout << "threadElemCount=" << threadElemExtent[0u] << "\tgridBlockCount=" << gridBlockCount
108  << std::endl;
109 # endif
110  auto argsD = m_args;
111  auto kernelFnObj = m_kernelFnObj;
112  dev.makeCurrent();
113 
114  std::uint32_t blocksLock[2] = {0u, 0u};
115  std::uint32_t* gridsLock = dev.gridsLock();
116 
117 # pragma acc parallel num_workers(blockThreadCount) copyin( \
118  threadElemExtent, \
119  blockThreadExtent, \
120  gridBlockExtent, \
121  argsD, \
122  blockSharedMemDynSizeBytes, \
123  blocksLock [0:2], \
124  kernelFnObj) default(present) deviceptr(gridsLock)
125  {
126  {
127 # pragma acc loop gang
128  for(TIdx b = 0u; b < gridBlockCount; ++b)
129  {
130  CtxBlockOacc<TDim, TIdx> blockShared(
131  gridBlockExtent,
132  blockThreadExtent,
133  threadElemExtent,
134  b,
135  blockSharedMemDynSizeBytes,
136  gridsLock,
137  blocksLock);
138 
139 // Execute the threads in parallel.
140 
141 // Parallel execution of the threads in a block is required because when
142 // syncBlockThreads is called all of them have to be done with their work up
143 // to this line. So we have to spawn one OS thread per thread in a block.
144 //! \warning The OpenACC is technically allowed to ignore the value in the num_workers clause
145 //! and could run fewer threads. The standard provides no way to check how many worker threads are running.
146 //! If fewer threads are run, syncBlockThreads will dead-lock. It is up to the developer/user
147 //! to choose a blockThreadCount which the runtime will respect.
148 # pragma acc loop worker
149  for(TIdx w = 0; w < blockThreadCount; ++w)
150  {
151  AccOacc<TDim, TIdx> acc(w, blockShared);
152 
153  core::apply(
154  [kernelFnObj, &acc](typename std::decay<TArgs>::type const&... args)
155  { kernelFnObj(acc, args...); },
156  argsD);
157  }
158  freeSharedVars(blockShared);
159  }
160  }
161  }
162  }
163 
164  private:
165  TKernelFnObj m_kernelFnObj;
166  core::Tuple<std::decay_t<TArgs>...> m_args;
167  };
