41  //! Maps a 1 dimensional index to a N dimensional index.
42  template<std::size_t TidxDimOut>
43  struct MapIdx<TidxDimOut, 1u, std::enable_if_t<TidxDimOut != 1u>>
44  {
45  // \tparam TElem Type of the index values.
46  // \param idx Idx to be mapped.
47  // \param extent Spatial size to map the index to
48  // \return A N dimensional vector.
49  ALPAKA_NO_HOST_ACC_WARNING
50  template<typename TElem>
51  ALPAKA_FN_HOST_ACC static auto mapIdx(
52  Vec<DimInt<1u>, TElem> const& idx,
53  Vec<DimInt<TidxDimOut>, TElem> const& extent) -> Vec<DimInt<TidxDimOut>, TElem>
54  {
55  auto idxNd = Vec<DimInt<TidxDimOut>, TElem>::all(0u);
56 
57  constexpr std::size_t lastIdx(TidxDimOut - 1u);
58 
59  // fast-dim
60  idxNd[lastIdx] = static_cast<TElem>(idx[0u] % extent[lastIdx]);
61 
62  // in-between
63  TElem hyperPlanesBefore = extent[lastIdx];
64 #if BOOST_COMP_PGI && (defined(ALPAKA_ACC_ANY_BT_OMP5_ENABLED) || defined(ALPAKA_ACC_ANY_BT_OACC_ENABLED)) \
65  && !defined(TPR30645)
66  for(std::size_t a(0u); a < lastIdx - 1; ++a)
67  {
68  const auto r = a + 1; // NVHPC does sometimes not understand, that loops which do not start at zero
69  // can have zero iterations
70 #else
71  for(std::size_t r(1u); r < lastIdx; ++r)
72  {
73 #endif
74  std::size_t const d = lastIdx - r;
75  idxNd[d] = static_cast<TElem>(idx[0u] / hyperPlanesBefore % extent[d]);
76  hyperPlanesBefore *= extent[d];
77  }
78 
79  // slow-dim
80  idxNd[0u] = static_cast<TElem>(idx[0u] / hyperPlanesBefore);
81 
82  return idxNd;
83  }
84  };

149  template<std::size_t TidxDimOut>
150  struct MapIdxPitchBytes<TidxDimOut, 1u, typename std::enable_if<TidxDimOut != 1u>::type>
151  {
152  // \tparam TElem Type of the index values.
153  // \param idx Idx to be mapped.
154  // \param pitch Spatial pitch (in elems) to map the index to
155  // \return N dimensional vector.
156  ALPAKA_NO_HOST_ACC_WARNING
157  template<typename TElem>
158  ALPAKA_FN_HOST_ACC static auto mapIdxPitchBytes(
159  Vec<DimInt<1u>, TElem> const& idx,
160  Vec<DimInt<TidxDimOut>, TElem> const& pitch) -> Vec<DimInt<TidxDimOut>, TElem>
161  {
162  auto idxNd = Vec<DimInt<TidxDimOut>, TElem>::all(0u);
163 
164  constexpr std::size_t lastIdx = TidxDimOut - 1u;
165 
166  TElem tmp = idx[0u];
167  for(std::size_t d(0u); d < lastIdx; ++d)
168  {
169  idxNd[d] = static_cast<TElem>(tmp / pitch[d + 1]);
170  tmp %= pitch[d + 1];
171  }
172  idxNd[lastIdx] = tmp;
173 
174  return idxNd;
175  }
176  };
