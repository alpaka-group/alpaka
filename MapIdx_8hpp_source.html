36  //! Maps a 1 dimensional index to a N dimensional index.
37  template<std::size_t TidxDimOut>
38  struct MapIdx<TidxDimOut, 1u, std::enable_if_t<(TidxDimOut > 1u)>>
39  {
40  // \tparam TElem Type of the index values.
41  // \param idx Idx to be mapped.
42  // \param extent Spatial size to map the index to
43  // \return A N dimensional vector.
44  ALPAKA_NO_HOST_ACC_WARNING
45  template<typename TElem>
46  ALPAKA_FN_HOST_ACC static auto mapIdx(
47  Vec<DimInt<1u>, TElem> const& idx,
48  Vec<DimInt<TidxDimOut>, TElem> const& extent) -> Vec<DimInt<TidxDimOut>, TElem>
49  {
50  auto idxNd = Vec<DimInt<TidxDimOut>, TElem>::all(0u);
51 
52  constexpr std::size_t lastIdx(TidxDimOut - 1u);
53 
54  // fast-dim
55  idxNd[lastIdx] = static_cast<TElem>(idx[0u] % extent[lastIdx]);
56 
57  // in-between
58  TElem hyperPlanesBefore = extent[lastIdx];
59  for(std::size_t r(1u); r < lastIdx; ++r)
60  {
61  std::size_t const d = lastIdx - r;
62  idxNd[d] = static_cast<TElem>(idx[0u] / hyperPlanesBefore % extent[d]);
63  hyperPlanesBefore *= extent[d];
64  }
65 
66  // slow-dim
67  idxNd[0u] = static_cast<TElem>(idx[0u] / hyperPlanesBefore);
68 
69  return idxNd;
70  }
71  };

163  // \tparam TElem Type of the index values.

165  // \param pitch Spatial pitch (in elems) to map the index to
166  // \return N dimensional vector.
167  ALPAKA_NO_HOST_ACC_WARNING
168  template<typename TElem>
169  ALPAKA_FN_HOST_ACC static auto mapIdxPitchBytes(
170  Vec<DimInt<1u>, TElem> const& idx,
171  Vec<DimInt<TidxDimOut>, TElem> const& pitch) -> Vec<DimInt<TidxDimOut>, TElem>
172  {
173  auto idxNd = Vec<DimInt<TidxDimOut>, TElem>::all(0u);
174 
175  constexpr std::size_t lastIdx = TidxDimOut - 1u;
176 
177  TElem tmp = idx[0u];
178  for(std::size_t d(0u); d < lastIdx; ++d)
179  {
180  idxNd[d] = static_cast<TElem>(tmp / pitch[d + 1]);
181  tmp %= pitch[d + 1];
182  }
183  idxNd[lastIdx] = tmp;
184 
185  return idxNd;
186  }
187  };
