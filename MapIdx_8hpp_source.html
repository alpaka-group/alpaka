15  //! Maps an N-dimensional index to an N-dimensional position. At least one dimension must always be 1 or zero.
16  //!
17  //! \tparam TDimOut Dimension of the index vector to map to.
18  //! \param in The index vector to map from.
19  //! \param extent The extents of the input or output space, whichever has more than 1 dimensions.
20  ALPAKA_NO_HOST_ACC_WARNING template<
21  std::size_t TDimOut,
22  std::size_t TDimIn,
23  std::size_t TDimExtents,
24  typename TElem>
25  ALPAKA_FN_HOST_ACC auto mapIdx(Vec<DimInt<TDimIn>, TElem> const& in, Vec<DimInt<TDimExtents>, TElem> const& extent)
26  -> Vec<DimInt<TDimOut>, TElem>
27  {
28  if constexpr(TDimOut == 0 || TDimIn == 0)
29  return Vec<DimInt<TDimOut>, TElem>::zeros();
30  else if constexpr(TDimOut == TDimIn)
31  return in;
32  else if constexpr(TDimOut == 1)
33  {
34  TElem out = in[0];
35  for(std::size_t d = 1; d < TDimIn; ++d)
36  out = static_cast<TElem>(out * extent[d] + in[d]);
37  return {out};
38  }
39  else if constexpr(TDimIn == 1)
40  {
41  auto flat = in.front();
42  Vec<DimInt<TDimOut>, TElem> out;
43  for(std::size_t d = TDimOut - 1u; d > 0; d--)
44  {
45  out[d] = static_cast<TElem>(flat % extent[d]);
46  flat /= extent[d];
47  }
48  out.front() = static_cast<TElem>(flat);
49  return out;
50  }
51  else
52  static_assert(!sizeof(TElem), "Not implemented");
53  }

56  //! byte view. At least one dimension must always be 1 or zero.
57  //!
58  //! \tparam TDimOut Dimension of the index vector to map to.

61  ALPAKA_NO_HOST_ACC_WARNING
62  template<std::size_t TDimOut, std::size_t TDimIn, std::size_t TidxDimPitch, typename TElem>
63  ALPAKA_FN_HOST_ACC auto mapIdxPitchBytes(
64  Vec<DimInt<TDimIn>, TElem> const& in,
65  Vec<DimInt<TidxDimPitch>, TElem> const& pitches) -> Vec<DimInt<TDimOut>, TElem>

68  return Vec<DimInt<TDimOut>, TElem>::zeros();

70  return in;
