109  //! This accelerator allows parallel kernel execution on an OpenMP target device.

119  >

184  //! The OpenMP 5.0 accelerator device properties get trait specialization.
185  template<typename TDim, typename TIdx>
186  struct GetAccDevProps<AccOmp5<TDim, TIdx>>
187  {
188  ALPAKA_FN_HOST static auto getAccDevProps(DevOmp5 const& /* dev */) -> AccDevProps<TDim, TIdx>
189  {
190 # if defined(ALPAKA_OFFLOAD_MAX_BLOCK_SIZE) && ALPAKA_OFFLOAD_MAX_BLOCK_SIZE > 0
191  auto const blockThreadCount = std::min(::omp_get_max_threads(), ALPAKA_OFFLOAD_MAX_BLOCK_SIZE);
192 # else
193  auto const blockThreadCount = ::omp_get_max_threads();
194 # endif
195 # ifdef ALPAKA_CI
196  auto const blockThreadCountMax(alpaka::core::clipCast<TIdx>(std::min(4, blockThreadCount)));
197  auto const gridBlockCountMax(alpaka::core::clipCast<TIdx>(std::min(4, ::omp_get_max_threads())));
198 # else
199  auto const blockThreadCountMax(alpaka::core::clipCast<TIdx>(blockThreadCount));
200  //! \todo for a later OpenMP (or when compilers work with a GPU target): fix max block size for target
201  //! On CPU we would want
202  //! gridBlockCountMax = ::omp_get_max_threads() / blockThreadCountMax
203  //! but this would lead to only one block running on GPU, or too small blocks (see
204  //! ALPAKA_OFFLOAD_MAX_BLOCK_SIZE). OpenMP 5.0 may actually mandate, that
205  //! ::omp_get_max_threads() == max_teams * threads_per_team ,
206  //! however with the maximum grid size (i.e. max_teams) being INT_MAX this may not work.
207  //! We actually want to set
208  //! gridBlockCountMax = ::omp_get_max_teams()
209  //! but there is no function ::omp_get_max_teams().
210  //! Instead we set ::omp_get_max_threads() again, to have a
211  //! number which does not kill CPUs and is reasonable
212  //! (::omp_get_max_threads() seems to return the block size)
213  //! for GPUs.
214  auto const gridBlockCountMax(alpaka::core::clipCast<TIdx>(::omp_get_max_threads()));
215 # endif
216  return {// m_multiProcessorCount
217  static_cast<TIdx>(gridBlockCountMax),
218  // m_gridBlockExtentMax
219  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
220  // m_gridBlockCountMax
221  std::numeric_limits<TIdx>::max(),
222  // m_blockThreadExtentMax
223  Vec<TDim, TIdx>::all(blockThreadCountMax),
224  // m_blockThreadCountMax
225  blockThreadCountMax,
226  // m_threadElemExtentMax
227  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
228  // m_threadElemCountMax
229  std::numeric_limits<TIdx>::max(),
230  // m_sharedMemSizeBytes
231  AccOmp5<TDim, TIdx>::staticAllocBytes()};
232  }
233  };
