180  //! The OpenMP 5.0 accelerator device properties get trait specialization.
181  template<typename TDim, typename TIdx>
182  struct GetAccDevProps<AccOmp5<TDim, TIdx>>
183  {
184  ALPAKA_FN_HOST static auto getAccDevProps(DevOmp5 const& /* dev */) -> AccDevProps<TDim, TIdx>
185  {
186 # if defined(ALPAKA_OFFLOAD_MAX_BLOCK_SIZE) && ALPAKA_OFFLOAD_MAX_BLOCK_SIZE > 0
187  auto const blockThreadCount = std::min(::omp_get_max_threads(), ALPAKA_OFFLOAD_MAX_BLOCK_SIZE);
188 # else
189  auto const blockThreadCount = ::omp_get_max_threads();
190 # endif
191 # ifdef ALPAKA_CI
192  auto const blockThreadCountMax(alpaka::core::clipCast<TIdx>(std::min(4, blockThreadCount)));
193  auto const gridBlockCountMax(alpaka::core::clipCast<TIdx>(std::min(4, ::omp_get_max_threads())));
194 # else
195  auto const blockThreadCountMax(alpaka::core::clipCast<TIdx>(blockThreadCount));
196  //! \todo for a later OpenMP (or when compilers work with a GPU target): fix max block size for target
197  //! On CPU we would want
198  //! gridBlockCountMax = ::omp_get_max_threads() / blockThreadCountMax
199  //! but this would lead to only one block running on GPU, or too small blocks (see
200  //! ALPAKA_OFFLOAD_MAX_BLOCK_SIZE). OpenMP 5.0 may actually mandate, that
201  //! ::omp_get_max_threads() == max_teams * threads_per_team ,
202  //! however with the maximum grid size (i.e. max_teams) being INT_MAX this may not work.
203  //! We actually want to set
204  //! gridBlockCountMax = ::omp_get_max_teams()
205  //! but there is no function ::omp_get_max_teams().
206  //! Instead we set ::omp_get_max_threads() again, to have a
207  //! number which does not kill CPUs and is reasonable
208  //! (::omp_get_max_threads() seems to return the block size)
209  //! for GPUs.
210  auto const gridBlockCountMax(alpaka::core::clipCast<TIdx>(::omp_get_max_threads()));
211 # endif
212  return {// m_multiProcessorCount
213  static_cast<TIdx>(gridBlockCountMax),
214  // m_gridBlockExtentMax
215  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
216  // m_gridBlockCountMax
217  std::numeric_limits<TIdx>::max(),
218  // m_blockThreadExtentMax
219  Vec<TDim, TIdx>::all(blockThreadCountMax),
220  // m_blockThreadCountMax
221  blockThreadCountMax,
222  // m_threadElemExtentMax
223  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
224  // m_threadElemCountMax
225  std::numeric_limits<TIdx>::max(),
226  // m_sharedMemSizeBytes
227  AccOmp5<TDim, TIdx>::staticAllocBytes()};
228  }
229  };
