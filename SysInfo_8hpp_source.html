80  //! \return The name of the CPU the code is running on.
81  inline auto getCpuName() -> std::string
82  {
83  // Get extended ids.
84  std::uint32_t ex[4] = {0};
85  cpuid(0x80000000, 0, ex);
86  std::uint32_t const nExIds(ex[0]);
87 
88  if(!nExIds)
89  {
90  switch(ex[1])
91  {
92  case UNKNOWN_COMPILER:
93  return "<unknown: compiler>";
94  case UNKNOWN_CPU:
95  return "<unknown: CPU>";
96  default:
97  return "<unknown>";
98  }
99  }
100 #if BOOST_ARCH_X86
101  // Get the information associated with each extended ID.
102  char cpuBrandString[0x40] = {0};
103  for(std::uint32_t i(0x80000000); i <= nExIds; ++i)
104  {
105  cpuid(i, 0, ex);
106 
107  // Interpret CPU brand string and cache information.
108  if(i == 0x80000002)
109  {
110  std::memcpy(cpuBrandString, ex, sizeof(ex));
111  }
112  else if(i == 0x80000003)
113  {
114  std::memcpy(cpuBrandString + 16, ex, sizeof(ex));
115  }
116  else if(i == 0x80000004)
117  {
118  std::memcpy(cpuBrandString + 32, ex, sizeof(ex));
119  }
120  }
121  return std::string(cpuBrandString);
122 #else
123  return std::string("unknown");
124 #endif
125  }
126  //! \return The total number of bytes of global memory.
127  //! Adapted from David Robert Nadeau:
128  //! http://nadeausoftware.com/articles/2012/09/c_c_tip_how_get_physical_memory_size_system
129  inline auto getTotalGlobalMemSizeBytes() -> std::size_t
130  {
131 #if BOOST_OS_WINDOWS
132  MEMORYSTATUSEX status;
133  status.dwLength = sizeof(status);
134  GlobalMemoryStatusEx(&status);
135  return static_cast<std::size_t>(status.ullTotalPhys);
136 
137 #elif BOOST_OS_CYGWIN
138  // New 64-bit MEMORYSTATUSEX isn't available.
139  MEMORYSTATUS status;
140  status.dwLength = sizeof(status);
141  GlobalMemoryStatus(&status);
142  return static_cast<std::size_t>(status.dwTotalPhys);
143 
144 #elif BOOST_OS_UNIX || BOOST_OS_MACOS
145  // Unix : Prefer sysctl() over sysconf() except sysctl() with HW_REALMEM and HW_PHYSMEM which are not
146  // always reliable
147 # if defined(CTL_HW) && (defined(HW_MEMSIZE) || defined(HW_PHYSMEM64))
148  int mib[2]
149  = { CTL_HW,
150 # if defined(HW_MEMSIZE) // OSX
151  HW_MEMSIZE
152 # elif defined(HW_PHYSMEM64) // NetBSD, OpenBSD.
153  HW_PHYSMEM64
154 # endif
155  };
156  std::uint64_t size(0);
157  std::size_t sizeLen{sizeof(size)};
158  if(sysctl(mib, 2, &size, &sizeLen, nullptr, 0) < 0)
159  {
160  throw std::logic_error("getTotalGlobalMemSizeBytes failed calling sysctl!");
161  }
162  return static_cast<std::size_t>(size);
163 
164 # elif defined(_SC_AIX_REALMEM) // AIX.
165  return static_cast<std::size_t>(sysconf(_SC_AIX_REALMEM)) * static_cast<std::size_t>(1024);
166 
167 # elif defined(_SC_PHYS_PAGES) && defined(_SC_PAGESIZE) // Linux, FreeBSD, OpenBSD, Solaris.
168  return static_cast<std::size_t>(sysconf(_SC_PHYS_PAGES))
169  * static_cast<std::size_t>(sysconf(_SC_PAGESIZE));
170 
171 # elif defined(_SC_PHYS_PAGES) && defined(_SC_PAGE_SIZE) // Legacy.
172  return static_cast<std::size_t>(sysconf(_SC_PHYS_PAGES))
173  * static_cast<std::size_t>(sysconf(_SC_PAGE_SIZE));
174 
175 # elif defined(CTL_HW) \
176  && (defined(HW_PHYSMEM) || defined(HW_REALMEM)) // FreeBSD, DragonFly BSD, NetBSD, OpenBSD, and OSX.
177  int mib[2]
178  = { CTL_HW,
179 # if defined(HW_REALMEM) // FreeBSD.
180  HW_REALMEM;
181 # elif defined(HW_PYSMEM) // Others.
182  HW_PHYSMEM;
183 # endif
184  };

187  if(sysctl(mib, 2, &size, &sizeLen, nullptr, 0) < 0)

190  }
191  return static_cast<std::size_t>(size);
192 # endif

198  //! \return The free number of bytes of global memory.
199  //! \throws std::logic_error if not implemented on the system and std::runtime_error on other errors.
200  inline auto getFreeGlobalMemSizeBytes() -> std::size_t
201  {
202 #if BOOST_OS_WINDOWS
203  MEMORYSTATUSEX status;
204  status.dwLength = sizeof(status);
205  GlobalMemoryStatusEx(&status);
206  return static_cast<std::size_t>(status.ullAvailPhys);
207 
208 #elif BOOST_OS_LINUX
209  std::string token;
210  std::ifstream file("/proc/meminfo");
211  if(file)
212  {
213  while(file >> token)
214  {
215  if(token == "MemFree:")
216  {
217  std::size_t freeGlobalMemSizeBytes(0);
218  if(file >> freeGlobalMemSizeBytes)
219  {
220  return freeGlobalMemSizeBytes * size_t(1024);
221  }
222  else
223  {
224  throw std::runtime_error("Unable to read MemFree value!");
225  }
226  }
227  }
228  throw std::runtime_error("Unable to find MemFree in '/proc/meminfo'!");
229  }
230  else
231  {
232  throw std::runtime_error("Unable to open '/proc/meminfo'!");
233  }
234 #elif BOOST_OS_MACOS
235  int free_pages = 0;
236  std::size_t len = sizeof(free_pages);
237  if(sysctlbyname("vm.page_free_count", &free_pages, &len, nullptr, 0) < 0)
238  {
239  throw std::logic_error("getFreeGlobalMemSizeBytes failed calling sysctl(vm.page_free_count)!");
240  }
241  int page_size = 0;
242  len = sizeof(page_size);
243  if(sysctlbyname("vm.pagesize", &page_size, &len, nullptr, 0) < 0)
244  {
245  throw std::logic_error("getFreeGlobalMemSizeBytes failed calling sysctl(vm.pagesize)!");
246  }
247  return static_cast<std::size_t>(free_pages) * static_cast<std::size_t>(page_size);
248 #else
249 # error "getFreeGlobalMemSizeBytes not implemented for this system!"
250 #endif
251  }
