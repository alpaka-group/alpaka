80  // Don't call this without locking first!
81  auto clean_dependencies() -> void
82  {
83  // Clean up completed events
84  auto const start = std::begin(m_dependencies);
85  auto const old_end = std::end(m_dependencies);
86  auto const new_end = std::remove_if(
87  start,
88  old_end,
89  [](sycl::event ev) {
90  return ev.get_info<sycl::info::event::command_execution_status>()
91  == sycl::info::event_command_status::complete;
92  });
93 
94  m_dependencies.erase(new_end, old_end);
95  }

124  template<bool TBlocking, typename TTask>
125  auto enqueue(TTask const& task) -> void
126  {
127  {
128  std::lock_guard<std::shared_mutex> lock{m_mutex};
129 
130  clean_dependencies();
131 
132  // Execute task
133  m_last_event = m_queue.submit(
134  [this, &task](sycl::handler& cgh)
135  {
136  if(!m_dependencies.empty())
137  cgh.depends_on(m_dependencies);
138 
139  if constexpr(is_sycl_kernel<TTask>) // Kernel
140  task(cgh, m_fence_dummy); // Will call cgh.parallel_for internally
141  else if constexpr(is_sycl_task<TTask>) // Copy / Fill
142  task(cgh); // Will call cgh.{copy, fill} internally
143  else // Host
144  cgh.host_task(task);
145  });
146 
147  m_dependencies.clear();
148  }
149 
150  if constexpr(TBlocking)
151  wait();
152  }
