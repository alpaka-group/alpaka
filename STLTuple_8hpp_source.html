61 /// \struct Tuple
62 /// \brief is a fixed-size collection of heterogeneous values
63 /// \ztparam Ts... - the types of the elements that the tuple stores.
64 /// Empty list is supported.
65 template <class... Ts>
66 struct Tuple {};

68 /// \brief specialisation of the \ref Tuple class when the tuple has at least
69 /// one element.
70 /// \tparam T : the type of the first element in the tuple.
71 /// \tparam Ts... the rest of the elements in the tuple. Ts... can be empty.
72 template <class T, class... Ts>
73 struct Tuple<T, Ts...> {
74  Tuple(T t, Ts... ts) : head(t), tail(ts...) {}
75  T head;
76  Tuple<Ts...> tail;
77 };

109 ///\ struct ElemTypeHolder
110 /// \brief ElemTypeHolder class is used to specify the types of the
111 /// elements inside the tuple
112 /// \tparam size_t the number of elements inside the tuple
113 /// \tparam class the tuple class
114 template <size_t, class>

117 /// \brief specialisation of the \ref ElemTypeHolder class when the number of
118 /// elements inside the tuple is 1
119 template <class T, class... Ts>
120 struct ElemTypeHolder<0, Tuple<T, Ts...>> {
121  typedef T type;
122 };
123 
124 /// \brief specialisation of the \ref ElemTypeHolder class when the number of
125 /// elements inside the tuple is bigger than 1. It recursively calls itself to
126 /// detect the type of each element in the tuple
127 /// \tparam T : the type of the first element in the tuple.
128 /// \tparam Ts... the rest of the elements in the tuple. Ts... can be empty.
129 /// \tparam K is the Kth element in the tuple
130 template <size_t k, class T, class... Ts>
131 struct ElemTypeHolder<k, Tuple<T, Ts...>> {
132  typedef typename ElemTypeHolder<k - 1, Tuple<Ts...>>::type type;
133 };

135 /// get
136 /// \brief Extracts the first element from the tuple.
137 /// K=0 represents the first element of the tuple. The tuple cannot be empty.
138 /// \tparam Ts... are the type of the elements in the tuple.
139 /// \param t is the tuple whose contents to extract
140 /// \return typename ElemTypeHolder<0, Tuple<Ts...> >::type &>::type
141 
142 #define TERMINATE_CONDS_TUPLE_GET(CVQual) \
143  template <size_t k, class... Ts> \
144  typename StaticIf<k == 0, CVQual \
145  typename ElemTypeHolder<0, Tuple<Ts...>>::type&>::type \
146  get(CVQual Tuple<Ts...>& t) { \
147  static_assert(sizeof...(Ts) != 0, \
148  "The requseted value is bigger than the size of the tuple"); \
149  return t.head; \
150  }
151 
152 TERMINATE_CONDS_TUPLE_GET(const)
153 TERMINATE_CONDS_TUPLE_GET()
154 #undef TERMINATE_CONDS_TUPLE_GET
155 /// get
156 /// \brief Extracts the Kth element from the tuple.
157 ///\tparam K is an integer value in [0,sizeof...(Types)).
158 /// \tparam T is the (sizeof...(Types) -(K+1)) element in the tuple
159 /// \tparam Ts... are the type of the elements in the tuple.
160 /// \param t is the tuple whose contents to extract
161 /// \return typename ElemTypeHolder<K, Tuple<Ts...> >::type &>::type
162 #define RECURSIVE_TUPLE_GET(CVQual) \
163  template <size_t k, class T, class... Ts> \
164  typename StaticIf<k != 0, CVQual \
165  typename ElemTypeHolder<k, Tuple<T, Ts...>>::type&>::type \
166  get(CVQual Tuple<T, Ts...>& t) { \
167  return utility::tuple::get<k - 1>(t.tail); \
168  }
169 RECURSIVE_TUPLE_GET(const)

171 #undef RECURSIVE_TUPLE_GET

200 /// \struct RangeBuilder
201 /// \brief Collects internal details for generating index ranges [MIN, MAX)
202 /// Declare primary template for index range builder
203 /// \tparam MIN is the starting index in the tuple
204 /// \tparam N represents sizeof..(elemens)- sizeof...(Is)
205 /// \tparam Is... are the list of generated index so far
206 template <size_t MIN, size_t N, size_t... Is>
207 struct RangeBuilder;
208 
209 /// \brief base Step: Specialisation of the \ref RangeBuilder when the
210 /// MIN==MAX. In this case the Is... is [0 to sizeof...(tuple elements))
211 /// \tparam MIN is the starting index of the tuple
212 /// \tparam Is is [0 to sizeof...(tuple elements))
213 template <size_t MIN, size_t... Is>
214 struct RangeBuilder<MIN, MIN, Is...> {
215  typedef IndexList<Is...> type;
216 };
217 
218 /// Induction step: Specialisation of the RangeBuilder class when N!=MIN
219 /// in this case we are recursively subtracting N by one and adding one
220 /// index to Is... list until MIN==N
221 /// \tparam MIN is the starting index in the tuple
222 /// \tparam N represents sizeof..(elemens)- sizeof...(Is)
223 /// \tparam Is... are the list of generated index so far
224 template <size_t MIN, size_t N, size_t... Is>
225 struct RangeBuilder : public RangeBuilder<MIN, N - 1, N - 1, Is...> {};

233 /// append_base
234 /// \brief unpacking the elements of the input tuple t and creating a new tuple
235 /// by adding element a at the end of it.
236 ///\tparam Args... the type of the elements inside the tuple t
237 /// \tparam T the type of the new element going to be added at the end of tuple
238 /// \tparam I... is the list of index from [0 to sizeof...(t))
239 /// \param t the tuple on which we want to append a.
240 /// \param a the new elements going to be added to the tuple
241 /// \return Tuple<Args..., T>
242 template <typename... Args, typename T, size_t... I>
243 Tuple<Args..., T> append_base(Tuple<Args...> t, T a, IndexList<I...>) {
244  return utility::tuple::make_tuple(get<I>(t)..., a);
245 }

247 /// append
248 /// \brief the deduction function for \ref append_base that automatically
249 /// generate the \ref IndexRange
250 ///\tparam Args... the type of the elements inside the tuple t
251 /// \tparam T the type of the new element going to be added at the end of tuple
252 /// \param t the tuple on which we want to append a.
253 /// \param a the new elements going to be added to the tuple
254 /// \return Tuple<Args..., T>
255 template <typename... Args, typename T>
256 Tuple<Args..., T> append(Tuple<Args...> t, T a) {
257  return utility::tuple::append_base(t, a, IndexRange<0, sizeof...(Args)>());
258 }

260 /// append_base
261 /// \brief This is a specialisation of \ref append_base when we want to
262 /// concatenate
263 /// tuple t2 at the end of the tuple t1. Here we unpack both tuples, generate
264 /// the IndexRange for each of them and create an output tuple T that contains
265 /// both elements of t1 and t2.
266 ///\tparam Args1... the type of the elements inside the tuple t1
267 ///\tparam Args2... the type of the elements inside the tuple t2
268 /// \tparam I1... is the list of index from [0 to sizeof...(t1))
269 /// \tparam I2... is the list of index from [0 to sizeof...(t2))
270 /// \param t1 is the tuple on which we want to append t2.
271 /// \param t2 is the tuple that is going to be added on t1.
272 /// \return Tuple<Args1..., Args2...>
273 template <typename... Args1, typename... Args2, size_t... I1, size_t... I2>
274 Tuple<Args1..., Args2...> append_base(Tuple<Args1...> t1, Tuple<Args2...> t2,
275  IndexList<I1...>, IndexList<I2...>) {
276  return utility::tuple::make_tuple(get<I1>(t1)..., get<I2>(t2)...);
277 }

279 /// append
280 /// \brief deduction function for \ref append_base when we are appending tuple
281 /// t1 by tuple t2. In this case the \ref IndexRange for both tuple are
282 /// automatically generated.
283 ///\tparam Args1... the type of the elements inside the tuple t1
284 ///\tparam Args2... the type of the elements inside the tuple t2
285 /// \param t1 is the tuple on which we want to append t2.
286 /// \param t2 is the tuple that is going to be added on t1.
287 /// \return Tuple<Args1..., Args2...>
288 template <typename... Args1, typename... Args2>
289 Tuple<Args1..., Args2...> append(Tuple<Args1...> t1, Tuple<Args2...> t2) {
290  return utility::tuple::append_base(t1, t2, IndexRange<0, sizeof...(Args1)>(),
291  IndexRange<0, sizeof...(Args2)>());
