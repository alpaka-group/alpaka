41  //#############################################################################
42  //! The CUDA RT device event implementation.
43  class EventCudaImpl final
44  {
45  public:
46  //-----------------------------------------------------------------------------
47  ALPAKA_FN_HOST EventCudaImpl(
48  dev::DevCudaRt const & dev,
49  bool bBusyWait) :
50  m_dev(dev),
51  m_CudaEvent()
52  {
53  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
54 
55  // Set the current device.
56  ALPAKA_CUDA_RT_CHECK(
57  cudaSetDevice(
58  m_dev.m_iDevice));
59  // Create the event on the current device with the specified flags. Valid flags include:
60  // - cudaEventDefault: Default event creation flag.
61  // - cudaEventBlockingSync : Specifies that event should use blocking synchronization.
62  // A host thread that uses cudaEventSynchronize() to wait on an event created with this flag will block until the event actually completes.
63  // - cudaEventDisableTiming : Specifies that the created event does not need to record timing data.
64  // Events created with this flag specified and the cudaEventBlockingSync flag not specified will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
65  ALPAKA_CUDA_RT_CHECK(
66  cudaEventCreateWithFlags(
67  &m_CudaEvent,
68  (bBusyWait ? cudaEventDefault : cudaEventBlockingSync) | cudaEventDisableTiming));
69  }
70  //-----------------------------------------------------------------------------
71  EventCudaImpl(EventCudaImpl const &) = delete;
72  //-----------------------------------------------------------------------------
73  EventCudaImpl(EventCudaImpl &&) = default;
74  //-----------------------------------------------------------------------------
75  auto operator=(EventCudaImpl const &) -> EventCudaImpl & = delete;
76  //-----------------------------------------------------------------------------
77  auto operator=(EventCudaImpl &&) -> EventCudaImpl & = delete;
78  //-----------------------------------------------------------------------------
79  ALPAKA_FN_HOST ~EventCudaImpl()
80  {
81  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
82 
83  // Set the current device. \TODO: Is setting the current device before cudaEventDestroy required?
84  ALPAKA_CUDA_RT_CHECK(cudaSetDevice(
85  m_dev.m_iDevice));
86  // In case event has been recorded but has not yet been completed when cudaEventDestroy() is called, the function will return immediately
87  // and the resources associated with event will be released automatically once the device has completed event.
88  // -> No need to synchronize here.
89  ALPAKA_CUDA_RT_CHECK(cudaEventDestroy(
90  m_CudaEvent));
91  }
92 
93  public:
94  dev::DevCudaRt const m_dev; //!< The device this event is bound to.
95  cudaEvent_t m_CudaEvent;
96  };
97  }
