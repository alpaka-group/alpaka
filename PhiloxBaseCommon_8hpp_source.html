38  /** Common class for Philox family engines
39  *
40  * Checks the validity of passed-in parameters and calls the \a TBackend methods to perform N rounds of the
41  * Philox shuffle.
42  *
43  * @tparam TBackend device-dependent backend, specifies the array types
44  * @tparam TParams Philox algorithm parameters \sa PhiloxParams
45  * @tparam TImpl engine type implementation (CRTP)
46  */
47  template<typename TBackend, typename TParams, typename TImpl>
48  class PhiloxBaseCommon
49  : public TBackend
50  , public PhiloxConstants<TParams>
51  {
52  static unsigned const numRounds = TParams::rounds;
53  static unsigned const vectorSize = TParams::counterSize;
54  static unsigned const numberWidth = TParams::width;
55 
56  static_assert(numRounds > 0, "Number of Philox rounds must be > 0.");
57  static_assert(vectorSize % 2 == 0, "Philox counter size must be an even number.");
58  static_assert(vectorSize <= 16, "Philox SP network is not specified for sizes > 16.");
59  static_assert(numberWidth % 8 == 0, "Philox number width in bits must be a multiple of 8.");
60 
61  // static_assert(TWidth == 32 || TWidth == 64, "Philox implemented only for 32 and 64 bit numbers.");
62  static_assert(numberWidth == 32, "Philox implemented only for 32 bit numbers.");
63 
64  public:
65  using Counter = typename TBackend::Counter;
66  using Key = typename TBackend::Key;
67 
68  protected:
69  /** Single round of the Philox shuffle
70  *
71  * @param counter state of the counter
72  * @param key value of the key
73  * @return shuffled counter
74  */
75  ALPAKA_FN_HOST_ACC auto singleRound(Counter const& counter, Key const& key)
76  {
77  std::uint32_t H0, L0, H1, L1;
78  multiplyAndSplit64to32(counter[0], this->MULTIPLITER_4x32_0, H0, L0);
79  multiplyAndSplit64to32(counter[2], this->MULTIPLITER_4x32_1, H1, L1);
80  return Counter{H1 ^ counter[1] ^ key[0], L1, H0 ^ counter[3] ^ key[1], L0};
81  }
82 
83  /** Bump the \a key by the Weyl sequence step parameter
84  *
85  * @param key the key to be bumped
86  * @return the bumped key
87  */
88  ALPAKA_FN_HOST_ACC auto bumpKey(Key const& key)
89  {
90  return Key{key[0] + this->WEYL_32_0, key[1] + this->WEYL_32_1};
91  }
92 
93  /** Advance the \a counter to the next state
94  *
95  * Increments the passed-in \a counter by one with a 128-bit carry.
96  *
97  * @param counter reference to the counter which is to be advanced
98  */
99  ALPAKA_FN_HOST_ACC void advanceCounter(Counter& counter)
100  {
101  counter[0]++;
102  /* 128-bit carry */
103  if(counter[0] == 0)
104  {
105  counter[1]++;
106  if(counter[1] == 0)
107  {
108  counter[2]++;
109  if(counter[2] == 0)
110  {
111  counter[3]++;
112  }
113  }
114  }
115  }
116 
117  /** Advance the internal state counter by \a offset N-vectors (N = counter size)
118  *
119  * Advances the internal value of this->state.counter
120  *
121  * @param offset number of N-vectors to skip
122  */
123  ALPAKA_FN_HOST_ACC void skip4(uint64_t offset)
124  {
125  Counter& counter = static_cast<TImpl*>(this)->state.counter;
126  Counter temp = counter;
127  counter[0] += low32Bits(offset);
128  counter[1] += high32Bits(offset) + (counter[0] < temp[0] ? 1 : 0);
129  counter[2] += (counter[0] < temp[1] ? 1u : 0u);
130  counter[3] += (counter[0] < temp[2] ? 1u : 0u);
131  }
132 
133  /** Advance the counter by the length of \a subsequence
134  *
135  * Advances the internal value of this->state.counter
136  *
137  * @param subsequence number of subsequences to skip
138  */
139  ALPAKA_FN_HOST_ACC void skipSubsequence(uint64_t subsequence)
140  {
141  Counter& counter = static_cast<TImpl*>(this)->state.counter;
142  Counter temp = counter;
143  counter[2] += low32Bits(subsequence);
144  counter[3] += high32Bits(subsequence) + (counter[2] < temp[2] ? 1 : 0);
145  }
146 
147  /** Performs N rounds of the Philox shuffle
148  *
149  * @param counter_in initial state of the counter
150  * @param key_in initial state of the key
151  * @return result of the PRNG shuffle; has the same size as the counter
152  */
153  ALPAKA_FN_HOST_ACC auto nRounds(Counter const& counter_in, Key const& key_in) -> Counter
154  {
155  Key key{key_in};
156  Counter counter = singleRound(counter_in, key);
157 
158  // TODO: Consider unrolling the loop for performance
159  for(unsigned int n = 0; n < numRounds; ++n)
160  {
161  key = bumpKey(key);
162  counter = singleRound(counter, key);
163  }
164  // TODO: Should the key be returned as well??? i.e. should the original key be bumped?
165 
166  return counter;
167  }
