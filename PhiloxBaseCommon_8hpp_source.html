19  /** Common class for Philox family engines
20  *
21  * Relies on `PhiloxStateless` to provide the PRNG and adds state to handling the counting.
22  *
23  * @tparam TBackend device-dependent backend, specifies the array types
24  * @tparam TParams Philox algorithm parameters \sa PhiloxParams
25  * @tparam TImpl engine type implementation (CRTP)
26  *
27  * static const data members are transformed into functions, because GCC
28  * assumes types with static data members to be not mappable and makes not
29  * exception for constexpr ones. This is a valid interpretation of the
30  * OpenMP <= 4.5 standard. In OpenMP >= 5.0 types with any kind of static
31  * data member are mappable.
32  */

43  /** Advance the \a counter to the next state
44  *
45  * Increments the passed-in \a counter by one with a 128-bit carry.
46  *
47  * @param counter reference to the counter which is to be advanced
48  */
49  ALPAKA_FN_HOST_ACC void advanceCounter(Counter& counter)
50  {
51  counter[0]++;
52  /* 128-bit carry */
53  if(counter[0] == 0)
54  {
55  counter[1]++;
56  if(counter[1] == 0)
57  {
58  counter[2]++;
59  if(counter[2] == 0)
60  {
61  counter[3]++;
62  }
63  }
64  }
65  }
66 
67  /** Advance the internal state counter by \a offset N-vectors (N = counter size)
68  *
69  * Advances the internal value of this->state.counter
70  *
71  * @param offset number of N-vectors to skip
72  */
73  ALPAKA_FN_HOST_ACC void skip4(uint64_t offset)
74  {
75  Counter& counter = static_cast<TImpl*>(this)->state.counter;
76  Counter temp = counter;
77  counter[0] += low32Bits(offset);
78  counter[1] += high32Bits(offset) + (counter[0] < temp[0] ? 1 : 0);
79  counter[2] += (counter[0] < temp[1] ? 1u : 0u);
80  counter[3] += (counter[0] < temp[2] ? 1u : 0u);
81  }
82 
83  /** Advance the counter by the length of \a subsequence
84  *
85  * Advances the internal value of this->state.counter
86  *
87  * @param subsequence number of subsequences to skip
88  */
89  ALPAKA_FN_HOST_ACC void skipSubsequence(uint64_t subsequence)
90  {
91  Counter& counter = static_cast<TImpl*>(this)->state.counter;
92  Counter temp = counter;
93  counter[2] += low32Bits(subsequence);
94  counter[3] += high32Bits(subsequence) + (counter[2] < temp[2] ? 1 : 0);
95  }
96  };
