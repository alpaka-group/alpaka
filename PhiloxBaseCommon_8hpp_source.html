34  /** Common class for Philox family engines
35  *
36  * Checks the validity of passed-in parameters and calls the \a TBackend methods to perform N rounds of the
37  * Philox shuffle.
38  *
39  * @tparam TBackend device-dependent backend, specifies the array types
40  * @tparam TParams Philox algorithm parameters \sa PhiloxParams
41  * @tparam TImpl engine type implementation (CRTP)
42  */
43  template<typename TBackend, typename TParams, typename TImpl>
44  class PhiloxBaseCommon
45  : public TBackend
46  , public PhiloxConstants<TParams>
47  {
48  static unsigned const numRounds = TParams::rounds;
49  static unsigned const vectorSize = TParams::counterSize;
50  static unsigned const numberWidth = TParams::width;
51 
52  static_assert(numRounds > 0, "Number of Philox rounds must be > 0.");
53  static_assert(vectorSize % 2 == 0, "Philox counter size must be an even number.");
54  static_assert(vectorSize <= 16, "Philox SP network is not specified for sizes > 16.");
55  static_assert(numberWidth % 8 == 0, "Philox number width in bits must be a multiple of 8.");
56 
57  // static_assert(TWidth == 32 || TWidth == 64, "Philox implemented only for 32 and 64 bit numbers.");
58  static_assert(numberWidth == 32, "Philox implemented only for 32 bit numbers.");
59 
60  public:
61  using Counter = typename TBackend::Counter;
62  using Key = typename TBackend::Key;
63 
64  protected:
65  /** Single round of the Philox shuffle
66  *
67  * @param counter state of the counter
68  * @param key value of the key
69  * @return shuffled counter
70  */
71  ALPAKA_FN_HOST_ACC auto singleRound(Counter const& counter, Key const& key)
72  {
73  std::uint32_t H0, L0, H1, L1;
74  multiplyAndSplit64to32(counter[0], this->MULTIPLITER_4x32_0, H0, L0);
75  multiplyAndSplit64to32(counter[2], this->MULTIPLITER_4x32_1, H1, L1);
76  return Counter{H1 ^ counter[1] ^ key[0], L1, H0 ^ counter[3] ^ key[1], L0};
77  }
78 
79  /** Bump the \a key by the Weyl sequence step parameter
80  *
81  * @param key the key to be bumped
82  * @return the bumped key
83  */
84  ALPAKA_FN_HOST_ACC auto bumpKey(Key const& key)
85  {
86  return Key{key[0] + this->WEYL_32_0, key[1] + this->WEYL_32_1};
87  }
88 
89  /** Advance the \a counter to the next state
90  *
91  * Increments the passed-in \a counter by one with a 128-bit carry.
92  *
93  * @param counter reference to the counter which is to be advanced
94  */
95  ALPAKA_FN_HOST_ACC void advanceCounter(Counter& counter)
96  {
97  counter[0]++;
98  /* 128-bit carry */
99  if(counter[0] == 0)
100  {
101  counter[1]++;
102  if(counter[1] == 0)
103  {
104  counter[2]++;
105  if(counter[2] == 0)
106  {
107  counter[3]++;
108  }
109  }
110  }
111  }
112 
113  /** Advance the internal state counter by \a offset N-vectors (N = counter size)
114  *
115  * Advances the internal value of this->state.counter
116  *
117  * @param offset number of N-vectors to skip
118  */
119  ALPAKA_FN_HOST_ACC void skip4(uint64_t offset)
120  {
121  Counter& counter = static_cast<TImpl*>(this)->state.counter;
122  Counter temp = counter;
123  counter[0] += low32Bits(offset);
124  counter[1] += high32Bits(offset) + (counter[0] < temp[0] ? 1 : 0);
125  counter[2] += (counter[0] < temp[1] ? 1u : 0u);
126  counter[3] += (counter[0] < temp[2] ? 1u : 0u);
127  }
128 
129  /** Advance the counter by the length of \a subsequence
130  *
131  * Advances the internal value of this->state.counter
132  *
133  * @param subsequence number of subsequences to skip
134  */
135  ALPAKA_FN_HOST_ACC void skipSubsequence(uint64_t subsequence)
136  {
137  Counter& counter = static_cast<TImpl*>(this)->state.counter;
138  Counter temp = counter;
139  counter[2] += low32Bits(subsequence);
140  counter[3] += high32Bits(subsequence) + (counter[2] < temp[2] ? 1 : 0);
141  }
142 
143  /** Performs N rounds of the Philox shuffle
144  *
145  * @param counter_in initial state of the counter
146  * @param key_in initial state of the key
147  * @return result of the PRNG shuffle; has the same size as the counter
148  */
149  ALPAKA_FN_HOST_ACC auto nRounds(Counter const& counter_in, Key const& key_in) -> Counter
150  {
151  Key key{key_in};
152  Counter counter = singleRound(counter_in, key);
153 
154  // TODO: Consider unrolling the loop for performance
155  for(unsigned int n = 0; n < numRounds; ++n)
156  {
157  key = bumpKey(key);
158  counter = singleRound(counter, key);
159  }
160  // TODO: Should the key be returned as well??? i.e. should the original key be bumped?
161 
162  return counter;
163  }
164  };
