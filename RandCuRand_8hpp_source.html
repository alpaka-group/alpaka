143  //-----------------------------------------------------------------------------
144  template<
145  typename TGenerator>
146  __device__ auto operator()(
147  TGenerator & generator)
148  -> float
149  {
150  // (0.f, 1.0f]
151  float const fUniformRand(curand_uniform(&generator.m_State));
152  // NOTE: (1.0f - curand_uniform) does not work, because curand_uniform seems to return denormalized floats around 0.f.
153  // [0.f, 1.0f)
154  return fUniformRand * static_cast<float>( fUniformRand != 1.0f );
155  }
156  };
