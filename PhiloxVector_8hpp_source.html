35  /** Philox engine generating a vector of numbers
36  *
37  * This engine's operator() will return a vector of numbers corresponding to the full size of its counter.
38  * This is a convenience vs. memory size tradeoff since the user has to deal with the output array
39  * themselves, but the internal state comprises only of a single counter and a key.
40  *
41  * @tparam TAcc Accelerator type as defined in alpaka/acc
42  * @tparam TParams Basic parameters for the Philox algorithm
43  */
44  template<typename TAcc, typename TParams>
45  class PhiloxVector : public trait::PhiloxBaseTraits<TAcc, TParams, PhiloxVector<TAcc, TParams>>::Base
46  {
47  public:
48  /// Specialization for different TAcc backends
49  using Traits = trait::PhiloxBaseTraits<TAcc, TParams, PhiloxVector<TAcc, TParams>>;
50 
51  using Counter = typename Traits::Counter; ///< Backend-dependent Counter type
52  using Key = typename Traits::Key; ///< Backend-dependent Key type
53  using State = PhiloxStateVector<Counter, Key>; ///< Backend-dependent State type
54  template<typename TDistributionResultScalar>
55  using ResultContainer = typename Traits::template ResultContainer<TDistributionResultScalar>;
56 
57  State state;
58 
59  protected:
60  /** Get the next array of random numbers and advance internal state
61  *
62  * @return The next array of random numbers
63  */
64  ALPAKA_FN_HOST_ACC auto nextVector()
65  {
66  this->advanceCounter(state.counter);
67  return this->nRounds(state.counter, state.key);
68  }
69 
70  /** Skips the next \a offset vectors
71  *
72  * Unlike its counterpart in \a PhiloxSingle, this function advances the state in multiples of the
73  * counter size thus skipping the entire array of numbers.
74  */
75  ALPAKA_FN_HOST_ACC void skip(uint64_t offset)
76  {
77  this->skip4(offset);
78  }
79 
80  public:
81  /** Construct a new Philox engine with vector output
82  *
83  * @param seed Set the Philox generator key
84  * @param subsequence Select a subsequence of size 2^64
85  * @param offset Skip \a offset numbers form the start of the subsequence
86  */
87  ALPAKA_FN_HOST_ACC PhiloxVector(uint64_t seed = 0, uint64_t subsequence = 0, uint64_t offset = 0)
88  : state{{0, 0, 0, 0}, {low32Bits(seed), high32Bits(seed)}}
89  {
90  this->skipSubsequence(subsequence);
91  skip(offset);
92  nextVector();
93  }
94 
95  /** Get the next vector of random numbers
96  *
97  * @return The next vector of random numbers
98  */
99  ALPAKA_FN_HOST_ACC auto operator()()
100  {
101  return nextVector();
102  }
103  };
