41  //! The OMP5 device memory set trait specialization.
42  template<typename TDim>
43  struct CreateTaskMemset<TDim, DevOmp5>
44  {
45  template<typename TExtent, typename TViewFwd>
46  ALPAKA_FN_HOST static auto createTaskMemset(TViewFwd&& view, std::uint8_t const& byte, TExtent const& extent)
47  {
48  using TView = std::remove_reference_t<TViewFwd>;
49  using Idx = typename alpaka::trait::IdxType<TExtent>::type;
50  auto pitch = getPitchBytesVec(view);
51  auto byteExtent = getExtentVec(extent);
52  constexpr auto lastDim = TDim::value - 1;
53  byteExtent[lastDim] *= static_cast<Idx>(sizeof(Elem<TView>));
54 
55  if(pitch[0] == 0)
56  {
57  return createTaskKernel<AccOmp5<TDim, Idx>>(
58  WorkDivMembers<TDim, Idx>(
59  Vec<TDim, Idx>::zeros(),
60  Vec<TDim, Idx>::zeros(),
61  Vec<TDim, Idx>::zeros()),
62  MemSetKernel(),
63  byte,
64  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view)),
65  byteExtent,
66  pitch); // NOP if size is zero
67  }
68 
69 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
70  std::cout << "Set TDim=" << TDim::value << " pitch=" << pitch << " byteExtent=" << byteExtent << std::endl;
71 # endif
72  auto elementsPerThread = Vec<TDim, Idx>::all(static_cast<Idx>(1u));
73  elementsPerThread[lastDim] = 4;
74  // Let alpaka calculate good block and grid sizes given our full problem extent
75  WorkDivMembers<TDim, Idx> const workDiv(getValidWorkDiv<AccOmp5<TDim, Idx>>(
76  getDev(view),
77  byteExtent,
78  elementsPerThread,
79  false,
80  alpaka::GridBlockExtentSubDivRestrictions::Unrestricted));
81  return createTaskKernel<AccOmp5<TDim, Idx>>(
82  workDiv,
83  MemSetKernel(),
84  byte,
85  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view)),
86  byteExtent,
87  pitch);
88  }
89  };
