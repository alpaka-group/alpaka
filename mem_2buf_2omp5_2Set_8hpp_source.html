40  //! The OMP5 device memory set trait specialization.
41  template<typename TDim>
42  struct CreateTaskMemset<TDim, DevOmp5>
43  {
44  template<typename TExtent, typename TView>
45  ALPAKA_FN_HOST static auto createTaskMemset(TView& view, std::uint8_t const& byte, TExtent const& extent)
46  {
47  using Idx = typename alpaka::traits::IdxType<TExtent>::type;
48  auto pitch = getPitchBytesVec(view);
49  auto byteExtent = extent::getExtentVec(extent);
50  constexpr auto lastDim = TDim::value - 1;
51  byteExtent[lastDim] *= static_cast<Idx>(sizeof(Elem<TView>));
52 
53  if(pitch[0] == 0)
54  {
55  return createTaskKernel<AccOmp5<TDim, Idx>>(
56  WorkDivMembers<TDim, Idx>(
57  Vec<TDim, Idx>::zeros(),
58  Vec<TDim, Idx>::zeros(),
59  Vec<TDim, Idx>::zeros()),
60  MemSetKernel(),
61  byte,
62  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view)),
63  byteExtent,
64  pitch); // NOP if size is zero
65  }
66 
67 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
68  std::cout << "Set TDim=" << TDim::value << " pitch=" << pitch << " byteExtent=" << byteExtent
69  << std::endl;
70 # endif
71  auto elementsPerThread = Vec<TDim, Idx>::all(static_cast<Idx>(1u));
72  elementsPerThread[lastDim] = 4;
73  // Let alpaka calculate good block and grid sizes given our full problem extent
74  WorkDivMembers<TDim, Idx> const workDiv(getValidWorkDiv<AccOmp5<TDim, Idx>>(
75  getDev(view),
76  byteExtent,
77  elementsPerThread,
78  false,
79  alpaka::GridBlockExtentSubDivRestrictions::Unrestricted));
80  return createTaskKernel<AccOmp5<TDim, Idx>>(
81  workDiv,
82  MemSetKernel(),
83  byte,
84  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view)),
85  byteExtent,
86  pitch);
87  }
88  };
